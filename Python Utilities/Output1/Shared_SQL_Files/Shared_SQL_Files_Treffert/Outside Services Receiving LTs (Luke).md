```sql
USE LIVE
GO

--SELECT * FROM SHOP_RESOURCE WHERE ([STATUS] IS NULL OR [STATUS] != 'O') AND USER_8 IN ('RECEIVING', 'INSPECTION')
--ID = 'P2M1-REC'

DECLARE @CLOSEDATE DATE = '1/1/2020';

WITH CTE_CONTRACTOR AS
(SELECT WORKORDER_BASE_ID
	, WORKORDER_LOT_ID
	, WORKORDER_SPLIT_ID
	, WORKORDER_SUB_ID
	, SEQUENCE_NO
FROM OPERATION O2
WHERE RESOURCE_ID = 'CONTRACTOR' 
	AND WORKORDER_TYPE = 'W'
	AND CLOSE_DATE >= @CLOSEDATE 
)
--SELECT * FROM CTE_CONTRACTOR WHERE WORKORDER_BASE_ID = '1538606'

SELECT O.WORKORDER_BASE_ID
	, O.WORKORDER_LOT_ID
	, O.WORKORDER_SPLIT_ID
	, O.WORKORDER_SUB_ID
	, O.SEQUENCE_NO
	, O.RESOURCE_ID
	, SR.USER_8
	, O.CLOSE_DATE
	--, LT.CLOCK_IN
	--, LT.CLOCK_OUT
FROM OPERATION O
INNER JOIN SHOP_RESOURCE SR
	ON SR.ID = O.RESOURCE_ID
	AND SR.USER_8 IN ('RECEIVING', 'INSPECTION')
INNER JOIN CTE_CONTRACTOR CTE
	ON CTE.WORKORDER_BASE_ID = O.WORKORDER_BASE_ID
	AND CTE.WORKORDER_LOT_ID = O.WORKORDER_LOT_ID
	AND CTE.WORKORDER_SPLIT_ID = O.WORKORDER_SPLIT_ID
	AND CTE.WORKORDER_SUB_ID = O.WORKORDER_SUB_ID
	AND CTE.SEQUENCE_NO <= O.SEQUENCE_NO
--LEFT OUTER JOIN LABOR_TICKET LT
--	ON LT.WORKORDER_TYPE = O.WORKORDER_TYPE
--	AND LT.WORKORDER_BASE_ID = O.WORKORDER_BASE_ID
--	AND LT.WORKORDER_LOT_ID = O.WORKORDER_LOT_ID
--	AND LT.WORKORDER_SPLIT_ID = O.WORKORDER_SPLIT_ID
--	AND LT.WORKORDER_SUB_ID = O.WORKORDER_SUB_ID
--	AND LT.OPERATION_SEQ_NO >= O.SEQUENCE_NO
ORDER BY O.WORKORDER_BASE_ID
	, O.WORKORDER_LOT_ID
	, O.WORKORDER_SPLIT_ID
	, O.WORKORDER_SUB_ID
	, O.SEQUENCE_NO
	, O.RESOURCE_ID
```
