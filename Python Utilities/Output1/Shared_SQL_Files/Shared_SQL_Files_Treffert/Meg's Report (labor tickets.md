SELECT LT.WORKORDER_BASE_ID
	, LT.EMPLOYEE_ID
	, LT.HOURS_WORKED
	, EE.FIRST_NAME
	, EE.LAST_NAME
	, LT.SHIFT_DATE
	, LT.ACT_CLOCK_IN
	, LT.ACT_CLOCK_OUT
	, WO.PART_ID
	, LT.RESOURCE_ID
FROM LABOR_TICKET LT
	INNER JOIN WORK_ORDER WO
		ON LT.WORKORDER_BASE_ID = WO.BASE_ID
	INNER JOIN EMPLOYEE EE
		ON LT.EMPLOYEE_ID = EE.ID
WHERE LEFT(WO.PART_ID, 4) IN ('100T', '112A', '112U', '112W', '113W', '114T', '115A', '116U', '140U', '142U', '143T', '143W', '144A', '144U', '144W', '146T', '146U', '149A', '151W', '152W', '212Z', '213A', '215Z', '218T', '218Z', '232T', '255U', '256T', '272T', '272U', '287A', '287T', '287U', '294W', '313A', '313T', '313W', '315Z', '317Z', '343A', '343W', '344A', '345A', '412A', '413U', '417A', '433W', '472W', '491Z', '492Z', '658Z', '678Z', '710Z', '711Z', '715Z', '832Z', '843Z')
	OR left(WO.PART_ID, 3) IN ('65B', '69B')
ORDER BY WO.PART_ID
