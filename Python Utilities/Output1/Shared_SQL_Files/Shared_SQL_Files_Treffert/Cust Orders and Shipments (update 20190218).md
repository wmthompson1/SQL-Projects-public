# Cust Orders and Shipments (update 20190218)

```sql
use DataWarehouse
go

DECLARE @StartDate DATE = DATEADD(YEAR,-1,DATEADD(DAY,1,(EOMONTH(DATEADD(MONTH,-1,GETDATE())))));
 
WITH CTE_BASE AS
(SELECT A.*
	, CASE
		WHEN DSL.SUPPLY_TYPE = 'CP' THEN 'CP'
		ELSE 'WO'
	END AS SUPPLY_TYPE
	, WO.BASE_ID
	, WO.LOT_ID
	, WO.SPLIT_ID
	, WO.SUB_ID
FROM 
(SELECT CO.CUSTOMER_ID 
	, CO.ID AS CUST_ORDER_ID
	, COL.LINE_NO AS COL_LINE_NO
	, CO.ORDER_DATE
	, CO.CREATE_DATE
	, ISNULL(COL.DESIRED_SHIP_DATE, CO.DESIRED_SHIP_DATE) AS DESIRED_SHIP_DATE
	, CO.[STATUS] AS CO_STATUS
	, COL.SERVICE_CHARGE_ID
	, CASE
		WHEN COL.SERVICE_CHARGE_ID IS NOT NULL THEN COL.MISC_REFERENCE
		ELSE ''
	END AS CHARGE_TYPE
	, CASE
		WHEN COL.SERVICE_CHARGE_ID IS NULL THEN COL.MISC_REFERENCE
		ELSE ''
	END AS CUST_PART_DESC
	, COL.PART_ID
	, P.[DESCRIPTION]
	, P.PRODUCT_CODE AS PART_PROD_CODE
	, P.COMMODITY_CODE AS PART_COMM_CODE
	, P.PLANNER_USER_ID
	, P.BUYER_USER_ID
	, COL.LINE_STATUS AS SHIP_SHORT
	, COL.ORDER_QTY
	, COL.TOTAL_SHIPPED_QTY
	, COL.UNIT_PRICE
	, UDF.STRING_VAL AS BUS_STREAM
	, UDF1.STRING_VAL AS MATERIAL_TYPE
	, S.PACKLIST_ID
	, SL.LINE_NO AS PACKLIST_LINE_NO
	, CASE
		WHEN LTRIM(RTRIM(RIGHT(R.INVOICE_ID,1))) = 'M' THEN 0
		ELSE SHIPPED_QTY
	END AS SHIPPED_QTY
	, S.SHIPPED_DATE AS SHIPPED_DATE
	, R.INVOICE_DATE
	, RL.INVOICE_ID
	, RL.LINE_NO AS INVOICE_LINE
	, RL.AMOUNT AS INVOICE_AMT
	, CASE
		WHEN LTRIM(RTRIM(RIGHT(R.INVOICE_ID,1))) = 'M' THEN 0
		ELSE RL.QTY
	END AS INVOICE_QTY
FROM CUST_ORDER_LINE COL
	INNER JOIN CUSTOMER_ORDER CO
		ON CO.ID = COL.CUST_ORDER_ID
	LEFT OUTER JOIN PART P
		ON COL.PART_ID = P.ID
	LEFT OUTER JOIN SHIPPER_LINE SL
		ON COL.CUST_ORDER_ID = SL.CUST_ORDER_ID
		AND COL.LINE_NO = SL.CUST_ORDER_LINE_NO
		AND SL.SHIPPED_QTY <> 0
	LEFT OUTER JOIN SHIPPER S
		ON SL.PACKLIST_ID = S.PACKLIST_ID
	LEFT OUTER JOIN RECEIVABLE_LINE RL
		ON SL.PACKLIST_ID = RL.PACKLIST_ID
		AND SL.LINE_NO = RL.PACKLIST_LINE_NO
	LEFT OUTER JOIN RECEIVABLE R
		ON RL.INVOICE_ID = R.INVOICE_ID
	LEFT OUTER JOIN USER_DEF_FIELDS UDF
		ON UDF.DOCUMENT_ID = CO.CUSTOMER_ID
		AND PROGRAM_ID = 'VMCUSMNT'
		AND UDF.ID = 'UDF-0000072'
	LEFT OUTER JOIN USER_DEF_FIELDS UDF1
		ON UDF1.DOCUMENT_ID = P.ID
		AND UDF1.PROGRAM_ID = 'VMPRTMNT'
		AND UDF1.ID = 'UDF-0000035'
WHERE CO.[STATUS] IN ('H', 'R', 'C')
	AND COALESCE(R.INVOICE_DATE, S.SHIPPED_DATE, COL.DESIRED_SHIP_DATE, CO.DESIRED_SHIP_DATE) >= @StartDate
) A
LEFT OUTER  JOIN DEMAND_SUPPLY_LINK DSL
	ON DSL.DEMAND_TYPE = 'CO'
	AND DSL.DEMAND_BASE_ID = A.CUST_ORDER_ID
	AND DSL.DEMAND_SEQ_NO = A.COL_LINE_NO
LEFT OUTER JOIN WORK_ORDER WO
	ON DSL.SUPPLY_TYPE = 'WO' 
	AND DSL.SUPPLY_BASE_ID = WO.BASE_ID
	AND DSL.SUPPLY_LOT_ID = WO.LOT_ID
	AND DSL.SUPPLY_SPLIT_ID = WO.SPLIT_ID
	AND DSL.SUPPLY_SUB_ID = WO.SUB_ID 
	AND WO.[TYPE] = 'W'
)

, CTE_REMAINING AS
(SELECT *
FROM CTE_BASE
where CO_STATUS = 'R'
	and SHIP_SHORT <> 'C'
	and ORDER_QTY > TOTAL_SHIPPED_QTY
	and TOTAL_SHIPPED_QTY > 0
	--and DESIRED_SHIP_DATE <> ISNULL(INVOICE_DATE, SHIPPED_DATE)
)
--SELECT * FROM CTE_REMAINING

SELECT 'BASE' AS CTE
	, CUSTOMER_ID
	, CUST_ORDER_ID
	, COL_LINE_NO
	, CREATE_DATE
	, ORDER_DATE
	, DESIRED_SHIP_DATE
	, CO_STATUS
	, SERVICE_CHARGE_ID
	, CHARGE_TYPE
	, CUST_PART_DESC
	, PART_ID
	, [DESCRIPTION]
	, PART_PROD_CODE
	, PART_COMM_CODE
	, PLANNER_USER_ID
	, BUYER_USER_ID
	, SHIP_SHORT
	, ORDER_QTY
	, TOTAL_SHIPPED_QTY
	, UNIT_PRICE
	, BUS_STREAM
	, MATERIAL_TYPE
	, PACKLIST_ID
	, PACKLIST_LINE_NO
	, SHIPPED_QTY
	, SHIPPED_DATE
	, INVOICE_DATE
	, INVOICE_ID
	, INVOICE_LINE
	, INVOICE_AMT
	, INVOICE_QTY
	, SUPPLY_TYPE
	, BASE_ID
	, LOT_ID
	, SPLIT_ID
	, SUB_ID
FROM CTE_BASE
UNION ALL
SELECT 'REMAINING' AS CTE
	, CUSTOMER_ID
	, CUST_ORDER_ID
	, COL_LINE_NO
	, CREATE_DATE
	, ORDER_DATE
	, DESIRED_SHIP_DATE
	, CO_STATUS
	, SERVICE_CHARGE_ID
	, CHARGE_TYPE
	, CUST_PART_DESC
	, PART_ID
	, [DESCRIPTION]
	, PART_PROD_CODE
	, PART_COMM_CODE
	, PLANNER_USER_ID
	, BUYER_USER_ID
	, SHIP_SHORT
	, ORDER_QTY
	, TOTAL_SHIPPED_QTY
	, UNIT_PRICE
	, BUS_STREAM
	, MATERIAL_TYPE
	, NULL as PACKLIST_ID
	, NULL as PACKLIST_LINE_NO
	, NULL as SHIPPED_QTY
	, NULL as SHIPPED_DATE
	, NULL as INVOICE_DATE
	, NULL as INVOICE_ID
	, NULL as INVOICE_LINE
	, NULL as INVOICE_AMT
	, NULL as INVOICE_QTY
	, SUPPLY_TYPE
	, BASE_ID
	, LOT_ID
	, SPLIT_ID
	, SUB_ID
FROM CTE_REMAINING
GROUP BY CUSTOMER_ID
	, CUST_ORDER_ID
	, COL_LINE_NO
	, DESIRED_SHIP_DATE
	, CREATE_DATE
	, ORDER_DATE
	, CO_STATUS
	, SERVICE_CHARGE_ID
	, CHARGE_TYPE
	, PART_ID
	, [DESCRIPTION]
	, CUST_PART_DESC
	, PART_PROD_CODE
	, PART_COMM_CODE
	, PLANNER_USER_ID
	, BUYER_USER_ID
	, SHIP_SHORT
	, ORDER_QTY
	, TOTAL_SHIPPED_QTY
	, UNIT_PRICE
	, BUS_STREAM
	, MATERIAL_TYPE
	, SUPPLY_TYPE
	, BASE_ID
	, LOT_ID
	, SPLIT_ID
	, SUB_ID
```
