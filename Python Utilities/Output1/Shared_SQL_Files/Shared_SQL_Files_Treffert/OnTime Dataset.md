# OnTime Dataset

```sql
USE DataWarehouse
GO

DECLARE @StartDate DATETIME = '1/1/2018'
;
WITH CTE_INITIAL AS
--select *
(SELECT CO.CUSTOMER_ID 
	, CASE
		WHEN CO.CUSTOMER_ID = 'BOE609' THEN 'CMP'
		ELSE 'Other'
	END AS CMP
	, COL.CUST_ORDER_ID
	, COL.LINE_NO AS COL_LINE_NO
	, ISNULL(COL.DESIRED_SHIP_DATE, CO.DESIRED_SHIP_DATE) AS DESIRED_SHIP_DATE
	, CO.ORDER_DATE
	, CO.[STATUS] AS CO_STATUS
	, COL.SERVICE_CHARGE_ID
	, CASE 
		WHEN WO.BASE_ID LIKE ('%R%') THEN 'Rework' 
		WHEN COL.PART_ID IS NOT NULL AND COL.TOTAL_SHIPPED_QTY <= 0 
			THEN 'Return' 
		WHEN WO.BASE_ID IS NOT NULL THEN 'Parts' 
		WHEN COL.MISC_REFERENCE LIKE 'Minimum%' 
			OR COL.SERVICE_CHARGE_ID in ('L', 'LP')
			THEN 'Min' 
		WHEN COL.MISC_REFERENCE LIKE ('NRE%') 
			OR COL.MISC_REFERENCE LIKE ('NON RECUR%') 
			OR COL.MISC_REFERENCE LIKE ('NON_RECUR%') 
			OR COL.MISC_REFERENCE LIKE ('COST%') 	
			OR COL.SERVICE_CHARGE_ID = 'NRE' 
			THEN 'NRE' 
		WHEN COL.MISC_REFERENCE LIKE 'Exp%'
			OR COL.SERVICE_CHARGE_ID = 'EX' 
			THEN 'Exp' 
		WHEN (COL.MISC_REFERENCE LIKE ('%pene%') OR COL.MISC_REFERENCE LIKE ('%pena%'))
			AND (COL.TOTAL_AMT_SHIPPED < 0 OR COL.TOTAL_AMT_ORDERED < 0)	
			THEN 'Pen Credit' 
		WHEN (COL.TOTAL_AMT_SHIPPED < 0 OR COL.TOTAL_AMT_ORDERED < 0) 
			THEN 'Credit' 
		WHEN COL.SERVICE_CHARGE_ID IN ('E', 'S')
			THEN 'Misc Charge'
	END AS [Type] 
	, case
		when o.RESOURCE_ID in ('pm', 'BAL-pm', 'AUB-pm', 'AUB-assyPM') 
			then 1
		else 0
	end as PMCount
	, case
		when o.resource_ID = 'BAL-Ship' and o.operation_type = 'REVISION' 
			then 1
		when o.resource_ID like '%-Ship' OR O.RESOURCE_ID LIKE 'rwec' 
			then 0
		else 1
	end as DivCount
	, CASE
		WHEN COL.SERVICE_CHARGE_ID IS NOT NULL THEN COL.MISC_REFERENCE
		ELSE ''
	END AS CHARGE_TYPE
	, COL.PART_ID
	, CASE
		WHEN COL.LINE_STATUS = 'C'
			THEN 'YES'
		ELSE 'NO'
	END AS SHIP_SHORT
	, COL.ORDER_QTY
	, COL.TOTAL_SHIPPED_QTY
	, COL.UNIT_PRICE
	, UDF.STRING_VAL AS BUS_STREAM
	, UDF1.STRING_VAL AS MATERIAL_TYPE
	, S.PACKLIST_ID
	, S.CREATE_DATE AS SHIP_CREATE_DATE
	, S.SHIPPED_DATE AS SHIPPED_DATE
	--, RL.INVOICE_ID
	--, R.INVOICE_DATE
	--, RL.LINE_NO AS INVOICE_LINE
	--, RL.AMOUNT AS INVOICE_AMT
	, CASE
		WHEN DSL.SUPPLY_TYPE = 'CP' THEN 'CP'
		ELSE 'WO'
	END AS SUPPLY_TYPE
	, WO.BASE_ID
	, WO.LOT_ID
	, WO.SPLIT_ID
	, WO.SUB_ID
	, WO.[STATUS] AS WO_STATUS
	, WO.CREATE_DATE AS WO_CREATE_DATE
	, WO.DESIRED_WANT_DATE
	, WO.CLOSE_DATE AS WO_CLOSE_DATE
	, O.SEQUENCE_NO AS OP_SEQ_NO
	, SR.ID AS RESOURCE_ID
	, O.OPERATION_TYPE
	, O.[STATUS] AS OP_STATUS
	, SR.USER_1 AS SKILLS_TYPE
FROM CUST_ORDER_LINE COL
	INNER JOIN CUSTOMER_ORDER CO
		ON CO.ID = COL.CUST_ORDER_ID
	INNER JOIN CUSTOMER C
		ON CO.CUSTOMER_ID = C.ID
		AND C.USER_9 NOT LIKE 'o%'
	LEFT OUTER JOIN SHIPPER_LINE SL
		ON COL.CUST_ORDER_ID = SL.CUST_ORDER_ID
		AND COL.LINE_NO = SL.CUST_ORDER_LINE_NO
		AND SL.SHIPPED_QTY <> 0
	LEFT OUTER JOIN SHIPPER S
		ON SL.PACKLIST_ID = S.PACKLIST_ID
	--LEFT OUTER JOIN RECEIVABLE_LINE RL
	--	ON SL.PACKLIST_ID = RL.PACKLIST_ID
	--	AND SL.LINE_NO = RL.PACKLIST_LINE_NO
	--LEFT OUTER JOIN RECEIVABLE R
	--	ON RL.INVOICE_ID = R.INVOICE_ID
	LEFT OUTER JOIN USER_DEF_FIELDS UDF
		ON UDF.DOCUMENT_ID = CO.CUSTOMER_ID
		AND PROGRAM_ID = 'VMCUSMNT'
		AND UDF.ID = 'UDF-0000072'
	LEFT OUTER JOIN USER_DEF_FIELDS UDF1
		ON UDF1.DOCUMENT_ID = COL.PART_ID
		AND UDF1.PROGRAM_ID = 'VMPRTMNT'
		AND UDF1.ID = 'UDF-0000035'
	LEFT OUTER  JOIN DEMAND_SUPPLY_LINK DSL
		ON DSL.DEMAND_TYPE = 'CO'
		AND DSL.DEMAND_BASE_ID = COL.CUST_ORDER_ID
		AND DSL.DEMAND_SEQ_NO = COL.LINE_NO
	LEFT OUTER JOIN WORK_ORDER WO
		ON DSL.SUPPLY_TYPE = 'WO' 
		AND DSL.SUPPLY_BASE_ID = WO.BASE_ID
		AND DSL.SUPPLY_LOT_ID = WO.LOT_ID
		AND DSL.SUPPLY_SPLIT_ID = WO.SPLIT_ID
		AND DSL.SUPPLY_SUB_ID = WO.SUB_ID 
		AND WO.[TYPE] = 'W'
	LEFT OUTER JOIN OPERATION O
		ON  WO.[TYPE] = O.WORKORDER_TYPE
		AND WO.BASE_ID = O.WORKORDER_BASE_ID 
		AND WO.LOT_ID = O.WORKORDER_LOT_ID 
		AND WO.SPLIT_ID = O.WORKORDER_SPLIT_ID
		AND WO.SUB_ID = O.WORKORDER_SUB_ID
	LEFT OUTER JOIN SHOP_RESOURCE SR 
		ON O.RESOURCE_ID = SR.ID
WHERE CO.[STATUS] IN ('H', 'R', 'C')
	AND WO.[STATUS] in ('C', 'R') 
	AND NOT (WO.BASE_ID LIKE '%R' OR WO.BASE_ID LIKE '%RW' OR WO.BASE_ID LIKE 'R%') 	
	AND COALESCE(COL.DESIRED_SHIP_DATE, CO.DESIRED_SHIP_DATE) >= @StartDate
	AND COL.PART_ID NOT LIKE 'UP%'
)
-- SELECT * FROM CTE_INITIAL

SELECT CUSTOMER_ID 
, CMP
, CUST_ORDER_ID
, COL_LINE_NO
, DESIRED_SHIP_DATE
, CO_STATUS
, SERVICE_CHARGE_ID
, [Type] 
, CHARGE_TYPE
, PART_ID
, SHIP_SHORT
, ORDER_QTY
, TOTAL_SHIPPED_QTY
, UNIT_PRICE
, BUS_STREAM
, MATERIAL_TYPE
, SUPPLY_TYPE
, BASE_ID
, LOT_ID
, SPLIT_ID
, SUB_ID
, WO_STATUS
, WO_CREATE_DATE
, DESIRED_WANT_DATE
, WO_CLOSE_DATE
, OP_SEQ_NO
, RESOURCE_ID
, OPERATION_TYPE
, OP_STATUS
, SKILLS_TYPE
, ORDER_DATE
, MAX(SHIPPED_DATE) AS LATEST_SHIP_DATE
, CASE
	WHEN max(SHIPPED_DATE) > MAX(DESIRED_SHIP_DATE) THEN 'No'
	ELSE 'Yes'
END AS ON_TIME,
--, TURN_DAYS
	(SELECT SUM(PMCOUNT) 
		FROM CTE_INITIAL PM 
		WHERE PM.BASE_ID = T1.BASE_ID
	) PMCOUNT,
	(SELECT SUM(DIVCOUNT) 
		FROM CTE_INITIAL A  
		WHERE SKILLS_TYPE = 'A'  
			AND A.BASE_ID = T1.BASE_ID
	) ACOUNT,
	(SELECT SUM(DIVCOUNT) 
		FROM CTE_INITIAL B  
		WHERE SKILLS_TYPE = 'B'  
		AND  B.BASE_ID = T1.BASE_ID
	) BCOUNT,
	(SELECT SUM(DIVCOUNT) 
		FROM CTE_INITIAL M  
		WHERE SKILLS_TYPE = 'M'  
			AND M.BASE_ID = T1.BASE_ID
	) MCOUNT,
	(SELECT SUM(DIVCOUNT) 
		FROM CTE_INITIAL O  
		WHERE SKILLS_TYPE = 'O'  
			AND O.BASE_ID = T1.BASE_ID
	) OCOUNT,
	(SELECT SUM(DIVCOUNT) 
		FROM CTE_INITIAL OS 
		WHERE SKILLS_TYPE = 'OS' 
			AND OS.BASE_ID = T1.BASE_ID
	) OSCOUNT
FROM CTE_INITIAL CTE
GROUP BY CUSTOMER_ID 
	, CMP
	, CUST_ORDER_ID
	, COL_LINE_NO
	, DESIRED_SHIP_DATE
	, CO_STATUS
	, SERVICE_CHARGE_ID
	, [Type] 
	, CHARGE_TYPE
	, PART_ID
	, SHIP_SHORT
	, ORDER_QTY
	, TOTAL_SHIPPED_QTY
	, UNIT_PRICE
	, BUS_STREAM
	, MATERIAL_TYPE
	, SUPPLY_TYPE
	, BASE_ID
	, LOT_ID
	, SPLIT_ID
	, SUB_ID
	, WO_STATUS
	, WO_CREATE_DATE
	, DESIRED_WANT_DATE
	, WO_CLOSE_DATE
	, OP_SEQ_NO
	, RESOURCE_ID
	, OPERATION_TYPE
	, OP_STATUS
	, SKILLS_TYPE
	, ORDER_DATE
```
