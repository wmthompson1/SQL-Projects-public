# Operation to Labor Ticket Information

```sql
use datawarehouse
go

--declare @BusinessUnit nvarchar(5) = 'FIN',
--@Department nvarchar(20) = 'MASKING',
--@PlantLocation nvarchar(5) = 'AP1',
--@TIMEFRAME nvarchar(20) = 'LAST WEEK'

set datefirst 1;

SELECT O.WORKORDER_BASE_ID
	, O.WORKORDER_LOT_ID
	, O.WORKORDER_SPLIT_ID
	, O.WORKORDER_SUB_ID
	, O.WORKORDER_TYPE
	, O.SEQUENCE_NO
	, O.RESOURCE_ID
	, O.OPERATION_TYPE
	, O.CLOSE_DATE
	, SR.USER_1 AS BUSINESSUNIT
	, SR.USER_2 AS DEPARTMENT
	, SR.USER_3 AS SUBDEPT
	, SR.USER_4 AS SUPERVISOR
	, SR.USER_5 AS FACILITY
	, DATEPART(YYYY, O.CLOSE_DATE) AS [YEAR]
	, DATEPART(MM, O.CLOSE_DATE) AS [MONTH]
	, DATEPART(QQ, O.CLOSE_DATE) AS [QUARTER]
	, DATEPART(WW, O.CLOSE_DATE) AS [WEEK]
	, CASE
		WHEN SR.USER_1 = 'A' THEN 'FIN'
		WHEN SR.USER_1 = 'M' THEN 'MFG'
		WHEN SR.USER_1 = 'O' THEN 'OUTREACH'
		WHEN SR.USER_1 = 'B' THEN 'BALLARD'
		WHEN SR.USER_1 = 'C' THEN 'CLOSED'
	END AS BUSUNIT
	--, LT.WORKORDER_BASE_ID
	--, LT.WORKORDER_LOT_ID
	--, LT.WORKORDER_SPLIT_ID
	--, LT.WORKORDER_SUB_ID
	--, LT.OPERATION_SEQ_NO
	--, lt.RESOURCE_ID
	, lt.CLOCK_IN
	, lt.CLOCK_OUT
	, LT.HOURS_WORKED
	, lt.EMPLOYEE_ID
	, E.FIRST_NAME
	, E.LAST_NAME
FROM OPERATION O
LEFT OUTER JOIN SHOP_RESOURCE_SITE SR
	ON O.RESOURCE_ID = SR.RESOURCE_ID
LEFT OUTER JOIN LABOR_TICKET LT
	ON LT.WORKORDER_TYPE = O.WORKORDER_TYPE
	AND LT.WORKORDER_BASE_ID = O.WORKORDER_BASE_ID
	AND LT.WORKORDER_LOT_ID = O.WORKORDER_LOT_ID
	AND LT.WORKORDER_SPLIT_ID = O.WORKORDER_SPLIT_ID
	AND LT.WORKORDER_SUB_ID = O.WORKORDER_SUB_ID
	AND LT.OPERATION_SEQ_NO = O.SEQUENCE_NO
Left outer Join EMPLOYEE E
	on  E.ID = lt.EMPLOYEE_ID
WHERE O.CLOSE_DATE >= GETDATE() - 365
AND O.CLOSE_DATE IS NOT NULL
AND O.[STATUS] = 'C'
```
