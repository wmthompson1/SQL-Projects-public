; WITH CTE_ALL AS
--> PULL ALL OPERATIONS FOR THE MASTERS
(SELECT O.WORKORDER_TYPE
	, O.WORKORDER_BASE_ID
	, O.WORKORDER_LOT_ID
	, O.WORKORDER_SPLIT_ID
	, O.WORKORDER_SUB_ID
	, O.SEQUENCE_NO
	, O.RESOURCE_ID
	, O.OPERATION_TYPE

	AND O.WORKORDER_SPLIT_ID = U.SPLIT_ID
	AND O.WORKORDER_SUB_ID = U.SUB_ID

	, WORKORDER_SPLIT_ID
	, WORKORDER_SUB_ID

FROM CTE_ALL
)

	, A.WORKORDER_SUB_ID

	, A.NEXT_OP

	, O.OPERATION_TYPE
FROM CTE_NEXT A

	ON A.WORKORDER_TYPE = U.[TYPE]
	AND A.WORKORDER_BASE_ID = U.BASE_ID

	AND A.WORKORDER_SPLIT_ID = U.SPLIT_ID
	AND A.WORKORDER_SUB_ID = U.SUB_ID
	AND A.SEQUENCE_NO = U.SEQ_NO
	AND A.RESOURCE_ID = U.RESOURCE_ID
	AND A.OPERATION_TYPE = U.OP_TYPE
LEFT OUTER JOIN OPERATION O
	ON A.WORKORDER_TYPE = O.WORKORDER_TYPE
	AND A.WORKORDER_BASE_ID = O.WORKORDER_BASE_ID
	AND A.WORKORDER_LOT_ID = O.WORKORDER_LOT_ID
	AND A.WORKORDER_SPLIT_ID = O.WORKORDER_SPLIT_ID
	AND A.WORKORDER_SUB_ID = O.WORKORDER_SUB_ID
	AND A.NEXT_OP = O.SEQUENCE_NO
