--> TABLE REFERENCES
SELECT *
FROM WORK_ORDER
INNER JOIN OPERATION
	ON WORK_ORDER.BASE_ID = OPERATION.WORKORDER_BASE_ID
	AND WORK_ORDER.LOT_ID = OPERATION.WORKORDER_LOT_ID
	AND WORK_ORDER.SPLIT_ID = OPERATION.WORKORDER_SPLIT_ID
	AND WORK_ORDER.SUB_ID = OPERATION.WORKORDER_SUB_ID
	AND WORK_ORDER.[TYPE] = OPERATION.WORKORDER_TYPE
WHERE WORK_ORDER.BASE_ID  = '1653308'

SELECT *
FROM WORK_ORDER WO
INNER JOIN OPERATION O
	ON WO.BASE_ID = O.WORKORDER_BASE_ID
	AND WO.LOT_ID = O.WORKORDER_LOT_ID
	and WO.SPLIT_ID = O.WORKORDER_SPLIT_ID
	AND wo.SUB_ID = O.WORKORDER_SUB_ID
	AND wo.[TYPE] = O.WORKORDER_TYPE
WHERE WO.BASE_ID  = '1653308'

SELECT *
FROM WORK_ORDER WO
INNER JOIN OPERATION B
	ON WO.BASE_ID = B.WORKORDER_BASE_ID
	AND WO.LOT_ID = B.WORKORDER_LOT_ID
	AND WO.SPLIT_ID = B.WORKORDER_SPLIT_ID
	AND WO.SUB_ID = B.WORKORDER_SUB_ID
	AND WO.[TYPE] = B.WORKORDER_TYPE
WHERE WO.BASE_ID  = '1653308'
	AND B.[STATUS] = 'R'
