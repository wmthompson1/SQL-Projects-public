SET NOCOUNT ON;

DECLARE @STARTDATE DATE 
, @ENDDATE DATE = CAST(DATEADD(dd, -DAY(GETDATE()), GETDATE()) AS DATE)

SELECT @STARTDATE = CAST(DATEADD(YEAR, -1, @ENDDATE) AS DATE)
--TRUNCATE TABLE Datamart.dbo.SK_SUPPLIER_PERFORMANCE_DETAILS;

--GET IQM NCM data
--contractor operation;  Has work order information to sequence
; WITH CTE_IQM AS
(SELECT NCM_ID
	, NCM.PRODUCT_REC_ID
	, DSL.SUPPLY_BASE_ID AS SupplierPO
	, DSL.SUPPLY_SEQ_NO AS SupplierPO_LINE
	, NCM.LOT_SIZE
	, NCM.ACCEPTED
	, NCM.REJECTED
	, NCM.NCM_DATE
FROM VQ_NONCONFORMANCE NCM
INNER JOIN VQ_PRODUCT P
	ON P.RECORD_ID = NCM.PRODUCT_REC_ID
INNER JOIN VQ_PRODUCTION_ORDER PO
	ON PO.RECORD_ID = NCM.PROD_ORDER_REC_ID
	AND PO.ALT_ID_1 = 'W'
INNER JOIN VQ_OPERATION O	
	ON O.RECORD_ID = NCM.OPERATION_REC_ID
INNER JOIN OPERATION OP
	ON  PO.ALT_ID_2 = OP.WORKORDER_BASE_ID
	AND PO.ALT_ID_3 = OP.WORKORDER_LOT_ID
	AND PO.ALT_ID_4 = OP.WORKORDER_SPLIT_ID
	AND PO.ALT_ID_5 = OP.WORKORDER_SUB_ID
	AND O.DISPLAY_ID = OP.SEQUENCE_NO
	AND OP.WORKORDER_TYPE = 'W'
--LEFT OUTER JOIN [SQL-IQM-9].[IQM_Baseline_App].DBO.VQ_SUPPLIER S
--	ON S.RECORD_ID = NCM.REPORTED_BY_SUPP_REC_ID 
INNER JOIN DBO.DEMAND_SUPPLY_LINK DSL
	ON OP.WORKORDER_BASE_ID = DSL.DEMAND_BASE_ID
	AND OP.WORKORDER_LOT_ID = DSL.DEMAND_LOT_ID
	AND OP.WORKORDER_SPLIT_ID = DSL.DEMAND_SPLIT_ID
	AND OP.WORKORDER_SUB_ID = DSL.DEMAND_SUB_ID
	AND OP.SEQUENCE_NO = DSL.DEMAND_SEQ_NO
	AND OP.WORKORDER_TYPE ='W'
--INNER JOIN DBO.WORK_ORDER WO
--	ON DSL.DEMAND_BASE_ID = WO.BASE_ID
--	AND DSL.DEMAND_LOT_ID = WO.LOT_ID
--	AND DSL.DEMAND_SPLIT_ID = WO.SPLIT_ID
--	AND DSL.DEMAND_SUB_ID = WO.SUB_ID
--	AND DSL.SUPPLY_PART_ID = WO.PART_ID
--	AND WO.[TYPE] ='W'
WHERE NCM.NCM_TYPE = 'SUPPLIER - MAJOR'
	AND O.CONST_RESOURCE_ID = 'CONTRACTOR'
	AND PO.ALT_ID_2 IS NOT NULL
UNION ALL
--not a contractor operation; not associated with a work order
SELECT NCM_ID
	, NCM.PRODUCT_REC_ID
	, CASE 
		WHEN CHARINDEX('/', NCM.PO_NUMBER,1) = 0 THEN NCM.PO_NUMBER
		ELSE SUBSTRING(NCM.PO_NUMBER, 1, CHARINDEX('/', NCM.PO_NUMBER, 1) -1) 
	END AS SupplierPO
	, SUBSTRING(NCM.PO_NUMBER, LEN(NCM.PO_NUMBER), CHARINDEX('/', NCM.PO_NUMBER, 1) +1) AS SupplierPO_LINE
	, NCM.LOT_SIZE
	, NCM.ACCEPTED
	, NCM.REJECTED
	, NCM.NCM_DATE
FROM VQ_NONCONFORMANCE NCM
LEFT OUTER JOIN VQ_PRODUCT P
	ON P.RECORD_ID = NCM.PRODUCT_REC_ID
LEFT OUTER JOIN VQ_PRODUCTION_ORDER PO
	ON PO.RECORD_ID = NCM.PROD_ORDER_REC_ID
	AND PO.ALT_ID_1 = 'W'
LEFT OUTER JOIN VQ_OPERATION O	
	ON O.RECORD_ID = NCM.OPERATION_REC_ID
--LEFT OUTER JOIN [SQL-IQM-9].[IQM_Baseline_App].DBO.VQ_SUPPLIER s
--	ON S.RECORD_ID = NCM.REPORTED_BY_SUPP_REC_ID 
WHERE NCM.NCM_TYPE = 'SUPPLIER - MAJOR'
	AND PO.ALT_ID_2 IS NULL
)

, CTE_RECEIVED AS 
(SELECT PO.VENDOR_ID
	, POL.PURC_ORDER_ID
	, POL.LINE_NO AS PURC_ORDER_LINE_NO
	, POL.LINE_STATUS AS PO_LINE_STATUS
	, PO.[STATUS] AS PO_STATUS
	, POL.ORDER_QTY AS PO_LINE_ORDERQTY
	, RL.RECEIVER_ID
	, RL.LINE_NO AS RECEIVER_LINE
	, R.RECEIVED_Date AS RDATE
	, RL.RECEIVED_Qty AS TotReceivedQty
	, CONVERT(DATE, POL.DESIRED_RECV_DATE + 5, 101) as DUE_DATE
	, CONVERT(NVARCHAR,LEFT(CONVERT(VARCHAR, R.RECEIVED_Date, 112), 6)) AS YrMth
	, CASE 
		WHEN R.RECEIVED_DATE <= POL.DESIRED_RECV_DATE + 5 THEN 1
		ELSE 0
	END AS OntimeQty
	, CASE 
		WHEN R.RECEIVED_DATE <= POL.DESIRED_RECV_DATE + 5 THEN 0
		ELSE 1
	END AS LateQty
	, 1 AS TotQty
	, POL.PART_ID
FROM PURC_ORDER_LINE POL
INNER JOIN PURCHASE_ORDER PO
	ON POL.PURC_ORDER_ID = PO.ID
INNER JOIN RECEIVER_LINE RL 
	ON POL.PURC_ORDER_ID = RL.PURC_ORDER_ID
	AND POL.LINE_NO = RL.PURC_ORDER_LINE_NO
	AND RL.RECEIVED_QTY > 0
INNER JOIN RECEIVER R 
	ON RL.RECEIVER_ID = R.ID
WHERE ISNULL(POL.VENDOR_PART_ID, '') != 'Tooling' 
	AND POL.PART_ID IS NOT NULL
	AND R.RECEIVED_Date >= @STARTDATE
	AND R.RECEIVED_Date <= @ENDDATE
	AND DATEDIFF(DAY, ISNULL(POL.DESIRED_RECV_DATE, PO.DESIRED_RECV_DATE), PO.ORDER_DATE) * (-1) >= 0
	AND EXISTS(SELECT 1
		FROM VENDOR V
		WHERE V.ID = PO.VENDOR_ID 
		AND V.USER_2 IN ('APPROVED', 'PENDING')
		)
)

SELECT VENDOR_ID
	, PURC_ORDER_ID
	, PURC_ORDER_LINE_NO
	, PO_LINE_STATUS
	, PO_STATUS
	, PO_LINE_ORDERQTY
	, RECEIVER_ID
	, RECEIVER_LINE
	, RDATE
	, TotReceivedQty 
	, SUM(ISNULL(NCM.REJECTED,0)) AS TotRejected	
	, DUE_DATE
	, YrMth
	, OntimeQty
	, LateQty
	, TotQty
	, PART_ID
FROM CTE_RECEIVED R
LEFT OUTER JOIN CTE_IQM NCM
	ON R.PURC_ORDER_ID = NCM.SupplierPO
	AND R.PURC_ORDER_LINE_NO = NCM.SupplierPO_LINE
GROUP BY VENDOR_ID
	, PURC_ORDER_ID
	, PURC_ORDER_LINE_NO
	, PO_LINE_STATUS
	, PO_STATUS
	, PO_LINE_ORDERQTY
	, RECEIVER_ID
	, RECEIVER_LINE
	, RDATE
	, TotReceivedQty 
	, DUE_DATE
	, YrMth
	, OntimeQty
	, LateQty
	, TotQty
	, PART_ID
ORDER BY VENDOR_ID
	, rdate
