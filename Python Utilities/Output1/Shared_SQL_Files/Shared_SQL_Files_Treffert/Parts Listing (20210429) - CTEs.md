# Parts Listing (20210429) - CTEs

```sql
USE DATAWAREHOUSE
GO

; WITH CTE_PARTS AS
(SELECT ID AS PART_ID
, [DESCRIPTION]
, [STATUS]
, CREATE_DATE
, PLANNER_USER_ID
, [BUYER_USER_ID]
, [COMMODITY_CODE]
, [STOCKED]
, [FABRICATED]
, [PURCHASED]
, [CONSUMABLE]
, [TOOL_OR_FIXTURE]
FROM PART P
WHERE (STATUS != 'O' OR STATUS IS NULL)
AND ID NOT LIKE '%REWORK%'
)

, CTE_MASTERS AS
(SELECT WO.PART_ID
, MAX(WO.CLOSE_DATE) AS CLOSE_DATE
FROM WORK_ORDER WO
WHERE WO.[TYPE] = 'M'
AND WO.CLOSE_DATE IS NOT NULL
GROUP BY PART_ID
)

, CTE_WOS AS
(SELECT WO.PART_ID
, MAX(WO.CLOSE_DATE) AS CLOSE_DATE
FROM WORK_ORDER WO
WHERE WO.[TYPE] = 'W'
AND WO.CLOSE_DATE IS NOT NULL
GROUP BY PART_ID
)

, CTE_CP AS
(SELECT CP.PART_ID
, MAX(WO.CLOSE_DATE) AS CLOSE_DATE
FROM CO_PRODUCT CP
INNER JOIN WORK_ORDER WO
ON WO.BASE_ID = CP.WORKORDER_BASE_ID
AND WO.LOT_ID = CP.WORKORDER_LOT_ID
AND WO.SPLIT_ID = CP.WORKORDER_SPLIT_ID
AND WO.SUB_ID = CP.WORKORDER_SUB_ID
AND WO.[TYPE] = CP.WORKORDER_TYPE
WHERE [TYPE] = 'W'
AND WO.CLOSE_DATE IS NOT NULL
GROUP BY CP.PART_ID
)

, CTE_REQ AS
(SELECT REQ.PART_ID
, MAX(WO.CLOSE_DATE) AS CLOSE_DATE
FROM REQUIREMENT REQ
INNER JOIN WORK_ORDER WO
ON WO.BASE_ID = REQ.WORKORDER_BASE_ID
AND WO.LOT_ID = REQ.WORKORDER_LOT_ID
AND WO.SPLIT_ID = REQ.WORKORDER_SPLIT_ID
AND WO.SUB_ID = REQ.WORKORDER_SUB_ID
AND WO.[TYPE] = REQ.WORKORDER_TYPE
WHERE [TYPE] = 'W'
AND WO.CLOSE_DATE IS NOT NULL
GROUP BY REQ.PART_ID
)

, CTE_CO AS
(SELECT COL.PART_ID
, MAX(ISNULL(COL.DESIRED_SHIP_DATE, CO.DESIRED_SHIP_DATE)) AS LAST_SHIP_DATE
FROM CUST_ORDER_LINE COL
INNER JOIN CUSTOMER_ORDER CO
ON COL.CUST_ORDER_ID = CO.ID
GROUP BY COL.PART_ID
)

SELECT P.*
, CO.LAST_SHIP_DATE AS LAST_CO_DATE
, M.CLOSE_DATE AS MASTER_DATE
, WO.CLOSE_DATE AS LAST_WO_DATE
, CP.CLOSE_DATE AS LAST_CP_DATE
, REQ.CLOSE_DATE AS LAST_REQ_DATE
FROM CTE_PARTS P
LEFT OUTER JOIN CTE_CO CO
ON P.PART_ID = CO.PART_ID
LEFT OUTER JOIN CTE_MASTERS M
ON P.PART_ID = M.PART_ID
LEFT OUTER JOIN CTE_WOS WO
ON P.PART_ID = WO.PART_ID
LEFT OUTER JOIN CTE_CP CP
ON P.PART_ID = CP.PART_ID
LEFT OUTER JOIN CTE_REQ REQ
ON P.PART_ID = REQ.PART_ID


SELECT * FROM CTE_PARTS
```
