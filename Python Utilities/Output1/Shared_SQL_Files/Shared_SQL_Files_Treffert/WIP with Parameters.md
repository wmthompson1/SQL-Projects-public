# WIP with Parameters

```sql
USE LIVE
GO

SELECT WO.BASE_ID
	, WO.LOT_ID
	, WO.SPLIT_ID
	, WO.SUB_ID
	, WO.CREATE_DATE
	, WO.[STATUS]
	, WO.DESIRED_WANT_DATE
	, WO.DESIRED_RLS_DATE
	, WO.PART_ID
	, WO.[TYPE]
	, WO.ACT_MATERIAL_COST + WO.ACT_LABOR_COST + WO.ACT_SERVICE_COST AS WIPVALUE
--	, 'OVER 30' AS DAYS30
	, T.LASTTKTTRANSID
	, T.TRANSACTION_DATE
	, T.LASTEMPNAME
	, T.OPERATION_SEQ_NO
	, T.RESOURCE_ID
	, CASE 
		WHEN SR.USER_1 IS NULL THEN 'NO OPERATIONS'
		WHEN SR.USER_1 = 'A' THEN 'AUBURN'
		WHEN SR.USER_1 = 'B' THEN 'BALLARD'
		WHEN SR.USER_1 = 'E' THEN 'MFG-AUBNORTH'
		WHEN SR.USER_1 = 'M' THEN 'MFG-AUBSOUTH'
		WHEN SR.USER_1 = 'O' THEN 'OUTREACH'
		ELSE 'CONTRACTOR'
	END AS [LOCATION]
	, CO.CUSTOMER_ID
	, SR.USER_1 AS SKILLS_TYPE
	, SR.USER_4 AS SUPERVISOR
	, SR.USER_5 AS FACILITY
	, SR.USER_6 AS PROCESS
	, SR.USER_7 AS SUBPROCESS
	, SR.USER_8 AS DEPARTMENT
FROM WORK_ORDER WO 
LEFT OUTER JOIN OPERATION SO 
	ON SO.WORKORDER_TYPE = WO.[TYPE]
	AND SO.WORKORDER_BASE_ID = WO.BASE_ID 
	AND SO.WORKORDER_LOT_ID = WO.LOT_ID 
	AND SO.WORKORDER_SPLIT_ID = WO.SPLIT_ID 
	AND SO.WORKORDER_SUB_ID = WO.SUB_ID 
	AND SO.SEQUENCE_NO = dbo.ufn_GetFirstOperation(WO.BASE_ID, WO.LOT_ID, WO.SUB_ID, WO.SPLIT_ID, WO.[TYPE])
LEFT OUTER JOIN SHOP_RESOURCE SR
	ON SO.RESOURCE_ID = SR.ID
LEFT OUTER JOIN svw_last_ticket_info_no_limit T 
	ON WO.BASE_ID = T.WORKORDER_BASE_ID
	AND WO.LOT_ID = T.WORKORDER_LOT_ID
	AND WO.SPLIT_ID = T.WORKORDER_SPLIT_ID
	AND WO.SUB_ID = T.WORKORDER_SUB_ID
    AND WO.[TYPE] = T.WORKORDER_TYPE
LEFT OUTER JOIN DEMAND_SUPPLY_LINK DSL
	ON DSL.SUPPLY_BASE_ID = WO.BASE_ID
	AND DSL.SUPPLY_LOT_ID = WO.LOT_ID
	AND DSL.SUPPLY_SPLIT_ID = WO.SPLIT_ID
	AND DSL.SUPPLY_SUB_ID = WO.SUB_ID
	AND DSL.SUPPLY_TYPE = 'WO'
LEFT OUTER JOIN CUST_ORDER_LINE COL
	ON DSL.DEMAND_BASE_ID = COL.CUST_ORDER_ID
	AND DSL.DEMAND_SEQ_NO = COL.LINE_NO
LEFT OUTER JOIN CUSTOMER_ORDER CO
	ON CO.ID = COL.CUST_ORDER_ID
WHERE WO.[STATUS] IN ('F','R','U')
	AND WO.[TYPE] = 'W'
	AND WO.DESIRED_WANT_DATE <= CONVERT(DATE, GETDATE()-31, 111)
	AND ISNULL(SR.USER_5, '_BLANK') IN (@FACILITY)
	AND ISNULL(SR.USER_6, '_BLANK') IN (@PROCESS)
	AND ISNULL(SR.USER_7, '_BLANK') IN (@SUBPROCESS)
	AND ISNULL(SR.USER_8, '_BLANK') IN (@DEPARTMENT)
	AND O.[STATUS] != 'X'
```
