# Labor Ticket Data-Driven Subscription

```sql
USE LIVE
go

SELECT ADP.[Reports To Name] AS [SUPERVISOR] 
	, CASE
		WHEN ADP.[Reports To Name] = 'Baker, W. P.' THEN 'RICKB@SKILLSINC.COM'
		WHEN ADP.[Reports To Name] = 'Robinson, Rodney' THEN 'RODR@SKILLSINC.COM'
		WHEN ADP.[Reports To Name] = 'Wentzke, M.' THEN 'AnnW@SKILLSINC.COM'
		ELSE ISNULL(SUPV.EMAIL_ADDR, 'tracit@skillsinc.com')
	END AS Email
	, *
FROM LABOR_TICKET AS LT 
INNER JOIN EMPLOYEE AS EE 
	ON LT.EMPLOYEE_ID = EE.ID
INNER JOIN LIVESupplemental.DBO.ADP_EMP AS ADP
	ON CONVERT(NVARCHAR(15), RIGHT('0000' + LT.EMPLOYEE_ID, 6)) = ADP.[File Number]
--	AND [Position Status] = 'Active'
LEFT OUTER JOIN EMPLOYEE AS SUPV
	ON ADP.[Reports To Name] = SUPV.[LAST_NAME] + ', ' + SUPV.[FIRST_NAME]
	AND SUPV.ACTIVE = 'Y'
WHERE LT.ACT_CLOCK_OUT IS NULL
	AND LT.HOURS_WORKED IS NULL
	AND LT.ACT_CLOCK_IN IS NOT NULL
	AND (LT.TRANSACTION_DATE >= DATEADD(DAY, 1, EOMONTH(DATEADD(MONTH, -2, GETDATE()), -0))	
		AND LT.TRANSACTION_DATE < CONVERT(DATE, GETDATE())
		)
GROUP BY ADP.[Reports To Name] 
	, ISNULL(SUPV.EMAIL_ADDR, 'tracit@skillsinc.com')
```
