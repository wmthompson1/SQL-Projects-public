# Labor Tickets Information

```sql
USE DATAWAREHOUSE
GO

SET DATEFIRST 1;
--> VISUAL HOURS
SELECT CO.CUSTOMER_ID 
	, COL.LINE_NO
	, COL.PART_ID
	, COL.MISC_REFERENCE
	, LT.TRANSACTION_ID
	, LT.WORKORDER_TYPE
	, LT.WORKORDER_BASE_ID
	, LT.WORKORDER_LOT_ID
	, LT.WORKORDER_SPLIT_ID
	, LT.WORKORDER_SUB_ID
	, LT.OPERATION_SEQ_NO
	, LT.[TYPE]
	, LT.EMPLOYEE_ID
	, LT.DEPARTMENT_ID
	, LT.RESOURCE_ID
	, LT.CLOCK_IN
	, LT.CLOCK_OUT
	, LT.HOURS_WORKED
	, LT.GOOD_QTY
	, LT.BAD_QTY
	, LT.TRANSACTION_DATE
	, LT.[DESCRIPTION]
	, LT.ACT_LABOR_COST
	, LT.ACT_BURDEN_COST


FROM DataWarehouse.dbo.LABOR_TICKET LT
LEFT OUTER JOIN DataWarehouse.dbo.DEMAND_SUPPLY_LINK DSL
ON DSL.SUPPLY_TYPE = 'WO'
AND DSL.SUPPLY_BASE_ID = LT.WORKORDER_BASE_ID
AND DSL.SUPPLY_LOT_ID = LT.WORKORDER_LOT_ID
AND DSL.SUPPLY_SPLIT_ID = LT.WORKORDER_SPLIT_ID
AND DSL.SUPPLY_SUB_ID = LT.WORKORDER_SUB_ID
AND LT.WORKORDER_TYPE = 'W'
LEFT OUTER JOIN CUST_ORDER_LINE COL
ON DSL.DEMAND_BASE_ID = COL.CUST_ORDER_ID
AND DSL.DEMAND_SEQ_NO = COL.LINE_NO
LEFT OUTER JOIN CUSTOMER_ORDER CO
ON COL.CUST_ORDER_ID = CO.ID
WHERE LT.TRANSACTION_DATE >= GETDATE()-365

```
