DECLARE @DATADATE DATE = '1/1/'+ convert(varchar(4), year(GETDATE())-1)

WITH CTE_LT AS
(SELECT WORKORDER_TYPE
	, WORKORDER_BASE_ID
	, WORKORDER_LOT_ID
	, WORKORDER_SPLIT_ID
	, WORKORDER_SUB_ID
	, OPERATION_SEQ_NO
	, LT.RESOURCE_ID
	, SR.USER_5 AS [LOCATION]
	, SHIFT_ID
	, EE.ID
	, EE.LAST_NAME
	, EE.FIRST_NAME
	, SUM(HOURS_WORKED) AS TOTAL_HOURS
	, MAX(CLOCK_OUT) AS CLOCK_OUT
FROM LABOR_TICKET LT
INNER JOIN EMPLOYEE EE
	ON LT.EMPLOYEE_ID = EE.ID
INNER JOIN SHOP_RESOURCE SR
	ON LT.RESOURCE_ID = SR.ID
WHERE --CONVERT(date, LT.CLOCK_OUT) >= @DATADATE
  CONVERT(date, LT.CLOCK_OUT) >= '3/1/2018'
  AND CONVERT(date, LT.CLOCK_OUT) <= '4/1/2018'
GROUP BY WORKORDER_TYPE
	, WORKORDER_BASE_ID
	, WORKORDER_LOT_ID
	, WORKORDER_SPLIT_ID
	, WORKORDER_SUB_ID
	, OPERATION_SEQ_NO
	, LT.RESOURCE_ID
	, SR.USER_5
	, SHIFT_ID
	, EE.ID
	, EE.LAST_NAME
	, EE.FIRST_NAME
)

SELECT LT.*
	, (WO.DESIRED_QTY) AS TOTAL_QTY
FROM CTE_LT LT
INNER JOIN WORK_ORDER WO
	ON WO.[TYPE] = LT.WORKORDER_TYPE
	AND WO.BASE_ID = LT.WORKORDER_BASE_ID
	AND WO.LOT_ID = LT.WORKORDER_LOT_ID
	AND WO.SPLIT_ID = LT.WORKORDER_SPLIT_ID
	AND WO.SUB_ID = LT.WORKORDER_SUB_ID
