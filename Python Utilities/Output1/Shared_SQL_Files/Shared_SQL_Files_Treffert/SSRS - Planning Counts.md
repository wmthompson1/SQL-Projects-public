# SSRS - Planning Counts

```sql
USE LIVE
GO

DECLARE @STARTDATE DATE = '5/1/2021'
, @ENDDATE DATE =  GETDATE()


; WITH CTE_CO AS 
--> TOTAL NUMBER OF JOBS CREATED (RECEIVED)
(SELECT CO.CUSTOMER_ID
	, CO.ID AS CUST_ORDER_ID
	, CONVERT(DATE, CO.CREATE_DATE) AS CREATE_DATE
	, LEFT(P.PLANNER_USER_ID, 3) AS BUS_STREAM
	, CASE
		WHEN DSL.SUPPLY_BASE_ID IS NULL THEN 'UNLINKED'
		ELSE 'LINKED'
	END AS CO_TYPE
FROM CUSTOMER_ORDER CO
INNER JOIN CUST_ORDER_LINE COL
	ON CO.ID = COL.CUST_ORDER_ID
INNER JOIN PART P
	ON P.ID = COL.PART_ID
INNER JOIN DEMAND_SUPPLY_LINK DSL
	ON COL.CUST_ORDER_ID = DSL.DEMAND_BASE_ID
	AND COL.LINE_NO = DSL.DEMAND_SEQ_NO
WHERE CO.[STATUS] NOT IN ('C', 'X')
)

, CTE_P2_REVIEW AS 
--> TOTAL NUMBER OF PLANNED 
(SELECT WO.BASE_ID
	, CO.CUSTOMER_ID
	, LEFT(P.PLANNER_USER_ID, 3) AS BUS_STREAM
	, CASE
		WHEN DSL.SUPPLY_BASE_ID IS NULL THEN 'UNLINKED'
		ELSE 'LINKED'
	END AS CO_TYPE
	, CONVERT(DATE, O.CLOSE_DATE) AS CLOSE_DATE
FROM WORK_ORDER WO
INNER JOIN OPERATION O
	ON WO.[TYPE] = O.WORKORDER_TYPE
	AND WO.BASE_ID = O.WORKORDER_BASE_ID
	AND WO.LOT_ID = O.WORKORDER_LOT_ID
	AND WO.SPLIT_ID = O.WORKORDER_SPLIT_ID
	AND WO.SUB_ID = O.WORKORDER_SUB_ID
INNER JOIN SHOP_RESOURCE SR
	ON O.RESOURCE_ID = SR.ID
LEFT OUTER JOIN DEMAND_SUPPLY_LINK DSL
	ON WO.[TYPE] = 'W'
	AND DSL.SUPPLY_TYPE = 'WO'
	AND WO.BASE_ID = DSL.SUPPLY_BASE_ID
	AND WO.LOT_ID = DSL.SUPPLY_LOT_ID
	AND WO.SPLIT_ID = DSL.SUPPLY_SPLIT_ID
	AND WO.SUB_ID = DSL.SUPPLY_SUB_ID
LEFT OUTER JOIN CUST_ORDER_LINE COL
	ON COL.CUST_ORDER_ID = DSL.DEMAND_BASE_ID
	AND COL.LINE_NO = DSL.DEMAND_SEQ_NO
LEFT OUTER JOIN CUSTOMER_ORDER CO
	ON CO.ID = COL.CUST_ORDER_ID
LEFT OUTER JOIN PART P
	ON P.ID = WO.PART_ID
WHERE WO.[STATUS] NOT IN ('C', 'X')
	AND WO.[TYPE] = 'W'
	AND O.[STATUS] = 'C'
	AND SR.DEPARTMENT_ID = 'REVIEW'
	AND O.OPERATION_TYPE NOT IN ('MGG-ORDER-ENTRY', 'FIN-ORDER-ENTRY')
)

, CTE_P2_STAGING AS 
--> TOTAL NUMBER OF CMP JOBS THROUGH STAGING (STAGING OPS COMPLETED)
(SELECT WO.BASE_ID
	, CO.CUSTOMER_ID
	, LEFT(P.PLANNER_USER_ID, 3) AS BUS_STREAM
	, CASE
		WHEN DSL.SUPPLY_BASE_ID IS NULL THEN 'UNLINKED'
		ELSE 'LINKED'
	END AS CO_TYPE
	, CONVERT(DATE, O.CLOSE_DATE) AS CLOSE_DATE
FROM WORK_ORDER WO
INNER JOIN OPERATION O
	ON WO.[TYPE] = O.WORKORDER_TYPE
	AND WO.BASE_ID = O.WORKORDER_BASE_ID
	AND WO.LOT_ID = O.WORKORDER_LOT_ID
	AND WO.SPLIT_ID = O.WORKORDER_SPLIT_ID
	AND WO.SUB_ID = O.WORKORDER_SUB_ID
INNER JOIN SHOP_RESOURCE SR
	ON O.RESOURCE_ID = SR.ID
LEFT OUTER JOIN DEMAND_SUPPLY_LINK DSL
	ON WO.[TYPE] = 'W'
	AND DSL.SUPPLY_TYPE = 'WO'
	AND WO.BASE_ID = DSL.SUPPLY_BASE_ID
	AND WO.LOT_ID = DSL.SUPPLY_LOT_ID
	AND WO.SPLIT_ID = DSL.SUPPLY_SPLIT_ID
	AND WO.SUB_ID = DSL.SUPPLY_SUB_ID
LEFT OUTER JOIN CUST_ORDER_LINE COL
	ON COL.CUST_ORDER_ID = DSL.DEMAND_BASE_ID
	AND COL.LINE_NO = DSL.DEMAND_SEQ_NO
LEFT OUTER JOIN CUSTOMER_ORDER CO
	ON CO.ID = COL.CUST_ORDER_ID
LEFT OUTER JOIN PART P
	ON P.ID = WO.PART_ID
WHERE WO.[STATUS] NOT IN ('C', 'X')
	AND WO.[TYPE] = 'W'
	AND O.[STATUS] = 'C'
	AND SR.DEPARTMENT_ID = 'STAGING'
)

, CTE_COUNTS AS
(SELECT 1 AS SORT
	, 'TOTAL JOBS RECEIVED' AS TITLE
	, CUSTOMER_ID
	, BUS_STREAM
	, CO_TYPE
	, COUNT(CO.CUST_ORDER_ID) AS CNT
	, CONVERT(DATE, CO.CREATE_DATE) AS [DATE]
FROM CTE_CO CO
--WHERE CONVERT(DATE, CO.CREATE_DATE) > GETDATE()-8
GROUP BY CUSTOMER_ID
	, BUS_STREAM
	, CO_TYPE
	, CO.CREATE_DATE
UNION ALL
SELECT 2 AS SORT
	, 'TOTAL JOBS PLANNED' AS TITLE
	, CUSTOMER_ID
	, BUS_STREAM
	, CO_TYPE
	, COUNT(WO.BASE_ID) AS CNT
	, CONVERT(DATE, WO.CLOSE_DATE) AS [DATE]
FROM CTE_P2_REVIEW WO
--WHERE CONVERT(DATE, WO.CLOSE_DATE) > GETDATE()-8
GROUP BY CUSTOMER_ID
	, BUS_STREAM
	, CO_TYPE
	, WO.CLOSE_DATE
UNION ALL
SELECT 3 AS SORT
	, 'TOTAL JOBS RELEASED' AS TITLE
	, CUSTOMER_ID
	, BUS_STREAM
	, CO_TYPE
	, COUNT(WO.BASE_ID) AS CNT
	, CONVERT(DATE, WO.CLOSE_DATE) AS [DATE]
FROM CTE_P2_STAGING WO
--WHERE CONVERT(DATE, WO.CLOSE_DATE) > GETDATE()-8
GROUP BY CUSTOMER_ID
	, BUS_STREAM
	, CO_TYPE
	, WO.CLOSE_DATE
 )

SELECT SORT AS CMP_SORT
	, TITLE AS CMP_TITLE
	, CUSTOMER_ID AS CMP_CUST_ID
	, BUS_STREAM AS CMP_BS
	, CO_TYPE AS CMP_TYPE
	, CNT AS CMP_CNT
	, [DATE] AS CMP_DATE
	, NULL AS NON_SORT
	, NULL AS NON_TITLE
	, NULL AS NON_CUST_ID
	, NULL AS NON_BS
	, NULL AS NON_TYPE
	, NULL AS NON_CNT
	, NULL AS NON_DATE
FROM CTE_COUNTS
WHERE [DATE] >= @STARTDATE
	AND [DATE] <= @ENDDATE
	AND BUS_STREAM IN (@BUSSTREAM)
	AND CUSTOMER_ID = 'BOE609'
UNION ALL 
SELECT NULL AS CMP_SORT
	, NULL AS CMP_TITLE
	, NULL AS CMP_CUST_ID
	, NULL AS CMP_BS
	, NULL AS CMP_TYPE
	, NULL AS CMP_CNT
	, NULL AS CMP_DATE
	, SORT AS NON_SORT
	, TITLE AS NON_TITLE
	, CUSTOMER_ID AS NON_CUST_ID
	, BUS_STREAM AS NON_BS
	, CO_TYPE AS NON_TYPE
	, CNT AS NON_CNT
	, [DATE] AS NON_DATE
FROM CTE_COUNTS
WHERE [DATE] >= @STARTDATE
	AND [DATE] <= @ENDDATE
	AND BUS_STREAM IN (@BUSSTREAM)
	AND CUSTOMER_ID != 'BOE609'
```
