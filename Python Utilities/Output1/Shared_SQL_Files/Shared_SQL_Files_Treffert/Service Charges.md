# Service Charges

```sql
USE DataWarehouse
GO

SELECT CO.ID
	, CO.CUSTOMER_ID
	, COL.MISC_REFERENCE
	, COL.UNIT_PRICE
	, S.SHIPPED_DATE
	, CO.DESIRED_SHIP_DATE
	, SL.USER_SHIPPED_QTY
	, COL.SERVICE_CHARGE_ID
FROM CUSTOMER_ORDER CO
INNER JOIN SHIPPER S
	ON CO.ID = S.CUST_ORDER_ID
INNER JOIN SHIPPER_LINE SL
	ON S.PACKLIST_ID = SL.PACKLIST_ID
INNER JOIN CUST_ORDER_LINE COL
	ON SL.CUST_ORDER_ID = COL.CUST_ORDER_ID
	AND SL.CUST_ORDER_LINE_NO = COL.LINE_NO
WHERE  COL.SERVICE_CHARGE_ID IS NOT NULL
	AND SL.USER_SHIPPED_QTY > 0 
	AND (S.SHIPPED_DATE >= '6/1/2018'
		AND S.SHIPPED_DATE < '7/1/2018')



```
