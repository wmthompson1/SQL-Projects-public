```sql
use DataWarehouse
go

drop table #wo
drop table #ops

declare @start date = '1/1/2019'
--@workorder nvarchar(10) = '1462340'

SELECT [TYPE]
	, BASE_ID
	, LOT_ID
	, SPLIT_ID
	, SUB_ID
	, ACT_MATERIAL_COST
	, ACT_LABOR_COST
	, ACT_BURDEN_COST
	, ACT_SERVICE_COST
INTO #WO
FROM WORK_ORDER WO
WHERE WO.CREATE_DATE >= @START
	AND WO.[TYPE] = 'W'
--BASE_ID = @workorder 

CREATE CLUSTERED INDEX CIDX_WO ON #WO ([TYPE], BASE_ID, LOT_ID, SPLIT_ID, SUB_ID);

--select * from #wo where split_id <> 0

SELECT O.WORKORDER_TYPE
	, O.WORKORDER_BASE_ID 
	, O.WORKORDER_LOT_ID
	, O.WORKORDER_SPLIT_ID
	, O.WORKORDER_SUB_ID
	, SUM(ACT_TTL_MAT_COST) AS MATERIAL_COSTS
	, SUM(ACT_ATL_LAB_COST) + SUM(ACT_TTL_LAB_COST) AS LABOR_COSTS
	, SUM(ACT_ATL_BUR_COST) + SUM(ACT_TTL_BUR_COST) AS BURDEN_COSTS
	, SUM(ACT_ATL_SER_COST) + SUM(ACT_TTL_SER_COST) AS SERVICE_COSTS
INTO #OPS
FROM OPERATION O
WHERE EXISTS (SELECT 1 FROM #WO WO
	WHERE O.WORKORDER_BASE_ID = WO.BASE_ID
	AND O.WORKORDER_LOT_ID = WO.LOT_ID
	AND O.WORKORDER_SPLIT_ID = WO.SPLIT_ID
	AND O.WORKORDER_SUB_ID = WO.SUB_ID
	AND O.WORKORDER_TYPE = WO.[TYPE]
)
--WORKORDER_BASE_ID = @workorder 
GROUP BY O.WORKORDER_TYPE
	, O.WORKORDER_BASE_ID 
	, O.WORKORDER_LOT_ID
	, O.WORKORDER_SPLIT_ID
	, O.WORKORDER_SUB_ID

CREATE CLUSTERED INDEX CIDX_OPS ON #OPS (WORKORDER_TYPE, WORKORDER_BASE_ID, WORKORDER_LOT_ID, WORKORDER_SPLIT_ID, WORKORDER_SUB_ID);

SELECT WO.[TYPE]
	, WO.BASE_ID
	, WO.LOT_ID
	, WO.SPLIT_ID
	, WO.SUB_ID
	, WO.ACT_MATERIAL_COST
	, O.MATERIAL_COSTS
	, WO.ACT_LABOR_COST
	, O.LABOR_COSTS
	, WO.ACT_BURDEN_COST
	, O.BURDEN_COSTS
	, WO.ACT_SERVICE_COST
	, O.SERVICE_COSTS
FROM #WO WO
LEFT OUTER JOIN #OPS O
ON O.WORKORDER_TYPE = WO.[TYPE]
AND O.WORKORDER_BASE_ID = WO.BASE_ID
AND O.WORKORDER_LOT_ID = WO.LOT_ID
AND O.WORKORDER_SPLIT_ID = WO.SPLIT_ID
AND O.WORKORDER_SUB_ID = WO.SUB_ID
WHERE WO.ACT_MATERIAL_COST <> O.MATERIAL_COSTS
	OR WO.ACT_LABOR_COST <> O.LABOR_COSTS
	OR WO.ACT_BURDEN_COST <> O.BURDEN_COSTS
	OR WO.ACT_SERVICE_COST <> O.SERVICE_COSTS
```
