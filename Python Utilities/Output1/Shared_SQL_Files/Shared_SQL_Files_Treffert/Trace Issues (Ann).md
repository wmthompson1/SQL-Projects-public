# Trace Issues (Ann)

```sql
SELECT P.ID AS PART_ID
	, P.[DESCRIPTION]
	, T.ID
	, T.PART_ID
	, P.COMMODITY_CODE
	, P.STOCK_UM
	, T.APROPERTY_1
	, T.APROPERTY_2
	, T.APROPERTY_3
	, T.APROPERTY_4
	, T.APROPERTY_5
	, T.EXPIRATION_DATE
	, T.LOT_ID
	, T.CREATE_DATE
	, T.IN_QTY
	, T.OUT_QTY
	, P.QTY_AVAILABLE_ISS
	, P.QTY_AVAILABLE_MRP
	, P.QTY_ON_HAND
FROM PART P
LEFT OUTER JOIN TRACE T
ON P.ID = T.PART_ID
WHERE T.PART_ID <> N'20243C-071 CMPN' 
	AND P.COMMODITY_CODE IN (N'Details', N'End Item', N'Raw Material', N'Standards')
ORDER BY P.COMMODITY_CODE
	, T.PART_ID
	, T.ID
```
