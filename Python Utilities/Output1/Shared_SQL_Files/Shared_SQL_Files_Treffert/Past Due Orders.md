# Past Due Orders

```sql
use DataWarehouse
go

select CO.SALESREP_ID
	, CO.CUSTOMER_ID
	, CO.ID
	, COL.LINE_NO
	, CO.[STATUS] AS COSTATUS
	, CASE
		WHEN COL.LINE_STATUS = 'C' THEN 'YES'
		ELSE 'NO'
	END AS SHIPSHORT
	, COL.PART_ID
	, COL.CUSTOMER_PART_ID
	, CO.ORDER_DATE
	, ISNULL(COL.DESIRED_SHIP_DATE, CO.DESIRED_SHIP_DATE) AS DESIRED_SHIP_DATE
	, WO.SCHED_FINISH_DATE
	, COL.ORDER_QTY
	, COL.TOTAL_SHIPPED_QTY
	, CASE
		WHEN EX.CUST_ORDER_ID IS NULL THEN ''
		ELSE 'EXPEDITE'
	END AS EXPEDITE
	, CO.TOTAL_AMT_ORDERED AS COTOTAL
	, CO.TOTAL_AMT_SHIPPED AS COBILLED
	, COL.TOTAL_AMT_ORDERED
	, COL.TOTAL_AMT_SHIPPED
	, WO.BASE_ID
	, WO.LOT_ID
	, WO.SPLIT_ID
	, WO.SUB_ID
	, WO.[STATUS] AS WOSTATUS
	, dbo.sfn_current_operation('W', WO.BASE_ID, WO.LOT_ID, WO.SPLIT_ID	, WO.SUB_ID) AS SEQ_NO
	, dbo.sfn_current_RESOURCE_ID('W', WO.BASE_ID, WO.LOT_ID, WO.SPLIT_ID, WO.SUB_ID) AS RESOURCE_ID
FROM CUSTOMER_ORDER CO 
INNER JOIN CUST_ORDER_LINE COL 
	ON  COL.CUST_ORDER_ID = CO.ID
LEFT OUTER JOIN 
	(SELECT TOP 1 CUST_ORDER_ID
	FROM CUST_ORDER_LINE COL
	WHERE SERVICE_CHARGE_ID = 'EX'
	) EX
	ON CO.ID = EX.CUST_ORDER_ID
LEFT OUTER JOIN DEMAND_SUPPLY_LINK AS DSL 
	ON  DSL.DEMAND_BASE_ID = COL.CUST_ORDER_ID
	AND DSL.DEMAND_SEQ_NO = COL.LINE_NO
LEFT OUTER JOIN WORK_ORDER WO
	ON  DSL.SUPPLY_BASE_ID = WO.BASE_ID 
	AND DSL.SUPPLY_LOT_ID = WO.LOT_ID 
	AND DSL.SUPPLY_SPLIT_ID = WO.SPLIT_ID 
	AND DSL.SUPPLY_SUB_ID = WO.SUB_ID 
	AND DSL.SUPPLY_TYPE  = 'WO'
WHERE CO.[STATUS] NOT IN ('C', 'X', 'H')
	AND COL.PART_ID IS NOT NULL
	AND ORDER_QTY - TOTAL_SHIPPED_QTY > 0
	AND isnull(COL.DESIRED_SHIP_DATE, CO.DESIRED_SHIP_DATE) < GETDATE()
	
order by co.id, col.line_no
```
