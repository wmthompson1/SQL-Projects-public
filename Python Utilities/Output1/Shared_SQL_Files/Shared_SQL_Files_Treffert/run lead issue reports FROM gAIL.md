# run lead issue reports FROM gAIL

```sql
SELECT 
	O.WORKORDER_BASE_ID, 
	O.WORKORDER_LOT_ID, 
	O.SEQUENCE_NO, 
	O.RESOURCE_ID, 
	O.SETUP_HRS, 
	O.RUN, 
	O.RUN_TYPE, 
	O.LOAD_SIZE_QTY, 
	O.TRANSIT_DAYS, 
	O.STATUS, 
	O.WORKORDER_TYPE, 
	O.SCHED_FINISH_DATE, 
	O.SCHED_START_DATE
FROM	OPERATION O
WHERE  
	O.STATUS in ('F', 'R', 'U') 
	AND O.RESOURCE_ID <> 'contractor' 
	AND O.WORKORDER_TYPE='W'
ORDER BY 
	O.SCHED_START_DATE


SELECT 
	O.WORKORDER_BASE_ID, 
	O.WORKORDER_LOT_ID, 
	O.SCHED_START_DATE, 
	O.SCHED_FINISH_DATE, 
	O.COULD_FINISH_DATE, 
	O.ISDETERMINANT, 
	O.MOVE_HRS, 
	O.SEQUENCE_NO, 
	O.RESOURCE_ID, 
	OS.COULD_START_DATE, 
	O.WORKORDER_TYPE, 
	O.STATUS,
	case when OS.COULD_START_DATE < O.SCHED_START_DATE then 'delta' end as Delta,
	O.SCHED_START_DATE - OS.COULD_START_DATE as Delta2,
	O.SCHED_FINISH_DATE - O.SCHED_START_DATE as Time
FROM	OPERATION O 
LEFT OUTER JOIN 
	OPERATION_SCHED OS 
	ON O.RESOURCE_ID = OS.RESOURCE_ID 
	AND O.WORKORDER_TYPE = OS.WORKORDER_TYPE 
	AND O.WORKORDER_BASE_ID = OS.WORKORDER_BASE_ID 
	AND O.WORKORDER_LOT_ID = OS.WORKORDER_LOT_ID
	AND O.WORKORDER_SPLIT_ID = OS.WORKORDER_SPLIT_ID
	AND O.WORKORDER_SUB_ID = OS.WORKORDER_SUB_ID
	AND O.SEQUENCE_NO = OS.SEQUENCE_NO
WHERE  
	O.WORKORDER_TYPE = 'W' 
	AND O.STATUS = 'R' 
	AND O.RESOURCE_ID = 'southaub-final'

```
