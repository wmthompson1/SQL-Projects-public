# Part Listing B4 20171225 (20210224)

```sql
USE LIVE
GO

SELECT P.ID AS PART_ID
, P.[STATUS]
, P.[DESCRIPTION]
, P.PLANNER_USER_ID
, P.BUYER_USER_ID
, P.COMMODITY_CODE
, P.CREATE_DATE
--, I.TRANSACTION_ID
--, R.WORKORDER_BASE_ID
FROM PART P
LEFT OUTER JOIN INVENTORY_TRANS I
ON P.ID = I.PART_ID
LEFT OUTER JOIN REQUIREMENT R
ON P.ID = R.PART_ID
WHERE (P.[STATUS] IS NULL OR P.[STATUS] != 'O')
AND P.CREATE_DATE <= '12/25/2017'
AND P.ID NOT LIKE '%REWORK%'
AND I.TRANSACTION_ID IS NULL
AND R.WORKORDER_BASE_ID IS NULL




```
