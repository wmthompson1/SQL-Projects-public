(SELECT CUSTOMER_ID

	, MAX(EFFECTIVE_DATE) AS EFF_DATE
	, MAX(ISNULL(DISCONTINUE_DATE, GETDATE() + 1)) AS DISC_DATE
	, DEFAULT_UNIT_PRICE
FROM CUST_PRICE_EFFECT 
WHERE CUSTOMER_ID = 'ROYMFG'
	AND ISNULL(DISCONTINUE_DATE, GETDATE() + 1) > GETDATE()
	AND ISNULL(EFFECTIVE_DATE, GETDATE() - 1) <= GETDATE()
GROUP BY CUSTOMER_ID, PART_ID
	, DEFAULT_UNIT_PRICE
)
