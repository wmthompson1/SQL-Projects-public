		WHEN DSL.SUPPLY_BASE_ID IS NULL THEN 'UNLINKED'

	, T.TRACE_ID

	ON SL.TRANSACTION_ID = T.TRANSACTION_ID
	AND T.QTY != 0
LEFT OUTER JOIN INVENTORY_TRANS ITOUT
	ON SL.TRANSACTION_ID = ITOUT.TRANSACTION_ID

	ON ITOUT.TRANSACTION_ID = ITD.OUT_TRANS_ID

	ON ITIN.TRANSACTION_ID = ITD.IN_TRANS_ID
LEFT OUTER JOIN DATAWAREHOUSE.dbo.WORK_ORDER WO
--	ON (T.TRACE_ID = WO.BASE_ID + '/' + WO.LOT_ID + '.' + WO.SPLIT_ID)
	ON ITIN.WORKORDER_BASE_ID = WO.BASE_ID
	AND ITIN.WORKORDER_LOT_ID = WO.LOT_ID
	AND ITIN.WORKORDER_SPLIT_ID = WO.SPLIT_ID
	AND ITIN.WORKORDER_SUB_ID = WO.SUB_ID
	AND ITIN.WORKORDER_TYPE = WO.[TYPE]
	AND WO.DESIRED_QTY <> 0
	AND WO.SUB_ID = '0'
	AND WO.[TYPE] = 'W'
	AND WO.[STATUS] = 'C'
LEFT OUTER JOIN DATAWAREHOUSE.dbo.DEMAND_SUPPLY_LINK DSL
	ON DSL.DEMAND_TYPE = 'CO'
	AND DSL.DEMAND_BASE_ID = COL.CUST_ORDER_ID
	AND DSL.DEMAND_SEQ_NO = COL.LINE_NO
WHERE (1 = 1)
--	AND RL.PACKLIST_ID IS NOT NULL
	AND S.SHIPPED_DATE >= @START
	AND COL.PART_ID IS NOT NULL
	AND SL.SHIPPED_QTY <> 0
--	AND DSL.SUPPLY_BASE_ID IS NOT NULL
	AND COL.ORDER_QTY > 0
)

, CTE_AMTRIB AS

	, SUM(ISNULL(REC_AMOUNT, 0)) AS AMT_REC_AMOUNT

WHERE CUSTOMER_ID = 'AMTRIB'
GROUP BY CUST_ORDER_ID
	, REPLACE(REPLACE(BASE_ID, 'A', ''), 'B', '')
	, REC_QTY
	, SHIPPED_QTY
)
--SELECT * FROM CTE_AMTRIB WHERE CUST_ORDER_ID = '429280'

, CTE_CHARGES AS
(SELECT COL.CUST_ORDER_ID
	, SUM(ISNULL(RL.AMOUNT, 0)) AS TOTAL_CHARGE
FROM DATAWAREHOUSE.dbo.CUST_ORDER_LINE COL
INNER JOIN DATAWAREHOUSE.dbo.CUSTOMER_ORDER CO
	ON COL.CUST_ORDER_ID = CO.ID
LEFT OUTER JOIN DATAWAREHOUSE.dbo.RECEIVABLE_LINE RL
	ON RL.CUST_ORDER_ID = COL.CUST_ORDER_ID
	AND RL.CUST_ORDER_LINE_NO = COL.LINE_NO
LEFT OUTER JOIN DATAWAREHOUSE.dbo.RECEIVABLE R
	ON RL.INVOICE_ID = R.INVOICE_ID
WHERE COL.PART_ID IS NULL
	AND COL.SERVICE_CHARGE_ID IS NOT NULL
	AND CO.CREATE_DATE >= '1/1/2015'
	AND CO.[STATUS] != 'X'
	AND COL.ORDER_QTY > 0
GROUP BY COL.CUST_ORDER_ID
)
--SELECT * FROM CTE_CHARGES

, CTE_PARTS AS
(SELECT COL.CUST_ORDER_ID
	, SUM(ISNULL(COL.ORDER_QTY, 0)) AS PART_QTY
FROM DATAWAREHOUSE.dbo.CUST_ORDER_LINE COL
INNER JOIN DATAWAREHOUSE.dbo.CUSTOMER_ORDER CO
	ON COL.CUST_ORDER_ID = CO.ID
INNER JOIN CTE_CHARGES CTE
	ON COL.CUST_ORDER_ID = CTE.CUST_ORDER_ID
LEFT OUTER JOIN DATAWAREHOUSE.dbo.DEMAND_SUPPLY_LINK DSL
	ON DSL.DEMAND_TYPE = 'CO'
	AND DSL.DEMAND_BASE_ID = COL.CUST_ORDER_ID
	AND DSL.DEMAND_SEQ_NO = COL.LINE_NO
WHERE COL.PART_ID IS NOT NULL
	AND COL.ORDER_QTY > 0
	AND (DSL.SUPPLY_BASE_ID NOT LIKE '%R%' 
		OR DSL.SUPPLY_BASE_ID IS NULL)
GROUP BY COL.CUST_ORDER_ID
)
--SELECT * FROM CTE_PARTS

, CTE_CHGAMTS AS
(SELECT CTE1.CUST_ORDER_ID
	, CTE1.TOTAL_CHARGE
	, CTE2.PART_QTY
	, CTE1.TOTAL_CHARGE / CTE2.PART_QTY AS UNIT_CHARGE_ALLOC
FROM CTE_CHARGES CTE1
INNER JOIN CTE_PARTS CTE2
	ON CTE1.CUST_ORDER_ID = CTE2.CUST_ORDER_ID
)
--SELECT * FROM CTE_CHGAMTS  where CUST_ORDER_ID = '346106'

, CTE_MERGE AS
(SELECT CMB.ORDER_TYPE
	, UDF.STRING_VAL AS BUS_STREAM
	, CASE
		WHEN CMB.CUSTOMER_ID = 'BOE609' AND P.PLANNER_USER_ID = 'FIN-HOURLY' THEN 'CMP-FIN'
		WHEN CMB.CUSTOMER_ID = 'BOE609' AND P.PLANNER_USER_ID = 'FIN' THEN 'CMP-FIN'
		WHEN CMB.CUSTOMER_ID = 'BOE609' AND P.PLANNER_USER_ID = 'MFG-HOURLY' THEN 'CMP-MFG'
		WHEN CMB.CUSTOMER_ID = 'BOE609' AND P.PLANNER_USER_ID = 'MFG' THEN 'CMP-MFG'
		ELSE UDF.STRING_VAL
	END AS BUS_UNIT
	, P.PLANNER_USER_ID
	, P.BUYER_USER_ID
	, B.Salesperson_ID
	, CMB.CO_CREATE_DATE
	, CMB.WO_CREATE_DATE
	, CMB.SHIPPED_DATE
	, CMB.CUSTOMER_ID
	, CMB.CUST_ORDER_ID
	, CMB.COL_LINE
	, CMB.BASE_ID
	, CMB.LOT_ID
	, CMB.SPLIT_ID
	, CMB.SUB_ID
	, CMB.WO_STATUS
	, CMB.PACKLIST_ID
	, CMB.TRANSACTION_ID
	, CMB.INVOICE_ID
	, CMB.INV_LINE
	, CMB.PART_ID
	, CMB.SHIP_SHORT
	, CMB.TRACE_ID
	, CMB.COL_UNIT_PRICE
	, CMB.REC_AMOUNT

----> QUANTITIES
	, CMB.TOTAL_SHIPPED_QTY
	, CMB.COL_ORDER_QTY
	, CMB.DESIRED_QTY
	, CMB.TRACE_QTY
	, CMB.SHIPPED_QTY
	, CMB.REC_QTY

-->  audit purposes
	, CTE01.TOTAL_CHARGE
	, CTE01.PART_QTY
	, CTE01.UNIT_CHARGE_ALLOC
	, BOE.[PART NUMBER]
	, BOE.SUM_EXTENDED_DOLLARS
	, BOE.TOTAL_HOURS
	, AMT.AMT_BASE_ID
	, AMT.AMT_REC_AMOUNT
	, AMT.AMT_BURDEN_COST
	, AMT.AMT_MATERIAL_COST
	, AMT.AMT_LABOR_COST
	, AMT.AMT_SERVICE_COST
-->
----> WO/TRACE COSTS
	, CMB.TOTAL_BURDEN_COST
	, CMB.TOTAL_MATERIAL_COST
	, CMB.TOTAL_LABOR_COST
	, CMB.TOTAL_SERVICE_COST
	, CMB.TOTAL_COST
	--, CMB.UNIT_BURDEN_COST
	--, CMB.UNIT_MATERIAL_COST
	--, CMB.UNIT_LABOR_COST
	--, CMB.UNIT_SERVICE_COST
	, CMB.UNIT_TOTAL_COST

	, CASE
		WHEN CMB.BASE_ID LIKE '%R%' THEN 0						--> REWORK
		WHEN CMB.INVOICE_ID LIKE '%DM%' THEN 0						--> DM
		WHEN CMB.INVOICE_ID LIKE '%CM%' THEN 0						--> CM
		WHEN CMB.COL_ORDER_QTY <= 0 THEN 0
		ELSE ISNULL(CTE01.UNIT_CHARGE_ALLOC, 0) * ISNULL(REC_QTY, SHIPPED_QTY)
	END AS TRACE_CHARGE_ALLOC

	, CASE
		WHEN AMT.CUST_ORDER_ID IS NOT NULL THEN AMT.AMT_BURDEN_COST
		WHEN CMB.CUSTOMER_ID = 'AMTRIB' AND CMB.BASE_ID != REPLACE(REPLACE(CMB.BASE_ID, 'A', ''), 'B', '') THEN 0
		ELSE CMB.UNIT_BURDEN_COST * ISNULL(TRACE_QTY, SHIPPED_QTY)
	END AS TRACE_BURDEN_COST
	, CASE
		WHEN AMT.CUST_ORDER_ID IS NOT NULL THEN AMT.AMT_SERVICE_COST
		WHEN CMB.CUSTOMER_ID = 'AMTRIB' AND CMB.BASE_ID != REPLACE(REPLACE(CMB.BASE_ID, 'A', ''), 'B', '') THEN 0
		ELSE CMB.UNIT_SERVICE_COST * ISNULL(TRACE_QTY, SHIPPED_QTY)
	END AS TRACE_SERVICE_COST
	, CASE
		WHEN AMT.CUST_ORDER_ID IS NOT NULL THEN AMT.AMT_LABOR_COST
		WHEN CMB.CUSTOMER_ID = 'AMTRIB' AND CMB.BASE_ID != REPLACE(REPLACE(CMB.BASE_ID, 'A', ''), 'B', '') THEN 0
		ELSE CMB.UNIT_LABOR_COST * ISNULL(TRACE_QTY, SHIPPED_QTY)
	END AS TRACE_LABOR_COST
	, CASE
		WHEN AMT.CUST_ORDER_ID IS NOT NULL THEN AMT.AMT_MATERIAL_COST
		WHEN CMB.CUSTOMER_ID = 'AMTRIB' AND CMB.BASE_ID != REPLACE(REPLACE(CMB.BASE_ID, 'A', ''), 'B', '') THEN 0
		ELSE CMB.UNIT_MATERIAL_COST * ISNULL(TRACE_QTY, SHIPPED_QTY)
	END AS TRACE_MATERIAL_COST

	, CASE
		WHEN AMT.CUST_ORDER_ID IS NOT NULL THEN AMT.AMT_BURDEN_COST / ISNULL(CMB.DESIRED_QTY, 1)
		WHEN CMB.CUSTOMER_ID = 'AMTRIB' AND CMB.BASE_ID != REPLACE(REPLACE(CMB.BASE_ID, 'A', ''), 'B', '') THEN 0
		WHEN INVOICE_ID LIKE '%DM%' THEN 0						--> DM
		WHEN INVOICE_ID LIKE '%CM%' THEN 0						--> CM
		ELSE CMB.UNIT_BURDEN_COST 
	END AS UNIT_BURDEN_COST
	, CASE
		WHEN AMT.CUST_ORDER_ID IS NOT NULL THEN AMT.AMT_MATERIAL_COST / ISNULL(CMB.DESIRED_QTY, 1)
		WHEN CMB.CUSTOMER_ID = 'AMTRIB' AND CMB.BASE_ID != REPLACE(REPLACE(CMB.BASE_ID, 'A', ''), 'B', '') THEN 0
		WHEN INVOICE_ID LIKE '%DM%' THEN 0						--> DM
		WHEN INVOICE_ID LIKE '%CM%' THEN 0						--> CM
		ELSE CMB.UNIT_MATERIAL_COST
	END AS UNIT_MATERIAL_COST
	, CASE
		WHEN AMT.CUST_ORDER_ID IS NOT NULL THEN AMT.AMT_LABOR_COST / ISNULL(CMB.DESIRED_QTY, 1)
		WHEN CMB.CUSTOMER_ID = 'AMTRIB' AND CMB.BASE_ID != REPLACE(REPLACE(CMB.BASE_ID, 'A', ''), 'B', '') THEN 0
		WHEN INVOICE_ID LIKE '%DM%' THEN 0						--> DM
		WHEN INVOICE_ID LIKE '%CM%' THEN 0						--> CM
		ELSE CMB.UNIT_LABOR_COST
	END AS UNIT_LABOR_COST
	, CASE
		WHEN AMT.CUST_ORDER_ID IS NOT NULL THEN AMT.AMT_SERVICE_COST / ISNULL(CMB.DESIRED_QTY, 1)
		WHEN CMB.CUSTOMER_ID = 'AMTRIB' AND CMB.BASE_ID != REPLACE(REPLACE(CMB.BASE_ID, 'A', ''), 'B', '') THEN 0
		WHEN INVOICE_ID LIKE '%DM%' THEN 0						--> DM
		WHEN INVOICE_ID LIKE '%CM%' THEN 0						--> CM
		ELSE CMB.UNIT_SERVICE_COST
	END AS UNIT_SERVICE_COST

	, CASE 
		WHEN AMT.CUST_ORDER_ID IS NOT NULL THEN AMT.AMT_REC_AMOUNT
		WHEN BOE.ID IS NOT NULL THEN BOE.SUM_EXTENDED_DOLLARS
		ELSE CMB.REC_AMOUNT
	END AS TOTAL_INVOICE_AMOUNT

	, CASE 
		WHEN AMT.CUST_ORDER_ID IS NOT NULL THEN AMT.AMT_REC_AMOUNT
		WHEN BOE.ID IS NOT NULL THEN BOE.SUM_EXTENDED_DOLLARS
		ELSE CMB.REC_AMOUNT / ISNULL(REC_QTY, SHIPPED_QTY)
	END AS UNIT_INVOICE_AMOUNT

	, CASE
		WHEN AMT.CUST_ORDER_ID IS NOT NULL THEN AMT.AMT_REC_AMOUNT
		WHEN BOE.ID IS NOT NULL THEN BOE.SUM_EXTENDED_DOLLARS
		ELSE (CMB.REC_AMOUNT / ISNULL(REC_QTY, SHIPPED_QTY)) * ISNULL(TRACE_QTY, SHIPPED_QTY)
	END AS TRACE_INVOICE_AMOUNT

FROM CTE_COMBINED CMB 
LEFT OUTER JOIN #BUS_UNIT UDF
	ON UDF.DOCUMENT_ID = CMB.CUSTOMER_ID
LEFT OUTER JOIN DATAWAREHOUSE.dbo.PART_SITE P
	ON P.PART_ID = CMB.PART_ID
LEFT OUTER JOIN LIVESupplemental.dbo.buyer_assn B
	ON LEFT(P.BUYER_USER_ID, 2) = B.Account_ID
LEFT OUTER JOIN CTE_CHGAMTS CTE01
	ON CMB.CUST_ORDER_ID = CTE01.CUST_ORDER_ID
LEFT OUTER JOIN #CTE_BOE BOE
	ON BOE.ID = CMB.CUST_ORDER_ID
--	AND BOE.BASE_ID = CMB.BASE_ID
LEFT OUTER JOIN CTE_AMTRIB AMT
	ON AMT.CUST_ORDER_ID = CMB.CUST_ORDER_ID
	AND AMT.AMT_BASE_ID = CMB.BASE_ID
)
--SELECT * FROM CTE_MERGE WHERE CUST_ORDER_ID = '459410'

SELECT CASE
		WHEN CUSTOMER_ID = 'AMTRIB' THEN 'AMTRIB'
		WHEN CUSTOMER_ID = 'SPIAER' THEN 'SPIAER'
		WHEN INVOICE_ID LIKE '%CM' OR INVOICE_ID LIKE '%DM' THEN 'DM/CM'
		WHEN BASE_ID LIKE '%R%' THEN 'REWORK'
		WHEN INVOICE_ID IS NULL THEN 'NO INVOICE YET'
		WHEN BASE_ID IS NULL AND REC_QTY < 0 THEN 'RETURN RECORD'
		WHEN TOTAL_SHIPPED_QTY = 0 AND REC_QTY < 0 THEN 'RETURN NO RESHIPMENT'
		WHEN TRACE_INVOICE_AMOUNT < 0 AND REC_QTY < 0 THEN 'RETURN'
		WHEN TRACE_INVOICE_AMOUNT < 0 AND REC_QTY > 0 THEN 'CREDIT'
		WHEN TRACE_INVOICE_AMOUNT = 0 AND TRACE_CHARGE_ALLOC > 0 THEN 'CHARGE'
		WHEN REC_AMOUNT = 0 AND TRACE_INVOICE_AMOUNT = 0 AND REC_QTY != 0 THEN '$0 INVOICE???'
--		WHEN REC_AMOUNT = 0 AND CHARGE_ALLOC = 0 AND MRG.[STATUS] = 'X' THEN 'CANCELLED'
		WHEN TRACE_INVOICE_AMOUNT > 0 THEN 'INVOICED'
		ELSE 'ISSUES'
	END AS REC_ISSUES
	, MRG.*
	, (TRACE_CHARGE_ALLOC + TRACE_INVOICE_AMOUNT) AS INV_TRACE_PRICE
	, (TRACE_CHARGE_ALLOC + TRACE_INVOICE_AMOUNT) / ISNULL(TRACE_QTY, SHIPPED_QTY) AS INV_TRACE_UNIT_PRICE
	, (TRACE_BURDEN_COST + TRACE_MATERIAL_COST + TRACE_LABOR_COST + TRACE_SERVICE_COST) AS TRACE_TOTAL_COST
	, UNIT_TOTAL_COST AS TRACE_UNIT_COST

INTO #RESULTS
FROM CTE_MERGE MRG
