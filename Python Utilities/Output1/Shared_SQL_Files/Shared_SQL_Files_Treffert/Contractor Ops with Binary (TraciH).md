# Contractor Ops with Binary (TraciH)

```sql
USE LIVE
GO

SELECT CO.CUSTOMER_ID
	, WO.BASE_ID
	, WO.LOT_ID
	, WO.SPLIT_ID
	, WO.SUB_ID
	, WO.DESIRED_WANT_DATE
	, WO.PART_ID
	, O.SEQUENCE_NO
	, O.RESOURCE_ID
	, O.OPERATION_TYPE
	, CAST(CAST(OB.BITS AS VARBINARY(MAX)) AS NVARCHAR(MAX)) OPERATION_BINARY
	, CAST(CAST(OTB.BITS AS VARBINARY(MAX)) AS NVARCHAR(MAX)) OPTYPE_BINARY
FROM WORK_ORDER WO
INNER JOIN OPERATION O
	ON WO.BASE_ID = O.WORKORDER_BASE_ID
	AND WO.LOT_ID = O.WORKORDER_LOT_ID
	AND WO.SPLIT_ID = O.WORKORDER_SPLIT_ID
	AND WO.SUB_ID = O.WORKORDER_SUB_ID
	AND WO.[TYPE] = O.WORKORDER_TYPE
LEFT OUTER JOIN DEMAND_SUPPLY_LINK  DSL
	ON  O.[WORKORDER_BASE_ID] = DSL.[SUPPLY_BASE_ID]
	AND O.[WORKORDER_LOT_ID] = DSL.[SUPPLY_LOT_ID]
	AND O.[WORKORDER_SPLIT_ID] = DSL.[SUPPLY_SPLIT_ID]
	AND O.[WORKORDER_SUB_ID] = DSL.[SUPPLY_SUB_ID]
	AND DSL.[SUPPLY_TYPE] = 'WO'
LEFT OUTER JOIN CUSTOMER_ORDER CO
	ON DSL.DEMAND_TYPE = 'CO'
	AND DSL.DEMAND_BASE_ID = CO.ID  
LEFT OUTER JOIN OPER_TYPE_BINARY OTB
	ON O.OPERATION_TYPE = OTB.OPERATION_TYPE_ID
	AND OTB.SITE_ID = 'SK01'
	AND OTB.[TYPE] = 'D'
LEFT OUTER JOIN OPERATION_BINARY OB
	ON WO.BASE_ID = OB.WORKORDER_BASE_ID
	AND WO.LOT_ID = OB.WORKORDER_LOT_ID
	AND WO.SPLIT_ID = OB.WORKORDER_SPLIT_ID
	AND WO.SUB_ID = OB.WORKORDER_SUB_ID
	AND WO.[TYPE] = OB.WORKORDER_TYPE
	AND O.SEQUENCE_NO = OB.SEQUENCE_NO
WHERE RESOURCE_ID = 'CONTRACTOR'
AND WO.DESIRED_WANT_DATE >= '5/1/2018'
AND WO.[TYPE] = 'W'

ORDER BY BASE_ID
```
