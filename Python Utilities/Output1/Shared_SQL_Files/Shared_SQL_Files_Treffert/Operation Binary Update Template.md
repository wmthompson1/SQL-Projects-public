# Operation Binary Update Template

```sql
USE LIVE
GO

DROP TABLE #T1
SELECT OB.WORKORDER_TYPE
	, OB.WORKORDER_BASE_ID
	, OB.WORKORDER_LOT_ID
	, OB.WORKORDER_SPLIT_ID
	, OB.WORKORDER_SUB_ID
	, OB.SEQUENCE_NO
	, O.RESOURCE_ID
	, O.OPERATION_TYPE
	, O.[STATUS]
	, CAST(CAST(OB.BITS AS VARBINARY(MAX)) AS NVARCHAR(MAX)) AS [BEFORE_OB]
	, REPLACE(CAST(CAST(OB.BITS AS VARBINARY(MAX)) AS NVARCHAR(MAX)) 
		, 'INDICATE MASKANT USED:  PLUGS / TAPE / LIQUID'
		, 'MASKANT USED: PLUGS'
	) AS [AFTER_OB]
	, OB.BITS AS [OB_BINARY]
	, OB.BITS_LENGTH AS [OB_BITLEN]
INTO #T1
FROM OPERATION O
INNER JOIN OPERATION_BINARY OB
	ON O.WORKORDER_TYPE = OB.WORKORDER_TYPE
	AND O.WORKORDER_BASE_ID = OB.WORKORDER_BASE_ID
	AND O.WORKORDER_LOT_ID = OB.WORKORDER_LOT_ID
	AND O.WORKORDER_SPLIT_ID = OB.WORKORDER_SPLIT_ID
	AND O.WORKORDER_SUB_ID = OB.WORKORDER_SUB_ID
	AND O.SEQUENCE_NO = OB.SEQUENCE_NO
WHERE (O.WORKORDER_TYPE = 'M')
	AND RESOURCE_ID = 'P1F1-PEN-MASK'
--	AND CAST(CAST(OB.BITS AS VARBINARY(MAX)) AS NVARCHAR(MAX)) LIKE '%MASKANT USED: PLUGS%'
	AND CAST(CAST(OB.BITS AS VARBINARY(MAX)) AS NVARCHAR(MAX)) LIKE '%INDICATE MASKANT USED:  PLUGS / TAPE / LIQUID%'
--	AND OB.WORKORDER_BASE_ID IN ()

SELECT * FROM #T1
```
