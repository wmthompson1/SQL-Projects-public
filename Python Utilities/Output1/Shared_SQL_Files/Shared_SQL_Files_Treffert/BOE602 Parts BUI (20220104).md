# BOE602 Parts BUI (20220104)

```sql
USE DataWarehouse
GO

; WITH CTE_PARTS AS
(SELECT COL.PART_ID
FROM CUST_ORDER_LINE COL
INNER JOIN CUSTOMER_ORDER CO
ON CO.ID = COL.CUST_ORDER_ID
WHERE CUSTOMER_ID = 'BOE602'
GROUP BY COL.PART_ID
)

SELECT CO.CUSTOMER_ID
--, COL.CUST_ORDER_ID
--, COL.LINE_NO
, COL.PART_ID
, P.BUYER_USER_ID
FROM CTE_PARTS CTE
LEFT OUTER JOIN PART P
ON P.ID = CTE.PART_ID
LEFT OUTER JOIN CUST_ORDER_LINE COL
ON COL.PART_ID = CTE.PART_ID
LEFT OUTER JOIN CUSTOMER_ORDER CO
ON CO.ID = COL.CUST_ORDER_ID
WHERE ISNULL(CTE.PART_ID, '') != ''
GROUP BY CO.CUSTOMER_ID
, COL.PART_ID
, P.BUYER_USER_ID
ORDER BY COL.PART_ID
, P.BUYER_USER_ID

SELECT CO.ID
, CO.CUSTOMER_ID
, CO.ORDER_DATE
, COL.PART_ID
FROM CUST_ORDER_LINE COL
INNER JOIN CUSTOMER_ORDER CO
ON CO.ID = COL.CUST_ORDER_ID
WHERE PART_ID IN ('112A2411-49', '140T2103-1778M-N1690100748')
```
