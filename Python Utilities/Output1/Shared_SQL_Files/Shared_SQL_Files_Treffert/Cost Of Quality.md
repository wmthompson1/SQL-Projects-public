# Cost Of Quality

```sql
USE [Datamart]
GO
/****** Object:  StoredProcedure [dbo].[usp_Cost_Of_Quality]    Script Date: 9/30/2019 8:41:26 AM ******/
SET ANSI_NULLS ON
GO
SET QUOTED_IDENTIFIER ON
GO
--ALTER PROCEDURE [dbo].[usp_Cost_Of_Quality]  
--AS
--BEGIN

--DROP TABLE DataWarehouse.dbo.Cost_Of_Quality

SET NOCOUNT ON;

SET TRANSACTION ISOLATION LEVEL READ UNCOMMITTED
SET DEADLOCK_PRIORITY LOW

--DECLARE @START DATE = DATEADD(YEAR,-1,DATEADD(DAY,1,(EOMONTH(DATEADD(MONTH,-1,GETDATE()))))),
DECLARE @START DATE = '1/1/2019',
@END DATE = GETDATE()-1
;
IF OBJECT_ID('tempdb..#WO') IS NOT NULL DROP TABLE #WO
IF OBJECT_ID('tempdb..#OPS') IS NOT NULL DROP TABLE #OPS
IF OBJECT_ID('tempdb..#COSTS') IS NOT NULL DROP TABLE #COSTS
IF OBJECT_ID('tempdb..#BUS_UNIT') IS NOT NULL DROP TABLE #BUS_UNIT

SELECT DOCUMENT_ID AS CUST_ID
	, STRING_VAL AS BUS_UNIT
INTO #BUS_UNIT
FROM [SQL-LAB-2].[LIVE].dbo.USER_DEF_FIELDS UDF
WHERE PROGRAM_ID = 'VMCUSMNT'
	AND UDF.ID = 'UDF-0000072'
	AND DOCUMENT_ID IS NOT NULL
--SELECT * FROM #BUS_UNIT

SELECT WO.[TYPE] 
	, WO.BASE_ID
	, WO.LOT_ID
	, WO.SPLIT_ID
	, WO.SUB_ID
	, WO.PART_ID AS WO_PART_ID
	, WO2.PART_ID AS WO2_PART_ID
	, P.[DESCRIPTION]
	, P.[TOOL_OR_FIXTURE]
	, P.COMMODITY_CODE
	, P.PLANNER_USER_ID
	, P.BUYER_USER_ID
	, P.PURCHASED
	, P.FABRICATED
	, P.[MAT_GL_ACCT_ID]
	, P.[LAB_GL_ACCT_ID]
	, P.[BUR_GL_ACCT_ID]
	, P.[SER_GL_ACCT_ID]
	, WO.DESIRED_QTY
	, WO.[STATUS] AS WO_STATUS
	, WO.CREATE_DATE AS WO_CREATE_DATE
	, WO.CLOSE_DATE AS WO_CLOSE_DATE
	, WO.ACT_BURDEN_COST
	, WO.ACT_MATERIAL_COST
	, WO.ACT_LABOR_COST
	, WO.ACT_SERVICE_COST
INTO #WO
FROM WORK_ORDER WO
INNER JOIN DATAWAREHOUSE.dbo.WORK_ORDER WO2
	ON WO.BASE_ID = WO2.BASE_ID
	AND WO.LOT_ID = WO2.LOT_ID
	AND WO.SPLIT_ID = WO2.SPLIT_ID
	AND WO2.SUB_ID = '0'
	AND WO.[TYPE] = 'W'
INNER JOIN PART P
	ON WO.PART_ID = P.ID
WHERE WO.CREATE_DATE >= @START
	AND WO.[STATUS] != 'F'
	AND WO.[TYPE] = 'W'

CREATE CLUSTERED INDEX CIDX_WO ON #WO ([TYPE], BASE_ID, LOT_ID, SPLIT_ID, SUB_ID);

SELECT O.WORKORDER_TYPE
	, O.WORKORDER_BASE_ID 
	, O.WORKORDER_LOT_ID
	, O.WORKORDER_SPLIT_ID
	, O.WORKORDER_SUB_ID
	, O.SEQUENCE_NO
	, O.RESOURCE_ID
	, O.CLOSE_DATE AS OPS_CLOSE_DATE
	, O.ACT_ATL_LAB_COST 
	, O.ACT_ATL_BUR_COST 
	, O.ACT_ATL_SER_COST 
	, O.ACT_TTL_MAT_COST
	, SR.USER_1 AS SKILLS_TYPE
	, SR.USER_2 AS DEPARTMENT
	, SR.USER_3 AS SUBDEPT
	, SR.USER_4 AS SUPERVISOR
	, SR.USER_5 AS FACILITY
	, CASE
			WHEN SR.USER_1 = 'A' THEN 'FIN'
			WHEN SR.USER_1 = 'M' THEN 'MFG'
			WHEN SR.USER_1 = 'O' THEN 'OUTREACH'
			WHEN SR.USER_1 = 'B' THEN 'FIN'
			WHEN SR.USER_1 = 'C' THEN 'CLOSED'
	END AS BUS_STREAM
	, DATEPART(Week, O.CLOSE_DATE) as WEEK_NUM
INTO #OPS
FROM OPERATION O
INNER JOIN SHOP_RESOURCE SR
ON SR.ID = O.RESOURCE_ID
WHERE EXISTS 
	(SELECT 1 FROM #WO WO2
	WHERE WO2.[TYPE] = O.WORKORDER_TYPE
	AND WO2.BASE_ID = O.WORKORDER_BASE_ID
	AND WO2.LOT_ID = O.WORKORDER_LOT_ID
	AND WO2.SPLIT_ID = O.WORKORDER_SPLIT_ID
	AND WO2.SUB_ID = O.WORKORDER_SUB_ID
	)	

CREATE CLUSTERED INDEX CIDX_OPS ON #OPS (WORKORDER_TYPE, WORKORDER_BASE_ID, WORKORDER_LOT_ID, WORKORDER_SPLIT_ID, WORKORDER_SUB_ID);
--SELECT * FROM #OPS
-- SELECT 1081311 - 1035945

SELECT O.WORKORDER_TYPE
	, O.WORKORDER_BASE_ID 
	, O.WORKORDER_LOT_ID
	, O.WORKORDER_SPLIT_ID
	, O.WORKORDER_SUB_ID
	, SUM(ACT_ATL_LAB_COST) AS LABOR_COSTS
	, SUM(ACT_ATL_BUR_COST) AS BURDEN_COSTS
	, SUM(ACT_ATL_SER_COST) AS SERVICE_COSTS
	, SUM(ACT_TTL_MAT_COST) AS MATERIAL_COSTS
INTO #COSTS
FROM #OPS O
GROUP BY O.WORKORDER_TYPE
	, O.WORKORDER_BASE_ID 
	, O.WORKORDER_LOT_ID
	, O.WORKORDER_SPLIT_ID
	, O.WORKORDER_SUB_ID

CREATE CLUSTERED INDEX CIDX_COSTS ON #COSTS (WORKORDER_TYPE, WORKORDER_BASE_ID, WORKORDER_LOT_ID, WORKORDER_SPLIT_ID, WORKORDER_SUB_ID);

SELECT CASE
		WHEN WO.COMMODITY_CODE = 'TOOLS' OR WO.PLANNER_USER_ID = 'TOOLING' THEN 'TOOLING'
		WHEN WO.WO_STATUS = 'X' AND SPLIT_ID <> '0' THEN 'SCRAP'
		WHEN WO.WO_STATUS = 'X' THEN 'CANCELLED'
		WHEN WO.BASE_ID LIKE '%R%' OR WO.WO_PART_ID LIKE '%REWORK%' THEN 'REWORK'
		WHEN (WO.ACT_BURDEN_COST + WO.ACT_LABOR_COST + WO.ACT_MATERIAL_COST + WO.ACT_SERVICE_COST) < 0 THEN 'NEG COSTS'
	END AS [DEFINITION]
	, BU.BUS_UNIT
	, CO.CUSTOMER_ID
	, COL.CUST_ORDER_ID
	, COL.LINE_NO
	, CASE
		WHEN WO.WO_PART_ID LIKE '%REWORK%' THEN WO2_PART_ID
		ELSE WO.WO_PART_ID
	END AS PART_ID
	, WO.*
	, OC.LABOR_COSTS
	, OC.BURDEN_COSTS
	, OC.SERVICE_COSTS
	, OC.MATERIAL_COSTS
	, OC.LABOR_COSTS + OC.BURDEN_COSTS + OC.SERVICE_COSTS + OC.MATERIAL_COSTS AS TOTAL_COSTS
	, O.SEQUENCE_NO
	, O.RESOURCE_ID
	, O.ACT_ATL_LAB_COST 
	, O.ACT_ATL_BUR_COST 
	, O.ACT_ATL_SER_COST 
	, O.ACT_TTL_MAT_COST
	, O.OPS_CLOSE_DATE
	, O.SKILLS_TYPE
	, O.DEPARTMENT
	, O.SUBDEPT
	, O.SUPERVISOR
	, O.BUS_STREAM
	, O.WEEK_NUM
	, CASE
		WHEN O.SKILLS_TYPE = 'M' THEN 1
		ELSE 0
	END AS MFG_COUNT
	, CASE
		WHEN O.SKILLS_TYPE IN ('A', 'B') THEN 1
		ELSE 0
	END AS FIN_COUNT
	, CASE
		WHEN O.SKILLS_TYPE NOT IN ('A', 'B', 'M') THEN 1
		ELSE 0
	END AS OTHER_COUNT
	, 1 AS TOTAL_COUNT
FROM #WO WO
INNER JOIN #OPS O
	ON O.WORKORDER_BASE_ID = WO.BASE_ID
	AND O.WORKORDER_LOT_ID = WO.LOT_ID
	AND O.WORKORDER_SPLIT_ID = WO.SPLIT_ID
	AND O.WORKORDER_SUB_ID = WO.SUB_ID
	AND O.WORKORDER_TYPE = WO.[TYPE]
LEFT OUTER JOIN #COSTS OC
	ON OC.WORKORDER_BASE_ID = O.WORKORDER_BASE_ID
	AND OC.WORKORDER_LOT_ID = O.WORKORDER_LOT_ID
	AND OC.WORKORDER_SPLIT_ID = O.WORKORDER_SPLIT_ID
	AND OC.WORKORDER_SUB_ID = O.WORKORDER_SUB_ID
	AND OC.WORKORDER_TYPE = O.WORKORDER_TYPE
LEFT OUTER JOIN DATAWAREHOUSE.dbo.DEMAND_SUPPLY_LINK DSL
	ON DSL.SUPPLY_TYPE = 'WO'
	AND DSL.SUPPLY_BASE_ID = WO.BASE_ID
	AND DSL.SUPPLY_LOT_ID = WO.LOT_ID
LEFT OUTER JOIN DATAWAREHOUSE.dbo.CUST_ORDER_LINE COL
	ON DSL.DEMAND_TYPE = 'CO'
	AND DSL.DEMAND_BASE_ID = COL.CUST_ORDER_ID
	AND DSL.DEMAND_SEQ_NO = COL.LINE_NO
LEFT OUTER JOIN DATAWAREHOUSE.dbo.CUSTOMER_ORDER CO
	ON COL.CUST_ORDER_ID = CO.ID
LEFT OUTER JOIN #BUS_UNIT BU
	ON CO.CUSTOMER_ID = BU.CUST_ID

where 
	WO.WO_STATUS = 'X' AND SPLIT_ID <> '0' 

ORDER BY BASE_ID
	, LOT_ID
	, SPLIT_ID
	, SUB_ID
	, SEQUENCE_NO

--END
```
