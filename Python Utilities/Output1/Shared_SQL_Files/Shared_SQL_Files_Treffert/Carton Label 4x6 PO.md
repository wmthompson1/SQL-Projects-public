# Carton Label 4x6 PO

```sql
USE LIVE
GO

DECLARE @PURCHASE_ORDER NVARCHAR(20) = '145124'
, @POLINE NVARCHAR(5) = '1'
, @LOCATION NVARCHAR(20) = ''

SELECT RL.RECEIVER_ID
, RL.LINE_NO
, RL.PURC_ORDER_ID
, RL.PURC_ORDER_LINE_NO
, PL.PART_ID
, CASE
	WHEN PS.PRIMARY_WHS_ID IS NULL THEN 'ALTERNATE'
	ELSE 'PRIMARY'
END + ' > ' + PL.WAREHOUSE_ID + '/' + PL.LOCATION_ID
AS [PART_LOCATION]
, ISNULL(UDF.STRING_VAL, DBO.SPLIT(P.[DESCRIPTION], 2, ' ')) AS MATL_TYPE
, T.APROPERTY_2
FROM RECEIVER_LINE RL
INNER JOIN PURC_ORDER_LINE POL
	ON POL.PURC_ORDER_ID = RL.PURC_ORDER_ID
	AND POL.LINE_NO = RL.PURC_ORDER_LINE_NO
LEFT OUTER JOIN PART P
	ON P.ID = POL.PART_ID
LEFT OUTER JOIN PART_LOCATION PL
	ON POL.PART_ID = PL.PART_ID
	AND PL.[STATUS] = 'A'
LEFT OUTER JOIN PART_SITE PS
	ON PS.PART_ID = POL.PART_ID
	AND PL.WAREHOUSE_ID = PS.PRIMARY_WHS_ID
	AND PL.LOCATION_ID = PS.PRIMARY_LOC_ID
LEFT OUTER JOIN USER_DEF_FIELDS UDF
	ON UDF.PROGRAM_ID = 'VMPRTMNT'
	AND UDF.ID = 'UDF-0000035'
	AND UDF.DOCUMENT_ID = POL.PART_ID
LEFT OUTER JOIN TRACE_INV_TRANS TI
	ON TI.TRANSACTION_ID = RL.TRANSACTION_ID
LEFT OUTER JOIN TRACE T
	ON TI.TRACE_ID = T.ID
WHERE RL.PURC_ORDER_ID = @PURCHASE_ORDER
AND RL.PURC_ORDER_LINE_NO = @POLINE
AND (CASE
	WHEN PS.PRIMARY_WHS_ID IS NULL THEN 'ALTERNATE'
	ELSE 'PRIMARY'
END + ' > ' + PL.WAREHOUSE_ID + '/' + PL.LOCATION_ID) = @LOCATION



SELECT CASE
	WHEN PS.PRIMARY_WHS_ID IS NULL THEN 'ALTERNATE'
	ELSE 'PRIMARY'
END + ' > ' + PL.WAREHOUSE_ID + '/' + PL.LOCATION_ID
AS [PART_LOCATION]
FROM RECEIVER_LINE RL
INNER JOIN PURC_ORDER_LINE POL
	ON POL.PURC_ORDER_ID = RL.PURC_ORDER_ID
	AND POL.LINE_NO = RL.PURC_ORDER_LINE_NO
LEFT OUTER JOIN PART P
	ON P.ID = POL.PART_ID
LEFT OUTER JOIN PART_LOCATION PL
	ON POL.PART_ID = PL.PART_ID
	AND PL.[STATUS] = 'A'
LEFT OUTER JOIN PART_SITE PS
	ON PS.PART_ID = POL.PART_ID
	AND PL.WAREHOUSE_ID = PS.PRIMARY_WHS_ID
	AND PL.LOCATION_ID = PS.PRIMARY_LOC_ID
WHERE RL.PURC_ORDER_ID = @PURCHASE_ORDER
AND RL.PURC_ORDER_LINE_NO = @POLINE


```
