USE LIVE
GO

DROP TABLE #T1

; WITH CTE_BASE AS		
(SELECT WO.BASE_ID + '/' + WO.LOT_ID + CASE WHEN WO.SPLIT_ID = 0 THEN '' ELSE '.' + WO.SPLIT_ID END AS JOB_ID		
	, WO.BASE_ID 	
	, WO.LOT_ID 	
	, WO.SPLIT_ID	
	, WO.SUB_ID	
	, WO.[STATUS] AS WO_STATUS	
	, dbo.sfn_current_operation(WO.[TYPE], WO.BASE_ID, WO.LOT_ID, WO.SPLIT_ID, WO.SUB_ID) AS CURRENT_OP	
	, MAX(O.SEQUENCE_NO) AS MAX_OPS
FROM OPERATION O
INNER JOIN WORK_ORDER WO
ON WO.BASE_ID = O.WORKORDER_BASE_ID
AND WO.LOT_ID = O.WORKORDER_LOT_ID
AND WO.SPLIT_ID = O.WORKORDER_SPLIT_ID
AND WO.SUB_ID = O.WORKORDER_SUB_ID
AND WO.[TYPE] = O.WORKORDER_TYPE
WHERE O.WORKORDER_TYPE = 'W'
AND O.[STATUS] IN ('C')
AND WO.[STATUS] IN ('R', 'F')
AND ISNULL(WO.[CLOSE_DATE], GETDATE()) > GETDATE() - 30
GROUP BY WO.[TYPE]
	, WO.BASE_ID 	
	, WO.LOT_ID 	
	, WO.SPLIT_ID	
	, WO.SUB_ID	
	, WO.[STATUS]
)
		
SELECT JOB_ID		
	, BASE_ID 	
	, LOT_ID 	
	, SPLIT_ID	
	, SUB_ID	
	, WO_STATUS	
	, CURRENT_OP	
	, SEQUENCE_NO	
	, O.RESOURCE_ID
	, O.[STATUS] AS OP_STATUS
INTO #T1
FROM CTE_BASE CTE		
LEFT OUTER JOIN OPERATION O		
	ON O.WORKORDER_TYPE = 'W'	
	AND CTE.BASE_ID = O.WORKORDER_BASE_ID	
	AND CTE.LOT_ID = O.WORKORDER_LOT_ID	
	AND CTE.SPLIT_ID = O.WORKORDER_SPLIT_ID	
	AND CTE.SUB_ID = O.WORKORDER_SUB_ID	
	AND CTE.CURRENT_OP > O.SEQUENCE_NO	
WHERE O.[STATUS] NOT IN ('C', 'X')	
ORDER BY JOB_ID		
	, BASE_ID 	
	, LOT_ID 	
	, SPLIT_ID	
	, SUB_ID	
	, SEQUENCE_NO	

SELECT * FROM #T1
