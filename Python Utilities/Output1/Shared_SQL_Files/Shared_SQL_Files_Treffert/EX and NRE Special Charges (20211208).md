# EX and NRE Special Charges (20211208)

```sql
SELECT R.INVOICE_ID
	, R.CUSTOMER_ID
	, R.[TYPE]
	, R.INVOICE_DATE
	, R.TOTAL_AMOUNT
	, R.[STATUS]
	, RL.CUST_ORDER_ID
	, RL.QTY
	, DSL.CUST_ORDER_ID
	, COL.SERVICE_CHARGE_ID
FROM RECEIVABLE R
INNER JOIN RECEIVABLE_LINE RL
	ON R.INVOICE_ID = RL.INVOICE_ID
INNER JOIN DEMAND_SUPPLY_LINK_3 DSL
	ON RL.CUST_ORDER_ID = DSL.CUST_ORDER_ID
	AND RL.CUST_ORDER_LINE_NO = DSL.LINE_NO
	AND RL.PACKLIST_ID = DSL.PACKLIST_ID 
	AND RL.PACKLIST_LINE_NO = DSL.PACK_LINE
LEFT OUTER JOIN CUST_ORDER_LINE COL
	ON RL.CUST_ORDER_ID = COL.CUST_ORDER_ID
	AND RL.CUST_ORDER_LINE_NO = COL.LINE_NO
WHERE R.CUSTOMER_ID IN (@CUSTID) -->(N'BOEDEF', N'BOEPOP') 
	AND COL.SERVICE_CHARGE_ID IS NOT NULL
	AND R.[STATUS] <> N'X' 
	AND R.INVOICE_DATE >= @STARTDATE
ORDER BY DSL.CUST_ORDER_ID
```
