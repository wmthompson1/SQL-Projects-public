```sql
USE LIVE
GO

--SELECT * FROM LIVESupplemental..[MasterPull_20220203]

; WITH CTE_COL AS 
(SELECT CO.CUSTOMER_ID
	, COL.PART_ID
	, LAST_ORDER_DATE
FROM (SELECT COL.PART_ID
		, MAX(CO.ORDER_DATE) AS LAST_ORDER_DATE
	FROM CUST_ORDER_LINE COL
	INNER JOIN CUSTOMER_ORDER CO
		ON COL.CUST_ORDER_ID = CO.ID
	WHERE EXISTS (SELECT 1 FROM LIVESupplemental..[MasterPull_20220203] M
		WHERE COL.PART_ID = M.BASE_ID
		)
	GROUP BY PART_ID
	) X
INNER JOIN CUST_ORDER_LINE COL
	ON COL.PART_ID = X.PART_ID
INNER JOIN CUSTOMER_ORDER CO
	ON COL.CUST_ORDER_ID = CO.ID
	AND X.LAST_ORDER_DATE = CO.ORDER_DATE
GROUP BY CO.CUSTOMER_ID
	, COL.PART_ID
	, LAST_ORDER_DATE
--ORDER BY CO.CUSTOMER_ID
--	, COL.PART_ID
--	, LAST_ORDER_DATE
)

, CTE_WOS AS 
(SELECT WO.PART_ID
	, MAX(WO.CREATE_DATE) AS LAST_CREATE_DATE
FROM WORK_ORDER WO
WHERE EXISTS (SELECT 1 FROM LIVESupplemental..[MasterPull_20220203] M
	WHERE WO.PART_ID = M.BASE_ID
	)
	AND WO.[TYPE] = 'W'
GROUP BY WO.PART_ID
)

, CTE_DOCS AS
(SELECT DRW.WORKORDER_BASE_ID
	, DRW.WORKORDER_LOT_ID 
	, DRW.WORKORDER_SPLIT_ID
	, DRW.WORKORDER_SUB_ID
	, DRW.WORKORDER_TYPE
	, DRW.SEQUENCE_NO
	, (SELECT D.ID + CASE
			WHEN D.REVISION_ID IS NULL THEN ' / REV: '
			ELSE ' / REV: ' + D.REVISION_ID
		END + CASE
			WHEN U1.STRING_VAL IS NULL THEN ' / ADCN: '
			ELSE ' / ADCN: ' + U1.STRING_VAL
		END 
	) AS DOCUMENT_ID
FROM WORK_ORDER WO
INNER JOIN DOCUMENT_REF_WO DRW
	ON DRW.WORKORDER_BASE_ID = WO.BASE_ID
	AND DRW.WORKORDER_SUB_ID = WO.SUB_ID
INNER JOIN DOCUMENT D
	ON DRW.DOCUMENT_ID = D.ID
LEFT JOIN USER_DEF_FIELDS U1  --> ADCN
	ON D.ID = U1.DOCUMENT_ID
	AND U1.PROGRAM_ID = 'VMDOCMNT'
	AND U1.ID = 'UDF-0000032'
WHERE EXISTS (SELECT 1 FROM LIVESupplemental..[MasterPull_20220203] M
	WHERE WO.BASE_ID = M.BASE_ID
		AND WO.LOT_ID = M.LOT_ID
		AND WO.SPLIT_ID = M.SPLIT_ID
		AND WO.SUB_ID = '0'
		AND WO.[TYPE] = 'M'
	)
	AND D.ID LIKE 'BAC%'
)
--SELECT * FROM CTE_DOCS

SELECT M.*
, COL.CUSTOMER_ID
, COL.LAST_ORDER_DATE
, WO.LAST_CREATE_DATE
, P.QTY_ON_HAND
, STUFF((SELECT CHAR(032) + T.DOCUMENT_ID
	FROM (SELECT WORKORDER_BASE_ID
			, WORKORDER_SUB_ID
			, SEQUENCE_NO
			, DOCUMENT_ID 
		FROM CTE_DOCS 
		GROUP BY WORKORDER_BASE_ID
			, WORKORDER_SUB_ID
			, SEQUENCE_NO
			, DOCUMENT_ID
	) T
	WHERE (T.WORKORDER_BASE_ID = M.BASE_ID)
	FOR XML PATH('')), 1, 1, '') AS DOCUMENT
FROM LIVESupplemental..[MasterPull_20220203] M
LEFT OUTER JOIN CTE_COL COL
	ON COL.PART_ID = M.BASE_ID
LEFT OUTER JOIN CTE_WOS WO
	ON WO.PART_ID = M.BASE_ID
LEFT OUTER JOIN PART P
	ON M.BASE_ID = P.ID

```
