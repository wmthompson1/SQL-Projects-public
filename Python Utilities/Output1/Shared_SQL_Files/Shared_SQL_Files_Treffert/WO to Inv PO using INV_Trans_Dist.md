# WO to Inv PO using INV_Trans_Dist

```sql
USE DataWarehouse
GO

SELECT WO.BASE_ID
, WO.LOT_ID
, IT2.PURC_ORDER_ID
, IT2.PURC_ORDER_LINE_NO
, IT2.PART_ID
, IT2.QTY
FROM WORK_ORDER WO
LEFT OUTER JOIN INVENTORY_TRANS IT
ON IT.WORKORDER_BASE_ID = WO.BASE_ID
AND IT.WORKORDER_LOT_ID = WO.LOT_ID
LEFT OUTER JOIN INV_TRANS_DIST ITD
ON IT.TRANSACTION_ID = ITD.OUT_TRANS_ID
LEFT OUTER JOIN INVENTORY_TRANS IT2
ON ITD.IN_TRANS_ID = IT2.TRANSACTION_ID
WHERE WO.BASE_ID IN ('1594233', '1342914', '1463115', '1581739', '1583115', '1584517')
```
