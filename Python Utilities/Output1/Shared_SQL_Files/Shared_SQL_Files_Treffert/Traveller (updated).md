DECLARE @ReportSettings AS TABLE (Resource NVARCHAR(30), barcode_etch INT, barcode_pre_etch INT, 
barcode_review INT, heightchange INT, printuser INT, QTY_IN int, QTY_NOSHOW INT, QTY_OUT INT, SHIP_VIA INT, STAMP int)

INSERT INTO @ReportSettings

DECLARE @PART_ID NVARCHAR(30) = '113T1141-25',
@BASE_ID NVARCHAR(15) = '1579073',
@LOT_ID NVARCHAR(1) = '1',
@SPLIT_ID NVARCHAR(1) = '0'

SELECT @PART_ID = PART_ID 
FROM WORK_ORDER WO
WHERE WO.BASE_ID = @BASE_ID
	AND WO.LOT_ID = @LOT_ID
	AND WO.SUB_ID = '0'
	AND WO.SPLIT_ID = @SPLIT_ID
	AND WO.TYPE = 'W'

;
WITH CTE_WORKORDERS AS
(SELECT W.*
	, CONVERT(NVARCHAR(MAX), CONVERT(VARBINARY(MAX), WB.BITS)) AS WO_SPECS
	, P.DESCRIPTION
FROM WORK_ORDER W
LEFT OUTER JOIN WORKORDER_BINARY AS WB
	ON WB.WORKORDER_TYPE = W.TYPE
	AND WB.WORKORDER_BASE_ID = W.BASE_ID
	AND WB.WORKORDER_LOT_ID = W.LOT_ID
	AND WB.WORKORDER_SPLIT_ID = W.SPLIT_ID
	AND WB.WORKORDER_SUB_ID = W.SUB_ID
JOIN PART P
	ON P.ID = W.PART_ID
WHERE W.BASE_ID = @BASE_ID
	AND W.LOT_ID = @LOT_ID
	AND W.SPLIT_ID = @SPLIT_ID
	AND W.TYPE = 'W'
)

, CTE_OPERATIONS AS
(SELECT O.*
	, R.PIECE_NO
	, R.PART_ID AS RQ_PART_ID
	, RB.BITS AS RQ_BITS
	, OB.BITS AS O_BITS
	, RP.DESCRIPTION AS RQ_PART_DESCRIPTION
	, R.REQUIRED_DATE AS RQ_REQD_DATE
	, RP.STOCK_UM AS RQ_PART_STOCK_UM
	, R.CALC_QTY AS RQ_CALC_QTY
	, O.WORKORDER_TYPE + '~' + O.WORKORDER_BASE_ID + '~' + O.WORKORDER_SPLIT_ID + '~' + O.WORKORDER_LOT_ID + '~' 
		+ O.WORKORDER_SUB_ID + '~' + CONVERT(VARCHAR(3), O.SEQUENCE_NO) AS O_UDFJOIN
	, R.SUBORD_WO_SUB_ID
FROM OPERATION O
LEFT JOIN REQUIREMENT R
	ON O.WORKORDER_BASE_ID = R.WORKORDER_BASE_ID
	AND O.WORKORDER_LOT_ID = R.WORKORDER_LOT_ID
	AND O.WORKORDER_SPLIT_ID = R.WORKORDER_SPLIT_ID
	AND O.WORKORDER_SUB_ID = R.WORKORDER_SUB_ID
	AND O.WORKORDER_TYPE = R.WORKORDER_TYPE
	AND O.SEQUENCE_NO = R.OPERATION_SEQ_NO
LEFT JOIN PART RP
	ON R.PART_ID = RP.ID
LEFT JOIN REQUIREMENT_BINARY RB
	ON R.WORKORDER_BASE_ID = RB.WORKORDER_BASE_ID
	AND R.WORKORDER_LOT_ID = RB.WORKORDER_LOT_ID
	AND R.WORKORDER_SPLIT_ID = RB.WORKORDER_SPLIT_ID
	AND R.WORKORDER_SUB_ID = RB.WORKORDER_SUB_ID
	AND R.WORKORDER_TYPE = RB.WORKORDER_TYPE
	AND R.OPERATION_SEQ_NO = RB.OPERATION_SEQ_NO
	AND R.PIECE_NO = RB.PIECE_NO
LEFT JOIN OPERATION_BINARY OB
	ON O.WORKORDER_BASE_ID = OB.WORKORDER_BASE_ID
	AND O.WORKORDER_LOT_ID = OB.WORKORDER_LOT_ID
	AND O.WORKORDER_SPLIT_ID = OB.WORKORDER_SPLIT_ID
	AND O.WORKORDER_SUB_ID = OB.WORKORDER_SUB_ID
	AND O.WORKORDER_TYPE = OB.WORKORDER_TYPE
	AND O.SEQUENCE_NO = OB.SEQUENCE_NO
WHERE O.WORKORDER_BASE_ID = @BASE_ID
	AND O.WORKORDER_LOT_ID = @LOT_ID
	AND O.WORKORDER_SPLIT_ID = @SPLIT_ID
	AND O.WORKORDER_TYPE = 'W'
)
--SELECT * FROM CTE_OPERATIONS

, CTE_DOCS AS
(SELECT DRW.* 
	, D.REVISION_ID
	, D.ECN_REVISION
	, U1.STRING_VAL AS ADCN
	, U2.STRING_VAL AS LOC
	, D.ID AS PART_ID
FROM DOCUMENT_REF_WO DRW
INNER JOIN DOCUMENT D
	ON DRW.DOCUMENT_ID = D.ID
LEFT JOIN USER_DEF_FIELDS U1
	ON DRW.DOCUMENT_ID = U1.DOCUMENT_ID
	AND U1.PROGRAM_ID = N'VMDOCMNT'
	AND U1.ID = N'UDF-0000032'
	AND U1.LABEL IS NULL
LEFT JOIN USER_DEF_FIELDS U2
	ON DRW.DOCUMENT_ID = U2.DOCUMENT_ID
	AND U2.PROGRAM_ID = N'VMDOCMNT'
	AND U2.ID = N'UDF-0000031'
	AND U2.LABEL IS NULL
WHERE EXISTS (SELECT 1 FROM CTE_WORKORDERS W
	WHERE W.PART_ID = DRW.WORKORDER_BASE_ID
	AND W.SUB_ID = N'0'
	)
)
--SELECT * FROM cte_docs

, CTE_WOLINK AS 
(SELECT DISTINCT DD.WORKORDER_BASE_ID, N'0' AS WORKORDER_SUB_ID
	, (SELECT ISNULL(D.DOCUMENT_ID, N'') + N' / REV: ' + ISNULL(D.REVISION_ID, N'') + N' / ADCN: ' 
		+ ISNULL(D.ADCN, N'') + N' / LOC: ' + ISNULL(D.LOC, N'') + CHAR(10)
		FROM CTE_DOCS D
		FOR XML PATH('')
	) AS VALS
FROM CTE_DOCS DD
UNION ALL 
SELECT DISTINCT DD.WORKORDER_BASE_ID
	, DD.WORKORDER_SUB_ID
	, (SELECT ISNULL(D.DOCUMENT_ID, N'') + N' / REV: ' + ISNULL(D.REVISION_ID, N'') + N' / ADCN: ' 
		+ ISNULL(D.ADCN, N'') + N' / LOC: ' + ISNULL(D.LOC, N'') + CHAR(10)
	FROM CTE_DOCS D
	WHERE (D.WORKORDER_BASE_ID= DD.WORKORDER_BASE_ID 
		AND D.WORKORDER_SUB_ID = DD.WORKORDER_SUB_ID) 
	FOR XML PATH('')
	) AS VALS
FROM CTE_DOCS DD
WHERE DD.WORKORDER_SUB_ID <> N'0'
GROUP BY DD.WORKORDER_BASE_ID
	, DD.WORKORDER_SUB_ID
)
--SELECT * FROM cte_WOLINK
 
, CTE_HIDE_ETCH_BARCODE AS 
(SELECT A.*
, CASE
	WHEN RS.REPORT_SECTION IS NOT NULL AND RS2.REPORT_SECTION IS NOT NULL THEN 1
	ELSE 0
END AS HIDE_ETCH_BARCODE
FROM
	(SELECT O.WORKORDER_TYPE
		, O.WORKORDER_BASE_ID
		, O.WORKORDER_LOT_ID
		, O.WORKORDER_SPLIT_ID
		, O.WORKORDER_SUB_ID
		, O.SEQUENCE_NO
		, O.RESOURCE_ID
		, LEAD(O.SEQUENCE_NO) OVER(PARTITION BY O.WORKORDER_BASE_ID ORDER BY O.WORKORDER_BASE_ID
		, O.WORKORDER_LOT_ID
		, O.WORKORDER_SPLIT_ID
		, O.WORKORDER_SUB_ID
		, O.SEQUENCE_NO) AS NEXTSEQ
		, LEAD(O.RESOURCE_ID) OVER(PARTITION BY O.WORKORDER_BASE_ID ORDER BY O.WORKORDER_BASE_ID
		, O.WORKORDER_LOT_ID
		, O.WORKORDER_SPLIT_ID
		, O.WORKORDER_SUB_ID
		, O.SEQUENCE_NO) AS NEXTRESOURCE
	FROM OPERATION AS O
		WHERE O.WORKORDER_BASE_ID = @BASE_ID
		AND O.WORKORDER_LOT_ID = @LOT_ID
		AND O.WORKORDER_SPLIT_ID = @SPLIT_ID
	) AS A
INNER JOIN LIVESUPPLEMENTAL.DBO.ResourceID_ReportSettings RS
	ON A.RESOURCE_ID = RS.RESOURCE_ID
	AND RS.REPORT_NAME = 'traveller'
INNER JOIN LIVESUPPLEMENTAL.DBO.ResourceID_ReportSettings rs2
	ON A.NEXTRESOURCE = RS2.RESOURCE_ID
	AND RS2.REPORT_NAME = 'traveller'
WHERE A.WORKORDER_BASE_ID = @BASE_ID
	AND A.WORKORDER_LOT_ID = @LOT_ID
	AND A.WORKORDER_SPLIT_ID = @SPLIT_ID
	AND A.WORKORDER_TYPE = 'W'
	AND RS.REPORT_SECTION IN ('BARCODE_ETCH')
	AND RS2.REPORT_SECTION IN ('BARCODE_PRE_ETCH')
)
-- End added needs optimizing

, CTE_WO_IMAGES AS
(SELECT CAST(IM.BASE_ID AS NVARCHAR(30)) AS PART_ID
	, CAST(IM.LOT_ID AS NVARCHAR(3)) AS LOT_ID 
	, IM.IMAGE_ID
	, I.IMAGELG
	, CAST(I.ORIGINALFILENAME AS NVARCHAR(255)) AS ORIGINALFILENAME
	, IM.ORDERNUM
	, PE.PERSPECTIVEDESCRIPTION
	, IA.ATTRIBUTETYPE 
	, CAST(IA.WORKORDER_BASE_ID AS NVARCHAR(30)) AS WORKORDER_BASE_ID 
	, CAST(IA.WORKORDER_LOT_ID AS NVARCHAR(3)) AS WORKORDER_LOT_ID
	, CAST(IA.WORKORDER_SPLIT_ID AS NVARCHAR(3)) AS WORKORDER_SPLIT_ID
	, CAST(IA.WORKORDER_SUB_ID AS NVARCHAR(3)) AS WORKORDER_SUB_ID 
	, IA.WORKORDER_TYPE
	, IA.OPERATION_SEQUENCE
	, IA.REQUIREMENT_PIECEID
FROM IMAGE_REPOSITORY..IMAGE_MASTER IM
INNER JOIN IMAGE_REPOSITORY..IMAGE_ATTRIBUTES IA
	ON IM.IMAGE_ID = IA.IMAGE_ID
INNER JOIN IMAGE_REPOSITORY..IMAGE I
	ON I.IMAGE_ID = IM.IMAGE_ID
INNER JOIN IMAGE_REPOSITORY..PERSPECTIVE PE 
	ON IM.PERSPECTIVEID = PE.PERSPECTIVEID
WHERE IM.BASE_ID = @PART_ID
--	AND IA.WORKORDER_SUB_ID IN (SELECT DISTINCT SUB_ID FROM CTE_WORKORDERS GROUP BY SUB_ID)
	AND IA.WORKORDER_SUB_ID IN (SELECT SUB_ID FROM CTE_WORKORDERS GROUP BY SUB_ID)
UNION ALL
--Images With No Attributes Apply to traveller header card
SELECT P.ID AS PART_ID
	, PS.ENGINEERING_MSTR
	, I.IMAGE_ID
	, I.IMAGELG
	, I.ORIGINALFILENAME
	, IM.ORDERNUM
	, PE.PERSPECTIVEDESCRIPTION
	, N'IMAGE_MASTER' AS ATTRIBUTETYPE
	, P.ID AS WORKORDER_BASE_ID
	, N'0' AS WORKORDER_LOT_ID
	, N'0' AS WORKORDER_SPLIT_ID
	, N'0' AS WORKORDER_SUB_ID
	, N'M' AS WORKORDER_TYPE
	, 0 AS OPERATION_SEQUENCE
	, 0 AS REQUIREMENT_PIECEID
FROM PART P 
INNER JOIN PART_SITE PS 
	ON PS.PART_ID = P.ID 
INNER JOIN IMAGE_REPOSITORY..IMAGE_MASTER IM
	ON CAST(IM.BASE_ID AS NVARCHAR(30)) = P.ID
	AND CAST(IM.LOT_ID AS NVARCHAR(30)) = PS.ENGINEERING_MSTR
INNER JOIN IMAGE_REPOSITORY..IMAGE I
	ON IM.IMAGE_ID = I.IMAGE_ID
INNER JOIN IMAGE_REPOSITORY..PERSPECTIVE PE 
	ON IM.PERSPECTIVEID = PE.PERSPECTIVEID
WHERE P.ID = @PART_ID
AND NOT EXISTS (SELECT 1 FROM IMAGE_REPOSITORY..IMAGE_ATTRIBUTES IA
	WHERE IA.IMAGE_ID = IM.IMAGE_ID
	)
)
--SELECT * FROM cte_WO_Images

SELECT O.WORKORDER_TYPE
	, O.WORKORDER_BASE_ID
	, O.WORKORDER_LOT_ID
	, O.WORKORDER_SPLIT_ID
	, O.WORKORDER_SUB_ID
	, O.SEQUENCE_NO
	, O.RESOURCE_ID
	, SR.DESCRIPTION AS RESOURCE_DESC
	, O.SERVICE_ID
	, O.OPERATION_TYPE
	, O.VENDOR_ID
	, O.VENDOR_SERVICE_ID
	, O.RQ_PART_ID
	, O.RQ_PART_DESCRIPTION
	, O.PIECE_NO
	, O.PIECE_NO AS RQ_PIECE
	, DBO.SFNWONUMFORMAT(O.WORKORDER_BASE_ID, O.WORKORDER_LOT_ID, O.WORKORDER_SPLIT_ID, O.WORKORDER_SUB_ID) AS WONUM
	, DBO.SFNWONUMFORMAT(O.WORKORDER_BASE_ID, O.WORKORDER_LOT_ID, O.WORKORDER_SPLIT_ID, O.WORKORDER_SUB_ID) AS WONUM_ID
	, O.WORKORDER_BASE_ID + N'/' + O.WORKORDER_LOT_ID + N'/' + O.WORKORDER_SPLIT_ID AS WO_ID
	, W.PART_ID
	, W.PART_ID AS WO_PART_ID
	, ISNULL(W.DESIRED_WANT_DATE, (SELECT TOP 1 W.DESIRED_WANT_DATE FROM CTE_WORKORDERS W)) AS DESIRED_WANT_DATE
	, ISNULL(W.DESIRED_RLS_DATE, (SELECT TOP 1 W.DESIRED_RLS_DATE FROM CTE_WORKORDERS W)) AS DESIRED_RLS_DATE
	, ISNULL(W.ENGINEERED_BY, (SELECT TOP 1 W.ENGINEERED_BY FROM CTE_WORKORDERS W)) AS ENGINEERED_BY
	, O.WORKORDER_BASE_ID AS BASE_ID
	, O.WORKORDER_LOT_ID AS LOT_ID
	, O.WORKORDER_SUB_ID AS SUB_ID
	, O.WORKORDER_SPLIT_ID AS SPLIT_ID
	, CASE WHEN RS.STAMP IS NOT NULL THEN 1 ELSE 0 END AS STAMP 
	, CASE WHEN RS.HEIGHTCHANGE IS NOT NULL THEN 1 ELSE 0 END AS HEIGHTCHANGE
	, CASE WHEN RS.SHIP_VIA IS NOT NULL THEN CO.SHIP_VIA ELSE NULL END AS SHIP_VIA 
	, CASE 
		WHEN RS.BARCODE_REVIEW IS NOT NULL AND O.OPERATION_TYPE NOT IN (N'QP-REVIEW', N'QP-NI') THEN 1 
		WHEN (W.PART_ID like N'Tooling%' OR W.PART_ID LIKE N'%Tooling') AND W.SUB_ID <> '0' THEN 1
		WHEN SR.AUTO_REPORTING = N'Y' THEN 1
		 --added 8/31/2018 vme needs to be optimized
		WHEN HEB.HIDE_ETCH_BARCODE = 1 THEN 1
		ELSE 0
	END AS BARCODE
	, CASE 
		WHEN O.OPERATION_TYPE IN (N'PRE-PULL', N'STAGING', N'REVIEW', N'QP-REVIEW', N'QP-NI', 'FAI', N'FAI-NI', N'PANEL', N'PROGM') THEN 1
		WHEN rs.QTY_IN IS NOT NULL THEN 0 
		ELSE 1
	END AS HIDE_QTY_IN
	, RS.QTY_OUT
	, CASE 
		WHEN RS.QTY_OUT IS NOT NULL THEN CAST(1 AS BIT)
		ELSE CAST(0 AS BIT)
	END AS HIDE_QTY_OUT
	, CASE 
		WHEN rs.QTY_NOSHOW IS NOT NULL THEN CAST(1 AS bit)
		ELSE CAST(0 AS bit)
	END AS HIDE_QTY_NOSHOW
	, CASE 
		WHEN RS.PRINTUSER IS NOT NULL THEN 1 
		ELSE 0 
	END AS PRINTUSER
	, O.USER_1 
	, N'*' + SUBSTRING(REPLACE(O.USER_1, N' ', N''), CHARINDEX(N':', REPLACE(O.USER_1, N' ', N'')) + 1, 80) + '*' AS OP_USER_1
	, O.USER_2
	, N'*' + SUBSTRING(REPLACE(O.USER_2, N' ', N''), CHARINDEX(N':', REPLACE(O.USER_2, N' ', N'')) + 1, 80) + '*' AS OP_USER_2
	, O.USER_3
	, N'*' + SUBSTRING(REPLACE(O.USER_3, N' ', N''), CHARINDEX(N':', REPLACE(O.USER_3, N' ', N'')) + 1, 80) + '*' AS OP_USER_3
	, O.USER_4
	, N'*' + SUBSTRING(REPLACE(O.USER_4, N' ', N''), CHARINDEX(N':', REPLACE(O.USER_4, N' ', N'')) + 1, 80) + '*' AS OP_USER_4
	, O.USER_5
	, N'*' + SUBSTRING(REPLACE(O.USER_5, N' ', N''), CHARINDEX(N':', REPLACE(O.USER_5, N' ', N'')) + 1, 80) + '*' AS OP_USER_5
	, O.USER_6
	, N'*' + SUBSTRING(REPLACE(O.USER_6, N' ', N''), CHARINDEX(N':', REPLACE(O.USER_6, N' ', N'')) + 1, 80) + '*' AS OP_USER_6
	, O.USER_7
	, N'*' + SUBSTRING(REPLACE(O.USER_7, N' ', N''), CHARINDEX(N':', REPLACE(O.USER_7, N' ', N'')) + 1, 80) + '*' AS OP_USER_7
	, O.USER_8
	, N'*' + SUBSTRING(REPLACE(O.USER_8, N' ', N''), CHARINDEX(N':', REPLACE(O.USER_8, N' ', N'')) + 1, 80) + '*' AS OP_USER_8
	, O.USER_9
	, N'*' + SUBSTRING(REPLACE(O.USER_9, N' ', N''), CHARINDEX(N':', REPLACE(O.USER_9, N' ', N'')) + 1, 80) + '*' AS OP_USER_9
	, O.USER_10
	, N'*' + SUBSTRING(REPLACE(O.USER_10, N' ', N''), CHARINDEX(N':', REPLACE(O.USER_10, N' ', N'')) + 1, 80) + '*' AS OP_USER_10 
	, CO.CUSTOMER_ID
	, CO.ID AS CUST_ORDER_ID
	, COL.CUSTOMER_PART_ID
	, C.[NAME] AS CUST_NAME
	, CA.[NAME] AS CUST_ADDRNAME
	, W.[DESCRIPTION] AS WO_PART_DESCRIPTION
	, ISNULL(W.WO_SPECS, CONVERT(NVARCHAR(MAX), CONVERT(VARBINARY(MAX), ORQ.RQ_BITS))) AS WO_LONG_DESCR
	, CONVERT(DECIMAL(14, 0), SA.SURFACE_AREA) AS SURFACE_AREA
	, SA.MATERIAL_TYPE + N' ' + SA.ALLOY + N' ' + CONVERT(VARCHAR(15), CONVERT(DECIMAL(5,2), SA.THICKNESS_VAL)) + ' X '
		+ CONVERT(VARCHAR(15), CONVERT(DECIMAL(5, 2), SA.WIDTH_VAL)) + ' X '
		+ CONVERT(VARCHAR(15), CONVERT(DECIMAL(5, 2), SA.LENGTH_VAL)) AS PART_DESCRIPTION2
	, CAST(CAST(O.O_BITS AS VARBINARY(MAX)) AS NVARCHAR(MAX)) AS OP_LONG_DESCR 
	, O.RQ_REQD_DATE AS RQ_REQD
	, O.RQ_PART_STOCK_UM
	, O.RQ_CALC_QTY
	, CASE 
		WHEN EXISTS (SELECT 1 FROM USER_DEF_FIELDS WHERE DOCUMENT_ID = w.PART_ID AND PROGRAM_ID = 'VMPRTMNT' AND ID = 'UDF-0000040' AND BOOL_VAL = 1)
			THEN 'ITAR Controlled'
		WHEN W.PART_ID LIKE '737RIB-KIT-PROD%' THEN 'Tags must remain on parts throughout entire process!'
		WHEN W.PART_ID LIKE '737RIB-MAX-KIT-PROD%' THEN 'Tags must remain on parts throughout entire process!'
		ELSE ''
	END AS MESSAGE
	, CASE 
		WHEN EXISTS (SELECT 1 
			FROM USER_DEF_FIELDS UDF 
			WHERE UDF.PROGRAM_ID ='VMMFGWIN_OP'
				AND UDF.ID = 'UDF-0000048'
				AND UDF.BOOL_VAL = 1 
				AND UDF.DOCUMENT_ID LIKE 'W~'+ o.WORKORDER_BASE_ID + '~'+ o.WORKORDER_SUB_ID + '~' + o.WORKORDER_LOT_ID 
					+'~'+ o.WORKORDER_SPLIT_ID + '%'
				AND LEFT(UDF.DOCUMENT_ID, 1) = 'W'
			) THEN 'TIME LIMIT'
		ELSE NULL
	END AS MESSAGE2
	, CASE 
		WHEN DSL.SUPPLY_TYPE = 'CP' AND COL.LINE_NO = (SELECT MIN(CL1.LINE_NO)
			FROM CUST_ORDER_LINE AS CL1
			INNER JOIN DEMAND_SUPPLY_LINK AS DSL1
				ON CL1.CUST_ORDER_ID = DSL1.DEMAND_BASE_ID
				AND CL1.PART_ID = DSL1.SUPPLY_PART_ID
				AND DSL1.SUPPLY_BASE_ID = @BASE_ID -- changed 6/10/2014
				AND DSL1.SUPPLY_LOT_ID = @LOT_ID
				AND DSL1.SUPPLY_SPLIT_ID = @SPLIT_ID
				AND DSL1.SUPPLY_TYPE = 'CP' 
			) THEN 0 
	END AS CP_FIRST_PART
	, CASE 
		WHEN DSL.SUPPLY_TYPE = 'CP' THEN COL.PART_ID
		ELSE NULL
	END AS CP_PART 
	, DSL.SUPPLY_TYPE
	, COL.LINE_NO
	, CASE 
		WHEN SUPPLY_TYPE = 'CP' THEN CAST( COL.ORDER_QTY AS INT)
		ELSE NULL
	END AS CP_QTY 
	, CASE 
		WHEN DSL.SUPPLY_TYPE = 'CP' THEN P.DESCRIPTION
		ELSE NULL
	END AS CP_DESCRIPTION
	, PSA.MATERIAL_TYPE + ' ' + PSA.ALLOY + ' ' + CONVERT(VARCHAR(15), CONVERT(DECIMAL(5, 2), PSA.THICKNESS_VAL)) + ' x '
		+ CONVERT(VARCHAR(15), CONVERT(DECIMAL(5, 2), PSA.WIDTH_VAL)) + ' x ' 
		+ CONVERT(VARCHAR(15), CONVERT(DECIMAL(5, 2), PSA.LENGTH_VAL)) 
	AS CP_DESCRIPTION2
	, CASE 
		WHEN ISNULL(SUPPLY_TYPE, 'WO') = 'WO' THEN COL.CUSTOMER_PART_ID
		ELSE NULL
	END AS WO_CUSTOMER_PART_ID
	, CASE 
		WHEN SUPPLY_TYPE = 'CP' THEN COL.CUSTOMER_PART_ID
		ELSE NULL
	END AS CP_CUSTOMER_PART_ID
	, SR.AUTO_REPORTING
	, OVD.LINE_NUM AS OVD_LINE_NO
	, OVD.LABEL_YN AS LABELYN
	, OVD.PROMPT AS OVD_PROMPT
	, O.SUBORD_WO_SUB_ID AS RQ_SUB_ID
	, CO.CUSTOMER_PO_REF
	, W.DESIRED_QTY
	, WL.VALS AS WO_LINK
	, CONVERT(NVARCHAR(MAX), CONVERT(VARBINARY(MAX), ORQ.RQ_BITS)) AS REQ_SPECS
	, IW.IMAGE_ID AS WO_IMAGEID
	, IW.IMAGELG AS WO_IMAGE
	, IW.ORDERNUM AS WO_ORDERNUM
	, IW.PERSPECTIVEDESCRIPTION AS WO_PERSPECTIVE
	, IOP.IMAGE_ID AS O_IMAGEID
	, IOP.IMAGELG AS O_IMAGE
	, IOP.ORDERNUM AS O_ORDERNUM
	, IOP.PERSPECTIVEDESCRIPTION AS O_PERSPECTIVE
	, IR.IMAGE_ID AS R_IMAGEID
	, IR.IMAGELG AS R_IMAGE
	, IR.ORDERNUM AS R_ORDERNUM
	, IR.PERSPECTIVEDESCRIPTION AS R_PERSPECTIVE
	, dbo.sfnGetCustSpecsString(O.WORKORDER_TYPE + N'~' + O.WORKORDER_BASE_ID + N'~' + O.WORKORDER_SUB_ID + N'~' + O.WORKORDER_LOT_ID + N'~' 
		+ O.WORKORDER_SPLIT_ID + N'~' + CAST(O.SEQUENCE_NO AS NVARCHAR(3))) AS CUSTSPECSSTRING
FROM CTE_WORKORDERS W
LEFT OUTER JOIN CTE_OPERATIONS O
	ON W.BASE_ID = O.WORKORDER_BASE_ID
	AND W.LOT_ID = O.WORKORDER_LOT_ID
	AND W.SPLIT_ID = O.WORKORDER_SPLIT_ID
	AND W.SUB_ID = O.WORKORDER_SUB_ID
	AND W.[TYPE] = O.WORKORDER_TYPE
LEFT JOIN SHOP_RESOURCE SR
	ON SR.ID = O.RESOURCE_ID
LEFT JOIN DEMAND_SUPPLY_LINK DSL
LEFT OUTER JOIN PART P
	ON DSL.SUPPLY_PART_ID = P.ID 
	ON DSL.SUPPLY_BASE_ID = W.BASE_ID
	AND DSL.SUPPLY_LOT_ID = W.LOT_ID
	AND DSL.SUPPLY_SPLIT_ID = W.SPLIT_ID
	--AND DSL.SUPPLY_SUB_ID = W.SUB_ID
	AND DSL.SUPPLY_TYPE IN ('WO', 'CP')
LEFT JOIN CUST_ORDER_LINE COL
	ON DSL.DEMAND_BASE_ID = COL.CUST_ORDER_ID
	AND DSL.DEMAND_SEQ_NO = COL.LINE_NO
LEFT JOIN CUSTOMER_ORDER CO
	ON COL.CUST_ORDER_ID = CO.ID
LEFT OUTER JOIN CUSTOMER AS C
	ON C.ID = CO.CUSTOMER_ID
LEFT OUTER JOIN CUST_ADDRESS AS CA
	ON CO.CUSTOMER_ID = CA.CUSTOMER_ID
	AND CO.SHIP_TO_ADDR_NO = CA.ADDR_NO
LEFT OUTER JOIN SKILLS_WO_SURFACE_AREA AS SA
	ON W.BASE_ID = SA.BASE_ID
	AND W.LOT_ID = SA.LOT_ID
	AND W.SPLIT_ID = SA.SPLIT_ID
-- ADDED 8/31/2018 VME -- NEEDS TO BE OPTIMIZED
LEFT OUTER JOIN CTE_HIDE_ETCH_BARCODE HEB
	ON O.WORKORDER_BASE_ID	= HEB.WORKORDER_BASE_ID
	AND O.WORKORDER_LOT_ID = HEB.WORKORDER_LOT_ID
	AND O.WORKORDER_SPLIT_ID = HEB.WORKORDER_SPLIT_ID
	AND O.WORKORDER_SUB_ID = HEB.WORKORDER_SUB_ID
	AND O.WORKORDER_TYPE = 'W'
	AND O.SEQUENCE_NO = HEB.SEQUENCE_NO
LEFT OUTER JOIN SKILLS_PART_UDF AS PSA
	ON PSA.PART_ID = CASE 
		WHEN SUPPLY_TYPE = 'CP' THEN COL.PART_ID
		ELSE NULL
	END
LEFT OUTER JOIN cte_WOLINK wl
	ON wl.WORKORDER_SUB_ID = w.SUB_ID
LEFT JOIN CTE_OPERATIONS ORQ
	ON ORQ.WORKORDER_BASE_ID = W.BASE_ID
	AND ORQ.WORKORDER_LOT_ID = W.LOT_ID
	AND ORQ.WORKORDER_SPLIT_ID = W.SPLIT_ID
	AND ORQ.SUBORD_WO_SUB_ID = W.SUB_ID
	AND ORQ.WORKORDER_TYPE = W.[TYPE]
LEFT JOIN CTE_WO_IMAGES IW
	ON IW.WORKORDER_SUB_ID = W.SUB_ID
	AND IW.OPERATION_SEQUENCE = 0
	AND IW.REQUIREMENT_PIECEID = 0
LEFT JOIN CTE_WO_IMAGES IOP
	ON IOP.WORKORDER_SUB_ID = O.WORKORDER_SUB_ID
	AND IOP.OPERATION_SEQUENCE = O.SEQUENCE_NO
	AND IOP.REQUIREMENT_PIECEID = 0
LEFT JOIN CTE_WO_IMAGES IR
	ON IR.WORKORDER_SUB_ID = O.WORKORDER_SUB_ID
	AND IR.OPERATION_SEQUENCE = O.SEQUENCE_NO
	AND IR.REQUIREMENT_PIECEID = O.PIECE_NO
LEFT JOIN @REPORTSETTINGS RS
	ON SR.ID = RS.RESOURCE 
LEFT JOIN LIVESupplemental.dbo.operator_variable_data OVD
	ON O.RESOURCE_ID = OVD.RESOURCE_ID
	AND O.OPERATION_TYPE = OVD.OPERATION_TYPE_ID
ORDER BY W.BASE_ID
	, W.LOT_ID
	, W.SPLIT_ID 
	, W.SUB_ID 
	, W.PART_ID
	, O.SEQUENCE_NO
	, O.PIECE_NO
	, CASE 
		WHEN W.WO_SPECS IS NULL THEN CAST(CAST(O.RQ_BITS AS VARBINARY(MAX))AS NVARCHAR(MAX))
		ELSE W.WO_SPECS
	END
	, COL.LINE_NO
