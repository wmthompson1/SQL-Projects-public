# PO Accural Report

```sql
USE LIVE
GO

SELECT V.ID AS VENDOR_ID
	, V.[NAME] AS VENDOR_NAME
	, POL.PURC_ORDER_ID
	, POL.LINE_NO AS PURC_ORDER_LINE_NO
	, R.RECEIVED_DATE
	, RL.RECEIVER_ID
	, RL.LINE_NO AS RECEIVER_LINE_NO
	, RL.RECEIVED_QTY
	, RL.UNIT_PRICE
	, RL.FIXED_CHARGE
	, (RL.RECEIVED_QTY * ISNULL(RL.UNIT_PRICE, 0)) + ISNULL(RL.FIXED_CHARGE, 0) AS RECEIVER_AMOUNT
FROM  RECEIVER_LINE RL
INNER JOIN RECEIVER R
	ON RL.RECEIVER_ID = R.ID
INNER JOIN PURC_ORDER_LINE POL
	ON RL.PURC_ORDER_ID = POL.PURC_ORDER_ID
	AND RL.PURC_ORDER_LINE_NO = POL.LINE_NO
INNER JOIN PURCHASE_ORDER PO
	ON POL.PURC_ORDER_ID = PO.ID
INNER JOIN VENDOR V
	ON PO.VENDOR_ID = V.ID
LEFT OUTER JOIN (PAYABLE_LINE PL
	INNER JOIN PAYABLE P
	ON P.VOUCHER_ID = PL.VOUCHER_ID
	)
	ON RL.RECEIVER_ID = PL.RECEIVER_ID
	AND RL.LINE_NO = PL.RECEIVER_LINE_NO
WHERE PL.VOUCHER_ID IS NULL
--	AND (RL.RECEIVED_QTY * ISNULL(RL.UNIT_PRICE, 0)) + ISNULL(RL.FIXED_CHARGE, 0) != 0
```
