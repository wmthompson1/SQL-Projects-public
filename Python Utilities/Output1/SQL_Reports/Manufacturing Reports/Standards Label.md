# Standards Label

```sql
SELECT *
FROM (SELECT IT.PART_ID
        , IT.WORKORDER_BASE_ID AS BASE_ID
        , IT.WORKORDER_LOT_ID AS LOT_ID
        , IT.WORKORDER_SPLIT_ID AS SPLIT_ID
        , IT.WORKORDER_SUB_ID as SUB_ID
        , IT.OPERATION_SEQ_NO as Starting_Seq_No
        , CASE
            WHEN T.ID IS NULL THEN 'NO TRACE  QTY: ' + + CONVERT(VARCHAR(10), CONVERT(DECIMAL, ABS(IT.QTY)))
            ELSE T.ID + '  QTY: ' + CONVERT(VARCHAR(10), CONVERT(DECIMAL, ABS(TI.QTY)))
        END AS LINE2
        , IT.LOCATION_ID
        , CASE 
            WHEN @OLDWOID = 'FALSE' THEN dbo.SFNWONUMFORMAT(IT.WORKORDER_BASE_ID, IT.WORKORDER_LOT_ID, IT.WORKORDER_SPLIT_ID, IT.WORKORDER_SUB_ID)
            ELSE dbo.SFNWONUMFORMAT_OLD(IT.WORKORDER_BASE_ID, IT.WORKORDER_LOT_ID, IT.WORKORDER_SPLIT_ID, IT.WORKORDER_SUB_ID) 
        END AS WONUM
        , SUM(ISNULL(TI.QTY, (IT.QTY * -1))) AS QTY
    FROM INVENTORY_TRANS IT
    LEFT OUTER JOIN TRACE_INV_TRANS TI
        ON IT.TRANSACTION_ID = TI.TRANSACTION_ID
    LEFT OUTER JOIN TRACE T
        ON TI.TRACE_ID = T.ID
        AND TI.PART_ID = T.PART_ID
    WHERE (IT.CLASS <> 'R') 
        AND (IT.WORKORDER_BASE_ID = @BASE_ID) 
        AND (IT.WORKORDER_SUB_ID IN (@SUB_ID)) 
        AND IT.WORKORDER_LOT_ID = @LOT_ID
        AND IT.WORKORDER_SPLIT_ID = @SPLIT_ID
        AND (IT.OPERATION_SEQ_NO IN (@Starting_Seq_No))
    GROUP BY IT.PART_ID
        , IT.WORKORDER_BASE_ID
        , IT.WORKORDER_LOT_ID 
        , IT.WORKORDER_SPLIT_ID
        , IT.WORKORDER_SUB_ID
        , IT.OPERATION_SEQ_NO
        , CASE
            WHEN T.ID IS NULL THEN 'NO TRACE  QTY: ' + + CONVERT(VARCHAR(10), CONVERT(DECIMAL, ABS(IT.QTY)))
            ELSE T.ID + '  QTY: ' + CONVERT(VARCHAR(10), CONVERT(DECIMAL, ABS(TI.QTY)))
        END
        , IT.LOCATION_ID
        , CASE 
            WHEN @OLDWOID = 'FALSE' THEN dbo.SFNWONUMFORMAT(IT.WORKORDER_BASE_ID, IT.WORKORDER_LOT_ID, IT.WORKORDER_SPLIT_ID, IT.WORKORDER_SUB_ID)
            ELSE dbo.SFNWONUMFORMAT_OLD(IT.WORKORDER_BASE_ID, IT.WORKORDER_LOT_ID, IT.WORKORDER_SPLIT_ID, IT.WORKORDER_SUB_ID) 
        END
    ) X
WHERE QTY < 0
ORDER BY PART_ID, LINE2

```
