-- VISUAL ENTERPRISE REPORTS > PART REPORTS > PARTS_CATALOG
--DECLARE @COMMODITY AS NVARCHAR(15) = 'END ITEM'
--, @PLANNER AS NVARCHAR(15) = 'MFG-MRP'
--, @PURCHASED AS NVARCHAR(1) = 'Y'
--, @STOCKED AS NVARCHAR(1)= 'N'
--, @CONTROLLED AS NVARCHAR(15) = ('QPL CONTROLLED')
--, @FABRICATED AS NVARCHAR(1) = 'N'
--, @BUYER AS NVARCHAR(15) = ''
--, @BUI AS NVARCHAR(15) = ''

SELECT P.ID AS PART_ID
    , ISNULL(P.[DESCRIPTION], '_BLANK') AS PART_DESCRIPTION
    , P.INSPECTION_REQD AS INSP_REQD
    , ISNULL(P.COMMODITY_CODE, '_BLANK') AS COMMODITY_CODE
    , (SELECT PUD.STRING_VAL AS EXPR1
        FROM dbo.PART AS P1 
        LEFT OUTER JOIN dbo.USER_DEF_FIELDS AS PUD 
        ON P1.ID = PUD.DOCUMENT_ID 
        AND PUD.PROGRAM_ID = 'VMPRTMNT' 
        AND PUD.ID = 'UDF-0000082'
        WHERE P1.ID = P.ID) AS CONTROLLED
    , (SELECT PUD.STRING_VAL AS EXPR1
        FROM dbo.PART AS P1 
        LEFT OUTER JOIN dbo.USER_DEF_FIELDS AS PUD 
        ON P1.ID = PUD.DOCUMENT_ID 
        AND PUD.PROGRAM_ID = 'VMPRTMNT' 
        AND PUD.ID = 'UDF-0000084'
        WHERE P1.ID = P.ID) AS CERTS_NEEDED
    , CAST(CAST(PB.BITS AS VARBINARY(MAX)) AS NVARCHAR(MAX)) AS PART_SPECIFICATIONS
    , P.STOCK_UM AS UM
    , STUFF((SELECT ', ' + d.id +' REV '+ d.REVISION_ID 
        FROM DOCUMENT D
        INNER JOIN DOCUMENT_REFERENCE DF
            ON DF.DOCUMENT_ID = D.ID
        AND D.CATEGORY_ID = 'SPECIFICATION'
        WHERE P.ID = DF.ID
        FOR XML PATH('')), 1, 1, '') AS SPEC_LIST
    , P.PREF_VENDOR_ID AS PREF_VENDOR
    , ISNULL(P.BUYER_USER_ID, '_BLANK') AS BUYER_USER_ID
    , ISNULL(B.SALESPERSON_ID, '_BLANK') AS BUYER_ID
    , P.PLANNER_USER_ID AS PLANNER_ID
    , ISNULL(P.STOCKED, '-') AS STOCKED
    , ISNULL(P.FABRICATED, '-') AS FABRICATED
    , ISNULL(P.PURCHASED, '-') AS PURCHASED
    , ISNULL(P.CONSUMABLE, '-') AS CONSUMABLE
    , P.QTY_ON_HAND
    , P.QTY_IN_DEMAND
    , P.QTY_ON_ORDER 
FROM PART P
--FROM PART_SITE_VIEW P
--LEFT OUTER JOIN SKILLS_PART_UDF UDF
--    ON P.PART_ID = UDF.PART_ID
LEFT OUTER JOIN PART_BINARY PB
    ON P.ID = PB.PART_ID
    AND PB.[TYPE] = 'D'
LEFT OUTER JOIN LIVESUPPLEMENTAL.DBO.BUYER_ASSN B
    ON LEFT(P.BUYER_USER_ID, 2) = B.ACCOUNT_ID
LEFT OUTER JOIN DOCUMENT_REFERENCE DF
    ON P.ID = DF.ID
LEFT OUTER JOIN DOCUMENT D
    ON DF.DOCUMENT_ID = D.ID
    AND D.CATEGORY_ID = 'SPECIFICATION'
WHERE (P.[STATUS] != 'O'
        OR P.[STATUS] IS NULL)
    AND P.COMMODITY_CODE IN (@COMMODITY)
    AND P.PLANNER_USER_ID IN (@PLANNER)
    AND ISNULL(B.SALESPERSON_ID, '_BLANK') IN (@BUYER)
    AND ISNULL(P.BUYER_USER_ID, '_BLANK') IN (@BUI)
    AND ISNULL(P.FABRICATED, '-') IN (@FABRICATED)
    AND ISNULL(P.PURCHASED, '-') IN (@PURCHASED)
    AND ISNULL(P.STOCKED, '-') IN (@STOCKED)
    AND ISNULL((SELECT PUD.STRING_VAL AS EXPR1
        FROM dbo.PART AS P1 
        LEFT OUTER JOIN dbo.USER_DEF_FIELDS AS PUD 
        ON P1.ID = PUD.DOCUMENT_ID 
        AND PUD.PROGRAM_ID = 'VMPRTMNT' 
        AND PUD.ID = 'UDF-0000082'
        WHERE P1.ID = P.ID)
        , '-') IN (@CONTROLLED)
