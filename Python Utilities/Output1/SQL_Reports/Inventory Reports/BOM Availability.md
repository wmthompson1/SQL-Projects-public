# BOM Availability

```sql
/**********************************************************************************************
DESCRIPTION:     MATERIALS AVAILABILITY REPORT
DEPENDENCY:         REPORT DEPENDS ON DATASET IN WORKORDER FOLDER FOR CUSTOMER DROP DOWN.    

DATE        MODIFIED BY         CHANGE DESCRIPTION
----------  ------------------  ------------------------------------------------------------
3/7/2020    VIVIAN ELIA            VIVIAN HAD CREATED, AND BEGAN TESTING, PRE-COVID19
10/23/2020    WILLIAM THOMPSON    ASSIGNED MYSELF TO THE PROJECT.
10/23/2020    WILLIAM THOMPSON    REMAINING OPS
20201207    traci treffert        CLEARED OUT UNNECESSARY CODE FOR CUSTOMER INFORMATION
                                LINKED STATUS AS PARAMETER

**********************************************************************************************/

SET TRANSACTION ISOLATION LEVEL READ UNCOMMITTED
SET DEADLOCK_PRIORITY LOW

; WITH CTE_WO AS 
(    SELECT WO.[TYPE], WO.BASE_ID, WO.LOT_ID, WO.SPLIT_ID, WO.SUB_ID
        , WO.[STATUS] AS WO_STATUS, WO.DESIRED_QTY AS WO_QTY, WO.PART_ID AS WO_PART_ID
        , ISNULL(WO.DESIRED_RLS_DATE, ( SELECT WO1.DESIRED_RLS_DATE 
            FROM WORK_ORDER WO1 WITH (NOLOCK) WHERE WO.BASE_ID = WO1.BASE_ID AND WO.LOT_ID = WO1.LOT_ID
                AND WO.SPLIT_ID = WO1.SPLIT_ID AND WO1.SUB_ID = '0' AND WO.[TYPE] = WO1.[TYPE] )
        ) AS WO_RELEASE_DATE
        , ISNULL(WO.DESIRED_WANT_DATE, ( SELECT WO1.DESIRED_WANT_DATE 
            FROM WORK_ORDER WO1 WITH (NOLOCK) WHERE WO.BASE_ID = WO1.BASE_ID AND WO.LOT_ID = WO1.LOT_ID
                AND WO.SPLIT_ID = WO1.SPLIT_ID AND WO1.SUB_ID = '0' AND WO.[TYPE] = WO1.[TYPE] )
        ) AS WO_DUE_DATE
        , ISNULL(dbo.sfn_remaining_operations(WO.[TYPE], WO.BASE_ID, WO.LOT_ID, WO.SPLIT_ID, WO.SUB_ID), N'NO REMAINING OPERATIONS') AS REMAINING_OPS
        , dbo.sfn_current_operation (WO.[TYPE], WO.BASE_ID, WO.LOT_ID, WO.SPLIT_ID, WO.SUB_ID) AS CURRENTOPERATION
    FROM WORK_ORDER WO WITH (NOLOCK)
    WHERE WO.[TYPE] = N'W' AND WO.[STATUS] IN ('F', 'R', 'U')
)
--SELECT * FROM CTE_WO

, CTE_RESULTS AS 
(    SELECT WO.WO_RELEASE_DATE, WO.WO_DUE_DATE, WO.WO_QTY, WO.WO_PART_ID, WP.[DESCRIPTION] AS WO_DESCRIPTION
        , P.BUYER_USER_ID AS WO_BUYER_ID, DBO.SFNWONUMFORMAT(WO.BASE_ID, WO.LOT_ID, WO.SPLIT_ID, WO.SUB_ID) AS WO_NUM
        , O.SEQUENCE_NO AS WO_OP_NUM, RQ.PIECE_NO, RQ.PART_ID AS BOM_PART_ID, P.[DESCRIPTION] AS BOM_ITEM_DESCRIPTION
        , SUM(RQ.CALC_QTY) OVER(PARTITION BY RQ.PART_ID ORDER BY RQ.PART_ID, WO.WO_RELEASE_DATE) AS QTY_REQD
        , P.QTY_AVAILABLE_ISS AS QTY_AVAILABLE, RQ.USAGE_UM, P.STOCK_UM
        , ISNULL(P.PRIMARY_LOC_ID, '') AS PRIMARY_LOC
        , WO.REMAINING_OPS, O.RESOURCE_ID
        , ISNULL(P.COMMODITY_CODE, '_BLANK') AS COMMODITY_CODE
        , SR.USER_5 AS FACILITY, SR.USER_7 AS PROCESS, WO.WO_STATUS
    FROM CTE_WO WO WITH (NOLOCK)
    LEFT OUTER JOIN DEMAND_SUPPLY_LINK DSL ON DSL.SUPPLY_BASE_ID = WO.BASE_ID
        AND DSL.SUPPLY_LOT_ID = WO.LOT_ID AND DSL.SUPPLY_SPLIT_ID = WO.SPLIT_ID
    LEFT OUTER JOIN OPERATION O ON WO.BASE_ID = O.WORKORDER_BASE_ID AND WO.LOT_ID = O.WORKORDER_LOT_ID 
        AND WO.SPLIT_ID = O.WORKORDER_SPLIT_ID AND WO.SUB_ID = O.WORKORDER_SUB_ID 
        AND WO.[TYPE] = O.WORKORDER_TYPE AND WO.CURRENTOPERATION = O.SEQUENCE_NO
    LEFT OUTER JOIN SHOP_RESOURCE SR ON SR.ID = O.RESOURCE_ID
    LEFT OUTER JOIN PART WP ON WO.WO_PART_ID = WP.ID
    LEFT OUTER JOIN REQUIREMENT RQ ON O.WORKORDER_TYPE = RQ.WORKORDER_TYPE AND O.SEQUENCE_NO = RQ.OPERATION_SEQ_NO
        AND O.WORKORDER_BASE_ID = RQ.WORKORDER_BASE_ID AND O.WORKORDER_LOT_ID = RQ.WORKORDER_LOT_ID
        AND O.WORKORDER_SPLIT_ID = RQ.WORKORDER_SPLIT_ID AND O.WORKORDER_SUB_ID = RQ.WORKORDER_SUB_ID
    LEFT OUTER JOIN PART_SITE_VIEW P ON RQ.PART_ID = P.PART_ID AND P.SITE_ID = 'SK01'
    --WHERE O.RESOURCE_ID IN (@RESOURCEID)
    WHERE CHARINDEX(@RESOURCEID, [REMAINING_OPS]) > 0 AND SR.USER_5 IN (@FACILITY)
    AND SR.USER_7 IN (@PROCESS) AND WO.WO_STATUS IN (@STATUS) AND ISNULL(P.COMMODITY_CODE, '_BLANK') IN (@COMMODITY)
)

SELECT WO_RELEASE_DATE, WO_DUE_DATE, WO_QTY, WO_PART_ID, WO_NUM, WO_DESCRIPTION,
    WO_BUYER_ID, WO_OP_NUM, PIECE_NO, BOM_PART_ID,
    BOM_ITEM_DESCRIPTION, QTY_REQD, QTY_AVAILABLE, USAGE_UM  ,
    CASE WHEN QTY_AVAILABLE >= QTY_REQD THEN 'YES'
        ELSE 'NO'
    END AS IN_STOCK,
    PRIMARY_LOC, REMAINING_OPS,
    IIF((CHARINDEX('CONTRACTOR', REMAINING_OPS) ) > 0, 1, 0) AS ISSERVICE,
    WO_STATUS, COMMODITY_CODE, FACILITY, PROCESS
FROM CTE_RESULTS A WITH (NOLOCK) WHERE (1=1)
```
