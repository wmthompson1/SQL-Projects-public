# WO Ops Run Times

```sql
SELECT WO.BASE_ID
    , WO.LOT_ID
    , WO.SPLIT_ID
    , WO.SUB_ID
    , WO.PART_ID
    , WO.DESIRED_QTY
    , WO.RECEIVED_QTY
    , WO.CREATE_DATE AS WO_CREATEDATE
    , WO.CLOSE_DATE AS WO_CLOSEDATE
    , O.SEQUENCE_NO
    , O.RESOURCE_ID
    , O.CLOSE_DATE AS OP_CLOSEDATE
    , O.SETUP_HRS
    , O.RUN
    , O.RUN_TYPE
    , O.RUN_HRS
    , O.ACT_SETUP_HRS
    , O.ACT_RUN_HRS
    , SR.USER_1
    , SR.USER_2
    , SR.USER_5
    , SR.USER_6
    , SR.USER_7
    , SR.USER_8
FROM OPERATION O
LEFT OUTER JOIN WORK_ORDER WO
    ON WO.[TYPE] = O.WORKORDER_TYPE
    AND WO.BASE_ID = O.WORKORDER_BASE_ID
    AND WO.LOT_ID = O.WORKORDER_LOT_ID
    AND WO.SPLIT_ID = O.WORKORDER_SPLIT_ID
    AND WO.SUB_ID = O.WORKORDER_SUB_ID 
LEFT OUTER JOIN SHOP_RESOURCE SR 
    ON O.RESOURCE_ID = SR.ID
WHERE (O.CLOSE_DATE >= @STARTDATE
        AND O.CLOSE_DATE <= @ENDDATE)
    AND CASE
            WHEN SR.USER_1 = 'A' THEN 'FIN'
            WHEN SR.USER_1 = 'M' THEN 'MFG'
            WHEN SR.USER_1 = 'O' THEN 'OUTREACH'
            WHEN SR.USER_1 = 'B' THEN 'BALLARD'
            ELSE 'OTHER'
    END IN (@TYPE)
    AND SR.USER_2 IN (@DEPT)
    AND SR.USER_5 IN (@FACILITY)
    AND SR.USER_6 IN (@GROUP)
    AND SR.USER_7 IN (@PROCESS)
    AND SR.USER_8 IN (@REPTDEPT)
    AND SR.ID IN (@RESOURCEID)

```
