# Open EE Labor Tickets

```sql
-- Visual Enterprise Reports > Labor Reports > Open EE Labor Tickets
SELECT LT.EMPLOYEE_ID
    , EE.LAST_NAME
    , EE.FIRST_NAME
    , LT.WORKORDER_BASE_ID 
    , LT.RESOURCE_ID
    , LT.TRANSACTION_DATE
    , LT.TRANSACTION_ID
    , CASE
        WHEN LT.WORKORDER_BASE_ID IS NULL THEN 'Indirect'
        ELSE 'Direct'
    END AS [Ticket Type]
FROM LABOR_TICKET AS LT
INNER JOIN EMPLOYEE AS EE
    ON LT.EMPLOYEE_ID = EE.ID
WHERE LT.ACT_CLOCK_OUT IS NULL
    AND LT.ACT_CLOCK_IN IS NOT NULL
    AND LT.HOURS_WORKED IS NULL
    AND (LT.TRANSACTION_DATE >= @StartDate 
    AND LT.TRANSACTION_DATE < @EndDate)
    AND LT.EMPLOYEE_ID IN (@EEID)

```
