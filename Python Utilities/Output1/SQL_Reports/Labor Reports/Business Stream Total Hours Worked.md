# Business Stream Total Hours Worked

```sql
--VISUAL ENTERPRISE > LABOR REPORTS > BUSINESS STREAM TOTAL HOURS WORKED
SELECT CASE
       WHEN USER_1 = 'M'  THEN 'MANUFACTURING'
       WHEN USER_1 = 'O'  THEN 'BUSINESS SOLUTIONS'
       ELSE 'FINISH'
    END AS [BUSINESS_STREAM]
    , USER_1 AS [SKILLSTYPE]
    , USER_5 AS [LOCATION]
    , USER_2 AS [PPD_PROCESS]
    , LT.RESOURCE_ID
    , SUM(LT.HOURS_WORKED) AS [TOTALHOURS]
FROM LABOR_TICKET LT
INNER JOIN SHOP_RESOURCE_SITE_VIEW SR
    ON SR.RESOURCE_ID = LT.RESOURCE_ID
    AND CASE
        WHEN SR.USER_1 = 'M'  THEN 'MANUFACTURING'
        WHEN SR.USER_1 = 'O'  THEN 'BUSINESS SOLUTIONS'
        ELSE 'FINISH'
    END IN (@SKILLSTYPE)
    AND SR.USER_5 IN (@LOCATION)
    AND SR.USER_2 IN (@PROCESS)
WHERE LT.TRANSACTION_DATE >= @STARTDATE
    AND LT.TRANSACTION_DATE <= @ENDDATE
    AND LT.RESOURCE_ID IN (@RESOURCEID)
GROUP BY USER_1
    , USER_5 
    , USER_2 
    , LT.RESOURCE_ID

```
