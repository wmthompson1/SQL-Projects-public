# Carton Label5

```sql
SELECT IT.TRANSACTION_ID, 
       IT.PART_ID, 
       ISNULL(TRT.TRACE_ID, dbo.sfnWONUMFormat(IT.WORKORDER_BASE_ID, IT.WORKORDER_LOT_ID, IT.WORKORDER_SPLIT_ID, IT.WORKORDER_SUB_ID)) AS WONUM,
       IT.WORKORDER_BASE_ID, 
       IT.WORKORDER_LOT_ID, 
       IT.QTY, 
       WAREHOUSE_ID, 
       LOCATION_ID
       , TRANSACTION_DATE
       
FROM INVENTORY_TRANS IT
    JOIN TRACE_INV_TRANS TRT
        ON IT.TRANSACTION_ID = TRT.TRANSACTION_ID

WHERE IT.TRANSACTION_ID = @TRANSACTION_ID

```
