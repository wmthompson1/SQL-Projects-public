--DECLARE @CORRECTIVE_ID AS NVARCHAR(16);
--DECLARE @site_id AS NVARCHAR(22);
--SET @corrective_id = '0000011';
--SET @site_id = 'SK01';
SELECT C.CORRECTIVE_ID,
      C.PROJECT_REC_ID,
      C.ASSIGNED_TO,
      C.ASSI_EMP_REC_ID,
      E1.NAME_1 ASSI_EMPNAME,
      C.EFFECTIVE,
      C.EFFECTIVE_DATE,
      C.CORRECTIVE_ACTION,
      C.CORRECTIVE_DATE,
      C.REFERENCE,
      C.DUE_DATE,
      C.STATUS,
      C.STATUS_EFFECTIVE_DATE,
      C.CORR_ACTION_DATE,
      C.REISSUED_FROM_REC_ID,
      C.USER_DEFINED_1,
      C.USER_DEFINED_2,
      C.BUSINESS_UNIT_ID,
      C.NC_TREND,
      C.IMMEDIATE_DATE,
      C.CLOSED_DATE,
      C.SUBJECT,
      C.CREATEDBY,
      C.CREATEDATE,
      C.SK_CORRECT_ACTION_TAKEN,
      C.SK_CORRECTION,
      C.SK_COMMUNICATION,
      C.SK_WIPVERIFIED,
      C.SK_FINGOODSVERIFY,
      C.SK_DIRECTCAUSESTMT,
      C.SK_ROOTCAUSESTMT,
      C.SK_PPDVERIFY,
      C.SK_CAFOLLOWUPQ2,
      C.SK_CAFOLLOWUPQ3,
      C.SK_CAFOLLOWUPQ4,
      C.SK_CAFOLLOWUPQ5,
      C.SK_CAFOLLOWUP_REJECTS,
      C.SK_CAFOLLOWUP_WO,
      C.SK_MAJOR_LOCATION,
      C.SK_MINOR_LOCATION,
      P.PROJECT_ID,
      P.PROJECT_TYPE,
      PT.EMPLOYEE_REC_ID,
      E.NAME_1, 
      PR.PRODUCT_ID, 
      PR.NAME, 
      SN.NoteContent, 
      SN.NoteDesc
FROM   DBO.VQ_CORRECTIVE C
      LEFT OUTER JOIN VQ_PROJECT P
        ON C.PROJECT_REC_ID = P.RECORD_ID
      LEFT OUTER JOIN VQ_PROJECT_TEAM PT
        ON PT.PROJECT_REC_ID = C.PROJECT_REC_ID
      LEFT OUTER JOIN VQ_EMPLOYEE E1
        ON E1.RECORD_ID = C.ASSI_EMP_REC_ID
      LEFT OUTER JOIN VQ_EMPLOYEE E
        ON E.RECORD_ID = PT.EMPLOYEE_REC_ID
      LEFT OUTER JOIN VQ_CORRECTIVE_PRODUCT CP
        ON C.RECORD_ID = CP.CORRECTIVE_REC_ID
      left outer JOIN VQ_PRODUCT PR
        ON CP.PRODUCT_REC_ID = PR.RECORD_ID
      LEFT OUTER JOIN  SpecificNotes SN
     ON C.NoteExistsFlag =  SN.SpecificNoteToken
WHERE  CORRECTIVE_ID = @CORRECTIVE_ID
    AND SITE_ID = @SITE_ID;
