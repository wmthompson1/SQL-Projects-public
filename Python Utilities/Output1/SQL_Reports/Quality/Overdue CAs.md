# Overdue CAs

```sql
SELECT C.CORRECTIVE_ID
    , C.[CAUSE]
    , CA.[DESCRIPTION]
    , E1.NAME_1 AS ASSIGNED_TO
    , C.EFFECTIVE_DATE
    , C.CORRECTIVE_DATE
    , C.DUE_DATE
    , C.[STATUS]
    , C.[USER_DEFINED_1] AS RISK_LEVEL
    , C.CLOSED_DATE
    , C.[SUBJECT] AS [SOURCE]
    , C.SK_MAJOR_LOCATION
    , C.SK_MINOR_LOCATION
FROM VQ_CORRECTIVE C
LEFT OUTER JOIN VQ_CAUSE CA
    ON C.[CAUSE] = CA.[CAUSE]
LEFT OUTER JOIN VQ_EMPLOYEE E1
    ON E1.RECORD_ID = C.ASSI_EMP_REC_ID
WHERE SITE_ID = @SITE_ID
    AND C.[STATUS] <> 'Closed'
    AND DUE_DATE >= @STARTDATE
    AND DUE_DATE <= @ENDDATE

```
