# Corrective Action___Tablix13

```sql
SELECT P.ENTITY_ID, 
       EI.PROJECT_REC_ID, 
       EI.INPUT_ID, 
      EI.INPUT_TEXT, 
       EI.EMPLOYEE_REC_ID, 
      EI.INPUT_TYPE, 
      EI.DUE_DATE, 
      EI.CLOSED, 
      EI.CLOSED_DATE, 
      DO.DOCUMENTNAME,
      DO.SEQUENCE,
      DO.DESCRIPTION,
      DO.DOCUMENTEXTENSION,
      DO.DOCUMENTOBJECT,
      DO.MEDIATYPE, 
      DO.ROWPOINTER
FROM  VQ_EMPLOYEE_INPUT EI
INNER JOIN VQ_PROJECT P
  ON EI.PROJECT_REC_ID = P.RECORD_ID
 LEFT OUTER JOIN[DBO].[DOCUMENTOBJECTREFERENCE] DOR
        ON EI.ROWPOINTER = DOR.TABLEROWPOINTER
         AND DOR.TABLENAME = 'VQ_EMPLOYEE_INPUT' 
     LEFT OUTER JOIN DOCUMENTOBJECT DO
        ON DOR.DOCUMENTOBJECTROWPOINTER = DO.ROWPOINTER
where
EXISTS (SELECT 1 FROM VQ_CORRECTIVE C
       WHERE C.PROJECT_REC_ID = EI.PROJECT_REC_ID
        AND C.CORRECTIVE_ID = @CORRECTIVE_ID
        AND C.SITE_ID = @SITE_ID
       )
```
