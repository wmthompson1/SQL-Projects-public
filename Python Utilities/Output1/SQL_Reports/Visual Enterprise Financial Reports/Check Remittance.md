# Check Remittance

```sql
SELECT R.CUSTOMER_ID
    , R.INVOICE_ID
    , COALESCE(COL.PART_ID, RL.REFERENCE, COL.MISC_REFERENCE, COL.SERVICE_CHARGE_ID) AS PART_ID
    , R.INVOICE_DATE 
    , R.TOTAL_AMOUNT 
    , R.PAID_AMOUNT
    , (R.TOTAL_AMOUNT - R.PAID_AMOUNT) AS 'OUTSTANDING_BALANCE'
    , RL.PACKLIST_ID, RL.PACKLIST_LINE_NO
    , SL.TRANSACTION_ID, I.WAREHOUSE_ID
    , IIF(I.LOCATION_ID IS NULL, '_blank', I.LOCATION_ID) AS 'LOCATION_ID'
FROM RECEIVABLE R
LEFT OUTER JOIN RECEIVABLE_LINE RL
    ON R.INVOICE_ID = RL.INVOICE_ID
LEFT OUTER JOIN CUST_ORDER_LINE COL
    ON RL.CUST_ORDER_ID = COL.CUST_ORDER_ID
    AND RL.CUST_ORDER_LINE_NO = COL.LINE_NO
LEFT OUTER JOIN SHIPPER_LINE SL ON RL.PACKLIST_ID = SL.PACKLIST_ID AND RL.PACKLIST_LINE_NO = SL.LINE_NO
LEFT OUTER JOIN (SELECT TRANSACTION_ID, WAREHOUSE_ID, IIF(LOCATION_ID IS NULL, '_blank', LOCATION_ID) AS 'LOCATION_ID' 
    FROM INVENTORY_TRANS with (nolock) ) I ON SL.TRANSACTION_ID = I.TRANSACTION_ID

WHERE R.[STATUS] != 'X'
    AND R.INVOICE_DATE >= @INVOICEDATE
    AND (R.TOTAL_AMOUNT - R.PAID_AMOUNT) <> 0
    AND R.CUSTOMER_ID IN (@CUSTOMER)
    
    --AND I.LOCATION_ID in ( IIF(I.LOCATION_ID IS NULL, '_blank', @LocationID) )
    AND I.LOCATION_ID in ('SIPTUL', 'SPIWIC', 'SPITUL') 
GROUP BY R.CUSTOMER_ID
    , R.INVOICE_ID
    , COALESCE(COL.PART_ID, RL.REFERENCE, COL.MISC_REFERENCE, COL.SERVICE_CHARGE_ID) 
    , R.INVOICE_DATE 
    , R.TOTAL_AMOUNT
    , R.PAID_AMOUNT
    , RL.PACKLIST_ID, RL.PACKLIST_LINE_NO, SL.TRANSACTION_ID
    , I.WAREHOUSE_ID, I.LOCATION_ID

```
