# FGA Ship List with DocRefs

```sql
--DECLARE @DAYS AS INT = 20

  IF OBJECT_ID('tempdb..#TMPMASTDOCS') IS NOT NULL
   BEGIN
     DROP TABLE #tmpmastdocs; 
   END;

--drop table #tmpmastdocs;

SELECT DRW.WORKORDER_BASE_ID, 
      DRW.WORKORDER_SUB_ID, 
      DRW.WORKORDER_TYPE, 
      PART_ID, 
      D.ID AS DOCUMENT,
      CASE
         WHEN D.REVISION_ID IS NULL
         THEN 'REV: '
         ELSE 'REV: ' + D.REVISION_ID
      END AS  REVISION,
      CASE
         WHEN U1.STRING_VAL IS NULL
         THEN 'ADCN: '
         ELSE 'ADCN: ' + U1.STRING_VAL
      END AS  ADCN
INTO #TMPMASTDOCS
FROM DOCUMENT_REF_WO DRW
    INNER JOIN WORK_ORDER W
       ON DRW.WORKORDER_BASE_ID = W.PART_ID
    INNER JOIN DOCUMENT D
       ON DRW.DOCUMENT_ID = D.ID
    LEFT JOIN USER_DEF_FIELDS U1
       ON D.ID = U1.DOCUMENT_ID
         AND U1.PROGRAM_ID = 'VMDOCMNT'
         AND U1.ID = 'UDF-0000032'
    LEFT JOIN USER_DEF_FIELDS U2
       ON D.ID = U2.DOCUMENT_ID
         AND U2.PROGRAM_ID = 'VMDOCMNT'
         AND U2.ID = 'UDF-0000031'
GROUP BY DRW.WORKORDER_BASE_ID, 
        DRW.WORKORDER_SUB_ID, 
        DRW.WORKORDER_TYPE, 
        PART_ID, 
        D.ID,
        CASE
           WHEN D.REVISION_ID IS NULL
           THEN 'REV: '
           ELSE 'REV: ' + D.REVISION_ID
        END,
        CASE
           WHEN U1.STRING_VAL IS NULL
           THEN 'ADCN: '
           ELSE 'ADCN: ' + U1.STRING_VAL
        END
ORDER BY WORKORDER_BASE_ID, 
        WORKORDER_SUB_ID, 
        DOCUMENT;


SELECT CO.ID + '/' + CONVERT(VARCHAR(3), CL.LINE_NO) AS CUSTIDLINE, 
      CO.CUSTOMER_ID, 
      CL.PART_ID, 
      DOC.DOCUMENT, 
      DOC.REVISION, 
      DOC.ADCN, 
      CL.USER_6, 
      CL.ORDER_QTY,
      CASE
         WHEN CL.DESIRED_SHIP_DATE < GETDATE()
         THEN DATEDIFF(day, GETDATE(), CL.DESIRED_SHIP_DATE)
         ELSE DATEDIFF(day, GETDATE(), CL.DESIRED_SHIP_DATE)
      END AS                                           DAYS, 
      CL.DESIRED_SHIP_DATE, 
      V.LOCATIONS AS                                   PART_LOC, 
      CL.ORDER_QTY - CL.TOTAL_SHIPPED_QTY AS           QTY_DUE
FROM CUST_ORDER_LINE AS CL
    INNER JOIN CUSTOMER_ORDER AS CO
       ON CL.CUST_ORDER_ID = CO.ID
         AND CO.STATUS IN('R')
    LEFT OUTER JOIN CUST_ADDRESS CA
       ON CA.CUSTOMER_ID = CO.CUSTOMER_ID
         AND CA.ADDR_NO = CO.SHIP_TO_ADDR_NO
    INNER JOIN PART_SITE_VIEW AS P
       ON CL.PART_ID = P.PART_ID
        -- AND P.PLANNER_USER_ID IN('MRP', 'PLAN', 'TTO', 'NEW')
         AND EXISTS
        (SELECT 1
        FROM LIVESupplemental.dbo.PLANNER_USER_ID_ReportSettings EX
        WHERE P.PLANNER_USER_ID = EX.PLANNER_USER_ID
            AND REPORT_NAME = 'FGA Ship List with DocRefs'
        )
    LEFT OUTER JOIN SKILLS_PART_LOC_LIST AS V
       ON P.PART_ID = V.ID
    LEFT OUTER JOIN DEMAND_SUPPLY_LINK DSL
       ON CL.CUST_ORDER_ID = DSL.DEMAND_BASE_ID
         AND CL.LINE_NO = DSL.DEMAND_SEQ_NO
    LEFT OUTER JOIN #TMPMASTDOCS DOC
       ON DOC.PART_ID = P.PART_ID
WHERE CL.ORDER_QTY > CL.TOTAL_SHIPPED_QTY
     AND CL.ORDER_QTY - CL.TOTAL_SHIPPED_QTY <> 0
     AND CO.CUSTOMER_ID NOT IN ('BOE609', 'BOE614')
    AND CL.LINE_STATUS NOT IN ('C')
AND CL.PART_ID NOT LIKE 'K12%'
AND CL.PART_ID NOT LIKE '%FAI'
AND CASE
       WHEN CL.DESIRED_SHIP_DATE < GETDATE()
       THEN DATEDIFF(day, GETDATE(), CL.DESIRED_SHIP_DATE)
       ELSE DATEDIFF(day, GETDATE(), CL.DESIRED_SHIP_DATE)
    END <= 10--@DAYS
--  and CL.part_id = '212A1214-13'
AND DSL.SUPPLY_BASE_ID IS NULL
ORDER BY DESIRED_SHIP_DATE;

```
