# FGA CBO Ship List

```sql
-- VISUAL ENTERPRISE > SHIPPING REPORTS > FGA CBO Ship List
SELECT CASE 
        WHEN CL.ORDER_QTY - CL.TOTAL_SHIPPED_QTY <= ISNULL(
            (SELECT SUM(QTY)AS SHIP_QTY FROM PART_LOCATION pl WHERE pl.PART_ID = P.PART_ID AND STATUS = 'A'),0) THEN 1 
        ELSE 2 
    END AS SORT
    , CASE 
        WHEN ADDR_2 LIKE '%HOLD%' THEN 6
        WHEN CITY LIKE '%HOLD%' THEN 6
        WHEN CITY LIKE '%RENTON%' AND 
            CASE 
            WHEN CL.DESIRED_SHIP_DATE < GETDATE() THEN DATEDIFF(DAY,GETDATE(), CL.DESIRED_SHIP_DATE)
            ELSE DATEDIFF(DAY,GETDATE(), CL.DESIRED_SHIP_DATE)
            END >= 5 THEN 5
        WHEN CITY LIKE '%EVERETT%' AND CASE 
            WHEN CL.DESIRED_SHIP_DATE < GETDATE() THEN DATEDIFF(DAY,GETDATE(), CL.DESIRED_SHIP_DATE)
            ELSE DATEDIFF(DAY,GETDATE(), CL.DESIRED_SHIP_DATE)
            END >= 7 THEN 4
        WHEN CO.CUSTOMER_ID NOT LIKE 'BOE%' AND CASE 
            WHEN CL.DESIRED_SHIP_DATE < GETDATE() THEN DATEDIFF(DAY,GETDATE(), CL.DESIRED_SHIP_DATE)
            ELSE DATEDIFF(DAY,GETDATE(), CL.DESIRED_SHIP_DATE)
            END >= 7 THEN 3
        WHEN STATE NOT LIKE 'WA%' AND CASE 
            WHEN CL.DESIRED_SHIP_DATE < GETDATE() THEN DATEDIFF(DAY,GETDATE(), CL.DESIRED_SHIP_DATE)
            ELSE DATEDIFF(DAY,GETDATE(), CL.DESIRED_SHIP_DATE)
            END>= 5 THEN 2
        WHEN TOTAL_SHIPPED_QTY + P.QTY_ON_HAND >= ORDER_QTY THEN 0
        ELSE 1
    END AS SORT2
    , CO.ID + '/'+ CONVERT(VARCHAR(3),CL.LINE_NO) AS CUSTIDLINE
    , CO.CUSTOMER_ID
    , CASE 
        WHEN ADDR_2 LIKE '%HOLD%' THEN 'HOLD'
        WHEN CITY LIKE '%HOLD%' THEN 'HOLD' 
        WHEN CA.STATE IS NULL  AND CA.COUNTRY <> 'USA'  THEN CA.COUNTRY
        ELSE CA.CITY +', ' + CA.STATE              
    END AS CO_OR_STATE
    , CL.PART_ID
    , CL.USER_6
    , CL.ORDER_QTY
    , CASE 
        WHEN CL.DESIRED_SHIP_DATE < GETDATE() THEN DATEDIFF(DAY,GETDATE(), CL.DESIRED_SHIP_DATE)
        ELSE DATEDIFF(DAY,GETDATE(), CL.DESIRED_SHIP_DATE)
    END AS DAYS
    , CL.DESIRED_SHIP_DATE
    , CL.TOTAL_SHIPPED_QTY
    , CO.[STATUS]
    , P.PLANNER_USER_ID
    , V.LOCATIONS AS PART_LOC
    , P.QTY_ON_ORDER AS QTY
    , CL.ORDER_QTY - CL.TOTAL_SHIPPED_QTY AS QTY_DUE
    , ISNULL((SELECT SUM(QTY) AS SHIP_QTY FROM PART_LOCATION pl WHERE pl.PART_ID = P.PART_ID AND [STATUS] = 'A'),0) AS SHIP_QTY
    , CL.LINE_STATUS
FROM CUST_ORDER_LINE AS CL
INNER JOIN CUSTOMER_ORDER AS CO
    ON CL.CUST_ORDER_ID = CO.ID
    AND CO.[STATUS] = 'R'
LEFT OUTER JOIN CUST_ADDRESS CA
    ON CA.CUSTOMER_ID = CO.CUSTOMER_ID
    AND CA.ADDR_NO = CO.SHIP_TO_ADDR_NO
INNER JOIN PART_SITE_VIEW P
    ON CL.PART_ID = P.PART_ID
--    AND P.PLANNER_USER_ID IN('MRP','PLAN','TTO', 'NEW')
    AND EXISTS
        (SELECT 1
        FROM LIVESupplemental.dbo.PLANNER_USER_ID_ReportSettings EX
        WHERE P.PLANNER_USER_ID = EX.PLANNER_USER_ID
            AND REPORT_NAME = 'FGA CBO Ship List'
        )

LEFT OUTER JOIN SKILLS_PART_LOC_LIST V
    ON P.PART_ID = V.ID
LEFT OUTER JOIN DEMAND_SUPPLY_LINK DSL
    ON CL.CUST_ORDER_ID = DSL.DEMAND_BASE_ID
    AND CL.LINE_NO = DSL.DEMAND_SEQ_NO
WHERE 
    P.SITE_ID = 'SK01'
    AND CL.ORDER_QTY > CL.TOTAL_SHIPPED_QTY
    AND CO.CUSTOMER_ID NOT IN ('BOE609','BOE614')
    AND CL.LINE_STATUS <> 'C'
    AND CL.PART_ID NOT LIKE 'K12%'
    AND CL.PART_ID NOT LIKE '%FAI'
    AND CASE 
        WHEN CL.DESIRED_SHIP_DATE < GETDATE() THEN DATEDIFF(DAY, GETDATE(), CL.DESIRED_SHIP_DATE)
        ELSE DATEDIFF(DAY, GETDATE(), CL.DESIRED_SHIP_DATE)
    END  <= @DAYS
    AND DSL.SUPPLY_BASE_ID IS NULL
    AND (CL.USER_6 LIKE ('CBO%') 
        OR CL.USER_6 LIKE ('VOI%') 
        OR CL.USER_6 LIKE ('KIT%')
    )

```
