SET NOCOUNT ON;

SELECT
  r.INVOICE_ID,
  r.INVOICE_DATE,
  r.CUSTOMER_ID,
  c.NAME AS CUSTOMER_NAME,
  r.TOTAL_AMOUNT,
  r.PAID_AMOUNT,
  r.POSTING_DATE,
  rl.LINE_NO,
  rl.PRODUCT_CODE,
  rl.QTY,
  rl.AMOUNT AS LINE_AMOUNT,
  rl.CUST_ORDER_ID,
  rl.PACKLIST_ID
FROM dbo.RECEIVABLE r
LEFT JOIN dbo.RECEIVABLE_LINE rl ON rl.INVOICE_ID = r.INVOICE_ID
LEFT JOIN dbo.CUSTOMER c ON c.ID = r.CUSTOMER_ID
WHERE r.INVOICE_DATE >= '2025-07-01'
ORDER BY r.INVOICE_DATE, r.INVOICE_ID, rl.LINE_NO;
