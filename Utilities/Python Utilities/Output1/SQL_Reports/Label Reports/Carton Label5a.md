# Carton Label5a

```sql
SELECT TRT.TRANSACTION_ID, WORKORDER_BASE_ID, WORKORDER_LOT_ID, WORKORDER_SUB_ID, WORKORDER_SPLIT_ID, TRT.QTY, IT.WAREHOUSE_ID, IT.LOCATION_ID, IT.PART_ID
    , ISNULL(TRT.TRACE_ID, dbo.sfnWONUMFormat(WORKORDER_BASE_ID, WORKORDER_LOT_ID, WORKORDER_SPLIT_ID, WORKORDER_SUB_ID)) AS TRACE_ID
FROM TRACE_INV_TRANS TRT
    JOIN INVENTORY_TRANS IT
        ON TRT.TRANSACTION_ID = IT.TRANSACTION_ID
WHERE TRT.TRANSACTION_ID = @TRANSACTION_ID
```
