```sql
SELECT DSL.SUPPLY_BASE_ID, DSL.SUPPLY_LOT_ID, DSL.SUPPLY_SUB_ID, DSL.SUPPLY_SPLIT_ID, CUSTOMER_ORDER.CUSTOMER_ID, CUSTOMER_ORDER.ID, CUSTOMER_ORDER.CUSTOMER_PO_REF, 
    CUST_ORDER_LINE.PART_ID, CUST_ORDER_LINE.CUSTOMER_PART_ID, CUST_ORDER_LINE.ORDER_QTY
    , dbo.sfnWONUMFormat(DSL.SUPPLY_BASE_ID, DSL.SUPPLY_LOT_ID, DSL.SUPPLY_SPLIT_ID, DSL.SUPPLY_SUB_ID) AS WONUM_ID
FROM CUST_ORDER_LINE 
    INNER JOIN CUSTOMER_ORDER 
        ON CUST_ORDER_LINE.CUST_ORDER_ID = CUSTOMER_ORDER.ID 
    INNER JOIN DEMAND_SUPPLY_LINK DSL
        ON CUST_ORDER_LINE.CUST_ORDER_ID = DSL.DEMAND_BASE_ID 
            AND CUST_ORDER_LINE.LINE_NO = DSL.DEMAND_SEQ_NO
WHERE (CUSTOMER_ORDER.ID = @ORDER_ID)
```
