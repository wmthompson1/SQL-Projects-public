# Standards Label 3.5

```sql
SELECT IT.PURC_ORDER_ID
    , IT.PURC_ORDER_LINE_NO
    , IT.PART_ID
    , TRT.TRACE_ID
    , T.APROPERTY_2
    , PL.PRIMARY_LOC_ID
FROM DBO.INVENTORY_TRANS IT
INNER JOIN PART_SITE PL
    ON IT.PART_ID = PL.PART_ID
INNER JOIN DBO.TRACE_INV_TRANS TRT
    ON IT.TRANSACTION_ID = TRT.TRANSACTION_ID
INNER JOIN DBO.TRACE T
    ON TRT.TRACE_ID = T.ID and PL.PART_ID=t.PART_ID
WHERE IT.PURC_ORDER_ID = @PO_ID  --'103309'--
    AND ISNULL(IT.PURC_ORDER_LINE_NO, 0) IN (SELECT LTRIM(ITEM) FROM LIVESUPPLEMENTAL.DBO.UFNSPLIT(@PO_LINE, ', ')
        )
    AND TRT.QTY > 0
ORDER BY IT.TRANSACTION_ID      
;

```
