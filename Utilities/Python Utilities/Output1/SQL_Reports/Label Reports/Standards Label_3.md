```sql
SELECT IT.PURC_ORDER_ID,
      IT.PURC_ORDER_LINE_NO,
      IT.PART_ID,
      T.APROPERTY_2,
      PL.PRIMARY_LOC_ID
FROM   DBO.INVENTORY_TRANS IT
      INNER JOIN PART_SITE PL
        ON IT.PART_ID = PL.PART_ID
      INNER JOIN DBO.TRACE_INV_TRANS TRT
        ON IT.TRANSACTION_ID = TRT.TRANSACTION_ID
      INNER JOIN DBO.TRACE T
        ON TRT.TRACE_ID = T.ID
           AND TRT.TRACE_ID = CASE
                             WHEN CHARINDEX(' / ',TRT.TRACE_ID) > 0
                             THEN CONVERT( VARCHAR(15),IT.PURC_ORDER_ID)+' / '+CONVERT(VARCHAR(3),IT.PURC_ORDER_LINE_NO)
                             WHEN CHARINDEX('/ ',TRT.TRACE_ID) > 0
                             THEN CONVERT(VARCHAR(15),IT.PURC_ORDER_ID)+'/ '+CONVERT(VARCHAR(3),IT.PURC_ORDER_LINE_NO)
                             WHEN CHARINDEX('-',TRT.TRACE_ID) > 0
                             THEN CONVERT(VARCHAR(15),IT.PURC_ORDER_ID)+'-'+CONVERT(VARCHAR(3),IT.PURC_ORDER_LINE_NO)
                             WHEN CHARINDEX('/',TRT.TRACE_ID) > 0
                             THEN CONVERT(VARCHAR(15),IT.PURC_ORDER_ID)+'/'+CONVERT(VARCHAR(3),IT.PURC_ORDER_LINE_NO)
                             ELSE CONVERT(VARCHAR(15),IT.PURC_ORDER_ID)
                          END
      INNER JOIN DBO.RECEIVER_LINE RL
        ON IT.PURC_ORDER_ID = RL.PURC_ORDER_ID
           AND IT.PURC_ORDER_LINE_NO = RL.PURC_ORDER_LINE_NO
           AND IT.PURC_ORDER_ID = @purchase_order --'103309'--
           AND isnull(IT.PURC_ORDER_LINE_NO,0) IN
(
    SELECT LTRIM(ITEM)
    FROM   LIVESUPPLEMENTAL.DBO.UFNSPLIT(@po_line,', ')
)--@PO_LINE
      AND TRT.QTY > 0;
```
