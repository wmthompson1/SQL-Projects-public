```sql
SELECT DISTINCT 
                  W.BASE_ID, W.PART_ID, W.DESIRED_QTY, CUSTOMER_ORDER.ORDER_DATE, CUSTOMER_ORDER.DESIRED_SHIP_DATE, 
                  CUSTOMER_ORDER.CUSTOMER_PO_REF, CUSTOMER_ORDER.ID, CUSTOMER_ORDER.CUSTOMER_ID, PART.DESCRIPTION, SKILL_WO_DATA.TOTAL_WO_QTY, 
                  CUST_ADDRESS.NAME
                  , dbo.sfnWONUMFormat(W.BASE_ID, W.LOT_ID, W.SPLIT_ID, W.SUB_ID) AS WONUM
FROM     WORK_ORDER AS W 
    INNER JOIN DEMAND_SUPPLY_LINK AS DEMAND_SUPPLY_LINK 
        ON W.BASE_ID = DEMAND_SUPPLY_LINK.SUPPLY_BASE_ID 
            AND W.LOT_ID = DEMAND_SUPPLY_LINK.SUPPLY_LOT_ID 
            AND W.SPLIT_ID = DEMAND_SUPPLY_LINK.SUPPLY_SPLIT_ID 
            AND W.SUB_ID = DEMAND_SUPPLY_LINK.SUPPLY_SUB_ID 
    INNER JOIN PART AS PART 
        ON W.PART_ID = PART.ID 
    LEFT OUTER JOIN SKILL_WO_DATA AS SKILL_WO_DATA 
        ON W.BASE_ID = SKILL_WO_DATA.BASE_ID 
            AND W.LOT_ID = SKILL_WO_DATA.LOT_ID 
            AND W.SPLIT_ID = SKILL_WO_DATA.SPLIT_ID 
    INNER JOIN CUST_ORDER_LINE AS CUST_ORDER_LINE 
        ON DEMAND_SUPPLY_LINK.DEMAND_BASE_ID = CUST_ORDER_LINE.CUST_ORDER_ID 
            AND DEMAND_SUPPLY_LINK.DEMAND_SEQ_NO = CUST_ORDER_LINE.LINE_NO 
    INNER JOIN CUSTOMER_ORDER AS CUSTOMER_ORDER 
        ON CUST_ORDER_LINE.CUST_ORDER_ID = CUSTOMER_ORDER.ID 
    INNER JOIN CUST_ADDRESS 
        ON CUSTOMER_ORDER.CUSTOMER_ID = CUST_ADDRESS.CUSTOMER_ID 
            AND CUSTOMER_ORDER.SHIP_TO_ADDR_NO = CUST_ADDRESS.ADDR_NO
WHERE W.TYPE = 'W'
    AND W.BASE_ID = @WORKORDER_ID
    AND W.LOT_ID = @LOT_ID
    AND W.SPLIT_ID = @SPLIT_ID

```
