# AR Aging

```sql
SELECT *
    , DATEDIFF(DD, DATEADD(DD, PERIOD_DAYS, INVOICE_DATE), GETDATE()) AS CATEGORY
    , CASE
        WHEN DATEDIFF(DD, DATEADD(DD, PERIOD_DAYS, INVOICE_DATE), GETDATE()) <= 0 THEN AMOUNT_OUTSTANDING
        ELSE 0
    END AS CURRENT_AMOUNT
    , CASE
        WHEN DATEDIFF(DD, DATEADD(DD, PERIOD_DAYS, INVOICE_DATE), GETDATE()) BETWEEN 1 AND 30 THEN AMOUNT_OUTSTANDING
        ELSE 0
    END AS [30 DAYS PAST DUE]
    , CASE
        WHEN DATEDIFF(DD, DATEADD(DD, PERIOD_DAYS, INVOICE_DATE), GETDATE()) BETWEEN 31 AND 45 THEN AMOUNT_OUTSTANDING
        ELSE 0
    END AS [45 DAYS PAST DUE]
    , CASE
        WHEN DATEDIFF(DD, DATEADD(DD, PERIOD_DAYS, INVOICE_DATE), GETDATE()) BETWEEN 46 AND 60 THEN AMOUNT_OUTSTANDING
        ELSE 0
    END AS [60 DAYS PAST DUE]
    , CASE
        WHEN DATEDIFF(DD, DATEADD(DD, PERIOD_DAYS, INVOICE_DATE), GETDATE()) > 60 THEN AMOUNT_OUTSTANDING
        ELSE 0
    END AS [90 DAYS PAST DUE]
FROM (SELECT CUSTOMER_ID
    , C.[NAME] AS CUSTOMER_NAME
    , R.INVOICE_ID
    , R.INVOICE_DATE
    , CASE
        WHEN R.TERMS_DESCRIPTION = 'Due on Receipt' THEN 0
        ELSE R.TERMS_NET_DAYS
    END AS PERIOD_DAYS
    , R.TERMS_NET_DAYS
    , R.TERMS_DESCRIPTION
    , R.TOTAL_AMOUNT
    , R.PAID_AMOUNT
    , R.TOTAL_AMOUNT - R.PAID_AMOUNT AS AMOUNT_OUTSTANDING
    ,(select STUFF((SELECT DISTINCT ',' +  (c.CUSTOMER_PO_REF)
FROM   shipper  s inner join  customer_order c on c.id=s.CUST_ORDER_ID where  s.INVOICE_ID=R.INVOICE_ID
              FOR XML PATH(''), TYPE).value('text()[1]', 'nvarchar(max)')
        , 1, LEN(','), '') ) as CUSTOMER_PO_REF
FROM RECEIVABLE R
LEFT OUTER JOIN CUSTOMER C
    ON R.CUSTOMER_ID = C.ID
WHERE R.[TYPE] NOT IN ('V')
    AND R.TOTAL_AMOUNT - R.PAID_AMOUNT != 0
    AND R.[STATUS] NOT IN ('X')
) X




```
