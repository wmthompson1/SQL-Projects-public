; WITH CTE_BASE AS
(SELECT IB.PART_ID
    , P.[DESCRIPTION]
    , P.COMMODITY_CODE 
    , SUM(IB.QTY) AS QTY
    , SUM(IB.MATERIAL_AMOUNT) AS MATERIAL_AMOUNT
    , SUM(IB.LABOR_AMOUNT) AS LABOR_AMOUNT
    , SUM(IB.BURDEN_AMOUNT) AS BURDEN_AMOUNT
    , SUM(IB.SERVICE_AMOUNT) AS SERVICE_AMOUNT
    , (SUM(IB.MATERIAL_AMOUNT) + SUM(IB.LABOR_AMOUNT) + SUM(IB.BURDEN_AMOUNT) + SUM(IB.SERVICE_AMOUNT)) AS TOTAL_AMOUNT
FROM INVENTORY_BALANCE IB
LEFT OUTER JOIN PART P
    ON P.ID = IB.PART_ID
WHERE IB.POSTING_DATE = @POSTINGDATE
--    AND P.COMMODITY_CODE IN (@COMMCODE)
GROUP BY IB.PART_ID
    , P.[DESCRIPTION]
    , P.COMMODITY_CODE 
)
--SELECT * FROM CTE_BASE

, CTE_CONTRACTS AS
(SELECT PART_ID
    , MAX(CPE.DEFAULT_UNIT_PRICE) AS DEFAULT_UNIT_PRICE
FROM CUST_PRICE_EFFECT CPE
WHERE CPE.DISCONTINUE_DATE >= GETDATE()
    AND CPE.EFFECTIVE_DATE <= GETDATE()
GROUP BY PART_ID
)

, CTE_IT AS
(SELECT PART_ID
     , MAX(IT.TRANSACTION_ID) AS LAST_SALE_TRANS
    , MAX(IT.TRANSACTION_DATE) AS LAST_SALES_DATE
FROM INVENTORY_TRANS IT
WHERE IT.[TYPE] = 'O'
    AND IT.CLASS = 'I'
GROUP BY PART_ID
)

SELECT CTE.PART_ID
    , CTE.[DESCRIPTION]
    , CTE.COMMODITY_CODE 
    , CTE.QTY
    , CTE.MATERIAL_AMOUNT
    , CTE.LABOR_AMOUNT
    , CTE.BURDEN_AMOUNT
    , CTE.SERVICE_AMOUNT
    , CTE.TOTAL_AMOUNT
    , CASE 
        WHEN CC.DEFAULT_UNIT_PRICE IS NULL THEN NULL
        ELSE CIT.LAST_SALE_TRANS
    END AS LAST_SALE_TRANS
    , CASE
        WHEN CC.DEFAULT_UNIT_PRICE IS NULL THEN NULL
        ELSE CIT.LAST_SALES_DATE
    END AS LAST_SALES_DATE
    , CASE
        WHEN CC.DEFAULT_UNIT_PRICE IS NULL THEN NULL
        ELSE DATEDIFF(MM, CIT.LAST_SALES_DATE, GETDATE())
    END AS MONTHS_SINCE_SALE
    , CASE    
        WHEN CC.DEFAULT_UNIT_PRICE IS NULL THEN NULL
        WHEN DATEDIFF(MM, CIT.LAST_SALES_DATE, GETDATE()) < 18 THEN 0
        WHEN DATEDIFF(MM, CIT.LAST_SALES_DATE, GETDATE()) < 25 THEN 0.20
        WHEN DATEDIFF(MM, CIT.LAST_SALES_DATE, GETDATE()) < 37 THEN 0.5
        WHEN DATEDIFF(MM, CIT.LAST_SALES_DATE, GETDATE()) >36 THEN 1
    END AS RESERVE_PCT
    , TOTAL_AMOUNT * CASE    
        WHEN CC.DEFAULT_UNIT_PRICE IS NULL THEN NULL
        WHEN DATEDIFF(MM, CIT.LAST_SALES_DATE, GETDATE()) < 18 THEN 0
        WHEN DATEDIFF(MM, CIT.LAST_SALES_DATE, GETDATE()) < 25 THEN 0.20
        WHEN DATEDIFF(MM, CIT.LAST_SALES_DATE, GETDATE()) < 37 THEN 0.5
        WHEN DATEDIFF(MM, CIT.LAST_SALES_DATE, GETDATE()) > 36 THEN 1
    END AS RESERVE_AMOUNT
    , CASE
        WHEN CC.DEFAULT_UNIT_PRICE IS NULL THEN NULL
        ELSE CPE.SELLING_UM
    END AS [SELLING_UM]
    , CASE
        WHEN CC.DEFAULT_UNIT_PRICE IS NULL THEN NULL
        ELSE CPE.CUSTOMER_PART_ID 
    END AS [CONTRACT]
    , CASE
        WHEN CC.DEFAULT_UNIT_PRICE IS NULL THEN NULL
        ELSE CO.CUSTOMER_ID
    END AS CUSTOMER_ID
     , CASE
        WHEN CC.DEFAULT_UNIT_PRICE IS NULL THEN NULL
        WHEN COMMODITY_CODE in ('Details', 'End Item') THEN CC.DEFAULT_UNIT_PRICE
        ELSE 0
    END AS 'Selling_Unit_Price'
    , CASE
        WHEN CC.DEFAULT_UNIT_PRICE IS NULL THEN NULL
        WHEN COMMODITY_CODE in ('Details', 'End Item') THEN TOTAL_AMOUNT
        ELSE 0
    END AS TOTAL_COSTS
    , CASE
        WHEN CC.DEFAULT_UNIT_PRICE IS NULL THEN NULL
        WHEN COMMODITY_CODE in ('Details', 'End Item') THEN (CTE.QTY * CC.DEFAULT_UNIT_PRICE) - TOTAL_AMOUNT
        ELSE NULL
    END AS Total_Price
    , CASE
        WHEN CC.DEFAULT_UNIT_PRICE IS NULL THEN NULL
        WHEN COMMODITY_CODE in ('Details', 'End Item') THEN (CTE.QTY * CC.DEFAULT_UNIT_PRICE) - TOTAL_AMOUNT - TOTAL_AMOUNT
        ELSE NULL
    END AS Change
FROM CTE_BASE CTE
LEFT OUTER JOIN CTE_IT CIT
    ON CTE.PART_ID = CIT.PART_ID
LEFT OUTER JOIN INVENTORY_TRANS IT
    ON IT.PART_ID = CTE.PART_ID
    AND IT.TRANSACTION_ID = CIT.LAST_SALE_TRANS
LEFT OUTER JOIN CUSTOMER_ORDER CO
    ON IT.CUST_ORDER_ID = CO.ID
LEFT OUTER JOIN CTE_CONTRACTS CC
    ON CC.PART_ID = CTE.PART_ID
LEFT OUTER JOIN CUST_PRICE_EFFECT CPE
    ON CPE.PART_ID = CTE.PART_ID
    AND CPE.DEFAULT_UNIT_PRICE = CC.DEFAULT_UNIT_PRICE
    AND CPE.DISCONTINUE_DATE >= GETDATE()
    AND CPE.EFFECTIVE_DATE <= GETDATE()
    AND CO.CUSTOMER_ID = CPE.CUSTOMER_ID
