# AP Aging

```sql
SELECT *
    , DATEDIFF(DD, DATEADD(DD, PERIOD_DAYS, INVOICE_DATE), GETDATE()) AS CATEGORY
    , CASE
        WHEN DATEDIFF(DD, DATEADD(DD, PERIOD_DAYS, INVOICE_DATE), GETDATE()) <= 0 THEN AMOUNT_OUTSTANDING
        ELSE 0
    END AS CURRENT_AMOUNT
    , CASE
        WHEN DATEDIFF(DD, DATEADD(DD, PERIOD_DAYS, INVOICE_DATE), GETDATE()) BETWEEN 1 AND 30 THEN AMOUNT_OUTSTANDING
        ELSE 0
    END AS [30 DAYS PAST DUE]
    , CASE
        WHEN DATEDIFF(DD, DATEADD(DD, PERIOD_DAYS, INVOICE_DATE), GETDATE()) BETWEEN 31 AND 60 THEN AMOUNT_OUTSTANDING
        ELSE 0
    END AS [60 DAYS PAST DUE]
    , CASE
        WHEN DATEDIFF(DD, DATEADD(DD, PERIOD_DAYS, INVOICE_DATE), GETDATE()) BETWEEN 61 AND 90 THEN AMOUNT_OUTSTANDING
        ELSE 0
    END AS [90 DAYS PAST DUE]
    , CASE
        WHEN DATEDIFF(DD, DATEADD(DD, PERIOD_DAYS, INVOICE_DATE), GETDATE()) > 90 THEN AMOUNT_OUTSTANDING
        ELSE 0
    END AS [120 DAYS PAST DUE]
FROM (SELECT VENDOR_ID
    , V.[NAME] AS VENDOR_NAME
    , P.INVOICE_ID
    , P.INVOICE_DATE
    , P.PAYMENT_DATE
    , CASE
        WHEN P.TERMS_DESCRIPTION = 'Due on Receipt' THEN 0
        ELSE P.TERMS_NET_DAYS
    END AS PERIOD_DAYS
    , P.TERMS_NET_DAYS
    , P.TERMS_DESCRIPTION
    , P.TOTAL_AMOUNT
    , P.PAID_AMOUNT
    , P.TOTAL_AMOUNT - P.PAID_AMOUNT AS AMOUNT_OUTSTANDING
FROM PAYABLE P
INNER JOIN VENDOR V
    ON P.VENDOR_ID = V.ID
WHERE P.PAY_STATUS IN ('H', 'P')
    AND P.TOTAL_AMOUNT - P.PAID_AMOUNT != 0
) X


```
