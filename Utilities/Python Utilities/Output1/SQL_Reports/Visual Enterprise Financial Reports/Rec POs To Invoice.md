# Rec POs To Invoice

```sql

; WITH CTE_INVOICE AS
(SELECT PO.PNO
    , F.COMPANYID AS VENDOR
    , F.DOCNO AS VENDOR_INVOICE
FROM TBLFILES F
LEFT OUTER JOIN TBLPO PO
    ON F.ID = PO.FILEID
LEFT OUTER JOIN LIVE.DBO.PAYABLE_LINE PL
    ON PL.PURC_ORDER_ID = PO.PNO 
WHERE F.SECTION = 'INVOICE'
    AND PL.VOUCHER_ID IS NULL
    AND F.VOUCHER# IS NULL
    AND DDATE > '1/1/2021'
    AND PO.PNO IN (SELECT PO.PNO
        FROM TBLFILES F
        LEFT OUTER JOIN TBLPO PO
            ON F.ID = PO.FILEID
        LEFT OUTER JOIN LIVE.DBO.PAYABLE_LINE PL
            ON PL.PURC_ORDER_ID = PO.PNO 
        WHERE F.SECTION = 'PACK SLIP'
            AND PL.RECEIVER_ID IS NULL
            AND F.RECEIVER# IS NOT NULL)
)

, CTE_PACKSLIP AS
(SELECT PO.PNO
    , F.COMPANYID AS VENDOR
    , F.DOCNO AS VENDOR_PACKLIST
FROM TBLFILES F
LEFT OUTER JOIN TBLPO PO
    ON F.ID = PO.FILEID
LEFT OUTER JOIN LIVE.DBO.PAYABLE_LINE PL
    ON PL.PURC_ORDER_ID = PO.PNO 
WHERE F.SECTION = 'PACK SLIP'
    AND PL.RECEIVER_ID IS NULL
    AND F.RECEIVER# IS NOT NULL
    AND DDATE > '1/1/2021'
    AND PO.PNO IN (SELECT PO.PNO
        FROM TBLFILES F
        LEFT OUTER JOIN TBLPO PO
            ON F.ID = PO.FILEID
        LEFT OUTER JOIN LIVE.DBO.PAYABLE_LINE PL
            ON PL.PURC_ORDER_ID = PO.PNO 
        WHERE F.SECTION = 'INVOICE'
            AND PL.VOUCHER_ID IS NULL
            AND F.VOUCHER# IS NULL
            AND DDATE > '1/1/2021')
)

SELECT I.*
, P.VENDOR_PACKLIST
,Live.dbo.Get_Inspect(i.pNo) as Inspect
FROM CTE_INVOICE I
INNER JOIN CTE_PACKSLIP P
    ON I.PNO = P.PNO
ORDER BY I.PNO


```
