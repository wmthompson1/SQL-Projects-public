# Packslip Receiver Audit

```sql
SELECT R.ID AS RECEIVER_ID
    , RL.LINE_NO AS RECEIVER_LINE
    , PO.VENDOR_ID
    , V.[NAME] AS VENDOR_NAME
    , R.RECEIVED_DATE
    , RL.PURC_ORDER_ID
    , RL.PURC_ORDER_LINE_NO
FROM dbo.RECEIVER R
INNER JOIN dbo.RECEIVER_LINE RL
ON R.ID = RL.RECEIVER_ID
LEFT OUTER JOIN PURC_ORDER_LINE POL
ON RL.PURC_ORDER_ID = POL.PURC_ORDER_ID
AND RL.PURC_ORDER_LINE_NO = POL.LINE_NO
LEFT OUTER JOIN PURCHASE_ORDER PO
ON RL.PURC_ORDER_ID = PO.ID
LEFT OUTER JOIN VENDOR V
ON PO.VENDOR_ID = V.ID
WHERE R.RECEIVED_DATE >= ISNULL(@STARTDATE, '1/1/2021')
    AND R.ID NOT IN (SELECT CAST(F.Receiver# AS NVARCHAR(25)) AS RECEIVER_ID
        FROM dbRMS.dbo.TBLFILES F
        WHERE ISNULL(F.Receiver#, '0') != '0'
            AND F.SECTION = 'Pack Slip'
        )
GROUP BY R.ID
    , RL.LINE_NO
    , PO.VENDOR_ID
    , V.[NAME]
    , R.RECEIVED_DATE
    , RL.PURC_ORDER_ID
    , RL.PURC_ORDER_LINE_NO

```
