IF OBJECT_ID('tempdb..#RESULTS') IS NOT NULL DROP TABLE #RESULTS
IF OBJECT_ID('tempdb..#TEMP1') IS NOT NULL DROP TABLE #TEMP1
IF OBJECT_ID('tempdb..#SPECS') IS NOT NULL DROP TABLE #SPECS

SELECT PR.ID AS REQUISITION_ID
    , PRL.LINE_NO AS REQUISITION_LINE
    , SPECS =  IIF(DR.DOCUMENT_ID IS NULL, '', DR.DOCUMENT_ID + '   ' ) + IIF(D.REVISION_ID IS NULL, '', 'Rev: ' + D.REVISION_ID)
INTO #TEMP1
FROM PURC_REQUISITION PR
INNER JOIN PURC_REQ_LINE PRL
    ON PR.ID = PRL.PURC_REQ_ID
LEFT JOIN PART P
    ON P.ID = PRL.PART_ID
LEFT OUTER JOIN DOCUMENT_REFERENCE DR
    ON DR.ID = PRL.PART_ID
LEFT OUTER JOIN DOCUMENT D
    ON D.ID = DR.DOCUMENT_ID
    AND DR.SOURCE_TYPE = 'P'
    AND D.SITE_ID = 'SK01'
WHERE DR.DOCUMENT_ID IS NOT NULL
    AND PR.ID = @REQ_ID

SELECT CAT.REQUISITION_ID
    , CAT.REQUISITION_LINE
    , NULLIF(REPLACE(STUFF((SELECT '~' + SUB.SPECS AS [text()]
        FROM #TEMP1 SUB
        WHERE SUB.REQUISITION_ID = CAT.REQUISITION_ID
            AND SUB.REQUISITION_LINE = CAT.REQUISITION_LINE
        FOR XML PATH('')
        ), 1, 1, '' )
        , '~', CHAR(13) + CHAR(10))
        , '') AS [SPECLIST]
INTO #RESULTS
FROM #TEMP1 CAT

SELECT REQUISITION_ID
    , REQUISITION_LINE
    , [SPECLIST]
INTO #SPECS
FROM #RESULTS
GROUP BY REQUISITION_ID
    , REQUISITION_LINE
    , [SPECLIST]

SELECT PR.ID AS REQ_ID
    , PRL.LINE_NO
--> PR INFORMATION
    , PR.REQUISITION_DATE
    , PR.REQUESTOR
    , EE.FIRST_NAME + ' ' + EE.LAST_NAME AS REQUESTOR_NAME
    , PR.PREPARER
    , PRE.FIRST_NAME + ' ' + PRE.LAST_NAME AS PREPARER_NAME
    , PR.BUYER
    , B.FIRST_NAME + ' ' + B.LAST_NAME AS BUYER_NAME
    , PR.SHIP_VIA
--    , PR.FREE_ON_BOARD
    , EE.EMAIL_ADDR BUYER_EMAIL
    , PR.TOTAL_AMT_ORDERED AS PR_TOTAL
    , CASE 
        WHEN ISNULL(PrePaidCC.BOOL_VAL,0) = 1 THEN 'Prepaid via CC' 
        ELSE T.[DESCRIPTION] 
    END AS TERMS
--> VENDOR INFORMATION
    , PR.VENDOR_ID
    , V.[NAME] AS VENDOR_NAME
    , V.ADDR_1 AS VENDOR_ADDR_1
    , V.ADDR_2 AS VENDOR_ADDR_2
    , V.ADDR_3 AS VENDOR_ADDR_3
    , CASE 
        WHEN V.COUNTRY IN ('US', 'USA', 'UNITED STATES') THEN V.CITY + ', ' + V.[STATE] + '  '+ V.ZIPCODE
        ELSE V.ADDR_3 + ' ' + V.COUNTRY + ' ' + V.ZIPCODE
    END AS VENDOR_CTSZ
    , PR.CONTACT_PHONE AS VENDOR_PHONE
    , PR.CONTACT_FAX AS VENDOR_FAX
    , UDF5.STRING_VAL AS VENDOR_QCNOTES
--> SHIP TO INFORMATION
    , STA.[NAME] AS SHIPTO_NAME
    , STA.ADDR_1 AS SHIPTO_ADDR_1
    , STA.CITY AS SHIPTO_CITY
    , STA.[STATE] AS SHIPTO_STATE
    , STA.ZIPCODE AS SHIPTO_ZIPCODE
    , STA.COUNTRY AS SHIPTO_COUNTRY
--> PO LINE INFORMATION
    , PRL.PART_ID
    , P.[DESCRIPTION] AS PART_DESCRIPTION
    , PRL.VENDOR_PART_ID
    , PRL.USER_ORDER_QTY
    , PRL.PURCHASE_UM
    , PRL.UNIT_PRICE 
    , ISNULL(PRL.DESIRED_RECV_DATE, PR.DESIRED_RECV_DATE) AS PRL_DESIRED_RECV_DATE
    , PRL.USER_ORDER_QTY * PRL.UNIT_PRICE AS TOTAL_LINE_AMT
    , PRL.TOTAL_AMT_ORDERED
    , PRL.MFG_NAME
    , PRL.MFG_PART_ID
    , PRL.USER_1 AS PRL_USER_1
    , PRL.USER_2 AS PRL_USER_2
    , PRL.GL_EXPENSE_ACCT_ID
    , CASE
        WHEN PRL.USER_2 = 'Y' THEN CASE
                WHEN PRL.TOTAL_AMT_ORDERED IS NULL THEN 0
                WHEN PRL.TOTAL_AMT_ORDERED = 0 THEN PRL.USER_ORDER_QTY * PRL.UNIT_PRICE * 0.1
                ELSE PRL.TOTAL_AMT_ORDERED * .1
            END
        ELSE 0
    END AS PRL_TAX
    , UDF6.STRING_VAL AS PART_QCNOTES
--> PO SPECIFICATIONS; REMOVE DUPLICATION BETWEEN VENDOR SPECS AND PO SPECS
    , CASE
        WHEN REPLACE(REPLACE(CAST(CAST(PRB.BITS AS VARBINARY(MAX)) AS NVARCHAR(MAX)), CHAR(13), ''), CHAR(10), '') = REPLACE(REPLACE(CAST(CAST(VB.BITS AS VARBINARY(MAX)) AS NVARCHAR(MAX)), CHAR(13), ''), CHAR(10), '') THEN NULL
        ELSE REPLACE(REPLACE(CAST(CAST(PRB.BITS AS VARBINARY(MAX)) AS NVARCHAR(MAX)), CHAR(13), ''), CHAR(10), '')
    END AS PO_LONG_DESCRIPTION
--> VENDOR SPECIFICATIONS; REMOVE DUPLICATION BETWEEN VENDOR SPECS AND VENDOR QCNOTES
    , CASE
        WHEN REPLACE(REPLACE(CAST(CAST(VB.BITS AS VARBINARY(MAX)) AS NVARCHAR(MAX)), CHAR(13), ''), CHAR(10), '') = UDF5.STRING_VAL THEN NULL
        ELSE REPLACE(REPLACE(CAST(CAST(VB.BITS AS VARBINARY(MAX)) AS NVARCHAR(MAX)), CHAR(13), ''), CHAR(10), '') 
    END AS SPECTAB_VERBIAGE
--> PRL SPECIFICATIONS; REMOVE DUPLICATION BETWEEN LINE SPECS AND PART SPECS -----------------------  look here if you get duplicate line specs
    , CASE
        WHEN REPLACE(REPLACE(CAST(CAST(PRLB.BITS AS VARBINARY(MAX)) AS NVARCHAR(MAX)), CHAR(13), ''), CHAR(10), '') = REPLACE(REPLACE(CAST(CAST(PAB.BITS AS VARBINARY(MAX)) AS NVARCHAR(MAX)), CHAR(13), ''), CHAR(10), '') THEN NULL
        ELSE REPLACE(REPLACE(CAST(CAST(PRLB.BITS AS VARBINARY(MAX)) AS NVARCHAR(MAX)), CHAR(13), ''), CHAR(10), '')
    END AS PRL_LONG_DESCRIPTION
, REPLACE(REPLACE(CAST(CAST(PAB.BITS AS VARBINARY(MAX)) AS NVARCHAR(MAX)), CHAR(13), ''), CHAR(10), '') AS PAB_LONG_DESCRIPTION

, UDF.BOOL_VAL AS ITAR_PART
    , CASE
        WHEN UDF.BOOL_VAL = 1 AND V.USER_1 = 'Y' AND ISNULL(V.USER_4, '') != '' THEN 'ITAR'
        ELSE NULL
    END AS ISITAR
    , CASE 
        WHEN UDF2.STRING_VAL = 'CONTROLLED' THEN 'Product is a Controlled Part'
        WHEN UDF2.STRING_VAL = 'QPL CONTROLLED' THEN 'Product must be from a QPL-approved manufacturer'
        ELSE NULL
    END AS CONTROLLED_TEXT
    , UDF3.STRING_VAL AS CERT_REQ
    , UDF4.STRING_VAL AS USED_ON
--> MESSAGES
    , CASE
        WHEN PRL.VENDOR_PART_ID LIKE '%EXP%' THEN 'EXPEDITE'
        ELSE NULL
    END AS EXPEDITE
    , CASE
        WHEN V.ACTIVE_FLAG = 'Y' AND ISNULL(V.USER_1, '') = '' AND ISNULL(V.USER_2, '') = '' THEN NULL
        WHEN V.USER_1 = 'Y' AND ISNULL(V.USER_2, '') != '' THEN V.USER_2
        ELSE 'Not Approved Vendor'
    END AS SUPPLIER_APPROVAL
    , CASE
        WHEN V.ACTIVE_FLAG = 'Y' AND ISNULL(V.USER_1, '') = '' AND ISNULL(V.USER_2, '') = '' THEN NULL
        WHEN V.USER_1 = 'Y' AND ISNULL(V.USER_4, '') != '' THEN V.USER_4
        ELSE 'Not ITAR Approved'
    END AS ITAR_APPROVAL
    , CASE
        WHEN V.ACTIVE_FLAG = 'Y' AND ISNULL(V.USER_1, '') = '' AND ISNULL(V.USER_2, '') = '' THEN NULL
        WHEN V.USER_1 = 'Y' AND ISNULL(V.USER_5, '') != '' THEN V.USER_5
        ELSE 'Not DPD Approved'
    END AS DPD_APPROVAL
    , HasPrtLimited = ISNULL(HasPrtLimited.Bool_Val, 0) 
    , HasVndLimited = ISNULL(HasVndLimited.Bool_Val, 0)
    , IsFlaggedLimited = IIF(ISNULL(HasVndLimited.Bool_Val, 0) = 1 AND ISNULL(HasPrtLimited.Bool_Val, 0) = 0, 1, 0) 
    , CASE
        WHEN UDF2.STRING_VAL IN ('CONTROLLED', 'QPL CONTROLLED') AND ISNULL(V.USER_2, '') = '' THEN 'Controlled Products must be from an approved manufacturer'
        WHEN P.FABRICATED = 'Y' AND ISNULL(V.USER_2, '') = '' THEN 'Fabricated Products must be from an approved manufacturer'
        WHEN V.USER_2 = 'Disapproved' THEN 'Check Vendor ASL information'
        WHEN ISNULL(HasVndLimited.Bool_Val, 0) = 1 AND ISNULL(HasPrtLimited.Bool_Val, 0) = 0 THEN 'Not a Limited Scope Part'
        WHEN UDF.BOOL_VAL = 1 AND ISNULL(V.USER_4, '') = '' THEN 'Not ITAR Approved'
        WHEN V.ACTIVE_FLAG = 'Y' AND V.USER_1 = 'Y' AND V.USER_2 = 'Approved' AND ISNULL(CAST(V.USER_4 AS DATE), GETDATE() + 1) <= GETDATE() THEN 'Check Vendor ASL information'
        WHEN V.ACTIVE_FLAG = 'Y' AND V.USER_1 = 'Y' AND V.USER_2 = 'Approved' AND ISNULL(CAST(V.USER_3 AS DATE), GETDATE() + 1) <= GETDATE() THEN 'Check Vendor ASL information'
        WHEN V.ACTIVE_FLAG = 'Y' AND V.USER_1 = 'Y' AND V.USER_2 = 'Approved' AND ISNULL(CAST(V.USER_4 AS DATE), GETDATE() + 1) <= GETDATE() THEN 'Check Vendor ASL information'
        WHEN V.ACTIVE_FLAG = 'Y' AND V.USER_1 = 'Y' AND V.USER_2 = 'Approved' AND ISNULL(CAST(V.USER_5 AS DATE), GETDATE() + 1) <= GETDATE() THEN 'Check Vendor ASL information'
        WHEN V.USER_2 = 'Pending' THEN NULL
        ELSE NULL
    END AS PRINTMESSAGE
    , SP.SPECLIST
FROM PURC_REQUISITION PR
LEFT OUTER JOIN PURC_REQ_LINE PRL
    ON PR.ID = PRL.PURC_REQ_ID
LEFT OUTER JOIN PART P
    ON PRL.PART_ID = P.ID
LEFT OUTER JOIN EMPLOYEE EE
    ON PR.REQUESTOR = EE.ID
LEFT OUTER JOIN EMPLOYEE PRE
    ON PR.PREPARER = PRE.[USER_ID]
LEFT OUTER JOIN EMPLOYEE B
    ON PR.BUYER = B.[USER_ID]
LEFT OUTER JOIN VENDOR V
    ON PR.VENDOR_ID = V.ID
LEFT OUTER JOIN TERMS T
    ON T.ID = V.DEF_TERMS_ID
LEFT OUTER JOIN SHIPTO_ADDRESS STA
    ON STA.ADDR_NO = PR.SHIPTO_ADDR_NO
--> DOCUMENTS!!!
LEFT OUTER JOIN #SPECS SP
    ON SP.REQUISITION_ID = PR.ID
    AND SP.REQUISITION_LINE = PRL.LINE_NO
--> ITAR INFO
LEFT OUTER JOIN USER_DEF_FIELDS UDF
    ON UDF.DOCUMENT_ID = PRL.PART_ID
    AND UDF.PROGRAM_ID = 'VMPRTMNT'
    AND UDF.ID = 'UDF-0000040'
    AND BOOL_VAL = 1
--> CONTROLLED INFO
LEFT OUTER JOIN USER_DEF_FIELDS UDF2
    ON UDF2.DOCUMENT_ID = PRL.PART_ID
    AND UDF2.PROGRAM_ID = 'VMPRTMNT'
    AND UDF2.ID = 'UDF-0000082'
--> CERT INFO
LEFT OUTER JOIN USER_DEF_FIELDS UDF3
    ON UDF3.DOCUMENT_ID = PRL.PART_ID
    AND UDF3.PROGRAM_ID = 'VMPRTMNT'
    AND UDF3.ID = 'UDF-0000084'
--> USED ON INFO
LEFT OUTER JOIN USER_DEF_FIELDS UDF4
    ON UDF4.DOCUMENT_ID = PRL.PART_ID
    AND UDF4.PROGRAM_ID = 'VMPRTMNT'
    AND UDF4.ID = 'UDF-0000085'
--> limited Scope; VMPRTMNT; HasPrtLimited
LEFT OUTER JOIN dbo.USER_DEF_FIELDS HasPrtLimited (nolock)
    ON HasPrtLimited.Document_ID = PRL.PART_ID
    AND HasPrtLimited.program_id = 'VMPRTMNT'
    AND HasPrtLimited.DOCUMENT_ID is NOT null
    AND ISNULL(HasPrtLimited.BOOL_VAL, 0) = 1
    AND HasPrtLimited.ID = ('UDF-0000091') 
--> limited Scope; VMVNDMNT; HasVndLimited
LEFT OUTER JOIN [dbo].[USER_DEF_FIELDS] HasVndLimited (nolock)
    ON HasVndLimited.Document_ID = PR.VENDOR_ID
    AND HasVndLimited.program_id = 'VMVNDMNT'
    AND HasVndLimited.DOCUMENT_ID is NOT null
    AND HasVndLimited.ID ='UDF-0000090'  
    AND HasVndLimited.BOOL_VAL = 1
--> limited Scope; VMPURENT; PrePaidCC
LEFT OUTER JOIN [dbo].[USER_DEF_FIELDS] PrePaidCC (nolock)
    ON PrePaidCC.Document_ID = PR.ID
    AND PrePaidCC.program_id = 'VMPURENT'
    AND PrePaidCC.DOCUMENT_ID is NOT null
    AND PrePaidCC.ID ='UDF-0000100'  
--> tt20210512 : UDF VENDOR QC NOTES
LEFT OUTER JOIN USER_DEF_FIELDS UDF5
    ON UDF5.DOCUMENT_ID = PR.VENDOR_ID
    AND UDF5.PROGRAM_ID = 'VMVNDMNT'
    AND UDF5.ID = 'UDF-0000102'
--> tt20210512 : UDF PART QC NOTES
LEFT OUTER JOIN USER_DEF_FIELDS UDF6
    ON UDF6.DOCUMENT_ID = PRL.PART_ID
    AND UDF6.PROGRAM_ID = 'VMPRTMNT'
    AND UDF6.ID = 'UDF-0000101'
LEFT OUTER JOIN PURC_REQ_BINARY PRB
    ON PR.ID = PRB.PURC_REQ_ID
    AND PRB.[TYPE] = 'D'
LEFT OUTER JOIN VENDOR_BINARY VB
    ON V.ID = VB.VENDOR_ID
    AND VB.[TYPE] = 'D'
LEFT OUTER JOIN PURC_REQ_LN_BINARY PRLB
    ON PRL.PURC_REQ_ID = PRLB.PURC_REQ_ID
    AND PRL.LINE_NO = PRLB.PURC_REQ_LINE_NO
    AND PRLB.[TYPE] = 'D'
LEFT OUTER JOIN PART_BINARY PAB
    ON PRL.PART_ID = PAB.PART_ID
    AND PAB.[TYPE] = 'D'
WHERE PR.ID = @REQ_ID;
