# Inventory Value (Latest Transaction)

```sql
--> Visual Enterprise Reports > Administrative Reports > Inventory Value (Latest Transaction)

SELECT P.ID
    , P.QTY_ON_HAND 
    , P.COMMODITY_CODE
    , IT.ACT_MATERIAL_COST
    , IT.ACT_LABOR_COST
    , IT.ACT_BURDEN_COST
    , IT.ACT_SERVICE_COST
    , IT.ACT_MATERIAL_COST + IT.ACT_LABOR_COST + IT.ACT_BURDEN_COST + IT.ACT_SERVICE_COST AS LAST_TRANSACTION_VALUE
    , IT.QTY
    , ((IT.ACT_MATERIAL_COST + IT.ACT_LABOR_COST + IT.ACT_BURDEN_COST + IT.ACT_SERVICE_COST) / IT.QTY) * P.QTY_ON_HAND AS TOTAL_VALUE
    , IT.TRANSACTION_DATE
FROM PART P 
INNER JOIN INVENTORY_TRANS IT
    ON P.ID = IT.PART_ID
INNER JOIN (SELECT PART_ID, MAX(TRANSACTION_ID) AS TRANSACTION_ID
    FROM INVENTORY_TRANS IT2
    WHERE IT2.QTY <> 0 
        AND IT2.[TYPE] IN ('I', 'O') 
        AND IT2.CLASS != 'A'
    GROUP BY IT2.PART_ID
    ) IT2
    ON IT.TRANSACTION_ID = IT2.TRANSACTION_ID
    AND IT.PART_ID = IT2.PART_ID
WHERE P.QTY_ON_HAND > 0 
ORDER BY IT.PART_ID
    , IT.TRANSACTION_DATE DESC

```
