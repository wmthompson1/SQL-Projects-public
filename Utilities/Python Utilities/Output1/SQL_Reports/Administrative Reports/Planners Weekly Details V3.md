Select A.* ,  P.[LOCATION]
    , P.LINE_OF_BUSINESS
    , P.EMPID
    , P.[NAME]    from 
(SELECT WO.[TYPE]
    , WO.BASE_ID
    , WO.LOT_ID
    , WO.SPLIT_ID
    , WO.SUB_ID
    , WO.[STATUS]
    , CAST(mAX(o.CLOSE_DATE)as  DATE) AS CREATE_DATE
    , WO.ENGINEERED_BY
    ,(select max (employee_id)  ) as eMploYEe
    , dbo.ufnINITIALS(WO.ENGINEERED_BY) AS INITIALS
    
    ,isnull( CO.CUSTOMER_ID,'NO CUSTOMER') CUSTOMER_ID
FROM WORK_ORDER WO  inner join OPERATION O 
ON O.WORKORDER_BASE_ID = WO.BASE_ID
    AND O.WORKORDER_LOT_ID= WO.LOT_ID
    AND O.WORKORDER_SPLIT_ID = WO.SPLIT_ID
    inner join LABOR_TICKET L On L.WORKORDER_BASE_ID=WO.BASE_ID
    and L.WORKORDER_LOT_ID=WO.LOT_ID
    and L.WORKORDER_SPLIT_ID= WO.SPLIT_ID
    and L.WORKORDER_SUB_ID= WO.SUB_ID
    and L.WORKORDER_TYPE=WO.TYPE
    and L.OPERATION_SEQ_NO= O.SEQUENCE_NO
    LEFT OUTER JOIN DEMAND_SUPPLY_LINK DSL
    ON DSL.SUPPLY_BASE_ID = WO.BASE_ID
    AND DSL.SUPPLY_LOT_ID = WO.LOT_ID
    AND DSL.SUPPLY_SPLIT_ID = WO.SPLIT_ID
--    AND DSL.SUPPLY_SUB_ID = WO.SUB_ID 
    AND DSL.SUPPLY_TYPE = 'WO'
LEFT OUTER JOIN CUSTOMER_ORDER CO
    ON DSL.DEMAND_BASE_ID = CO.ID
    AND DSL.DEMAND_TYPE = 'CO'

WHERE 
     convert(date,L.TRANSACTION_DATE,103) BETWEEN  convert(date,@STARTDATE,103) and convert(date,@ENDDATE,103)
    AND WO.[TYPE] IN (@WOTYPE)
    AND WO.[STATUS] IN (@WOSTATUS)
    AND WO.SUB_ID = '0'  -- eliminate non 0 legs as known issue for updating
    AND isnull( CO.CUSTOMER_ID,'NO CUSTOMER') IN (@CUST_ID)
    AND O.opeRAtioN_tYPe  in  ('AREVIEW','AREVIEW2','REVIEW','REVIEW2','REVIEW-FAI','CM-REVIEW','CM-REVIEW1','CMPFL-REVIEW')  and O.status='C'
    Group by 
     WO.[TYPE]
    , WO.BASE_ID
    , WO.LOT_ID
    , WO.SPLIT_ID
    , WO.SUB_ID
    , WO.[STATUS]
    ,o.SEQUENCE_NO
    , CAST(WO.CREATE_DATE AS DATE)
    , WO.ENGINEERED_BY
    , CO.CUSTOMER_ID) A LEFT OUTER JOIN LIVESupplemental.dbo.PlannersList P
    ON a.employee = P.EMPID
