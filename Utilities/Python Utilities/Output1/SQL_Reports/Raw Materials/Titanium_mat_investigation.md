```sql
SELECT c.ID, c.NAME as 'VENDOR', b.ID AS 'PO', a.LINE_NO as 'PO_LN', a.PART_ID, a.ORDER_QTY, 
TOTAL_RECEIVED_QTY as 'RECEIVED_QTY', a.LAST_RECEIVED_DATE AS 'RECEIVED_DATE'
FROM PURC_ORDER_LINE a WITH (NOLOCK) 
INNER JOIN PURCHASE_ORDER b WITH (NOLOCK) ON a.PURC_ORDER_ID = b.ID
INNER JOIN VENDOR c  WITH (NOLOCK) on b.VENDOR_ID = c.ID
WHERE a.PART_ID in (@PartID)
    AND b.ID in (@PO)
    AND b.VENDOR_ID in (@Vendor)
    /* Exclude any POs where the vendor is TMXDIV */
ORDER BY c.NAME, b.ID, a.LINE_NO asc
```
