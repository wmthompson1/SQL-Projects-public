# Boeing Unprinted Work Orders

```sql
-- VISUAL ENTERPRISE > CUSTOMER SERVICE REPORTS > BOEING REPORTS > BOEING UNPRINTED WORK ORDERS
SET TRANSACTION ISOLATION LEVEL READ UNCOMMITTED
SET DEADLOCK_PRIORITY LOW

/********************************
temp TABLES FOR UDF fields
*/
IF OBJECT_ID('tempdb..#FAI_COMPLETE') IS NOT NULL DROP TABLE #FAI_COMPLETE
IF OBJECT_ID('tempdb..#FAI_INITIALS') IS NOT NULL DROP TABLE #FAI_INITIALS
IF OBJECT_ID('tempdb..#FAI_DATE') IS NOT NULL DROP TABLE #FAI_DATE

SELECT * 
INTO #FAI_COMPLETE
FROM [SQL-LAB-2].[LIVE].dbo.USER_DEF_FIELDS UDF
WHERE PROGRAM_ID = 'VMMFGWIN_WO'
    AND UDF.ID = 'UDF-0000086'
    AND LABEL IS NULL
--SELECT * FROM #FAI_COMPLETE

SELECT * 
INTO #FAI_INITIALS
FROM [SQL-LAB-2].[LIVE].dbo.USER_DEF_FIELDS UDF
WHERE PROGRAM_ID = 'VMMFGWIN_WO'
    AND UDF.ID = 'UDF-0000087'
    AND LABEL IS NULL
--SELECT * FROM #FAI_INITIALS

SELECT * 
INTO #FAI_DATE
FROM [SQL-LAB-2].[LIVE].dbo.USER_DEF_FIELDS UDF
WHERE PROGRAM_ID = 'VMMFGWIN_WO'
    AND UDF.ID = 'UDF-0000088'
     AND LABEL IS NULL
--SELECT * FROM #FAI_DATE


SELECT CO.CUSTOMER_ID
    , WO.[TYPE] --NOT SHOWN ON REPORT
    , WO.BASE_ID
    , COL.CUST_ORDER_ID AS SFC
    , WO.LOT_ID  --NOT SHOWN ON REPORT
    , WO.SPLIT_ID --NOT SHOWN ON REPORT
    , WO.SUB_ID  --NOT SHOWN ON REPORT
    , O.sequence_no
    , O.expr1 AS RESOURCE_ID
    , WO.PART_ID
    , P.[DESCRIPTION]
    , WO.DESIRED_QTY
    , WO.[STATUS]
    , WO.CREATE_DATE
    , WO.DESIRED_WANT_DATE
    , WO.PRINTED_DATE
    , CASE 
        WHEN WO.PRINTED_DATE IS NULL THEN 'NO_PRINT_DATE'
        ELSE 'HAS_PRINT_DATE'
    END AS PRINTDATEFILTER
    , P.PLANNER_USER_ID
    , CASE 
        WHEN COL.USER_10 LIKE '%ETRAC%' OR COL.USER_10 LIKE '%IMPAC%' THEN COL.USER_10 
        ELSE NULL
    END AS USER_10
    , CASE 
        WHEN COL.USER_10 LIKE '%ETRAC%' OR COL.USER_10 LIKE '%IMPAC%'  THEN 'COLORLINE'
        ELSE 'TRANSPARENT'
    END AS LINECOLOR
    , '' AS COMMENTS
    , WO.DESIRED_RLS_DATE --NOT SHOWN ON REPORT
    , WO.ACT_MATERIAL_COST --NOT SHOWN ON REPORT
    , WO.ACT_LABOR_COST --NOT SHOWN ON REPORT
    , WO.ACT_SERVICE_COST  --NOT SHOWN ON REPORT
    , CASE 
        WHEN WO.PRINTED_DATE IS NULL THEN 'NO_PRINT_DATE'
        ELSE 'HAS_PRINT_DATE'
    END AS PRINTSTATUS
    , (SELECT BOOL_VAL FROM #FAI_COMPLETE UDF
        WHERE 'M'+'~'+ WO.PART_ID +'~0~0~0' = UDF.DOCUMENT_ID
       ) AS IS_FAI_COMPLETE
    , (SELECT STRING_VAL FROM #FAI_INITIALS UDF
       WHERE 'M'+'~'+ WO.PART_ID +'~0~0~0' = UDF.DOCUMENT_ID) AS QEA_EMPID
    , (SELECT DATE_VAL FROM #FAI_DATE UDF
        WHERE 'M'+'~'+ WO.PART_ID +'~0~0~0' = UDF.DOCUMENT_ID) AS FAI_APPV_DATE  
FROM WORK_ORDER WO
INNER JOIN [dbo].[WO_CURRENTLY_AT] O 
ON WO.BASE_ID = O.WORKORDER_BASE_ID
    AND WO.LOT_ID = O.WORKORDER_LOT_ID
    AND WO.SPLIT_ID = O.WORKORDER_SPLIT_ID
    AND WO.SUB_ID = O.WORKORDER_SUB_ID
    AND WO.[STATUS] IN ('F', 'U')
    AND WO.[TYPE] = 'W'
INNER JOIN DEMAND_SUPPLY_LINK DSL 
    ON WO.BASE_ID = DSL.SUPPLY_BASE_ID
    AND WO.LOT_ID = DSL.SUPPLY_LOT_ID
    AND WO.SPLIT_ID = DSL.SUPPLY_SPLIT_ID
    AND WO.SUB_ID = DSL.SUPPLY_SUB_ID
    AND WO.PART_ID = DSL.SUPPLY_PART_ID
    AND WO.[TYPE] = 'W'
    AND WO.[STATUS] IN ('F', 'U')
    AND WO.ACT_MATERIAL_COST = 0
    AND WO.ACT_SERVICE_COST = 0
INNER JOIN CUST_ORDER_LINE COL
    ON COL.CUST_ORDER_ID = DSL.DEMAND_BASE_ID 
    AND COL.LINE_NO = DSL.DEMAND_SEQ_NO
INNER JOIN CUSTOMER_ORDER CO
    ON CO.ID = COL.CUST_ORDER_ID
    AND CO.[STATUS] = 'R'
    AND COL.ORDER_QTY - COL.TOTAL_SHIPPED_QTY > 0
    AND COL.PART_ID IS NOT NULL -- AND COL.USER_10 IS NOT NULL
    AND CO.CUSTOMER_ID LIKE 'BOE%'
INNER JOIN PART P
    ON WO.PART_ID = P.ID
    AND EXISTS
        (SELECT 1
        FROM LIVESupplemental.dbo.PLANNER_USER_ID_ReportSettings EX
        WHERE P.PLANNER_USER_ID = EX.PLANNER_USER_ID
        AND REPORT_NAME = 'BOEING UNPRINTED WORK ORDERS'
        )

```
