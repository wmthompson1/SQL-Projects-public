```sql
SELECT NCM_ID
    , NCM.RECORD_ID
    , NCM.[ENTITY_ID]
    , NCM.SITE_ID
-- INFORMATION
    , PO.DISPLAY_ID AS PRODUCTION_ORDER_ID
    , NCM.PO_NUMBER AS VENDOR_PO_NUM
    , O.DISPLAY_ID AS OPERATION_ID
    , O.NEXT_ID AS [NEXT_OPERATION_ID]
    , O.CONST_RESOURCE_ID
    , NCM.REFERENCE_5 AS [CUSTOMER_REFERENCE]
    , NCM.MADE_BY_EMP_REC_ID
    , MADE_EE.EMPLOYEE_ID AS MADE_BY_EEID
    , MADE_EE.FIRST_NAME AS MADE_BY_EE_FNAME
    , MADE_EE.LAST_NAME AS MADE_BY_EE_LNAME
    , NCM.ASSIGNED_TO_REC_ID
    , ASSGN_EE.EMPLOYEE_ID AS ASSGN_TO_EEID
    , ASSGN_EE.FIRST_NAME AS ASSGN_TO_EE_FNAME
    , ASSGN_EE.LAST_NAME AS ASSGN_TO_EE_LNAME
    , NCM.LOT_NUMBER AS CERT_NUM
    , NCM.LOT_SIZE
    , NCM.ACCEPTED
    , NCM.REJECTED
-- HEADER
    , NCM.NCM_TYPE
    , NCM.NCM_DATE
    , NCM.[STATUS]
    , NCM.CLOSED_DATE
    , NCM.DUE_DATE
    , NCM.BUSINESS_UNIT_ID
    , P.PRODUCT_ID
    , P.[NAME] AS PROD_ID_NAME
-- DESCRIPTION
    , NCM.REFERENCE_1 AS [SPEC_SECT]
    , NCM.REFERENCE_2 AS [DRAW_REF]
    , NCM.SK_AS_IS AS [ASIS_DIM]
    , NCM.SK_SHOULD_BE AS [SHOULD_BE_DIM]
    , NCM.[DESCRIPTION] AS NCM_DESC
-- MFG/FIN
    , ISNULL(NCM.USER_DEFINED_1, NCM.USER_DEFINED_5) AS [PROCESS]
    , ISNULL(NCM.USER_DEFINED_2, NCM.USER_DEFINED_6) AS [SUBPROCESS]
    , NCM.USER_DEFINED_3 AS [CATEGORY]
    , ISNULL(NCM.USER_DEFINED_4, NCM.USER_DEFINED_7) AS [DEFECT]
-- DISPOSITION
    , NCM.DISPOSITION_TYPE
    , DT.[DESCRIPTION] AS DISP_DESC
    , NCM.DISP_DATE
    , NCM.DISP_DUE_DATE
    , NCM.DISP_STATUS
    , NCM.DISP_CLOSED_DATE
    , NCM.DISP_ASSG_TO_REC_ID
    , DISP_ASSG.EMPLOYEE_ID AS DISP_ASSN_BY_EEID
    , DISP_ASSG.FIRST_NAME AS DISP_ASSN_BY_EE_FNAME
    , DISP_ASSG.LAST_NAME AS DISP_ASSN_BY_EE_LNAME
    , NCM.DISP_EMP_REC_ID
    , DISP_EE.EMPLOYEE_ID AS DISP_BY_EEID
    , DISP_EE.FIRST_NAME AS DISP_BY_EE_FNAME
    , DISP_EE.LAST_NAME AS DISP_BY_EE_LNAME
    , NCM.REFERENCE_6 AS [REWORK_QTY]
    , NCM.REFERENCE_7 AS [SCRAP_QTY]
    , NCM.REFERENCE_8 AS [OVERAGE_QTY]
    , NCM.REFERENCE_9 AS [RETURN_QTY]
    , NCM.REFERENCE_10 AS [ASIS_QTY]
    , NCM.DISPOSITION
    , SK_MFG_NCM_PROCESS
    , SK_MFG_NCM_SUBPROCESS
    , SK_FIN_NCM_PROCESS
    , SK_FIN_NCM_SUBPROCESS
    , SK_ROOT_CAUSE_ANALYSIS AS CORRECTION
    , SK_ROOT_CAUSE AS CAUSE
FROM VQ_NONCONFORMANCE NCM
LEFT OUTER JOIN VQ_PRODUCT P
    ON P.RECORD_ID = NCM.PRODUCT_REC_ID
LEFT OUTER JOIN VQ_PRODUCTION_ORDER PO
    ON PO.RECORD_ID = NCM.PROD_ORDER_REC_ID
LEFT OUTER JOIN VQ_OPERATION O
    ON O.RECORD_ID = NCM.OPERATION_REC_ID
LEFT OUTER JOIN VQ_DISPOSITION_TYPE DT
    ON DT.[DISPOSITION_TYPE] = NCM.[DISPOSITION_TYPE]
LEFT OUTER JOIN VQ_EMPLOYEE MADE_EE
    ON MADE_EE.RECORD_ID = NCM.MADE_BY_EMP_REC_ID
LEFT OUTER JOIN VQ_EMPLOYEE ASSGN_EE
    ON ASSGN_EE.RECORD_ID = NCM.ASSIGNED_TO_REC_ID
LEFT OUTER JOIN VQ_EMPLOYEE DISP_EE
    ON DISP_EE.RECORD_ID = NCM.DISP_EMP_REC_ID
LEFT OUTER JOIN VQ_EMPLOYEE DISP_ASSG
    ON DISP_ASSG.RECORD_ID = NCM.DISP_ASSG_TO_REC_ID
WHERE NCM.NCM_ID = @NCM_ID
    AND NCM.SITE_ID = @SITE_ID
```
