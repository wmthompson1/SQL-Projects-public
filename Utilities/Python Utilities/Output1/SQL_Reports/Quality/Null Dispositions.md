# Null Dispositions

```sql
--HOME > QUALITY > NULL DISPOSITIONS
SELECT NCM_ID
    , PO.DISPLAY_ID AS PRODUCTION_ORDER_ID
    , NCM.PO_NUMBER AS VENDOR_PO_NUM
    , O.DISPLAY_ID AS OPERATION_ID
    , NCM.NCM_TYPE
    , NCM.[STATUS]
    , NCM.CLOSED_DATE
    , P.PRODUCT_ID
    , P.[NAME] AS PROD_ID_NAME
    , NCM.REFERENCE_1 AS [SPEC_SECT]
    , NCM.[DESCRIPTION] AS NCM_DESC
    , ISNULL(NCM.USER_DEFINED_4, NCM.USER_DEFINED_7) AS [DEFECT]
    , NCM.DISPOSITION_TYPE
    , NCM.DISPOSITION
    , DT.[DESCRIPTION] AS DISP_DESC
    , NCM.DISP_DATE
    , NCM.DISP_DUE_DATE
    , NCM.DISP_STATUS
    , NCM.DISP_CLOSED_DATE
FROM DataWarehouse.dbo.VQ_NONCONFORMANCE NCM
LEFT OUTER JOIN DataWarehouse.dbo.VQ_PRODUCT P
    ON P.RECORD_ID = NCM.PRODUCT_REC_ID
LEFT OUTER JOIN DataWarehouse.dbo.VQ_PRODUCTION_ORDER PO
    ON PO.RECORD_ID = NCM.PROD_ORDER_REC_ID
LEFT OUTER JOIN DataWarehouse.dbo.VQ_OPERATION O
    ON O.RECORD_ID = NCM.OPERATION_REC_ID
LEFT OUTER JOIN DataWarehouse.dbo.VQ_DISPOSITION_TYPE DT
    ON DT.[DISPOSITION_TYPE] = NCM.[DISPOSITION_TYPE]
WHERE NCM.DISP_STATUS = 'Closed'
    AND NCM.DISPOSITION_TYPE IS NULL
    AND NCM.DISPOSITION IS NULL
    AND NCM.SK_ROOT_CAUSE_ANALYSIS IS NULL
    AND NCM.SK_ROOT_CAUSE IS NULL

```
