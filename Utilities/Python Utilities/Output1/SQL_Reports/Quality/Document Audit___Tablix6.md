# Document Audit___Tablix6

```sql
--declare @document_audit_id as nvarchar(16)
--set @document_audit_id = '0000004'

SELECT DA.SK_AUDIT_MANAGER_ID,
       DA.DOCUMENT_AUDIT_ID,
      CASE WHEN DA.AUDITED_BY = 'I' 
          THEN 'INTERNAL'
          WHEN DA.AUDITED_BY = 'C' 
          THEN 'CUSTOMER'
          WHEN DA.AUDITED_BY = 'S' 
          THEN 'SUPPLIER'
          ELSE 'NO SELECTION' 
       END AS AUDITED_BY,
      CASE WHEN DA.RESULT = 'P'
          THEN 'Passed'
          WHEN DA.RESULT ='N'
          THEN 'Pending'
          WHEN DA.RESULT = 'F'
          THEN 'Failed'
          ELSE 'STATUS'
          END AS RESULT,
      DA.REVISION_LEVEL,
      DA.ISSUE_DATE,
      DA.NEXT_AUDIT_DATE,
      DA.DUE_DATE,
      DA.COMPLETION_DATE,
      DA.BUSINESS_UNIT_ID,
      DA.COMMENTS,
      DA.CREATEDBY,
      DA.CREATEDATE,
      DS.STANDARD_INTERVAL, 
      DS.TIME_ESTIMATE,
      DS.RETENTION, 
      DS.USER_DEFINED_1,
      DS.BUSINESS_UNIT_ID ASCDS_DOC_SITE, 
      E.NAME_1 AS INTERNALAUDITORNAME,
      E.EMPLOYEE_ID AS INTERNALAUDITOREMPID,
      D.DOCUMENT,
      D.DOCUMENT_ID,
      D.NAME AS DOC_NAME,
      D.STATUS,
      D.STATUS_EFFECTIVE_DATE,
      D.REVISION_LEVEL DOC_REVISION_LEVEL,
      AF.FINDING_ID,
      AF.FINDING_TEXT,
      AF.FINDING_TYPE,
      AF.FOLLOWUP_TEXT,
      AF.CLOSED_DATE

--, d.*

FROM   VQ_DOCUMENT_AUDIT DA
      LEFT JOIN VQ_EMPLOYEE E
        ON DA.EMPLOYEE_REC_ID = E.RECORD_ID
      LEFT JOIN VQ_DOCUMENT_SITE DS
        ON DS.RECORD_ID = DA.DOCUMENT_SITE_REC_ID
           AND DS.SITE_ID = DA.SITE_ID
           AND DS.RECORD_ID = DA.DOCUMENT_SITE_REC_ID
      LEFT JOIN VQ_DOCUMENT D
        ON DS.DOCUMENT_REC_ID = D.RECORD_ID
      LEFT JOIN VQ_AUDIT_FINDINGS AF
        ON AF.SK_DOC_AUDIT_ID = DA.DOCUMENT_AUDIT_ID
WHERE  DA.DOCUMENT_AUDIT_ID = @document_audit_id
```
