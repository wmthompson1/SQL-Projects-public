# Audit Manager___Tablix1

```sql
SELECT am.SITE_ID, am.ENTITY_ID, am.AUDIT_MANAGER_ID, am.AUDIT_ID, am.STANDARD, am.SUBJECT, am.SCHEDULED_DATE, laE.EMPLOYEE_ID AS LeadAuditorEMPID, laE.NAME_1 AS LeadAuditorName, laE.WORK_EMAIL_ADDRESS AS LeadAuditorEmail
    , aaE1.EMPLOYEE_ID AS AsstAuditor1EMPID, aaE1.NAME_1 AS AsstAuditor1Name, aaE1.WORK_EMAIL_ADDRESS AS AsstAuditor1Email
    , aaE2.EMPLOYEE_ID AS AsstAuditor2EMPID, aaE2.NAME_1 AS AsstAuditor2Name, aaE2.WORK_EMAIL_ADDRESS AS AsstAuditor2Email
    , aaE3.EMPLOYEE_ID AS AsstAuditor3EMPID, aaE3.NAME_1 AS AsstAuditor3Name, aaE3.WORK_EMAIL_ADDRESS AS AsstAuditor3Email
    , A1E.EMPLOYEE_ID AS Auditee1EMPID, A1E.NAME_1 AS Auditee1Name, A1E.WORK_EMAIL_ADDRESS AS Auditee1Email
    , A2E.EMPLOYEE_ID AS Auditee2EMPID, A2E.NAME_1 AS Auditee2Name, A2E.WORK_EMAIL_ADDRESS AS Auditee2Email
    , am.SCORE, am.RESULT, am.COMPLETION_DATE, am.BUSINESS_UNIT_ID, am.COMMENTS, am.CreatedBy, am.UpdatedBy, am.CreateDate
    , af.FINDING_ID, af.FINDING_TYPE, af.FINDING_TEXT, af.CLOSED_DATE FindingClosedDate, af.CreatedBy FindingCreatedBy, af.CreateDate FindingCreateDate, af.FOLLOWUP_TEXT
    , afE.EMPLOYEE_ID AS FindingFollowUpEMPID, afE.NAME_1 AS FindingFollowUpName, afE.WORK_EMAIL_ADDRESS AS FindingFollowUpEmail
    --, af.*
    --, *
FROM VQ_AUDIT_MANAGER am
    LEFT JOIN VQ_EMPLOYEE laE
        ON laE.RECORD_ID = am.LEAD_AUDITOR_REC_ID
    LEFT JOIN VQ_EMPLOYEE aaE1
        ON aaE1.RECORD_ID = am.ASST_AUDITOR1_REC_ID
    LEFT JOIN VQ_EMPLOYEE aaE2
        ON aaE2.RECORD_ID = am.ASST_AUDITOR2_REC_ID
    LEFT JOIN VQ_EMPLOYEE aaE3
        ON aaE3.RECORD_ID = am.ASST_AUDITOR3_REC_ID
    LEFT JOIN VQ_EMPLOYEE A1E
        ON A1E.RECORD_ID = am.AUDITEE1_REC_ID
    LEFT JOIN VQ_EMPLOYEE A2E
        ON A2E.RECORD_ID = am.AUDITEE2_REC_ID
    LEFT JOIN VQ_AUDIT_FINDINGS af
        ON am.RECORD_ID = af.AUDIT_REC_ID
    LEFT JOIN VQ_EMPLOYEE afE
        ON af.FOLLOWUP_BY_REC_ID = afE.RECORD_ID
WHERE am.AUDIT_MANAGER_ID = @AUDIT_MANAGER_ID
    AND am.SITE_ID = @SITE_ID
```
