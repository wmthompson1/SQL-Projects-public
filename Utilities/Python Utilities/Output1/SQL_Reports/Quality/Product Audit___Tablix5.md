```sql
--DECLARE @prod_audit_ID AS NVARCHAR(16)
----DECLARE @product_site_rec_ID AS bigint

--SET @prod_audit_ID = '0000002'
--SET @product_site_rec_ID = 8

SELECT PA.AUDIT_ID,
      PA.PRODUCT_SITE_REC_ID,
      PA.EMPLOYEE_REC_ID,
      PA.AUDIT_DATE,
       CASE WHEN PA.AUDITED_BY = 'I' 
          THEN 'INTERNAL'
          WHEN PA.AUDITED_BY = 'C' 
          THEN 'CUSTOMER'
          WHEN PA.AUDITED_BY = 'S' 
          THEN 'SUPPLIER'
          ELSE 'NO SELECTION' 
       END AS AUDITED_BY,
      PA.BUSINESS_UNIT_ID,
      PA.COMMENTS,
      PA.COMPLETION_DATE,
      PA.DOC_ISSUE_DATE,
      PA.DOC_NEXT_AUD_DATE,
     CASE WHEN PA.RESULT = 'P'
               THEN 'Passed'
               WHEN PA.RESULT ='N'
               THEN 'Pending'
               WHEN PA.RESULT = 'F'
               THEN 'Failed'
               ELSE 'STATUS'
      END AS RESULT,
      AM.AUDIT_MANAGER_ID,
      AM.STANDARD AS AMSTANDARD,
      E.NAME_1 AS AUDITEDBYEMPNAME,
      E.EMPLOYEE_ID,
      E.WORK_EMAIL_ADDRESS,
      AF.FINDING_ID,
      AF.FINDING_TEXT,
      AF.FINDING_TYPE,
      AF.FOLLOWUP_TEXT,
      AF.CREATEDATE,
      AF.CREATEDBY, 
      AF.CLOSED_DATE,
      P.PRODUCT_ID,
      P.STATUS, 
      p.STATUS_EFFECTIVE_DATE,
      P.REFERENCE,
      P.REVISION_LEVEL,
      P.COMMODITY_TYPE,
      P.BUSINESS_UNIT_ID AS PRODUCTBIZUNIT,
      P.DRAWING_NUMBER,
      P.ISSUE_DATE,
      P.LID,
      P.MAJOR_LOCATION,
      P.MINOR_LOCATION,
      P.NAME AS PRODUCTNAME, 
      PS.LAST_AUDIT_DATE, 
      PS.NEXT_AUDIT_DATE, 
      PS.STANDARD_INTERVAL, 
      PS.SPECIAL_INTERVAL
FROM   VQ_PRODUCT_AUDIT PA
      LEFT JOIN VQ_AUDIT_MANAGER AM
        ON PA.SITE_ID = AM.SITE_ID
           AND PA.SK_AUDIT_MANAGER_ID = AM.AUDIT_MANAGER_ID
      LEFT JOIN VQ_EMPLOYEE E
        ON PA.EMPLOYEE_REC_ID = E.RECORD_ID
      LEFT JOIN VQ_AUDIT_FINDINGS AF
        ON AF.SK_PROD_AUDIT_ID = PA.AUDIT_ID
      LEFT JOIN VQ_PRODUCT_SITE PS
        ON PA.PRODUCT_SITE_REC_ID = PS.RECORD_ID
      LEFT JOIN VQ_PRODUCT P
        ON P.RECORD_ID = PS.PRODUCT_REC_ID
WHERE  PA.SITE_ID = 'SK01'
      AND PA.ENTITY_ID = 'SK01'
      AND PA.AUDIT_ID = @prod_audit_id;
```
