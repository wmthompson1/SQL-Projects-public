# Department Defects

```sql
-- QUALITY > DEPARTMENT DEFECTS
SELECT NCM.NCM_ID
    , P.PRODUCT_ID
    , CO.CUSTOMER_ID
    , PO.DISPLAY_ID AS PRODUCTION_ORDER_ID
    , NCM.LOT_SIZE
    , NCM.ACCEPTED
    , NCM.REJECTED
    , NCM.NCM_DATE
    , NCM.REFERENCE_1 AS [SPEC_SECT]
    , NCM.SK_AS_IS AS [ASIS_DIM]
    , NCM.SK_SHOULD_BE AS [SHOULD_BE_DIM]
    , COALESCE(NCM.SK_FIN_NCM_PROCESS, NCM.SK_MFG_NCM_PROCESS, '') AS [PROCESS]
    , COALESCE(NCM.SK_FIN_NCM_SUBPROCESS, NCM.SK_MFG_NCM_SUBPROCESS, '') AS [SUBPROCESS]
    , ISNULL(NCM.USER_DEFINED_3, '') AS [FEATURE]
    , COALESCE(NCM.USER_DEFINED_7, NCM.USER_DEFINED_4, '') AS [DEFECT]
    , ISNULL(NCM.DISPOSITION_TYPE, '') AS DISPOSITION_TYPE
    , NCM.DISPOSITION
    , ISNULL(DISP_EE.[FIRST_NAME] + ' ' + DISP_EE.[LAST_NAME], '_Blank') AS [DISP_BY]
    , ISNULL(EE.[FIRST_NAME] + ' ' + EE.[LAST_NAME], '_Blank') AS [Assigned_to]
    , ISNULL(NCM.[STATUS], '') AS [NCM_STATUS]
    , ISNULL(NCM.BUSINESS_UNIT_ID, '') AS BUSINESS_UNIT
    , ISNULL(NCM.NCM_TYPE, '') AS NCM_TYPE
    , NCM.SK_ROOT_CAUSE
    , NCM.SK_ROOT_CAUSE_ANALYSIS
    , PS.PLANNER_USER_ID
    
FROM DataWarehouse.dbo.VQ_NONCONFORMANCE NCM
LEFT OUTER JOIN DataWarehouse.dbo.VQ_PRODUCT P
    ON P.RECORD_ID = NCM.PRODUCT_REC_ID
LEFT OUTER JOIN DataWarehouse.dbo.VQ_PRODUCTION_ORDER PO
    ON PO.RECORD_ID = NCM.PROD_ORDER_REC_ID
LEFT OUTER JOIN PART PS
    ON PS.ID = P.PRODUCT_ID
LEFT OUTER JOIN VQ_EMPLOYEE DISP_EE
    ON DISP_EE.RECORD_ID = NCM.[DISP_EMP_REC_ID]
LEFT OUTER JOIN DEMAND_SUPPLY_LINK DSL
    ON PO.DISPLAY_ID = DSL.SUPPLY_BASE_ID + '/' + DSL.SUPPLY_LOT_ID
LEFT OUTER JOIN CUSTOMER_ORDER CO
    ON CO.ID = DSL.DEMAND_BASE_ID
    AND DSL.[DEMAND_TYPE] = 'CO'
LEFT OUTER JOIN VQ_EMPLOYEE EE
    ON NCM.DISP_ASSG_TO_REC_ID = EE.RECORD_ID

WHERE 
    ISNULL(NCM.NCM_TYPE, '_Blank') IN (@NCMTYPE)
    AND ISNULL(NCM.[STATUS], '_Blank') IN (@NCMSTATUS)
    AND ISNULL(NCM.DISPOSITION_TYPE, '_Blank') IN (@DISPTYPE)
    AND COALESCE(NCM.SK_FIN_NCM_PROCESS, NCM.SK_MFG_NCM_PROCESS, '_BLANK') IN (@PROCESS)
    AND COALESCE(NCM.SK_FIN_NCM_SUBPROCESS, NCM.SK_MFG_NCM_SUBPROCESS, '_BLANK') IN (@SUBPROCESS)
    AND ISNULL(NCM.BUSINESS_UNIT_ID, '_Blank') IN (@BUSUNIT)
    AND ISNULL(NCM.USER_DEFINED_3, '_Blank') IN (@FEATURE)
    AND COALESCE(NCM.USER_DEFINED_7, NCM.USER_DEFINED_4, '_Blank') IN (@DEFECT)
    AND NCM.NCM_DATE >= @STARTDATE
    AND NCM.NCM_DATE <= @ENDDATE
    AND PS.PLANNER_USER_ID IN (@PLANNER)
    AND ISNULL(CO.CUSTOMER_ID, '_Blank') IN (@CUSTID)
    AND ISNULL(EE.[FIRST_NAME] + ' ' + EE.[LAST_NAME],  '_Blank') IN (@Assigned_TO)

```
