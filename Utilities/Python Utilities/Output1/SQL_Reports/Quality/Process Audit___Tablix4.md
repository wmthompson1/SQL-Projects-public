```sql
SELECT pa.SK_AUDIT_MANAGER_ID AUDIT_MANAGER_ID, af.FINDING_ID, af.FINDING_TYPE, af.FINDING_TEXT, af.CLOSED_DATE FindingClosedDate, af.CreatedBy FindingCreatedBy, af.CreateDate FindingCreateDate, af.FOLLOWUP_TEXT
    , afE.EMPLOYEE_ID AS FindingFollowUpEMPID, afE.NAME_1 AS FindingFollowUpName, afE.WORK_EMAIL_ADDRESS AS FindingFollowUpEmail 
    , af.SK_PROC_AUDIT_ID
FROM VQ_AUDIT_FINDINGS af
    LEFT JOIN VQ_EMPLOYEE afE
        ON af.FOLLOWUP_BY_REC_ID = afE.RECORD_ID
    LEFT JOIN VQ_PROCESS_AUDIT pa
        ON af.SK_PROC_AUDIT_ID = pa.AUDIT_ID
WHERE af.SK_PROC_AUDIT_ID = @AUDIT_ID
--am.AUDIT_MANAGER_ID = @AUDIT_MANAGER_ID
    AND pa.SITE_ID = @SITE_ID
```
