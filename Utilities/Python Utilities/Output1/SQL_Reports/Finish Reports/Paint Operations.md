# Paint Operations

```sql
SELECT O.DESIRED_WANT_DATE, O.WOSTATUS, O.OPSTATUS, CO.CUSTOMER_ID
    , O.BASE_ID, O.LOT_ID, O.SPLIT_ID, O.SUB_ID, O.SEQUENCE_NO
    , O.RESOURCE_ID, O.OPERATION_SEQ_NO, O.PART_ID
    , O.OPERATION_TYPE, O.DESIRED_QTY
    , CAST(CAST(BITS AS VARBINARY(MAX)) AS NVARCHAR(MAX)) AS SPECS
    , OT.[DESCRIPTION]
    , CASE 
        WHEN CAST(CAST(BITS AS VARBINARY(MAX)) AS NVARCHAR(MAX)) LIKE '%ONE COAT%' THEN 1
        WHEN CAST(CAST(BITS AS VARBINARY(MAX)) AS NVARCHAR(MAX)) LIKE '%TWO COAT%' THEN 2
        WHEN CAST(CAST(BITS AS VARBINARY(MAX)) AS NVARCHAR(MAX)) LIKE '%THREE COAT%' THEN 3
        ELSE 0
    END AS NUMBER_OF_COATS

FROM SKILLS_WO_OPS_OPEN O WITH (NOLOCK)
INNER JOIN SHOP_RESOURCE SR WITH (NOLOCK) ON O.RESOURCE_ID = SR.ID
INNER JOIN OPER_TYPE_BINARY OTB WITH (NOLOCK) ON OTB.OPERATION_TYPE_ID = O.OPERATION_TYPE
INNER JOIN OPERATION_TYPE OT WITH (NOLOCK) ON O.OPERATION_TYPE = OT.ID
LEFT OUTER JOIN DEMAND_SUPPLY_LINK DSL ON O.BASE_ID = DSL.SUPPLY_BASE_ID
    AND O.LOT_ID = DSL.SUPPLY_LOT_ID AND O.SPLIT_ID = DSL.SUPPLY_SPLIT_ID
    AND O.SUB_ID = DSL.SUPPLY_SUB_ID
LEFT OUTER JOIN CUST_ORDER_LINE COL ON DSL.DEMAND_BASE_ID = COL.CUST_ORDER_ID
    AND DSL.DEMAND_SEQ_NO = COL.LINE_NO
LEFT OUTER JOIN CUSTOMER_ORDER CO ON COL.CUST_ORDER_ID = CO.ID

WHERE SR.USER_6 = 'PAINT' AND SR.USER_7 IN ('PRIMER', 'TOPCOAT') AND O.OPSTATUS != 'C'

```
