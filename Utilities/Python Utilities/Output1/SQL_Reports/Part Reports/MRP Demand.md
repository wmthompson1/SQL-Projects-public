# MRP Demand

```sql
; WITH CTE_PLANNED AS
(SELECT PART_ID
, SUM(ORDER_QTY) AS PLANNED_QTY
FROM [PLANNED_ORDER]
GROUP BY PART_ID
)

SELECT P.ID AS PART_ID
, P.[DESCRIPTION]
, P.[PLANNER_USER_ID]
, P.[BUYER_USER_ID]
, P.[COMMODITY_CODE]
, P.[FABRICATED]
, P.[PURCHASED]
, P.[PLANNING_LEADTIME]
, P.[QTY_ON_HAND]
, P.QTY_IN_DEMAND AS [QUANTITY_IN_DEMAND]
, P.QTY_ON_ORDER AS [QUANTITY_ON_ORDER]
, CTE.PLANNED_QTY
, P.DRAWING_ID AS [EXP DATE]
, UDF35.STRING_VAL AS [Matl]
, UDF36.STRING_VAL AS [Alloy]
, UDF37.NUMBER_VAL AS [Length]
, UDF38.NUMBER_VAL AS [Width_val]
, UDF39.NUMBER_VAL AS [Thick]
, P.USER_2 AS [MFG PROCESSES]
, P.USER_3 AS [FIN PROCESSES]
, P.USER_7 AS [MODEL]
, P.USER_9 AS [GEN PROCESS]
, P.USER_10 AS [CUST ROLT]
FROM PART P
LEFT OUTER JOIN CTE_PLANNED CTE
ON P.ID = CTE.PART_ID
LEFT OUTER JOIN USER_DEF_FIELDS UDF35
    ON UDF35.DOCUMENT_ID = P.ID
    AND UDF35.PROGRAM_ID = 'VMPRTMNT'
    AND UDF35.[ID] = 'UDF-0000035'
LEFT OUTER JOIN USER_DEF_FIELDS UDF36
    ON UDF36.DOCUMENT_ID = P.ID
    AND UDF36.PROGRAM_ID = 'VMPRTMNT'
    AND UDF36.[ID] = 'UDF-0000036'
LEFT OUTER JOIN USER_DEF_FIELDS UDF37
    ON UDF37.DOCUMENT_ID = P.ID
    AND UDF37.PROGRAM_ID = 'VMPRTMNT'
    AND UDF37.[ID] = 'UDF-0000037'
LEFT OUTER JOIN USER_DEF_FIELDS UDF38
    ON UDF38.DOCUMENT_ID = P.ID
    AND UDF38.PROGRAM_ID = 'VMPRTMNT'
    AND UDF38.[ID] = 'UDF-0000038'
LEFT OUTER JOIN USER_DEF_FIELDS UDF39
    ON UDF39.DOCUMENT_ID = P.ID
    AND UDF39.PROGRAM_ID = 'VMPRTMNT'
    AND UDF39.[ID] = 'UDF-0000039'
WHERE (P.[STATUS] IS NULL OR P.[STATUS] != 'O')
    AND P.ID NOT LIKE '%REWORK%'
    AND P.PLANNER_USER_ID IN (@PLANNER)
    AND P.COMMODITY_CODE IN (@COMMODITY)
    AND P.FABRICATED IN (@FABRICATED)
    AND P.PURCHASED IN (@PURCHASED)

```
