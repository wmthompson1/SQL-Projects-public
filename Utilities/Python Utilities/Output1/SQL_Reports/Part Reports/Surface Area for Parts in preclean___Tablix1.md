# Surface Area for Parts in preclean___Tablix1

```sql
-- VISUAL ENTERPRISE > PART REPORTS > SURFACE AREA FOR PARTS IN PRECLEAN
SELECT CAST(LT.TRANSACTION_DATE AS DATE) TRANSACTION_DATE
    , DATEPART(YYYY, LT.TRANSACTION_DATE) TRANSACTION_DATE_YEAR
    , DATEPART(WK, LT.TRANSACTION_DATE) TRANSACTION_DATE_WEEK
    , DATEPART(D, LT.TRANSACTION_DATE) TRANSACTION_DATE_DAY
    , W.BASE_ID
    , W.LOT_ID
    , W.SPLIT_ID
    , W.SUB_ID
    , W.[STATUS]
    , W.ENGINEERED_BY
    , PL.FILTER_STRING AS [NAME]
    , SR.USER_5 AS [Location]
    , PSA.PART_ID
    , CASE 
        WHEN PSA.MATERIAL_TYPE IS NULL THEN 'NEED DIMENSIONS' 
        ELSE PSA.MATERIAL_TYPE
    END AS MATERIAL_TYPE
    , PSA.ALLOY
    , PSA.THICKNESS_VAL
    , PSA.LENGTH_VAL
    , PSA.WIDTH_VAL
    , SUM(W.DESIRED_QTY * ISNULL(PSA.SURFACE_AREA, 0))
        + ISNULL(
        (SELECT SUM(CP.DESIRED_QTY * PSA.SURFACE_AREA) AS EXPR1
        FROM CO_PRODUCT AS CP
        INNER JOIN PART AS SP
            ON CP.PART_ID = SP.ID
        INNER JOIN SKILLS_PART_SURFACE_AREA AS PSA
            ON CP.PART_ID = PSA.PART_ID
        WHERE CP.WORKORDER_TYPE = 'W'
            AND CP.WORKORDER_BASE_ID = W.BASE_ID
            AND CP.WORKORDER_LOT_ID = W.LOT_ID
            AND CP.WORKORDER_SPLIT_ID = W.SPLIT_ID
        ), 0) AS SURFACE_AREA
FROM WORK_ORDER W
INNER JOIN OPERATION O
    ON W.[TYPE] = O.WORKORDER_TYPE
    AND W.BASE_ID = O.WORKORDER_BASE_ID
    AND W.LOT_ID = O.WORKORDER_LOT_ID
    AND W.SPLIT_ID = O.WORKORDER_SPLIT_ID
    AND W.SUB_ID = O.WORKORDER_SUB_ID
INNER JOIN SHOP_RESOURCE SR
    ON SR.ID = O.RESOURCE_ID
    AND SR.USER_6 = 'CHEMICAL PROCESSING'
INNER JOIN LABOR_TICKET LT
    ON O.WORKORDER_TYPE = LT.WORKORDER_TYPE
    AND O.WORKORDER_BASE_ID = LT.WORKORDER_BASE_ID
    AND O.WORKORDER_LOT_ID = LT.WORKORDER_LOT_ID
    AND O.WORKORDER_SPLIT_ID = LT.WORKORDER_SPLIT_ID
    AND O.WORKORDER_SUB_ID = LT.WORKORDER_SUB_ID
    AND LT.TRANSACTION_DATE >= @STARTDATE 
    AND LT.TRANSACTION_DATE < @ENDDATE 
LEFT OUTER JOIN PART P
    ON W.PART_ID = P.ID
LEFT OUTER JOIN SKILLS_PART_SURFACE_AREA PSA
    ON W.PART_ID = PSA.PART_ID
LEFT OUTER JOIN LIVESupplemental.dbo.PlannersList PL
    ON COALESCE(
        PARSENAME(REPLACE(ENGINEERED_BY,'/','.'),1) 
        , PARSENAME(REPLACE(ENGINEERED_BY,'/','.'),2)
        , PARSENAME(REPLACE(ENGINEERED_BY,'/','.'),3)
        , PARSENAME(REPLACE(ENGINEERED_BY,'/','.'),4)
        , CASE
            WHEN RIGHT(ENGINEERED_BY, 1) = '/' AND LEN(ENGINEERED_BY) = 4 THEN REPLACE(ENGINEERED_BY,'/','')
            WHEN RIGHT(ENGINEERED_BY, 1) = '/' AND LEN(ENGINEERED_BY) = 3 THEN REPLACE(ENGINEERED_BY,'/','')
            ELSE ENGINEERED_BY
        END
    ) = PL.FILTER_STRING
WHERE W.[TYPE] = 'W'
    AND W.SUB_ID = 0
GROUP BY CAST(LT.TRANSACTION_DATE AS DATE)
    , DATEPART(YYYY, LT.TRANSACTION_DATE)
    , DATEPART(WK, LT.TRANSACTION_DATE)
    , DATEPART(D, LT.TRANSACTION_DATE)
    , W.BASE_ID
    , W.LOT_ID
    , W.SPLIT_ID
    , W.SUB_ID
    , W.[STATUS]
    , W.ENGINEERED_BY
    , PL.FILTER_STRING
    , SR.USER_5
    , PSA.PART_ID
    , PSA.MATERIAL_TYPE
    , PSA.ALLOY
    , PSA.THICKNESS_VAL
    , PSA.LENGTH_VAL
    , PSA.WIDTH_VAL

```
