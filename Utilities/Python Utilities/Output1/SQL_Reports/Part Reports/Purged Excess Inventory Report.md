# Purged Excess Inventory Report

```sql
SELECT PS.PART_ID, P.DESCRIPTION, P.COMMODITY_CODE, P.PURCHASED, P.FABRICATED, (PS.UNIT_MATERIAL_COST + PS.UNIT_LABOR_COST + PS.FIXED_COST +
PS.UNIT_BURDEN_COST + PS.UNIT_SERVICE_COST) AS 'TOTAL UNIT COST', P.BUYER_USER_ID, P.DRAWING_ID,
PL.WAREHOUSE_ID, PL.LOCATION_ID, PL.QTY
,(SELECT MAX(st1.TRANSACTION_DATE) FROM INVENTORY_TRANS st1 WHERE st1.PART_ID=ps.PART_ID AND st1.CLASS='I')'LAST TRANSACTION DATE'
FROM PART_SITE PS
INNER JOIN PART P ON P.ID = PS.PART_ID
LEFT JOIN REQUIREMENT R ON PS.PART_ID = R.PART_ID
LEFT JOIN PART_LOCATION PL ON PL.PART_ID = PS.PART_ID
WHERE PS.ENGINEERING_MSTR IS NULL
AND R.PART_ID IS NULL
and ps.status!='O'
and p.status!='O'

```
