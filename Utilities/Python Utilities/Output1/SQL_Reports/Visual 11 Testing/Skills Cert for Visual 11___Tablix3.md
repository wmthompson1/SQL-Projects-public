```sql
-- Visual Enterprise Reports > CERT Reports > Skills Cert for Laser Printers
--DECLARE @PACKLIST AS NVARCHAR(15) 
--SET @PACKLIST = '1236443'


SELECT WO.BASE_ID
    , WO.LOT_ID
    , WO.SPLIT_ID
    , dbo.sfnWONUMFormat(WO.BASE_ID,WO.LOT_ID,WO.SPLIT_ID,WO.SUB_ID) AS WONUM
    , dbo.sfnWONUMFormat_OLD(WO.BASE_ID,WO.LOT_ID,WO.SPLIT_ID,WO.SUB_ID) AS WONUM_Old
    , SL.SHIPPED_QTY
                  , ISNULL(SL.USER_4,
                  --CASE WHEN TS.MOVED_TO = 'Level II' 
         --                   THEN '2' 
                        --    WHEN TS.MOVED_TO = 'LEVEL III' 
                        --    THEN '3' 
                        --    ELSE SL.USER_4 END
                        TS.MOVED_TO) AS PENETRANT_LEVEL
    , COL.PART_ID
    , PSV.DESCRIPTION AS PART_DESCRIPTION
    , S.PACKLIST_ID
    , CASE
        WHEN C.USER_8 = 'Y' THEN COL.USER_2
        ELSE NULL
    END AS CUST_PO_LN
    , CASE
        WHEN C.USER_8 = 'Y' THEN 'Y'
        ELSE 'N'
    END AS CUST_POLN_PRINT
    , CO.CUSTOMER_PO_REF
    , CO.ID
    , S.SHIPPED_DATE
    , CASE
        WHEN CHARINDEX('<>', SOB.SPECS) = 0 THEN SOB.SPECS
        ELSE LEFT(SOB.specs, CHARINDEX('<>', SOB.SPECS)-1)
    END AS SPECS
    , SCD.DESCRIPTION
    , SCD.PRINT_
    , COL.CUSTOMER_PART_ID
    , DBO.SFNGETLASTWOLOCATION(WO.BASE_ID, WO.LOT_ID, WO.SUB_ID, WO.SPLIT_ID) AS SHIP_FROM
    -- ISNULL(CS.ADDR_1,'') + CHAR(10) + CHAR(13) + ISNULL(CS.CITY,'') + ', ' +  ISNULL(CS.STATE,'') + ' ' + ISNULL(CS.ZIPCODE,'') + ' ' + ISNULL(CS.COUNTRY,'') + CHAR(10) + CHAR(13) + CHAR(10) + CHAR(13) + ISNULL(CS.CONTACT_PHONE,'') + '  FAX ' + ISNULL(CS.CONTACT_FAX,'') AS SHIPFROM
    , C.ID AS cust_id
    , C.NAME
    , C.ADDR_1
    , C.ADDR_2
    , C.STATE
    , C.ZIPCODE
    , C.CITY
    , CA.NAME AS ShipName
    , CA.ADDR_1 AS ShipAddr1
    , CA.ADDR_2 AS ShipAddr2
    , CA.CITY AS ShipCity
    , CA.STATE AS ShipState
    , CA.ZIPCODE AS ShipZip
    , CA.COUNTRY
    , COL.LINE_NO
    , SOB.SEQUENCE_NO
    , O.OPERATION_TYPE
    , S.USER_1
    , S.USER_2
    , S.USER_3
    , S.USER_4
    , S.USER_5
    , S.USER_6
    , S.USER_7
    , S.USER_8
    , S.USER_9
    , S.USER_10
FROM CUSTOMER AS C
    INNER JOIN CUSTOMER_ORDER AS CO
        ON C.ID = CO.CUSTOMER_ID
    INNER JOIN CUST_ORDER_LINE AS COL
        ON CO.ID = COL.CUST_ORDER_ID
     INNER JOIN PART_SITE_VIEW PSV
         ON COL.PART_ID = PSV.PART_ID
    INNER JOIN SHIPPER AS S
        ON CO.ID = S.CUST_ORDER_ID
    INNER JOIN SHIPPER_LINE AS SL
        ON S.PACKLIST_ID = SL.PACKLIST_ID
        AND COL.CUST_ORDER_ID = SL.CUST_ORDER_ID
        AND COL.LINE_NO = SL.CUST_ORDER_LINE_NO
    INNER JOIN DEMAND_SUPPLY_LINK AS DSL
        ON COL.CUST_ORDER_ID = DSL.DEMAND_BASE_ID
        AND COL.LINE_NO = DSL.DEMAND_SEQ_NO
    INNER JOIN WORK_ORDER AS WO
        ON DSL.SUPPLY_BASE_ID = WO.BASE_ID
        AND DSL.SUPPLY_LOT_ID = WO.LOT_ID
        AND DSL.SUPPLY_SPLIT_ID = WO.SPLIT_ID
        AND DSL.SUPPLY_SUB_ID = WO.SUB_ID
    INNER JOIN OPERATION AS O
        ON WO.TYPE = O.WORKORDER_TYPE
        AND WO.BASE_ID = O.WORKORDER_BASE_ID
        AND WO.LOT_ID = O.WORKORDER_LOT_ID
        AND WO.SPLIT_ID = O.WORKORDER_SPLIT_ID
        AND WO.SUB_ID = O.WORKORDER_SUB_ID
    INNER JOIN SKILL_OPERATION_BINARY AS SOB
        ON O.WORKORDER_TYPE = SOB.WORKORDER_TYPE
        AND O.WORKORDER_BASE_ID = SOB.WORKORDER_BASE_ID
        AND O.WORKORDER_LOT_ID = SOB.WORKORDER_LOT_ID
        AND O.WORKORDER_SPLIT_ID = SOB.WORKORDER_SPLIT_ID
        AND O.WORKORDER_SUB_ID = SOB.WORKORDER_SUB_ID
        AND O.SEQUENCE_NO = SOB.SEQUENCE_NO
    INNER JOIN SKILLS_CERT_DESC AS SCD
        ON O.OPERATION_TYPE = SCD.OP_TYPE
    INNER JOIN CUST_ADDRESS AS CA
        ON CO.CUSTOMER_ID = CA.CUSTOMER_ID
        AND CO.SHIP_TO_ADDR_NO = CA.ADDR_NO
    -- gets penetrant tank information
    LEFT OUTER JOIN LIVESupplemental.dbo.Skills_Finish_Open_Orders_EastWest TS
         ON WO.TYPE = 'W'
        AND WO.BASE_ID = TS.BASE_ID
        AND WO.LOT_ID = TS.LOT_ID
        AND WO.SPLIT_ID = TS.SPLIT_ID
        AND WO.SUB_ID = TS.SUB_ID
WHERE       (
                     SCD.PRINT_ = 'Y'
      AND SL.SHIPPED_QTY > 0
      AND NOT(COL.PART_ID IN ('E','EX', 'EXP','L'))
      AND S.PACKLIST_ID =  @PACKLIST
      AND S.USER_10 IS NULL
      )
      OR 
      (
                      SCD.PRINT_ = 'Y'
       AND SL.SHIPPED_QTY > 0
       AND NOT(COL.PART_ID IN ('E','EX', 'EXP','L'))
       AND S.PACKLIST_ID =  @PACKLIST
       AND SOB.SEQUENCE_NO <= CONVERT(INT,S.USER_10)
       )
ORDER BY COL.LINE_NO
    , SOB.SEQUENCE_NO

```
