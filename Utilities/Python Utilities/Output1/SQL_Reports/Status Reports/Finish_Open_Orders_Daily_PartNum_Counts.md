# Finish_Open_Orders_Daily_PartNum_Counts

```sql
SELECT CO.CUSTOMER_ID
    , C.NAME
    , CASE 
        WHEN C.SALESREP_ID IS NULL THEN 'NO REP ASSN'
        ELSE C.SALESREP_ID 
    END AS CUSREP
    , CO.ID AS CUST_ORDER_ID
    , W.WORKORDER_BASE_ID
    , W.WORKORDER_LOT_ID
    , W.WORKORDER_SUB_ID
    , W.WORKORDER_SPLIT_ID
    , CL.PART_ID
    , CONVERT(DATE, ISNULL(CL.DESIRED_SHIP_DATE, CO.DESIRED_SHIP_DATE), 101) AS DUEDATE
    , CASE 
        WHEN CONVERT(DATE, ISNULL(CL.DESIRED_SHIP_DATE, CO.DESIRED_SHIP_DATE), 101) < CONVERT(DATE, GETDATE())
            THEN SUM(CL.ORDER_QTY )
        ELSE 0 
    END AS PASTDUE
    , CASE 
        WHEN CONVERT(DATE, ISNULL(CL.DESIRED_SHIP_DATE, CO.DESIRED_SHIP_DATE), 101) = CONVERT(DATE, GETDATE()+10)
            THEN SUM(CL.ORDER_QTY)
        ELSE 0 END AS DAYS10
    , CASE 
        WHEN CONVERT(DATE, ISNULL(CL.DESIRED_SHIP_DATE, CO.DESIRED_SHIP_DATE), 101) = CONVERT(DATE, GETDATE()+9)
            THEN SUM(CL.ORDER_QTY) 
        ELSE 0 
    END AS DAYS9
    , CASE 
        WHEN CONVERT(DATE, ISNULL(CL.DESIRED_SHIP_DATE, CO.DESIRED_SHIP_DATE), 101) = CONVERT(DATE, GETDATE()+8)
            THEN SUM(CL.ORDER_QTY) 
        ELSE 0 
    END AS DAYS8
    , CASE 
        WHEN CONVERT(DATE, ISNULL(CL.DESIRED_SHIP_DATE, CO.DESIRED_SHIP_DATE), 101) = CONVERT(DATE, GETDATE()+7)
            THEN SUM(CL.ORDER_QTY) 
        ELSE 0 
    END AS DAYS7
    , CASE 
        WHEN CONVERT(DATE, ISNULL(CL.DESIRED_SHIP_DATE, CO.DESIRED_SHIP_DATE), 101) = CONVERT(DATE, GETDATE()+6)
            THEN SUM(CL.ORDER_QTY) 
        ELSE 0 
    END AS DAYS6
    , CASE 
        WHEN CONVERT(DATE, ISNULL(CL.DESIRED_SHIP_DATE, CO.DESIRED_SHIP_DATE), 101) = CONVERT(DATE, GETDATE()+5)
            THEN SUM(CL.ORDER_QTY) 
        ELSE 0 
    END AS DAYS5
    , CASE 
        WHEN CONVERT(DATE, ISNULL(CL.DESIRED_SHIP_DATE, CO.DESIRED_SHIP_DATE), 101) = CONVERT(DATE, GETDATE()+4)
            THEN SUM(CL.ORDER_QTY) 
        ELSE 0 
    END AS DAYS4
    , CASE 
        WHEN CONVERT(DATE, ISNULL(CL.DESIRED_SHIP_DATE, CO.DESIRED_SHIP_DATE), 101) = CONVERT(DATE, GETDATE()+3)
            THEN SUM(CL.ORDER_QTY) 
        ELSE 0 
    END AS DAYS3
    , CASE 
        WHEN CONVERT(DATE, ISNULL(CL.DESIRED_SHIP_DATE, CO.DESIRED_SHIP_DATE), 101) = CONVERT(DATE, GETDATE()+2)
            THEN SUM(CL.ORDER_QTY) 
        ELSE 0 
    END AS DAYS2
    , CASE 
        WHEN CONVERT(DATE, ISNULL(CL.DESIRED_SHIP_DATE, CO.DESIRED_SHIP_DATE), 101) = CONVERT(DATE, GETDATE()+1)
            THEN SUM(CL.ORDER_QTY) 
        ELSE 0 
    END AS DAYS1
    , CASE 
        WHEN CONVERT(DATE, ISNULL(CL.DESIRED_SHIP_DATE, CO.DESIRED_SHIP_DATE), 101) = CONVERT(DATE, GETDATE())
            THEN SUM(CL.ORDER_QTY) 
        ELSE 0 
    END AS DAYS0
    , SUM(CL.ORDER_QTY) AS SUMQTY                     
FROM  CUST_ORDER_LINE CL
    INNER JOIN CUSTOMER_ORDER CO
        ON CL.CUST_ORDER_ID = CO.ID
        AND CO.STATUS NOT IN ('X', 'C')
    INNER JOIN CUSTOMER C
        ON CO.CUSTOMER_ID = C.ID     
    INNER JOIN DEMAND_SUPPLY_LINK DSL
        ON DSL.DEMAND_BASE_ID = CL.CUST_ORDER_ID
        AND DSL.DEMAND_SEQ_NO = CL.LINE_NO
        AND DSL.DEMAND_TYPE = 'CO'
    INNER JOIN dbo.SKILLS_WO_FINISH_ONLY w
        ON W.WORKORDER_BASE_ID = DSL.SUPPLY_BASE_ID
        AND W.WORKORDER_LOT_ID = DSL.SUPPLY_LOT_ID
        AND W.WORKORDER_SPLIT_ID = DSL.SUPPLY_SPLIT_ID
        AND W.WORKORDER_SUB_ID = DSL.SUPPLY_SUB_ID
        AND W.STATUS NOT IN ('X', 'C')
        AND DSL.SUPPLY_TYPE IN ('WO', 'CP')
WHERE CL.ORDER_QTY <> 0
    AND CONVERT(DATE, ISNULL(CL.DESIRED_SHIP_DATE, CO.DESIRED_SHIP_DATE), 101) 
        BETWEEN CONVERT(DATE,GETDATE()-365, 101) 
        AND GETDATE()+10
GROUP BY CO.CUSTOMER_ID
    , C.NAME
    , C.SALESREP_ID
    , CO.ID
    , CONVERT(DATE, ISNULL(CL.DESIRED_SHIP_DATE, CO.DESIRED_SHIP_DATE), 101)
    , W.WORKORDER_BASE_ID
    , W.WORKORDER_LOT_ID
    , W.WORKORDER_SUB_ID
    , W.WORKORDER_SPLIT_ID
    , CL.PART_ID

```
