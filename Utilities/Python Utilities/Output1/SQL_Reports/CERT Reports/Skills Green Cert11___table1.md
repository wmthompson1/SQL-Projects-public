WITH CTE_SHIPPERCONTENTS AS 
(SELECT ISNULL(DSL.SUPPLY_BASE_ID    
    , DBO.SFNGETBASEFROMTRACEID(TRT.TRACE_ID)) AS WORKORDER_BASE_ID
    , ISNULL(DSL.SUPPLY_LOT_ID, DBO.SFNGETLOTIDFROMTRACEID(TRT.TRACE_ID)) AS WORKORDER_LOT_ID
    , COALESCE(DSL.SUPPLY_SPLIT_ID, DBO.SFNGETSPLITIDFROMTRACEID(TRT.TRACE_ID),'0') AS WORKORDER_SPLIT_ID
    , ISNULL(CASE 
            WHEN CHARINDEX('-',TRT.TRACE_ID) <> '0'  THEN SUBSTRING(TRT.TRACE_ID, CHARINDEX('/',TRT.TRACE_ID)+1, 1)
            ELSE '0'
        END, SUPPLY_SUB_ID) AS WORKORDER_SUB_ID
    , DBO.SFNWONUMFORMAT_OLD(SUPPLY_BASE_ID, SUPPLY_LOT_ID, SUPPLY_SPLIT_ID, SUPPLY_SUB_ID) AS OLD_WO_ID
    , TRT.TRANSACTION_ID
    , TRT.TRACE_ID
    , COL.PART_ID
    , PSV.DESCRIPTION AS PART_DESCRIPTION
    , S.USER_1
    , S.USER_2
    , S.USER_3
    , S.USER_4
    , S.USER_5
    , S.USER_6
    , S.USER_7
    , S.USER_8
    , S.USER_9
    , S.USER_10
    , SL.CUST_ORDER_ID
    , SL.CUST_ORDER_LINE_NO
    , COL.CUSTOMER_PART_ID  AS CUST_PART_ID
    , CO.CUSTOMER_PO_REF
    , CO.CUSTOMER_ID
    , DBO.FNFORMATDATE(S.SHIPPED_DATE,'M/D/YYYY') AS SHIPPED_DATE
    , CO.SHIP_TO_ADDR_NO
    , CASE
        WHEN COL.USER_2 IS NOT NULL THEN 'Y'
        ELSE 'N'
    END AS CUST_POLN_PRINT
    , COL.USER_2 AS CUST_PO_LN
    , ISNULL(ABS(TRT.QTY), SL.SHIPPED_QTY) AS WO_SHIPPED_QTY
FROM SHIPPER_LINE SL
    INNER JOIN CUST_ORDER_LINE COL
        ON SL.CUST_ORDER_ID = COL.CUST_ORDER_ID
        AND SL.CUST_ORDER_LINE_NO = COL.LINE_NO
    INNER JOIN SHIPPER S
        ON SL.PACKLIST_ID = S.PACKLIST_ID
    INNER JOIN CUSTOMER_ORDER AS CO
        ON COL.CUST_ORDER_ID = CO.ID
     LEFT OUTER JOIN PART_SITE_VIEW PSV
      ON COL.PART_ID = PSV.PART_ID
    LEFT JOIN DEMAND_SUPPLY_LINK AS DSL
        ON COL.CUST_ORDER_ID = DSL.DEMAND_BASE_ID
        AND COL.LINE_NO = DSL.DEMAND_SEQ_NO
    LEFT JOIN TRACE_INV_TRANS AS TRT
        ON TRT.TRANSACTION_ID = SL.TRANSACTION_ID
WHERE  SL.PACKLIST_ID = @PACKLIST
    AND ISNULL(ABS(TRT.QTY), SL.SHIPPED_QTY) > 0
)

, SPLIT_TREE AS
(SELECT WO.BASE_ID
    , WO.LOT_ID
    , WO.SUB_ID
    , WO.SPLIT_ID
    , WO.COPY_FROM_SPLIT_ID 
    , DBO.UFN_GETFIRSTOPERATION(WO.BASE_ID, WO.LOT_ID, WO.SUB_ID, WO.SPLIT_ID, 'W') AS FIRST_SEQUENCE
    , 1 AS [DISTANCE FROM CHILD]
FROM WORK_ORDER WO
    JOIN CTE_WOINSHIPPER SC
        ON WO.BASE_ID = SC.WORKORDER_BASE_ID
        AND WO.LOT_ID = SC.WORKORDER_LOT_ID
        AND WO.SPLIT_ID = SC.WORKORDER_SPLIT_ID
        AND WO.SUB_ID = SC.WORKORDER_SUB_ID
        AND WO.TYPE = 'W'
WHERE WO.[COPY_FROM_SPLIT_ID] IS NOT NULL
    AND WO.BASE_ID = SC.WORKORDER_BASE_ID
    AND WO.LOT_ID = SC.WORKORDER_LOT_ID
    AND WO.SUB_ID = SC.WORKORDER_SUB_ID
    AND WO.SPLIT_ID = SC.WORKORDER_SPLIT_ID
UNION ALL
SELECT WO2.BASE_ID
    , WO2.LOT_ID
    , WO2.SUB_ID
    , WO2.SPLIT_ID
    , WO2.COPY_FROM_SPLIT_ID
    , DBO.UFN_GETFIRSTOPERATION(WO2.BASE_ID, WO2.LOT_ID, WO2.SUB_ID, WO2.SPLIT_ID, 'W')
    , [DISTANCE FROM CHILD] + 1
FROM WORK_ORDER WO2
    JOIN SPLIT_TREE S  
        ON WO2.BASE_ID = S.BASE_ID
        AND WO2.SUB_ID = S.SUB_ID
        AND WO2.LOT_ID = S.LOT_ID
        AND WO2.SPLIT_ID = S.COPY_FROM_SPLIT_ID
)

, CTE_ALL_WOS AS
(SELECT * 
FROM SPLIT_TREE
UNION ALL 
SELECT * 
    , NULL
    , 0
    , 0
FROM CTE_WOINSHIPPER W
WHERE NOT EXISTS

, CTE_SPLITHIERARCHYOPS AS 
(SELECT ST.BASE_ID
    , ST.LOT_ID
    , ST.SUB_ID
    , ST.SPLIT_ID
    , ST.[DISTANCE FROM CHILD]
    , COALESCE(O.SEQUENCE_NO, O2.SEQUENCE_NO, O3.SEQUENCE_NO) AS SEQUENCENO
    , COALESCE(O.OPERATION_TYPE, O2.OPERATION_TYPE, O3.OPERATION_TYPE) AS OPERATION_TYPE
    , COALESCE(O.RESOURCE_ID, O2.RESOURCE_ID, O3.RESOURCE_ID) AS RESOURCE_ID
    , COALESCE(O.SERVICE_ID, O2.SERVICE_ID, O3.SERVICE_ID) AS SERVICE_ID
    , ST.[DISTANCE FROM CHILD] AS STEPSFROMCHILD
FROM CTE_ALL_WOS ST
LEFT JOIN OPERATION O --SPLIT CHILDREN OPS
    ON ST.BASE_ID = O.WORKORDER_BASE_ID
    AND ST.LOT_ID = O.WORKORDER_LOT_ID
    AND ST.SUB_ID = O.WORKORDER_SUB_ID
    AND ST.SPLIT_ID = O.WORKORDER_SPLIT_ID
    AND O.WORKORDER_TYPE = 'W'
    AND ST.COPY_FROM_SPLIT_ID IS NOT NULL
    AND EXISTS
        (SELECT 1 
        FROM SPLIT_TREE S
        WHERE S.BASE_ID = O.WORKORDER_BASE_ID
            AND S.LOT_ID = O.WORKORDER_LOT_ID
            AND S.SPLIT_ID = O.WORKORDER_SPLIT_ID
            AND S.SUB_ID = O.WORKORDER_SUB_ID
        )
LEFT JOIN OPERATION O2 --SPLIT PARENT OPS
    ON ST.BASE_ID = O2.WORKORDER_BASE_ID
    AND ST.LOT_ID = O2.WORKORDER_LOT_ID
    AND ST.SUB_ID = O2.WORKORDER_SUB_ID
    AND ST.SPLIT_ID = O2.WORKORDER_SPLIT_ID
    AND O2.WORKORDER_TYPE = 'W'
    AND ST.COPY_FROM_SPLIT_ID IS NULL
    AND O2.SEQUENCE_NO < ISNULL(
        (SELECT TOP 1 ST.FIRST_SEQUENCE FROM SPLIT_TREE ST WHERE COPY_FROM_SPLIT_ID IS NOT NULL),0)
    AND ST.FIRST_SEQUENCE <> 0
LEFT JOIN OPERATION O3 --NON SPLIT OPS
    ON ST.BASE_ID = O3.WORKORDER_BASE_ID
    AND ST.LOT_ID = O3.WORKORDER_LOT_ID
    AND ST.SPLIT_ID = O3.WORKORDER_SPLIT_ID
    AND ST.SUB_ID = O3.WORKORDER_SUB_ID
    AND O3.WORKORDER_TYPE = 'W'
    AND ST.FIRST_SEQUENCE = 0
)

    , C.ADDR_1 AS SOLDTO_ADDR1
    , C.ADDR_2 AS SOLDTO_ADDR2

    , C.[STATE] AS SOLDTO_STATE
    , C.ZIPCODE AS SOLDTO_ZIPCODE

        , SHIP_TO_ADDR_NO
    FROM CTE_SHIPPERCONTENTS
    ) AS SC
JOIN CUST_ADDRESS AS CA
    ON SC.CUSTOMER_ID = CA.CUSTOMER_ID
    AND CA.ADDR_NO = SC.SHIP_TO_ADDR_NO
JOIN CUSTOMER AS C
    ON SC.CUSTOMER_ID = C.ID
)

SELECT W.*
    , R.PART_ID AS RESOURCEPART_ID
    , DBO.SFNSKILLS_WO_DOC_LINK_NO_LOC(W.BASE_ID, W.LOT_ID, W.SUB_ID, W.WORKORDER_SPLIT_ID, 'W') AS SKILLS_WO_DOC_NO_LOC
    , DBO.SFNSKILLS_WO_DOC_LINK(W.BASE_ID, W.LOT_ID, W.SUB_ID, W.WORKORDER_SPLIT_ID, 'W') AS SKILLS_WO_DOC
    , CASE
        WHEN R.PART_ID IS NULL THEN CASE
            WHEN W.RESOURCE_ID IN 
                (SELECT RESOURCE_ID 
                FROM LIVESUPPLEMENTAL.DBO.RESOURCEID_REPORTSETTINGS 
                WHERE REPORT_NAME = 'Skills Green Cert11' 
                AND REPORT_SECTION = 'Specs'
                )
                THEN REPLACE(CONVERT(NVARCHAR(MAX), CONVERT(VARBINARY(MAX), OB.BITS)), 'INSPECT', 'INSPECTED')
            ELSE CONVERT(NVARCHAR(MAX), CONVERT(VARBINARY(MAX), OB.BITS))
        END
        ELSE NULL
    END AS SPECS
    , CONVERT(NVARCHAR(MAX), CONVERT(VARBINARY(MAX), OB.BITS)) AS OPERATIONDESCRIPTION
    , CASE
        WHEN SCD.DESCRIPTION IS NULL
            THEN 'SEE ATTACHED SPECIAL PROCESS CERT FOR SKILLS '
            -- add vendor id here
            +'PO: '+ W.PO_BASE_ID
        ELSE SCD.DESCRIPTION
    END AS DESCRIPTION
    , CASE
        WHEN W.RESOURCE_ID IN 
            (SELECT RESOURCE_ID 
            FROM LIVESUPPLEMENTAL.DBO.RESOURCEID_REPORTSETTINGS 
            WHERE REPORT_NAME = 'Skills Green Cert11' 
            AND REPORT_SECTION = 'PRINT_'
            )
            AND SER.ID IS NOT NULL
            AND LEFT(SER.DESCRIPTION, 3) IN ('FIN', 'SPL')
            THEN 'Y'
        WHEN SCD.DESCRIPTION IS NULL
            AND LEFT(SER.DESCRIPTION, 3) NOT IN('FIN', 'SPL')
            THEN 'N'
        WHEN SCD.DESCRIPTION IS NULL
            AND SER.ID IS NULL
            THEN 'N'
        ELSE SCD.PRINT_
    END AS 'PRINT_'
    , ISNULL(CI.NAME, '') + CHAR(10) + CHAR(13) + ISNULL(CI.ADDR_1, '') + ' ' + ISNULL(CI.ADDR_2, '') + CHAR(10) + CHAR(13) + ISNULL(CI.CITY, '') + ' ' + ISNULL(CI.STATE, '') + ' ' + ISNULL(CI.ZIPCODE, '') + ' ' + ISNULL(CI.COUNTRY, '') AS SHIPTO
    , ISNULL(CI.SOLDTO_NAME, '') + CHAR(10) + CHAR(13) + ISNULL(CI.SOLDTO_ADDR1, '') + ' ' + ISNULL(CI.SOLDTO_ADDR2, '') + CHAR(10) + CHAR(13) + ISNULL(CI.SOLDTO_CITY, '') + ' ' + ISNULL(CI.SOLDTO_STATE, '') + ' ' + ISNULL(CI.SOLDTO_ZIPCODE, '') AS SOLDTO
    , CASE 
        WHEN CI.CUSTOMER_ID LIKE 'BOE%' THEN 
            CASE 
                WHEN W.OPERATION_TYPE = 'BOENOTE' THEN 'SHOW'
                ELSE 'HIDE'
            END
        ELSE 'SHOW'
    END AS BOECONTROL
    , CASE 
        WHEN W.OP_STATUS <> 'X' AND W.OPERATION_TYPE IN ('COC','REVISION','CHKMATL','RW','RW-CRES','AVG','EC', 'HARDNESS', 'BOENOTE')
            THEN CASE 
                WHEN R.PART_ID IS NULL THEN 'SHOW'
                ELSE 'HIDE'
            END
        ELSE 'HIDE'
    END AS OPTYPCONTROL
    , ISNULL(SCD.CUST_FILTER_APPLIED, 'N') AS OP_FILTER
    , ISNULL(CI.USER_7, 'N') AS CUST_FILTER
    , (SELECT SUM(ABS(TRT.QTY)) 
    FROM SHIPPER_LINE SL
    INNER JOIN TRACE_INV_TRANS AS TRT
        ON SL.TRANSACTION_ID = TRT.TRANSACTION_ID
        AND SL.PACKLIST_ID = @PACKLIST
    ) AS TOTQTY
    , CI.NAME
    , CI.ADDR_1
    , CI.ADDR_2
    , CI.ADDR_3
    , CI.CITY
    , CI.STATE
    , CI.ZIPCODE
    , CI.COUNTRY
    , CI.SALESREP_ID
    , T.REQ_TRACE_ID
    , CASE 
        WHEN T.REQ_TRACE_ID IS NULL THEN NULL
        ELSE RP.ID + ' ' + RP.DESCRIPTION
    END AS COMPONENT
    , CASE 
        WHEN T.REQ_TRACE_ID IS NULL THEN NULL
        ELSE R.PIECE_NO
    END AS PIECE_NO
    , ISNULL(T.APROPERTY_LABEL_1, TP.APROPERTY_LABEL_1) + ': ' + ISNULL(T.APROPERTY_1, TT.APROPERTY_1) AS PROPERTY_1

    , ISNULL(T.APROPERTY_LABEL_4, TP.APROPERTY_LABEL_4) + ': ' + ISNULL(T.APROPERTY_4, TT.APROPERTY_4) AS PROPERTY_4
    , CASE 
        WHEN ISNULL(T.APROPERTY_5, TT.APROPERTY_5) IS NOT NULL THEN CASE 
                WHEN ISNULL(T.APROPERTY_5, TT.APROPERTY_5) = 'N/A' OR ISNULL(T.APROPERTY_5, TT.APROPERTY_5) = 'NA'

            ELSE ISNULL(T.APROPERTY_LABEL_5, TP.APROPERTY_LABEL_5) + ': '+ ISNULL(T.APROPERTY_5, TT.APROPERTY_5)

        ELSE NULL

                   + w.ADDR_1 +  CHAR(10)+CHAR(13)  
                  + w.city + ', ' +  w.STATE + ' ' + w.zipcode + ' ' + 'USA' + CHAR(10)+CHAR(13) 
                  +  CHAR(10)+CHAR(13) + w.ADDR_3 
              FROM SHIPPER_LINE SL
             INNER JOIN  INVENTORY_TRANS it
                on SL.TRANSACTION_ID = it.TRANSACTION_ID
             INNER join WAREHOUSE W 
              ON IT.WAREHOUSE_ID = W.ID
              WHERE PACKLIST_ID = @PACKLIST
           ) AS SHIPFROM_aub
    , DBO.SFNGETLASTWOLOCATION(W.BASE_ID, W.LOT_ID, W.SUB_ID, W.SPLIT_ID) AS SHIP_FROM

    ON CI.CUSTOMER_ID = W.CUSTOMER_ID

    AND W.SUB_ID = R.WORKORDER_SUB_ID
    AND W.CURRENT_SPLIT_ID = R.WORKORDER_SPLIT_ID

    AND W.RESOURCE_ID IN
        (SELECT RESOURCE_ID 
        FROM LIVESUPPLEMENTAL.DBO.RESOURCEID_REPORTSETTINGS 
        WHERE REPORT_NAME = 'Skills Green Cert11' 
        AND REPORT_SECTION = 'requirement'
        )
    AND R.OPERATION_SEQ_NO = W.SEQUENCE_NO
    AND R.SUBORD_WO_SUB_ID IS NULL
    AND EXISTS
        (SELECT 1
        FROM PART
        WHERE ID = R.PART_ID
        )
LEFT JOIN PART AS RP
    ON R.PART_ID = RP.ID
LEFT JOIN TRACE TT
    ON TT.ID = W.TRACE_ID
LEFT JOIN TRACE_PROFILE TP
    ON TT.PART_ID = TP.PART_ID
LEFT JOIN
    (SELECT SUM(TRT.QTY) * -1 AS QTY
        , IT.WORKORDER_BASE_ID
        , IT.WORKORDER_LOT_ID
        , IT.WORKORDER_SUB_ID
        , IT.WORKORDER_SPLIT_ID
        , IT.WORKORDER_TYPE
        , IT.OPERATION_SEQ_NO
        , IT.REQ_PIECE_NO
        , CASE
            WHEN LEFT(TP.TRACE_ID_LABEL, 5) = 'Cert#' THEN 'Cert#'
            ELSE TP.TRACE_ID_LABEL
        END + ': ' + TRT.TRACE_ID AS REQ_TRACE_ID
        , TRT.PART_ID
