# Customer_Receiving_FIN

```sql
SELECT GETDATE() AS THISDATE,
      W.STATUS,
      W.TYPE AS WORKORDER_TYPE,
      W.BASE_ID AS WORKORDER_BASE_ID,
      W.LOT_ID AS WORKORDER_LOT_ID,
      W.SPLIT_ID AS WORKORDER_SPLIT_ID,
      W.SUB_ID AS WORKORDER_SUB_ID,
      [dbo].[sfnWONUMFormat](w.BASE_ID, w.LOT_ID, w.SPLIT_ID, w.SUB_ID)    AS WONUM_ID,
      CONVERT(DATE,CO.DESIRED_SHIP_DATE) AS WANT_DATE,
      --convert(date,w.DESIRED_WANT_DATE) as WANT_DATE,
      W.DESIRED_QTY,
      W.PART_ID,
      P.DESCRIPTION,
      SA.MATERIAL_TYPE AS MATERIAL_TYPE,
      SA.ALLOY AS ALLOY,
      CAST(
        CAST(ISNULL(SA.THICKNESS_VAL, 0) AS    FLOAT) 
        AS VARCHAR(10))
        +' x '+
        CAST(
            CAST(ISNULL(SA.WIDTH_VAL, 0) AS FLOAT) 
        AS VARCHAR(10))+' x '+
        CAST(
            CAST(ISNULL(SA.LENGTH_VAL, 0) AS FLOAT) 
        AS VARCHAR(10)) AS PART_SIZE,
      [DBO].[UFN_GETFIRSTOPERATION](W.BASE_ID,W.LOT_ID,W.SUB_ID,W.SPLIT_ID,W.TYPE) AS SEQUENCE_NO,
--  w.FirstRES as RESOURCE_ID,
      DSL.DEMAND_BASE_ID,
      DSL.DEMAND_SEQ_NO,
      CO.CUSTOMER_ID,
       CASE WHEN CO.CUSTOMER_ID = 'BOE609'
                 THEN DSL.DEMAND_BASE_ID
                 ELSE CO.CUSTOMER_PO_REF
        END as CUSTOMER_PO_REF,
      CO.ORDER_DATE,
      CASE WHEN EXISTS(
            SELECT 1 FROM USER_DEF_FIELDS UDF
            WHERE UDF.DOCUMENT_ID = W.PART_ID
                AND UDF.PROGRAM_ID = 'VMPRTMNT'
                AND UDF.ID = 'UDF-0000040'
                AND BOOL_VAL = 1
                )
            THEN 'Yes'
            ELSE 'No'
            END AS ITAR_PART

FROM   WORK_ORDER W
      INNER JOIN PART P
        ON P.ID = W.PART_ID
     INNER JOIN DEMAND_SUPPLY_LINK DSL
        ON W.BASE_ID = DSL.SUPPLY_BASE_ID
           AND W.LOT_ID = DSL.SUPPLY_LOT_ID
           AND W.SPLIT_ID = DSL.SUPPLY_SPLIT_ID
           AND W.SUB_ID = DSL.SUPPLY_SUB_ID
      INNER JOIN CUST_ORDER_LINE AS COL
        ON DSL.DEMAND_BASE_ID = COL.CUST_ORDER_ID
           AND DSL.DEMAND_SEQ_NO = COL.LINE_NO
      INNER JOIN CUSTOMER_ORDER CO
        ON DSL.DEMAND_BASE_ID = CO.ID
      INNER JOIN CUSTOMER C
        ON CO.CUSTOMER_ID = C.ID
      LEFT OUTER JOIN SKILLS_WO_SURFACE_AREA AS SA
        ON W.BASE_ID = SA.BASE_ID
           AND W.LOT_ID = SA.LOT_ID
           AND W.SPLIT_ID = SA.SPLIT_ID
WHERE  DSL.DEMAND_BASE_ID = @order_id;
```
