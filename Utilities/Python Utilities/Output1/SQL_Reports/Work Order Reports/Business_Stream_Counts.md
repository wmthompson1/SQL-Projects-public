# Business_Stream_Counts

```sql
-- VISUAL ENTERPRISE > WORK_ORDER REPORTS > BUSINESS_STREAM_COUNTS
WITH CTE_SUPERVISOR AS 
(SELECT WO.BASE_ID
    , P.[NAME]
FROM WORK_ORDER AS WO
INNER JOIN LIVESUPPLEMENTAL.DBO.PLANNERS AS P
    ON SUBSTRING(WO.ENGINEERED_BY, CHARINDEX(P.FILTER_STRING, WO.ENGINEERED_BY), LEN(WO.ENGINEERED_BY)) = P.FILTER_STRING
WHERE WO.[STATUS] NOT IN ('X', 'C')
    AND WO.[TYPE] = 'W'
GROUP BY WO.BASE_ID
    , P.[NAME]
)

, CTE_BASE AS
(SELECT C.PART_ID
    , C.BASE_ID
    , C.LOT_ID
    , C.SUB_ID
    , C.SPLIT_ID
    , ISNULL(C.CUSTOMER_ID, 'NO CUSTOMER') AS CUSTOMER_ID
    , C.[NAME]
    , (SELECT SR.USER_1 AS SKILLS_TYPE
    FROM OPERATION O1
    INNER JOIN SHOP_RESOURCE SR
        ON SR.ID = O1.RESOURCE_ID
    WHERE O1.WORKORDER_TYPE = 'W'
        AND O1.WORKORDER_BASE_ID = C.BASE_ID
        AND O1.WORKORDER_LOT_ID = C.LOT_ID
        AND O1.WORKORDER_SPLIT_ID = C.SPLIT_ID
        AND O1.WORKORDER_SUB_ID = C.SUB_ID
        AND O1.SEQUENCE_NO = C.MIN_OP
    ) AS STARTLOC
    , C.MIN_OP
    , (SELECT SR.USER_1 AS SKILLS_TYPE
    FROM OPERATION O1
    INNER JOIN SHOP_RESOURCE SR
        ON SR.ID = O1.RESOURCE_ID
    WHERE O1.WORKORDER_TYPE = 'W'
        AND O1.WORKORDER_BASE_ID = C.BASE_ID
        AND O1.WORKORDER_LOT_ID = C.LOT_ID
        AND O1.WORKORDER_SPLIT_ID = C.SPLIT_ID
        AND O1.WORKORDER_SUB_ID = C.SUB_ID
        AND O1.SEQUENCE_NO = C.MAX_OP
    ) AS ENDLOC
    , C.MAX_OP
    , TOTAL
    , AP1COUNT
    , AP1MCOUNT
    , BP1COUNT
    , AP2COUNT
    , AP3COUNT
    , BSCOUNT
FROM
    (SELECT W.BASE_ID
        , W.LOT_ID
        , W.SPLIT_ID
        , W.SUB_ID
        , W.PART_ID
        , CO.CUSTOMER_ID
        , W.ENGINEERED_BY
    --    , CTE.PLANNING_SUPERVISOR
        , CTE.[NAME]
        , MIN(SEQUENCE_NO) AS MIN_OP
        , MAX(SEQUENCE_NO) AS MAX_OP
        , SUM(CASE
            WHEN SR.USER_5 IN ('AP1', 'BP1', 'AP2', 'AP3') THEN 1
            ELSE 0
        END) AS TOTAL
        , SUM(CASE
            WHEN SR.USER_5 = 'AP1' THEN 1
            ELSE 0
        END) AS AP1COUNT
        , SUM(CASE
            WHEN SR.USER_5 = 'AP1' AND SR.USER_1 = 'M' THEN 1
            ELSE 0
        END) AS AP1MCOUNT
        , SUM(CASE
            WHEN SR.USER_5 = 'BP1' THEN 1
            ELSE 0
        END) AS BP1COUNT
        , SUM(CASE
            WHEN SR.USER_5 = 'AP2' THEN 1
            ELSE 0
        END) AS AP2COUNT
        , SUM(CASE
            WHEN SR.USER_5 = 'AP3' THEN 1
            ELSE 0
        END) AS AP3COUNT
        , SUM(CASE
            WHEN SR.USER_1 = 'O' THEN 1
            ELSE 0
        END) AS BSCOUNT
    FROM WORK_ORDER AS W
    LEFT OUTER JOIN CTE_SUPERVISOR CTE
        ON W.BASE_ID = CTE.BASE_ID
    INNER JOIN OPERATION AS O
        ON W.[TYPE] = O.WORKORDER_TYPE
        AND W.BASE_ID = O.WORKORDER_BASE_ID
        AND W.LOT_ID = O.WORKORDER_LOT_ID
        AND W.SPLIT_ID = O.WORKORDER_SPLIT_ID
        AND W.SUB_ID = O.WORKORDER_SUB_ID
    INNER JOIN SHOP_RESOURCE AS SR
        ON SR.ID = O.RESOURCE_ID
    LEFT OUTER JOIN DEMAND_SUPPLY_LINK AS DSL
        ON W.BASE_ID = DSL.SUPPLY_BASE_ID
        AND W.LOT_ID = DSL.SUPPLY_LOT_ID
        AND W.SPLIT_ID = DSL.SUPPLY_SPLIT_ID
        AND W.SUB_ID = DSL.SUPPLY_SUB_ID
    LEFT OUTER JOIN CUST_ORDER_LINE AS COL
        ON DSL.DEMAND_BASE_ID = COL.CUST_ORDER_ID
        AND DSL.DEMAND_SEQ_NO = COL.LINE_NO
    LEFT OUTER JOIN CUSTOMER_ORDER AS CO
        ON COL.CUST_ORDER_ID = CO.ID
    WHERE W.[STATUS] = 'R'
        AND W.PART_ID IS NOT NULL
        AND W.PART_ID <> 'TOOLING'
        AND W.PART_ID NOT LIKE '%REWORK%'
        AND W.PART_ID NOT LIKE '%BOLTBOARD%'
    GROUP BY W.BASE_ID
        , W.LOT_ID
        , W.SPLIT_ID
        , W.SUB_ID
        , W.PART_ID
        , CO.CUSTOMER_ID
        , W.ENGINEERED_BY
        , CTE.[NAME]
    ) AS C
)

SELECT *
    , CASE
        WHEN STARTLOC = 'M'  THEN 'Shannen Helle'
        WHEN STARTLOC = 'O'  THEN 'Dmitriy Mandelman'
        ELSE 'Jeff Thiessen'
    END AS PLANNING_SUPERVISOR
FROM CTE_BASE 
WHERE CASE
        WHEN STARTLOC = 'M'  THEN 'Shannen Helle'
        WHEN STARTLOC = 'O'  THEN 'Dmitriy Mandelman'
        ELSE 'Jeff Thiessen'
    END IN (@SUPERVISOR_FILTER)
```
