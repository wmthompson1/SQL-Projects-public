# Work Order Costs

```sql
SELECT P.PART_ID
    , P.PLANNER_USER_ID
    , P.BUYER_USER_ID
    , P.UNIT_PRICE
    , WO.BASE_ID
    , WO.LOT_ID
    , WO.SPLIT_ID
    , WO.SUB_ID
    , WO.CLOSE_DATE
    , WO.DESIRED_QTY
    , WO.ACT_MATERIAL_COST
    , WO.ACT_MATERIAL_COST / WO.DESIRED_QTY  AS [MATERIAL UNIT COST]
    , WO.ACT_LABOR_COST
    , WO.ACT_LABOR_COST / WO.DESIRED_QTY AS [LABOR UNIT COST]
    , WO.ACT_BURDEN_COST
    , WO.ACT_BURDEN_COST / WO.DESIRED_QTY AS [BURDEN UNIT COST]
    , WO.ACT_SERVICE_COST
    , WO.ACT_SERVICE_COST / WO.DESIRED_QTY AS [SERVICE UNIT COST]
FROM WORK_ORDER WO
INNER JOIN PART_SITE P
    ON WO.PART_ID = P.PART_ID
WHERE WO.[TYPE] = 'W'
    AND WO.[STATUS] = 'C'
    AND WO.DESIRED_QTY > 0
    AND (WO.CLOSE_DATE >= @STARTDATE AND WO.CLOSE_DATE >= '1/1/2019')
    AND P.PLANNER_USER_ID IN (@PLANNER)
    AND P.BUYER_USER_ID IN (@BUYER)


```
