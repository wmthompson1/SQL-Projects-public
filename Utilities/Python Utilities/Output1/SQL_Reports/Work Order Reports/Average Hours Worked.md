# Average Hours Worked

```sql
--> VISUAL ENTERPRISE > WORK_ORDER > AVERAGE HOURS WORKED
SELECT PART_ID
    , P.[DESCRIPTION]
    , LT.RESOURCE_ID
    , SR.USER_6 AS PROCESS
    , SR.USER_7 AS SUBPROCESS
    , CASE
        WHEN LT.[TYPE] = 'S' THEN 'SETUP'
        ELSE 'RUN'
    END AS LT_TYPE
    , COUNT(WO.BASE_ID) AS WO_COUNT
    , AVG(WO.DESIRED_QTY) AS AVG_QTY
    , AVG(ISNULL(LT.HOURS_WORKED, 0)) AS AVG_HRS_WORKED
    , SUM(ISNULL(LT.HOURS_WORKED, 0)) / SUM(WO.DESIRED_QTY) AS AVG_HRS_PER_PIECE
FROM WORK_ORDER WO
LEFT OUTER JOIN OPERATION O
    ON WO.BASE_ID = O.WORKORDER_BASE_ID
    AND WO.LOT_ID = O.WORKORDER_LOT_ID
    AND WO.SPLIT_ID = O.WORKORDER_SPLIT_ID
    AND WO.SUB_ID = O.WORKORDER_SUB_ID
    AND WO.[TYPE] = O.WORKORDER_TYPE
LEFT OUTER JOIN LABOR_TICKET LT
    ON WO.BASE_ID = LT.WORKORDER_BASE_ID
    AND WO.LOT_ID = LT.WORKORDER_LOT_ID
    AND WO.SPLIT_ID = LT.WORKORDER_SPLIT_ID
    AND WO.SUB_ID = LT.WORKORDER_SUB_ID
    AND WO.[TYPE] = LT.WORKORDER_TYPE
    AND O.SEQUENCE_NO = LT.OPERATION_SEQ_NO
LEFT OUTER JOIN SHOP_RESOURCE SR
    ON LT.RESOURCE_ID = SR.ID
LEFT OUTER JOIN PART P
    ON WO.PART_ID = P.ID
WHERE ISNULL(SR.USER_6, '_BLANK') IN (@PROCESS)
    AND ISNULL(SR.USER_7, '_BLANK') IN (@SUBPROCESS)
    AND SR.ID IN (@RESOURCEID)
    AND ISNULL(LT.TRANSACTION_DATE, '1/1/2021') >= @STARTDATE
    AND ISNULL(LT.[TYPE], '_BLANK') IN (@LTTYPE)
    AND WO.PART_ID LIKE REPLACE((N'%' + ISNULL(@PARTID, '') + N'%'), '%%', '%')
GROUP BY WO.PART_ID
    , P.[DESCRIPTION]
    , LT.RESOURCE_ID
    , LT.[TYPE]
    , SR.USER_6
    , SR.USER_7


```
