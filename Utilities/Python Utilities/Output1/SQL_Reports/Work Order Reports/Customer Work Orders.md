# Customer Work Orders

```sql
SELECT  
CO.CUSTOMER_ID
, WO.BASE_ID AS [Work Order Number]
, WO.PART_ID AS [Part Number]
, WO.DESIRED_RLS_DATE AS [Release Date] 
, WO.DESIRED_WANT_DATE AS [Due Date]
, WO.DESIRED_QTY AS [Work Order QTY]
, CO.ID AS [Sales Order #] 
, COL.LINE_NO
, CO.ORDER_DATE
, S.PACKLIST_ID
, S.SHIPPED_DATE AS [Ship Date] 
FROM WORK_ORDER WO
INNER JOIN DEMAND_SUPPLY_LINK DSL
ON DSL.SUPPLY_BASE_ID = WO.BASE_ID
AND DSL.SUPPLY_LOT_ID = WO.LOT_ID
AND DSL.SUPPLY_SPLIT_ID = WO.SPLIT_ID
AND DSL.SUPPLY_SUB_ID = WO.SUB_ID
AND DSL.SUPPLY_TYPE = 'WO'
AND WO.TYPE = 'W'
INNER JOIN CUST_ORDER_LINE COL
ON DSL.DEMAND_BASE_ID = COL.CUST_ORDER_ID
AND DSL.DEMAND_SEQ_NO = COL.LINE_NO
AND DSL.DEMAND_TYPE = 'CO'
INNER JOIN CUSTOMER_ORDER CO
ON CO.ID = COL.CUST_ORDER_ID
LEFT OUTER JOIN SHIPPER_LINE SL
ON SL.CUST_ORDER_ID = COL.CUST_ORDER_ID
AND SL.CUST_ORDER_LINE_NO = COL.LINE_NO
LEFT OUTER JOIN SHIPPER S
ON S.PACKLIST_ID = SL.PACKLIST_ID
WHERE WO.DESIRED_RLS_DATE >= @STARTDATE
AND WO.DESIRED_RLS_DATE <= @ENDDATE
AND CO.CUSTOMER_ID IN (@CUSTID)
--AND WO.PART_ID IN (@PARTID)


```
