# Work Order Specs Text Search with dates

```sql
WITH OP_BINARY AS
(
    SELECT oB.WORKORDER_BASE_ID, oB.WORKORDER_LOT_ID, oB.WORKORDER_SPLIT_ID, oB.WORKORDER_SUB_ID, oB.WORKORDER_TYPE, oB.TYPE, oB.SEQUENCE_NO, oB.ROWID, oB.BITS_LENGTH
    , CONVERT(NVARCHAR(MAX),CONVERT(VARBINARY(MAX),oB.BITS)) AS SPECS
    FROM OPERATION_BINARY oB
    WHERE oB.WORKORDER_TYPE IN (@WO_Type)
        AND CONVERT(NVARCHAR(MAX), CONVERT(VARBINARY(MAX), oB.BITS)) LIKE @SPECS
        AND oB.BITS IS NOT NULL
        AND EXISTS(
            SELECT 1 
            FROM WORK_ORDER w 
            WHERE w.STATUS IN (@WO_Status)
                AND w.BASE_ID = oB.WORKORDER_BASE_ID
                AND w.LOT_ID = oB.WORKORDER_LOT_ID
                AND w.SUB_ID = oB.WORKORDER_SUB_ID
                AND w.SPLIT_ID = oB.WORKORDER_SPLIT_ID
                AND w.TYPE = oB.WORKORDER_TYPE
                AND w.PART_ID LIKE ISNULL(@part_id,'%')
                AND w.CREATE_DATE >= @startdate
                AND w.CREATE_DATE <= @enddate
        )
), WO_BINARY AS
(
    SELECT wB.WORKORDER_BASE_ID, wB.WORKORDER_LOT_ID, wB.WORKORDER_SPLIT_ID, wB.WORKORDER_SUB_ID, wB.WORKORDER_TYPE, wB.TYPE, wB.ROWID, wB.BITS_LENGTH
        , CONVERT(NVARCHAR(MAX),CONVERT(VARBINARY(MAX),wB.BITS)) AS SPECS 
    FROM WORKORDER_BINARY wB
    WHERE wB.WORKORDER_TYPE IN (@WO_Type)
        AND CONVERT(NVARCHAR(MAX), CONVERT(VARBINARY(MAX), wB.BITS)) LIKE @SPECS
        
        AND EXISTS (
            SELECT 1 
            FROM WORK_ORDER wO
            WHERE wO.PART_ID LIKE ISNULL(@part_id,'%')
                AND wO.STATUS IN (@WO_Status)
                AND wB.WORKORDER_BASE_ID = wO.BASE_ID
                AND wB.WORKORDER_LOT_ID = wO.LOT_ID
                AND wB.WORKORDER_SPLIT_ID = wO.SPLIT_ID
                AND wB.WORKORDER_SUB_ID = wO.SUB_ID
                AND Wb.WORKORDER_TYPE = wO.TYPE
                AND wO.CREATE_DATE >= @startdate
                AND wO.CREATE_DATE <= @enddate
            )
)

SELECT DISTINCT 
       'WORK ORDER SPECS' AS CAMEFROM,
       W.TYPE,
       W.STATUS,
       W.BASE_ID,
       W.LOT_ID,
       W.SUB_ID,
       W.SPLIT_ID,
       W.PART_ID,
       '' AS SEQUENCE_NO,
       '' AS RESOURCE_ID, 
       '' AS OPERATION_TYPE, 
       B2.SPECS
  FROM dbo.WORK_ORDER w 
  JOIN WO_BINARY B2
    ON W.TYPE            = B2.WORKORDER_TYPE 
   AND W.BASE_ID         = B2.WORKORDER_BASE_ID
   AND W.LOT_ID          = B2.WORKORDER_LOT_ID
   AND W.SPLIT_ID        = B2.WORKORDER_SPLIT_ID  
   AND W.SUB_ID = B2.WORKORDER_SUB_ID  
UNION
SELECT DISTINCT 
       'OPERATION SPECS' AS CAMEFROM,
       W.TYPE,
       W.STATUS,
       W.BASE_ID,
       W.LOT_ID,
       W.SUB_ID,
       W.SPLIT_ID,
       W.PART_ID,
       O.SEQUENCE_NO,
       O.RESOURCE_ID, 
       O.OPERATION_TYPE, 
       B.SPECS
  FROM dbo.WORK_ORDER w 
  INNER JOIN dbo.OPERATION o
     ON w.BASE_ID         = o.WORKORDER_BASE_ID
    AND w.LOT_ID          = o.WORKORDER_LOT_ID
    AND w.SPLIT_ID        = o.WORKORDER_SPLIT_ID
    AND w.SUB_ID          = o.WORKORDER_SUB_ID
    AND w.TYPE = o.WORKORDER_TYPE
  INNER JOIN OP_BINARY B
     ON B.WORKORDER_TYPE            = O.WORKORDER_TYPE 
    AND B.WORKORDER_BASE_ID         = o.WORKORDER_BASE_ID
    AND B.WORKORDER_LOT_ID          = o.WORKORDER_LOT_ID
    AND B.WORKORDER_SPLIT_ID        = o.WORKORDER_SPLIT_ID
    AND B.WORKORDER_SUB_ID          = o.WORKORDER_SUB_ID
    AND B.SEQUENCE_NO               = O.SEQUENCE_NO
UNION
SELECT DISTINCT 
       'WORK ORDER SPECS' AS CAMEFROM,
       W.TYPE,
       W.STATUS,
       W.BASE_ID,
       W.LOT_ID,
       W.SUB_ID,
       W.SPLIT_ID,
       CP.PART_ID as part_id,
       '' AS SEQUENCE_NO,
       '' AS RESOURCE_ID, 
       '' AS OPERATION_TYPE, 
       B2.SPECS
  FROM dbo.WORK_ORDER w 
  JOIN WO_BINARY B2
    ON W.TYPE            = B2.WORKORDER_TYPE 
   AND W.BASE_ID         = B2.WORKORDER_BASE_ID
   AND W.LOT_ID          = B2.WORKORDER_LOT_ID
   AND W.SUB_ID = B2.WORKORDER_SUB_ID
   AND W.SPLIT_ID        = B2.WORKORDER_SPLIT_ID    
  join dbo.CO_PRODUCT     cp
    ON W.TYPE            = cp.WORKORDER_TYPE 
   AND W.BASE_ID         = cp.WORKORDER_BASE_ID
   AND W.LOT_ID          = cp.WORKORDER_LOT_ID
   AND W.SPLIT_ID        = cp.WORKORDER_SPLIT_ID  
   AND W.SUB_ID          = CP.WORKORDER_SUB_ID
```
