# Corrective Action - 10-79 paint type

```sql
use DataWarehouse
go

select lt.TRANSACTION_DATE
, o.WORKORDER_BASE_ID
, o.WORKORDER_lot_ID
, o.WORKORDER_split_ID
, o.WORKORDER_SUB_ID
, lt.RESOURCE_ID
, sr.user_2
, o.OPERATION_TYPE
, CAST(CAST(OB.BITS AS VARBINARY(MAX)) AS NVARCHAR(MAX)) OPERATION_BINARY
, CAST(CAST(OTB.BITS AS VARBINARY(MAX)) AS NVARCHAR(MAX)) OPTYPE_BINARY
from operation o
left outer join LABOR_TICKET lt
	on o.WORKORDER_TYPE = lt.WORKORDER_TYPE
	and o.WORKORDER_BASE_ID = lt.WORKORDER_BASE_ID
	and o.WORKORDER_lot_ID = lt.WORKORDER_lot_ID
	and o.WORKORDER_split_ID = lt.WORKORDER_split_ID
	and o.WORKORDER_sub_ID = lt.WORKORDER_sub_ID
left outer join SHOP_RESOURCE sr
	on sr.id = lt.RESOURCE_ID
LEFT OUTER JOIN OPER_TYPE_BINARY OTB	
	ON o.OPERATION_TYPE = OTB.OPERATION_TYPE_ID
	AND OTB.SITE_ID = 'SK01'
	AND OTB.[TYPE] = 'D'
LEFT OUTER JOIN OPERATION_BINARY OB	
	ON O.WORKORDER_BASE_ID = OB.WORKORDER_BASE_ID
	AND O.WORKORDER_LOT_ID = OB.WORKORDER_LOT_ID
	AND O.WORKORDER_SPLIT_ID = OB.WORKORDER_SPLIT_ID
	AND O.WORKORDER_SUB_ID = OB.WORKORDER_SUB_ID
	AND o.SEQUENCE_NO = OB.SEQUENCE_NO
where lt.TRANSACTION_DATE >= '9/23/2019'
and lt.TRANSACTION_DATE < '9/24/2019'
and sr.user_2 in ('takedown', 'paint')

order by lt.TRANSACTION_DATE 
```
