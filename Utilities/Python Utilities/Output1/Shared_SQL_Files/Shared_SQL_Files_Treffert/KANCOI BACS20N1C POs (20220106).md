# KANCOI BACS20N1C POs (20220106)

```sql
USE LIVE
GO

SELECT PO.VENDOR_ID
, PO.ORDER_DATE
, POL.PURC_ORDER_ID
, POL.LINE_NO
, POL.PART_ID
, POL.VENDOR_PART_ID
, POL.ORDER_QTY
, POL.UNIT_PRICE
, POL.TOTAL_AMT_ORDERED
, POL.TOTAL_AMT_RECVD
FROM PURC_ORDER_LINE POL
INNER JOIN PURCHASE_ORDER PO
ON PO.ID = POL.PURC_ORDER_ID
WHERE POL.PART_ID = 'BACS20N1C'
AND PO.VENDOR_ID = 'KANCOI'
ORDER BY PO.ORDER_DATE
, POL.PURC_ORDER_ID
, POL.LINE_NO
```
