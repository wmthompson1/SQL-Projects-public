# Visual Barcoded Hours

```sql
SELECT 
	EE.DEPARTMENT_ID as Department, 
	EE.LAST_NAME, 
	EE.FIRST_NAME, 
	EE.ID, 
	sum(LT.HOURS_WORKED) as Hours_Worked, 
	ADP.[Job Title]
FROM   
	LABOR_TICKET LT 
LEFT OUTER JOIN 
	EMPLOYEE EE 
	ON LT.EMPLOYEE_ID=EE.ID
LEFT OUTER JOIN 
	LIVESupplemental.dbo.emp_ADP ADP
	ON EE.id = ADP.empID
WHERE  
	LT.TYPE <> 'I' 
	AND (LT.TRANSACTION_DATE >= @StartDate
	AND LT.TRANSACTION_DATE < @EndDate
group by 
	EE.DEPARTMENT_ID, 
	EE.LAST_NAME, 
	EE.FIRST_NAME, 
	EE.ID, 
	ADP.[Job Title]

ORDER BY 
	EE.ID, 
	EE.DEPARTMENT_ID, 
	EE.LAST_NAME, 
	EE.FIRST_NAME, 
	ADP.[Job Title]

```
