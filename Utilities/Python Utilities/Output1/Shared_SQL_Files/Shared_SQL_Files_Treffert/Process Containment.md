# Process Containment

```sql
USE Datamart
GO

DECLARE @BASE_ID NVARCHAR(15) = '1605191'
, @SEQNUM INT = 380
, @MINUTES INT = 10

; WITH CTE_INITIAL AS
(SELECT LT.WORKORDER_BASE_ID
	, LT.OPERATION_SEQ_NO
	, LT.RESOURCE_ID
	, CONVERT(DATE, LT.TRANSACTION_DATE) AS TRANSACTION_DATE
	, LT.CLOCK_IN
	, LT.CLOCK_OUT
	, LT.EMPLOYEE_ID
FROM DataWarehouse.dbo.LABOR_TICKET LT
WHERE LT.WORKORDER_BASE_ID = @BASE_ID
	AND LT.OPERATION_SEQ_NO = @SEQNUM
)
--SELECT * FROM CTE_INITIAL

SELECT LT.WORKORDER_BASE_ID
	, LT.WORKORDER_LOT_ID
	, LT.WORKORDER_SPLIT_ID
	, LT.WORKORDER_SUB_ID
	, LT.OPERATION_SEQ_NO
	, LT.RESOURCE_ID
	, SR.USER_6 AS DEPARTMENT
	, SR.USER_7 AS PROCESS
	, LT.CLOCK_IN
	, LT.CLOCK_OUT
	, LT.EMPLOYEE_ID
	, EE.[FIRST_NAME] + ' ' + EE.[LAST_NAME] AS EMPLOYEE_NAME
	, OP.OPERATION_TYPE
	, CAST(CAST(OB.BITS AS VARBINARY(MAX)) AS NVARCHAR(MAX)) OPERATION_BINARY
	, CAST(CAST(OTB.BITS AS VARBINARY(MAX)) AS NVARCHAR(MAX)) OPTYPE_BINARY
FROM DataWarehouse.dbo.LABOR_TICKET LT
INNER JOIN DataWarehouse.dbo.SHOP_RESOURCE SR
	ON LT.RESOURCE_ID = SR.ID
INNER JOIN CTE_INITIAL CTE
	ON CTE.EMPLOYEE_ID = LT.EMPLOYEE_ID
	AND CTE.TRANSACTION_DATE = CONVERT(DATE, LT.TRANSACTION_DATE)
	AND DATEADD(MINUTE, -@MINUTES, CTE.CLOCK_IN) <= LT.CLOCK_IN
	AND DATEADD(MINUTE, @MINUTES, CTE.CLOCK_IN) >= LT.CLOCK_IN
	AND DATEADD(MINUTE, -@MINUTES, CTE.CLOCK_OUT) <= LT.CLOCK_OUT
	AND DATEADD(MINUTE, @MINUTES, CTE.CLOCK_OUT) >= LT.CLOCK_OUT
INNER JOIN DataWarehouse.dbo.EMPLOYEE EE
	ON LT.EMPLOYEE_ID = EE.ID
INNER JOIN DataWarehouse.dbo.OPERATION OP
	ON LT.WORKORDER_BASE_ID = OP.WORKORDER_BASE_ID
	AND LT.WORKORDER_LOT_ID = OP.WORKORDER_LOT_ID
	AND LT.WORKORDER_SPLIT_ID = OP.WORKORDER_SPLIT_ID
	AND LT.WORKORDER_SUB_ID = OP.WORKORDER_SUB_ID
	AND LT.WORKORDER_TYPE = OP.WORKORDER_TYPE
	AND LT.OPERATION_SEQ_NO = OP.SEQUENCE_NO
INNER JOIN OPER_TYPE_BINARY OTB	
	ON OP.OPERATION_TYPE = OTB.OPERATION_TYPE_ID
	AND OTB.SITE_ID = 'SK01'
	AND OTB.[TYPE] = 'D'
INNER JOIN OPERATION_BINARY OB	
	ON OP.WORKORDER_BASE_ID = OB.WORKORDER_BASE_ID
	AND OP.WORKORDER_LOT_ID = OB.WORKORDER_LOT_ID
	AND OP.WORKORDER_SPLIT_ID = OB.WORKORDER_SPLIT_ID
	AND OP.WORKORDER_SUB_ID = OB.WORKORDER_SUB_ID
	AND OP.SEQUENCE_NO = OB.SEQUENCE_NO
```
