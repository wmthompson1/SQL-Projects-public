# UDF Listing

```sql
; WITH CTE_MENU AS
(SELECT PROGRAM_ID, ID, TAB_ID, [LABEL], SEQUENCE_NO, LINE_NO, STRING_VAL
FROM USER_DEF_FIELDS UDF
WHERE PROGRAM_ID != 'STATIC_DATA'
AND DOCUMENT_ID IS NULL
)
--SELECT * FROM CTE_MENU ORDER BY ID

, CTE_ITEMS AS
(SELECT 'ZZ' AS PROGRAM_ID, ID, NULL AS TAB_ID, NULL AS [LABEL], NULL AS SEQUENCE_NO, LINE_NO, STRING_VAL
FROM USER_DEF_FIELDS UDF
WHERE (1=1)
AND DOCUMENT_ID IS NULL
AND PROGRAM_ID = 'STATIC_DATA'
)

, CTE_EXP AS
(SELECT 'ZZ' AS PROGRAM_ID, 'UDF-0000070' AS ID, 'Skills' AS TAB_ID, NULL AS [LABEL], NULL AS SEQUENCE_NO, NULL AS LINE_NO, [ExpediteType] AS STRING_VAL
FROM [SK_ExpediteOptions] UDF
)

, CTE_PARTS AS
(SELECT 'ZZ' AS PROGRAM_ID, 'UDF-0000082' AS ID, 'SK-Purchasing' AS TAB_ID, NULL AS [LABEL], NULL AS SEQUENCE_NO, NULL AS LINE_NO, [CONTROLLED_LIST] AS STRING_VAL
FROM [SK_PART_CONTROLLED_LIST] UDF
)

, CTE_CERTS AS
(SELECT 'ZZ' AS PROGRAM_ID, 'UDF-0000084' AS ID, 'SK-Purchasing' AS TAB_ID, NULL AS [LABEL], ID AS SEQUENCE_NO, NULL AS LINE_NO, [Certifications] AS STRING_VAL
FROM [SK_PartCertificationsList] UDF
)

, CTE_VENDOR93 AS
(SELECT 'ZZ' AS PROGRAM_ID, 'UDF-0000093' AS ID, 'SK-Purchasing-ASL' AS TAB_ID, NULL AS [LABEL], ID AS SEQUENCE_NO, NULL AS LINE_NO, [VendorCertification] AS STRING_VAL
FROM [SK_VendorCertificationList] UDF
)

, CTE_VENDOR95 AS
(SELECT 'ZZ' AS PROGRAM_ID, 'UDF-0000095' AS ID, 'SK-Purchasing-ASL' AS TAB_ID, NULL AS [LABEL], ID AS SEQUENCE_NO, NULL AS LINE_NO, [VendorCertification] AS STRING_VAL
FROM [SK_VendorCertificationList] UDF
)

, CTE_VENDOR97 AS
(SELECT 'ZZ' AS PROGRAM_ID, 'UDF-0000097' AS ID, 'SK-Purchasing-ASL' AS TAB_ID, NULL AS [LABEL], ID AS SEQUENCE_NO, NULL AS LINE_NO, [VendorCertification] AS STRING_VAL
FROM [SK_VendorCertificationList] UDF
)


SELECT * FROM CTE_MENU
UNION ALL
SELECT * FROM CTE_ITEMS
UNION ALL
SELECT * FROM CTE_EXP
UNION ALL
SELECT * FROM CTE_PARTS
UNION ALL
SELECT * FROM CTE_CERTS
UNION ALL
SELECT * FROM CTE_VENDOR93
UNION ALL
SELECT * FROM CTE_VENDOR95
UNION ALL
SELECT * FROM CTE_VENDOR97
ORDER BY ID, PROGRAM_ID, SEQUENCE_NO, LINE_NO
```
