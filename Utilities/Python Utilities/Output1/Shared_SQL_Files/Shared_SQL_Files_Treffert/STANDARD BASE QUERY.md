# STANDARD BASE QUERY

```sql
SELECT *
FROM DataWarehouse.dbo.WORK_ORDER WO
INNER JOIN DataWarehouse.dbo.LABOR_TICKET LT
ON WO.BASE_ID = LT.WORKORDER_BASE_ID
AND WO.LOT_ID = LT.WORKORDER_LOT_ID
AND WO.SPLIT_ID = LT.WORKORDER_SPLIT_ID
AND WO.SUB_ID = LT.WORKORDER_SUB_ID
AND WO.[TYPE] = LT.WORKORDER_TYPE
INNER JOIN DataWarehouse.dbo.SHOP_RESOURCE_SITE SR
ON LT.RESOURCE_ID = SR.RESOURCE_ID
LEFT OUTER JOIN DataWarehouse.dbo.DEMAND_SUPPLY_LINK DSL
ON WO.BASE_ID = DSL.SUPPLY_BASE_ID
AND WO.LOT_ID = DSL.SUPPLY_LOT_ID
AND WO.SPLIT_ID = DSL.SUPPLY_SPLIT_ID
AND WO.SUB_ID = DSL.SUPPLY_SUB_ID
LEFT OUTER JOIN DataWarehouse.dbo.CUST_ORDER_LINE COL
ON COL.CUST_ORDER_ID = DSL.DEMAND_BASE_ID
AND COL.LINE_NO = DSL.DEMAND_SEQ_NO
LEFT OUTER JOIN DataWarehouse.dbo.CUSTOMER_ORDER CO
ON COL.CUST_ORDER_ID = CO.ID
WHERE LT.TRANSACTION_DATE >= GETDATE() - 365
```
