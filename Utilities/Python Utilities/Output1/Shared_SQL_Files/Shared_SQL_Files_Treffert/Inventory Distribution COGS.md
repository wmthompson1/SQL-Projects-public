# Inventory Distribution COGS

```sql
USE LIVE
GO

DECLARE @STARTDATE DATE = '2/1/2020'
, @ENDDATE DATE = '3/1/2020'
, @REPORT NVARCHAR(10) = 'BOTH'
, @TYPE NVARCHAR(10) = 'BOTH'

SELECT SR.USER_1 AS SKILLS_TYPE
	, 'BURDEN' AS MSG
	, CASE 
		WHEN ID.DEMAND_SUPPLY_ID IS NULL THEN 'UNLINKED'
		WHEN ID.DEMAND_SUPPLY_ID IS NOT NULL THEN 'LINKED'
		ELSE 'BOTH'
	END AS [ORDER]
	, SUM(CASE WHEN C.USER_9 = '2' THEN (O.ACT_ATL_BUR_COST/S.RECEIVED_QTY) * ID.DIST_QTY ELSE 0 END) AS '2'
	, SUM(CASE WHEN C.USER_9 = '3' THEN (O.ACT_ATL_BUR_COST/S.RECEIVED_QTY) * ID.DIST_QTY ELSE 0 END) AS '3'
	, SUM(CASE WHEN C.USER_9 = '4' THEN (O.ACT_ATL_BUR_COST/S.RECEIVED_QTY) * ID.DIST_QTY ELSE 0 END) AS '4'
FROM LIVE.dbo.INVENTORY_TRANS IT 
INNER JOIN LIVE.dbo.INV_TRANS_DIST ID
	ON IT.TRANSACTION_ID = ID.OUT_TRANS_ID 
INNER JOIN LIVE.dbo.CUSTOMER_ORDER CO
	ON IT.CUST_ORDER_ID = CO.ID
INNER JOIN LIVE.dbo.CUSTOMER C
	ON CO.CUSTOMER_ID = C.ID
INNER JOIN LIVE.dbo.INVENTORY_TRANS I1
	ON ID.IN_TRANS_ID = I1.TRANSACTION_ID
 LEFT OUTER JOIN LIVE.dbo.OPERATION O
	ON I1.WORKORDER_TYPE = O.WORKORDER_TYPE
	AND I1.WORKORDER_BASE_ID = O.WORKORDER_BASE_ID
	AND I1.WORKORDER_LOT_ID = O.WORKORDER_LOT_ID
	AND I1.WORKORDER_SPLIT_ID = O.WORKORDER_SPLIT_ID
 LEFT OUTER JOIN LIVE.dbo.SHOP_RESOURCE SR
	ON O.RESOURCE_ID = SR.ID
 LEFT OUTER JOIN LIVE.dbo.SKILLS_WO_REC_QTY S
	ON O.WORKORDER_BASE_ID = S.BASE_ID
	AND O.WORKORDER_LOT_ID = S.LOT_ID
	AND O.WORKORDER_SPLIT_ID = S.SPLIT_ID
WHERE IT.CUST_ORDER_ID IS NOT NULL 
	AND (IT.TRANSACTION_DATE >= @STARTDATE
		AND IT.TRANSACTION_DATE < @ENDDATE)
	AND S.RECEIVED_QTY > 0
	AND IT.CLASS = N'I'
	AND IT.[TYPE] = N'O'
	AND @REPORT != 'LABOR' 
	AND CASE 
		WHEN @TYPE = 'UNLINKED' AND ID.DEMAND_SUPPLY_ID IS NULL THEN 'OK'
		WHEN @TYPE = 'LINKED' AND ID.DEMAND_SUPPLY_ID IS NOT NULL THEN 'OK'
		WHEN @TYPE = 'BOTH' THEN 'OK'
		ELSE 'NOPE'
	END = 'OK'
GROUP BY SR.USER_1, CASE 
		WHEN ID.DEMAND_SUPPLY_ID IS NULL THEN 'UNLINKED'
		WHEN ID.DEMAND_SUPPLY_ID IS NOT NULL THEN 'LINKED'
		ELSE 'BOTH'
	END
UNION ALL
SELECT SR.USER_1 AS SKILLS_TYPE
	, 'LABOR' AS MSG
	, CASE 
		WHEN ID.DEMAND_SUPPLY_ID IS NULL THEN 'UNLINKED'
		WHEN ID.DEMAND_SUPPLY_ID IS NOT NULL THEN 'LINKED'
		ELSE 'BOTH'
	END AS [ORDER]
	, SUM(CASE WHEN C.USER_9 = '2' THEN (O.ACT_ATL_LAB_COST/S.RECEIVED_QTY) * ID.DIST_QTY ELSE 0 END) AS '2'
	, SUM(CASE WHEN C.USER_9 = '3' THEN (O.ACT_ATL_LAB_COST/S.RECEIVED_QTY) * ID.DIST_QTY ELSE 0 END) AS '3'
	, SUM(CASE WHEN C.USER_9 = '4' THEN (O.ACT_ATL_LAB_COST/S.RECEIVED_QTY) * ID.DIST_QTY ELSE 0 END) AS '4'
FROM LIVE.dbo.INVENTORY_TRANS IT 
INNER JOIN LIVE.dbo.INV_TRANS_DIST ID
	ON IT.TRANSACTION_ID = ID.OUT_TRANS_ID 
INNER JOIN LIVE.dbo.CUSTOMER_ORDER CO
	ON IT.CUST_ORDER_ID = CO.ID
INNER JOIN LIVE.dbo.CUSTOMER C
	ON CO.CUSTOMER_ID = C.ID
INNER JOIN LIVE.dbo.INVENTORY_TRANS I1
	ON ID.IN_TRANS_ID = I1.TRANSACTION_ID
 LEFT OUTER JOIN LIVE.dbo.OPERATION O
	ON I1.WORKORDER_TYPE = O.WORKORDER_TYPE
	AND I1.WORKORDER_BASE_ID = O.WORKORDER_BASE_ID
	AND I1.WORKORDER_LOT_ID = O.WORKORDER_LOT_ID
	AND I1.WORKORDER_SPLIT_ID = O.WORKORDER_SPLIT_ID
--	AND I1.WORKORDER_SUB_ID = O.WORKORDER_SUB_ID
 LEFT OUTER JOIN LIVE.dbo.SHOP_RESOURCE SR
	ON O.RESOURCE_ID = SR.ID
 LEFT OUTER JOIN LIVE.dbo.SKILLS_WO_REC_QTY S
	ON O.WORKORDER_BASE_ID = S.BASE_ID
	AND O.WORKORDER_LOT_ID = S.LOT_ID
	AND O.WORKORDER_SPLIT_ID = S.SPLIT_ID
WHERE IT.CUST_ORDER_ID IS NOT NULL 
	AND (IT.TRANSACTION_DATE >= @STARTDATE
		AND IT.TRANSACTION_DATE < @ENDDATE)
	AND S.RECEIVED_QTY > 0
	AND IT.CLASS = N'I'
	AND IT.[TYPE] = N'O'
	AND @REPORT != 'BURDEN' 
	AND CASE 
		WHEN @TYPE = 'UNLINKED' AND ID.DEMAND_SUPPLY_ID IS NULL THEN 'OK'
		WHEN @TYPE = 'LINKED' AND ID.DEMAND_SUPPLY_ID IS NOT NULL THEN 'OK'
		WHEN @TYPE = 'BOTH' THEN 'OK'
		ELSE 'NOPE'
	END = 'OK'
GROUP BY SR.USER_1, CASE 
		WHEN ID.DEMAND_SUPPLY_ID IS NULL THEN 'UNLINKED'
		WHEN ID.DEMAND_SUPPLY_ID IS NOT NULL THEN 'LINKED'
		ELSE 'BOTH'
	END

```
