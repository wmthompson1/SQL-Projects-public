# Cust Orders and Shipments (update 20181012)

```sql
USE Datamart
GO

SELECT CUSTID
		, COL.CUST_ORDER_ID
		, COL.LINE_NO
		, COL.PART_ID
		, WO.BASE_ID
		, WO.LOT_ID
		, WO.SPLIT_ID
		, WO.SUB_ID
		, 
FROM
	(SELECT CO.CUSTOMER_ID AS CUSTID
		, COL.CUST_ORDER_ID
		, COL.LINE_NO
		, COL.PART_ID
		, WO.BASE_ID
		, WO.LOT_ID
		, WO.SPLIT_ID
		, WO.SUB_ID
		, O.SEQUENCE_NO
		, O.RESOURCE_ID
		, CASE
			WHEN O.RESOURCE_ID
		SR.USER_1 AS [LOCATION]
	--	, COUNT(*) AS LOC_TOTAL
	FROM CUST_ORDER_LINE COL 
	INNER JOIN CUSTOMER_ORDER CO 
		ON COL.CUST_ORDER_ID = CO.ID 
	LEFT OUTER JOIN DEMAND_SUPPLY_LINK DSL 
		ON COL.CUST_ORDER_ID = DSL.DEMAND_BASE_ID 
		AND COL.LINE_NO = DSL.DEMAND_SEQ_NO 
		AND DSL.DEMAND_TYPE = 'CO' 
	LEFT OUTER JOIN WORK_ORDER WO 
		ON DSL.SUPPLY_BASE_ID = WO.BASE_ID 
		AND DSL.SUPPLY_LOT_ID = WO.LOT_ID 
		AND DSL.SUPPLY_SPLIT_ID = WO.SPLIT_ID 
		AND DSL.SUPPLY_SUB_ID = WO.SUB_ID 
		AND DSL.SUPPLY_TYPE IN ('WO', 'CP') 
	LEFT OUTER JOIN OPERATION O 
		ON O.WORKORDER_TYPE = WO.[TYPE]
		AND WO.BASE_ID = O.WORKORDER_BASE_ID 
		AND WO.LOT_ID = O.WORKORDER_LOT_ID 
		AND O.WORKORDER_SPLIT_ID = '0' 
		AND O.WORKORDER_SUB_ID = '0' 
	LEFT OUTER JOIN SHOP_RESOURCE_SITE_VIEW SR 
		ON O.RESOURCE_ID = SR.RESOURCE_ID
	WHERE COL.PART_ID NOT LIKE 'UP%' 
		AND COALESCE(COL.DESIRED_SHIP_DATE, CO.DESIRED_SHIP_DATE) >= '1/1/2018'
	) AS A
```
