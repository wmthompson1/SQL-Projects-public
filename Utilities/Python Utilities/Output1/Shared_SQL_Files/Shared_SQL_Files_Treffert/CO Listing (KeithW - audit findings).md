# CO Listing (KeithW - audit findings)

```sql
USE LIVE
GO

SELECT C.[NAME] AS CUSTOMER_NAME
, C.ID AS CUSTOMER_ID
, C.ACTIVE_FLAG
, UDF.STRING_VAL AS CUSTOMER_TYPE
, CO.ORDER_DATE 
, CO.ID AS CUST_ORDER
, COL.LINE_NO AS CO_LINE
, COL.PART_ID
, P.PLANNER_USER_ID
FROM CUSTOMER_ORDER CO
INNER JOIN CUST_ORDER_LINE COL
ON CO.ID = COL.CUST_ORDER_ID
INNER JOIN CUSTOMER C
ON CO.CUSTOMER_ID = C.ID
LEFT OUTER JOIN USER_DEF_FIELDS UDF
ON C.ID = UDF.DOCUMENT_ID
AND UDF.ID = 'UDF-0000072'
AND UDF.PROGRAM_ID = 'VMCUSMNT'
LEFT OUTER JOIN PART P
ON COL.PART_ID = P.ID
WHERE CO.ORDER_DATE >= '1/1/2019'
AND COL.PART_ID IS NOT NULL
AND (P.PLANNER_USER_ID LIKE 'FIN%'
	OR UDF.STRING_VAL = 'FIN'
	)
ORDER BY CO.ORDER_DATE ASC

```
