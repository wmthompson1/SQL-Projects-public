USE DATAWAREHOUSE
GO

WITH CTE_LINKED AS
(SELECT CO.CUSTOMER_ID
	, CO.CUSTOMER_PO_REF
	, COL.CUST_ORDER_ID
	, COL.LINE_NO
	, COL.PART_ID
	, T.TRACE_ID
	, T.QTY * -1 AS TRACE_QTY
	, S.SHIPPED_DATE
	, SL.PACKLIST_ID
	, COL.TOTAL_SHIPPED_QTY
	, SL.SHIPPED_QTY
	, WO.[STATUS]
	, WO.CREATE_DATE AS WO_CREATE_DATE
	, WO.BASE_ID
	, WO.LOT_ID
	, WO.SPLIT_ID
	, WO.SUB_ID
	, WO.DESIRED_QTY
FROM SHIPPER_LINE SL
INNER JOIN SHIPPER S
	ON SL.PACKLIST_ID = S.PACKLIST_ID 
INNER JOIN CUST_ORDER_LINE COL
	ON SL.CUST_ORDER_ID = COL.CUST_ORDER_ID
	AND SL.CUST_ORDER_LINE_NO = COL.LINE_NO
INNER JOIN CUSTOMER_ORDER CO
	ON COL.CUST_ORDER_ID = CO.ID
INNER JOIN DEMAND_SUPPLY_LINK DSL
	ON DSL.DEMAND_TYPE = 'CO'
	AND DSL.DEMAND_BASE_ID = COL.CUST_ORDER_ID
	AND DSL.DEMAND_SEQ_NO = COL.LINE_NO
INNER JOIN WORK_ORDER WO
    ON DSL.SUPPLY_TYPE = 'WO' 
    AND DSL.SUPPLY_BASE_ID = WO.BASE_ID
    AND DSL.SUPPLY_LOT_ID = WO.LOT_ID
    AND DSL.SUPPLY_SPLIT_ID = WO.SPLIT_ID
    AND DSL.SUPPLY_SUB_ID = WO.SUB_ID 
	AND WO.[TYPE] = 'W'
LEFT OUTER JOIN TRACE_INV_TRANS T
	ON SL.TRANSACTION_ID = T.TRANSACTION_ID
WHERE SL.PACKLIST_ID IS NOT NULL
	AND SL.SHIPPED_QTY <> 0
	AND WO.PART_ID IN ('475Z3619-131', '475Z3619-132', '475Z3619-133', '475Z3619-135', '475Z3619-136', '475Z3619-137')
)

, CTE_UNLINKED AS

	, COL.LINE_NO
	, COL.PART_ID
	, T.TRACE_ID
	, T.QTY * -1 AS TRACE_QTY
	, S.SHIPPED_DATE
	, SL.PACKLIST_ID
	, COL.TOTAL_SHIPPED_QTY
	, SL.SHIPPED_QTY
	, WO.[STATUS]
	, WO.CREATE_DATE AS WO_CREATE_DATE
	, WO.BASE_ID
	, WO.LOT_ID
	, WO.SPLIT_ID
	, WO.SUB_ID
	, WO.DESIRED_QTY
FROM SHIPPER_LINE SL
INNER JOIN TRACE_INV_TRANS  T
	ON SL.TRANSACTION_ID = T.TRANSACTION_ID
INNER JOIN CUST_ORDER_LINE COL
	ON SL.CUST_ORDER_ID = COL.CUST_ORDER_ID
	AND SL.CUST_ORDER_LINE_NO = COL.LINE_NO
	AND COL.PART_ID IS NOT NULL
LEFT OUTER JOIN SHIPPER S
	ON SL.PACKLIST_ID = S.PACKLIST_ID 
LEFT OUTER JOIN CUSTOMER_ORDER CO
	ON COL.CUST_ORDER_ID = CO.ID
INNER JOIN WORK_ORDER WO
	ON (T.TRACE_ID = WO.BASE_ID + '/' + WO.LOT_ID)
	AND WO.DESIRED_QTY <> 0
	AND WO.SPLIT_ID = '0'
	AND WO.SUB_ID = '0'
	AND WO.[TYPE] = 'W'
	AND WO.[STATUS] = 'C'
LEFT OUTER JOIN DEMAND_SUPPLY_LINK DSL
	ON DSL.DEMAND_TYPE = 'CO'
	AND DSL.DEMAND_BASE_ID = COL.CUST_ORDER_ID
	AND DSL.DEMAND_SEQ_NO = COL.LINE_NO
	AND DSL.SUPPLY_BASE_ID IS NULL
WHERE WO.PART_ID IN ('475Z3619-131', '475Z3619-132', '475Z3619-133', '475Z3619-135', '475Z3619-136', '475Z3619-137')
	AND SL.SHIPPED_QTY <> 0
