# Part_ID Containment

```sql
USE DATAWAREHOUSE
GO

DECLARE @STARTDATE DATE = GETDATE()-30
, @PART_ID NVARCHAR(50) = '140T2502-11'
, @NCM NVARCHAR(10) = 'ONLY'
;

--> QUALITY > CONTAINMENT REPORTS > PART_ID CONTAINMENT
-- NCM#, PT#, WO#, QTY Rej, Process, Sub Process, Defect, Category, 

SELECT WO.PART_ID
	, WO.CLOSE_DATE
	, WO.[STATUS] 
	, WO.BASE_ID
	, WO.LOT_ID
	, WO.SPLIT_ID
	, WO.SUB_ID
	, WO.DESIRED_QTY
	, NCM.NCM_ID
	, NCM.NCM_DATE
	, NCM.ACCEPTED
	, NCM.REJECTED
	, ISNULL(SK_MFG_NCM_PROCESS, SK_FIN_NCM_PROCESS) AS [PROCESS]
	, ISNULL(SK_MFG_NCM_SUBPROCESS, SK_FIN_NCM_SUBPROCESS) AS [SUBPROCESS]
	, ISNULL(NCM.USER_DEFINED_4, NCM.USER_DEFINED_7) AS [DEFECT]
	, NCM.USER_DEFINED_3 AS [CATEGORY]
FROM DATAWAREHOUSE.DBO.WORK_ORDER WO
LEFT OUTER JOIN DATAWAREHOUSE.DBO.VQ_PRODUCTION_ORDER PO
	ON PO.[ALT_ID_2] = WO.BASE_ID
	AND PO.[ALT_ID_3] = WO.LOT_ID
	AND PO.[ALT_ID_4] = WO.SPLIT_ID
	AND PO.[ALT_ID_5] = WO.SUB_ID
	AND PO.[ALT_ID_1] = WO.[TYPE]
LEFT OUTER JOIN DATAWAREHOUSE.DBO.VQ_NONCONFORMANCE NCM
	ON PO.RECORD_ID = NCM.PROD_ORDER_REC_ID
WHERE WO.PART_ID IN (@PART_ID)
	AND WO.[STATUS] != 'X'
	AND ((@NCM = 'ONLY' AND NCM_ID IS NOT NULL)
		OR (@NCM != 'ONLY'
			AND EXISTS (SELECT 1 FROM DATAWAREHOUSE.DBO.OPERATION OP
				WHERE OP.CLOSE_DATE >= @STARTDATE
					AND OP.WORKORDER_BASE_ID = WO.BASE_ID
					AND OP.WORKORDER_LOT_ID = WO.LOT_ID
					AND OP.WORKORDER_SPLIT_ID = WO.SPLIT_ID
					AND OP.WORKORDER_SUB_ID = WO.SUB_ID
					AND OP.WORKORDER_TYPE = WO.[TYPE]
			)
		)
	)
ORDER BY WO.[STATUS] 
	, WO.BASE_ID
	, WO.LOT_ID
	, WO.SPLIT_ID
	, WO.SUB_ID
```
