use DataWarehouse

select CUSTOMER_ID
, CUSTOMER_PO_REF
, CUST_ORDER_ID
, LINE_NO
, CREATE_DATE
, ORDER_DATE
, isnull(col.DESIRED_SHIP_DATE, co.DESIRED_SHIP_DATE) as DESIRED_SHIP_DATE
, [STATUS]
, co.TOTAL_AMT_ORDERED
, co.TOTAL_AMT_SHIPPED
, PART_ID
, CUSTOMER_PART_ID
, MISC_REFERENCE
, ORDER_QTY
, SELLING_UM
, UNIT_PRICE
, TOTAL_SHIPPED_QTY
, TOTAL_USR_SHIP_QTY
, col.TOTAL_AMT_SHIPPED as LINE_TOTAL_SHIPPED
, col.TOTAL_AMT_ORDERED AS LINE_TOTAL_ORDERED
, SERVICE_CHARGE_ID
