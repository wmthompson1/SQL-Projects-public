# Part Warehouse Location (Luke)

```sql
use DataWarehouse
go

SELECT PL.PART_ID
	, PL.WAREHOUSE_ID
	, PL.LOCATION_ID
	, PL.QTY
	, CCP.LAST_COUNT_DATE
	, PS.UNIT_MATERIAL_COST
	, PS.UNIT_LABOR_COST
	, PS.BURDEN_PER_UNIT
	, PS.UNIT_SERVICE_COST
	, PL.QTY * (PS.UNIT_MATERIAL_COST + PS.UNIT_LABOR_COST + PS.BURDEN_PER_UNIT + PS.UNIT_SERVICE_COST) TOTAL_COST
FROM PART_SITE PS
INNER JOIN PART_LOCATION PL
	ON PS.PART_ID = PL.PART_ID
LEFT OUTER JOIN CYCLE_COUNT_PART CCP
	ON PL.WAREHOUSE_ID = CCP.WAREHOUSE_ID
	AND PL.PART_ID = CCP.PART_ID
where LEFT(PL.WAREHOUSE_ID , 7) <> 'BALLARD'
		AND (PL.WAREHOUSE_ID <> 'FGA' AND PL.LOCATION_ID <> 'FGA')
		AND (PL.WAREHOUSE_ID <> 'FGA' AND PL.LOCATION_ID <> 'EBAY')
		AND (PL.WAREHOUSE_ID <> 'FGA' AND PL.LOCATION_ID <> 'CFM')
		AND (PL.WAREHOUSE_ID <> 'WIP' AND PL.LOCATION_ID <> 'WIP')
		AND (PL.WAREHOUSE_ID <> 'SLLIKS' AND PL.LOCATION_ID <> 'BALLARD')
		AND (PL.WAREHOUSE_ID <> 'SLLIKS' AND PL.LOCATION_ID <> 'STORE')
		AND (PL.WAREHOUSE_ID <> 'Auburn Paint' AND PL.LOCATION_ID <> 'East Locker')
		AND (PL.WAREHOUSE_ID <> 'Arlington' AND PL.LOCATION_ID <> 'Overrun')
		AND (PL.WAREHOUSE_ID <> 'Arlington' AND PL.LOCATION_ID <> 'PLANT 1')
		AND (PL.WAREHOUSE_ID <> 'Auburn' AND PL.LOCATION_ID <> 'EBAY')
		AND (PL.WAREHOUSE_ID <> '1' AND PL.LOCATION_ID <> 'SHIPPING')
ORDER BY PL.PART_ID
	, PL.LOCATION_ID



```
