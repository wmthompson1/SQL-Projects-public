USE DATAWAREHOUSE
GO

WITH CTE_MAXLTS AS
(SELECT O.WORKORDER_TYPE
	, O.WORKORDER_BASE_ID
	, O.WORKORDER_LOT_ID
	, O.WORKORDER_SPLIT_ID
	, O.WORKORDER_SUB_ID
	, MAX(LT.OPERATION_SEQ_NO) AS MAXOPSEQ
FROM OPERATION O
	INNER JOIN WORK_ORDER WO
		ON WO.BASE_ID = O.WORKORDER_BASE_ID
		AND WO.LOT_ID = O.WORKORDER_LOT_ID
		AND WO.SPLIT_ID = O.WORKORDER_SPLIT_ID
		AND WO.SUB_ID = O.WORKORDER_SUB_ID
		AND WO.[TYPE] = O.WORKORDER_TYPE
	INNER JOIN LABOR_TICKET LT
		ON O.WORKORDER_BASE_ID = LT.WORKORDER_BASE_ID
		AND O.WORKORDER_LOT_ID = LT.WORKORDER_LOT_ID
		AND O.WORKORDER_SPLIT_ID = LT.WORKORDER_SPLIT_ID
		AND O.WORKORDER_SUB_ID = LT.WORKORDER_SUB_ID
		AND O.WORKORDER_TYPE = LT.WORKORDER_TYPE
		AND LT.OPERATION_SEQ_NO = O.SEQUENCE_NO
WHERE WO.DESIRED_RLS_DATE >= '1/1/2018'
	AND WO.CREATE_DATE >= '1/1/2018'
	AND O.WORKORDER_TYPE = 'W'
	AND WO.[STATUS] IN ('U', 'F', 'R', 'X')
GROUP BY O.WORKORDER_TYPE
	, O.WORKORDER_BASE_ID
	, O.WORKORDER_LOT_ID
	, O.WORKORDER_SPLIT_ID
	, O.WORKORDER_SUB_ID
)
--SELECT * FROM CTE_MAXLTS



SELECT O.WORKORDER_TYPE
	, O.WORKORDER_BASE_ID
	, O.WORKORDER_LOT_ID
	, O.WORKORDER_SPLIT_ID
	, O.WORKORDER_SUB_ID
	, O.SEQUENCE_NO
	, O.RESOURCE_ID
	, O.[STATUS] AS OSTATUS
	, O.CLOSE_DATE
	, O.OPERATION_TYPE
	, SR.[USER_1] AS SK_TYPE
	, SR.[USER_2] AS GROUP_PRI
	, SR.[USER_3] AS GROUP_SEC
	, SR.[USER_4] AS RES_SUPR
	, SR.[USER_5] AS PLANT_ID
	, WO.PART_ID
	, WO.DESIRED_QTY
	, WO.CREATE_DATE
	, WO.DESIRED_RLS_DATE
	, WO.DESIRED_WANT_DATE
	, WO.CLOSE_DATE
	, WO.[STATUS] AS WOSTATUS
	, LT.TRANSACTION_ID
	, LT.[TYPE] AS LTTYPE
	, LT.SETUP_COMPLETED
	, LT.EMPLOYEE_ID
	, LT.DEPARTMENT_ID
	, LT.RESOURCE_ID
	, LT.CLOCK_IN
	, LT.CLOCK_OUT
	, LT.HOURS_WORKED
	, LT.GOOD_QTY
	, LT.BAD_QTY
	, LT.TRANSACTION_DATE
	, LT.[DESCRIPTION]
	, ADP.[First Name] AS FNAME
	, ADP.[Last Name] AS LNAME
	, ADP.[Reports To] AS REPORTS_TO
	, ADP.[LOCATION] AS ADP_LOC
	, ADP.[Shift] AS ADP_SHIFT
FROM OPERATION O
	INNER JOIN SHOP_RESOURCE SR
		ON O.RESOURCE_ID = SR.ID
	INNER JOIN WORK_ORDER WO
		ON WO.BASE_ID = O.WORKORDER_BASE_ID
		AND WO.LOT_ID = O.WORKORDER_LOT_ID
		AND WO.SPLIT_ID = O.WORKORDER_SPLIT_ID
		AND WO.SUB_ID = O.WORKORDER_SUB_ID
		AND WO.[TYPE] = O.WORKORDER_TYPE
	LEFT OUTER JOIN LABOR_TICKET LT
		ON O.WORKORDER_BASE_ID = LT.WORKORDER_BASE_ID
		AND O.WORKORDER_LOT_ID = LT.WORKORDER_LOT_ID
		AND O.WORKORDER_SPLIT_ID = LT.WORKORDER_SPLIT_ID
		AND O.WORKORDER_SUB_ID = LT.WORKORDER_SUB_ID
		AND O.WORKORDER_TYPE = LT.WORKORDER_TYPE
		AND LT.OPERATION_SEQ_NO = O.SEQUENCE_NO
	LEFT OUTER JOIN CTE_MAXLTS MLT
		ON  O.WORKORDER_BASE_ID = LT.WORKORDER_BASE_ID
		AND O.WORKORDER_LOT_ID = LT.WORKORDER_LOT_ID
		AND O.WORKORDER_SPLIT_ID = LT.WORKORDER_SPLIT_ID
		AND O.WORKORDER_SUB_ID = LT.WORKORDER_SUB_ID
		AND O.WORKORDER_TYPE = LT.WORKORDER_TYPE
		AND LT.OPERATION_SEQ_NO = O.SEQUENCE_NO
	LEFT OUTER JOIN LIVESupplemental.DBO.emp_ADP ADP
		ON LT.EMPLOYEE_ID = ADP.empID
WHERE WO.DESIRED_RLS_DATE >= '1/1/2018'
	AND WO.CREATE_DATE >= '1/1/2018'
	AND O.WORKORDER_TYPE = 'W'
	AND WO.[STATUS] NOT IN ('U', 'F')
ORDER BY O.WORKORDER_TYPE
	, O.WORKORDER_BASE_ID
	, O.WORKORDER_LOT_ID
	, O.WORKORDER_SPLIT_ID
	, O.WORKORDER_SUB_ID
	, O.SEQUENCE_NO
	, O.RESOURCE_ID
	, LT.TRANSACTION_DATE
