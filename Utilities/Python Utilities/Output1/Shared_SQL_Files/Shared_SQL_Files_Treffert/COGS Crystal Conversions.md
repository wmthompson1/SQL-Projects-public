# COGS Crystal Conversions

```sql
USE LIVE
GO

--> MATCHES CRYSTAL REPORT
SELECT IT.TRANSACTION_ID
, IT.[TYPE]
, IT.CLASS
, ID.IN_TRANS_ID
, IT.WORKORDER_BASE_ID
, ID.DIST_QTY
, O.SEQUENCE_NO
, O.RESOURCE_ID
, O.ACT_ATL_LAB_COST
, O.ACT_ATL_BUR_COST
, SR.SKILLS_TYPE
, IT.TRANSACTION_DATE
, IT.CUST_ORDER_ID
, IT. CUST_ORDER_LINE_NO
, ID.DEMAND_SUPPLY_ID
, WO.RECEIVED_QTY
, WO.BASE_ID
, WO.SUB_ID
, S.RECEIVED_QTY
, C.USER_9
, (O.ACT_ATL_LAB_COST/S.RECEIVED_QTY) * ID.DIST_QTY AS LABOR
, (O.ACT_ATL_BUR_COST/S.RECEIVED_QTY) * ID.DIST_QTY AS BURDEN
FROM LIVE.dbo.INVENTORY_TRANS IT 
INNER JOIN LIVE.dbo.INV_TRANS_DIST ID
ON IT.TRANSACTION_ID = ID.OUT_TRANS_ID
INNER JOIN LIVE.dbo.CUSTOMER_ORDER CO
ON IT.CUST_ORDER_ID = CO.ID
INNER JOIN LIVE.dbo.CUSTOMER C
ON CO.CUSTOMER_ID = C.ID
INNER JOIN LIVE.dbo.INVENTORY_TRANS I1
ON ID.IN_TRANS_ID = I1.TRANSACTION_ID
LEFT OUTER JOIN LIVE.dbo.OPERATION O
ON I1.WORKORDER_TYPE = O.WORKORDER_TYPE
AND I1.WORKORDER_BASE_ID = O.WORKORDER_BASE_ID
AND I1.WORKORDER_LOT_ID = O.WORKORDER_LOT_ID
AND I1.WORKORDER_SPLIT_ID = O.WORKORDER_SPLIT_ID
LEFT OUTER JOIN LIVE.dbo.SHOP_RESOURCE SR
ON O.RESOURCE_ID = SR.ID
LEFT OUTER JOIN LIVE.dbo.WORK_ORDER WO
ON O.WORKORDER_TYPE = WO.[TYPE]
AND O.WORKORDER_BASE_ID = WO.BASE_ID
AND O.WORKORDER_LOT_ID = WO.LOT_ID
AND O.WORKORDER_SPLIT_ID = WO.SPLIT_ID
AND O.WORKORDER_SUB_ID = WO.SUB_ID
LEFT OUTER JOIN LIVE.dbo.SKILLS_WO_REC_QTY S
ON O.WORKORDER_BASE_ID = S.BASE_ID
AND O.WORKORDER_LOT_ID = S.LOT_ID
AND O.WORKORDER_SPLIT_ID = S.SPLIT_ID
WHERE IT.CUST_ORDER_ID IS NOT NULL 
AND S.RECEIVED_QTY > 0
AND IT.CLASS = N'I'
AND IT.[TYPE] = N'O'
AND IT.TRANSACTION_DATE >= '2020-10-01 00:00:00'
AND IT.TRANSACTION_DATE < '2020-11-01 00:00:00'
--> FILTERED UPON REQUEST AS OO AND OR ARE ASSOCIATED TO BUSINESS SOLUTIONS
AND C.USER_9 IN ('1', '2', '3', '4')


```
