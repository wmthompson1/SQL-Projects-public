SELECT CREATE_DATE
	, TRANSACTION_ID
	, TRANSACTION_DATE
	, CLOCK_IN
	, CLOCK_OUT
	, EMPLOYEE_ID
	, FIRST_NAME
	, LAST_NAME
	, RESOURCE_ID
	, [DESCRIPTION]
	, HOURS_WORKED
	, CASE
		WHEN CAST(CLOCK_IN AS TIME) >= '11:15:00' AND CAST(CLOCK_IN AS TIME) < '11:45:00' THEN 'CLOCK IN DURING LUNCH'
		WHEN CAST(CLOCK_OUT AS TIME) >= '11:15:00' AND CAST(CLOCK_OUT AS TIME) < '11:45:00' THEN 'CLOCK IN DURING LUNCH'
		ELSE NULL
	END AS NOTES
FROM LABOR_TICKET LT
LEFT OUTER JOIN EMPLOYEE EE
ON EE.ID = LT.EMPLOYEE_ID
WHERE TRANSACTION_DATE > DATEADD(MONTH, DATEDIFF(MONTH, 0, GETDATE()), 0) --> GETDATE() - 30
	AND (TRANSACTION_DATE > GETDATE()
		OR CLOCK_IN > GETDATE()
		OR CLOCK_OUT > GETDATE()
		OR HOURS_WORKED < 0
		)

SELECT WO.BASE_ID
	, WO.LOT_ID
	, WO.SPLIT_ID
	, WO.[STATUS] AS WO_STATUS
	, WO2.SUB_ID
	, WO2.[STATUS] AS SUB_STATUS
FROM WORK_ORDER WO
INNER JOIN WORK_ORDER WO2
	ON  WO.[TYPE] = WO2.[TYPE]
	AND WO.BASE_ID = WO2.BASE_ID
	AND WO.LOT_ID = WO2.LOT_ID
	AND WO.SPLIT_ID = WO2.SPLIT_ID
	AND WO.SUB_ID = '0'
	AND WO2.SUB_ID <> '0'
	AND WO.[STATUS] IN ('C', 'X')
	AND WO2.[STATUS] NOT IN ('C', 'X')
WHERE WO.CREATE_DATE >= '1/1/2018'

--> PARTS CHECK
SELECT *
FROM (SELECT CASE
		WHEN P.[DESCRIPTION] LIKE '%OBSOLETE%' AND P.[STATUS] IS NULL THEN 'CHECK STATUS'
		ELSE NULL
	END AS ERROR_MSG01
	, CASE
		WHEN P.ANNUAL_USAGE_QTY != 0 THEN 'ANNUAL_USAGE_QTY <> 0'
		ELSE NULL
	END AS ERROR_MSG02
	, CASE
		WHEN P.[DESCRIPTION] LIKE '%TITAN%' AND UDF.ID IS NULL AND P.[STATUS] = 'A'	THEN 'MATERIAL TYPE: TITANIUM'
		ELSE NULL
	END AS ERROR_MSG03
	, CASE
		WHEN P.FABRICATED = 'Y' AND P.INSPECTION_REQD = 'Y'	THEN 'FABRICATED & INSP_REQD'
		ELSE NULL
	END AS ERROR_MSG04
	, P.ID AS PART_ID
	, P.[DESCRIPTION]
	, P.[STATUS]
	, UDF.STRING_VAL AS MATERIAL_TYPE
	, P.FABRICATED
	, P.INSPECTION_REQD
FROM PART P
LEFT OUTER JOIN USER_DEF_FIELDS AS UDF 
	ON P.ID = UDF.DOCUMENT_ID
	AND UDF.ID = 'UDF-0000035' 
	AND UDF.STRING_VAL LIKE '%TITAN%'
WHERE ([STATUS] IS NULL OR [STATUS] != 'O')
	AND P.ID NOT LIKE '%REWORK%'
) X
WHERE COALESCE(ERROR_MSG04, ERROR_MSG03, ERROR_MSG02, ERROR_MSG01, '') != ''


--> SHIPPED RELEASED COS
SELECT B.Salesperson_ID
	, C.ID
	, CO.ORDER_DATE
	, COL.CUST_ORDER_ID 
FROM CUST_ORDER_LINE COL 
INNER JOIN PART_SITE_VIEW P
	ON P.PART_ID = COL.PART_ID
INNER JOIN LIVESupplemental.dbo.buyer_assn B
	ON B.Account_ID = LEFT(P.BUYER_USER_ID, 2)
INNER JOIN 
	(SELECT CUST_ORDER_ID, SUM(ORDER_QTY) AS OrdQty, SUM(TOTAL_SHIPPED_QTY) as ShipQty
	FROM CUST_ORDER_LINE COL2
	GROUP BY CUST_ORDER_ID
	HAVING SUM(TOTAL_SHIPPED_QTY) - SUM(ORDER_QTY) >= 0
	) COL2
	ON  col2.CUST_ORDER_ID = col.CUST_ORDER_ID
INNER JOIN CUSTOMER_ORDER CO 
	ON  COL.CUST_ORDER_ID = CO.ID
INNER JOIN CUSTOMER C
	ON  CO.CUSTOMER_ID = C.ID
LEFT OUTER JOIN DEMAND_SUPPLY_LINK AS DSL 
	ON  DSL.DEMAND_BASE_ID = COL.CUST_ORDER_ID
	AND DSL.DEMAND_SEQ_NO = COL.LINE_NO
LEFT OUTER JOIN WORK_ORDER WO 
	ON  DSL.SUPPLY_BASE_ID = WO.BASE_ID 
	AND DSL.SUPPLY_LOT_ID = WO.LOT_ID 
	AND DSL.SUPPLY_SPLIT_ID = WO.SPLIT_ID 
	AND DSL.SUPPLY_SUB_ID = WO.SUB_ID 
	AND DSL.SUPPLY_TYPE  = 'WO'
LEFT OUTER JOIN WORK_ORDER RWO 
	ON  RWO.BASE_ID = 'RW' + CAST(WO.BASE_ID AS VARCHAR)
	AND RWO.LOT_ID = WO.LOT_ID
	AND RWO.SPLIT_ID = WO.SPLIT_ID
	AND RWO.SUB_ID = WO.SUB_ID
	AND RWO.[TYPE]  = 'WO'
LEFT OUTER JOIN WORK_ORDER WOR 
	ON  WOR.BASE_ID = CAST(WO.BASE_ID AS VARCHAR) + 'R'
	AND WOR.LOT_ID = WO.LOT_ID
	AND WOR.SPLIT_ID = WO.SPLIT_ID
	AND WOR.SUB_ID = WO.SUB_ID
	AND WOR.[TYPE]  = 'WO'
LEFT OUTER JOIN 
	(SELECT CUST_ORDER_ID, CUST_ORDER_LINE_NO, SUM(QTY) as InvoiceQty
	FROM RECEIVABLE_LINE RL
	GROUP BY CUST_ORDER_ID, CUST_ORDER_LINE_NO
	) RL
	ON  COL.CUST_ORDER_ID = RL.CUST_ORDER_ID
	AND COL.LINE_NO = RL.CUST_ORDER_LINE_NO
	AND RL.INVOICEQTY > 0
WHERE RL.CUST_ORDER_ID IS NOT NULL
	AND CO.[STATUS] = 'R' 
	AND COL.PART_ID IS NOT NULL
	AND SHIPQTY - ORDQTY >= 0
	AND WO.[STATUS] = 'C'
	AND (RWO.[STATUS] = 'C' OR RWO.[STATUS] IS NULL)
	AND (WOR.[STATUS] = 'C' OR WOR.[STATUS] IS NULL)
GROUP BY B.Salesperson_ID
	, C.ID
	, CO.ORDER_DATE
	, COL.CUST_ORDER_ID

--> SHIPPING STATUS
SELECT CO.CUSTOMER_ID
	, ISNULL(CL.DESIRED_SHIP_DATE, CO.DESIRED_SHIP_DATE) AS DESIRED_SHIP_DATE
	, S.PACKLIST_ID
	, S.CUST_ORDER_ID
	, CL.PART_ID
	, S.[STATUS]
	, SL.LINE_NO
	, SL.CUST_ORDER_LINE_NO
	, CL.ORDER_QTY
	, CL.UNIT_PRICE
	, S.SHIPPED_DATE
FROM SHIPPER S
LEFT OUTER JOIN SHIPPER_LINE SL
	ON S.PACKLIST_ID = SL.PACKLIST_ID
LEFT OUTER JOIN CUST_ORDER_LINE CL
	ON SL.CUST_ORDER_ID = CL.CUST_ORDER_ID
	AND SL.CUST_ORDER_LINE_NO = CL.LINE_NO
LEFT OUTER JOIN CUSTOMER_ORDER CO
	ON CO.ID = CL.CUST_ORDER_ID
WHERE S.STATUS <> 'A'
	AND SL.LINE_NO > 0
