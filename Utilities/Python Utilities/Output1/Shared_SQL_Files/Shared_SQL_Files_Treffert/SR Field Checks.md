# SR Field Checks

```sql
use DataWarehouse
GO

/*
DROP TABLE [LIVESupplemental].[dbo].[SR_CHECK]

SELECT SR.ID AS [RESOURCE_ID]
      , SR.[DESCRIPTION]
      , SR.[USER_1]
      , SR.[USER_2]
      , SR.[USER_3]
      , SR.[USER_4]
      , SR.[USER_5]
      , SR.[USER_6]
      , SR.[USER_7]
      , SR.[USER_8]
      , SR.[USER_9]
      , SR.[USER_10]
      , SRS.[SETUP_COST_PER_HR]
      , SRS.[RUN_COST_PER_HR]
      , SRS.[BUR_PERCENT_SETUP]
      , SRS.[BUR_PERCENT_RUN]
      , SRS.[AFC_GL_ACCT_ID]
      , SRS.[AFB_GL_ACCT_ID]
  INTO [LIVESupplemental].[dbo].[SR_CHECK]
  FROM DataWarehouse.dbo.SHOP_RESOURCE_SITE SRS
  INNER JOIN DataWarehouse.dbo.SHOP_RESOURCE SR
  ON SR.ID = SRS.RESOURCE_ID
 */
SELECT *
FROM 
(SELECT SRS.RESOURCE_ID
, CK.RESOURCE_ID AS CK_RESOURCEID
, CASE WHEN SRS.USER_1 != CK.USER_1 THEN SRS.USER_1 + ' -> ' + CK.USER_1 ELSE '' END AS USER_1
, CASE WHEN SRS.USER_4 != CK.USER_4 THEN SRS.USER_4 + ' -> ' + CK.USER_4 ELSE '' END AS USER_4
, CASE WHEN SRS.USER_5 != CK.USER_5 THEN SRS.USER_5 + ' -> ' + CK.USER_5 ELSE '' END AS USER_5
, CASE WHEN SRS.USER_6 != CK.USER_6 THEN SRS.USER_6 + ' -> ' + CK.USER_6 ELSE '' END AS USER_6
, CASE WHEN SRS.USER_7 != CK.USER_7 THEN SRS.USER_7 + ' -> ' + CK.USER_7 ELSE '' END AS USER_7
, CASE WHEN SRS.USER_8 != CK.USER_8 THEN SRS.USER_8 + ' -> ' + CK.USER_8 ELSE '' END AS USER_8
, CASE WHEN SRS.USER_9 != CK.USER_9 THEN SRS.USER_9 + ' -> ' + CK.USER_9 ELSE '' END AS USER_9
, CASE WHEN SRS.USER_10 != CK.USER_10 THEN SRS.USER_10 + ' -> ' + CK.USER_10 ELSE '' END AS USER_10
, CASE WHEN SRS.SETUP_COST_PER_HR != CK.SETUP_COST_PER_HR THEN CONVERT(NVARCHAR(MAX), CK.SETUP_COST_PER_HR) + ' -> ' + CONVERT(NVARCHAR(MAX), SRS.SETUP_COST_PER_HR) ELSE '' END AS SETUP_COST_PER_HR
, CASE WHEN SRS.RUN_COST_PER_HR != CK.RUN_COST_PER_HR THEN CONVERT(NVARCHAR(MAX), CK.RUN_COST_PER_HR) + ' -> ' + CONVERT(NVARCHAR(MAX), SRS.RUN_COST_PER_HR) ELSE '' END AS RUN_COST_PER_HR
, CASE WHEN SRS.BUR_PERCENT_SETUP != CK.BUR_PERCENT_SETUP THEN CONVERT(NVARCHAR(MAX), CK.BUR_PERCENT_SETUP) + ' -> ' + CONVERT(NVARCHAR(MAX), SRS.BUR_PERCENT_SETUP) ELSE '' END AS BUR_PERCENT_SETUP
, CASE WHEN SRS.BUR_PERCENT_RUN != CK.BUR_PERCENT_RUN THEN CONVERT(NVARCHAR(MAX), CK.BUR_PERCENT_RUN) + ' -> ' + CONVERT(NVARCHAR(MAX), SRS.BUR_PERCENT_RUN) ELSE '' END AS BUR_PERCENT_RUN
, CASE WHEN SRS.AFC_GL_ACCT_ID != CK.AFC_GL_ACCT_ID THEN CONVERT(NVARCHAR(MAX), CK.AFC_GL_ACCT_ID) + ' -> ' + CONVERT(NVARCHAR(MAX), SRS.AFC_GL_ACCT_ID) ELSE '' END AS AFC_GL_ACCT_ID
, CASE WHEN SRS.AFB_GL_ACCT_ID != CK.AFB_GL_ACCT_ID THEN CONVERT(NVARCHAR(MAX), CK.AFB_GL_ACCT_ID) + ' -> ' + CONVERT(NVARCHAR(MAX), SRS.AFB_GL_ACCT_ID) ELSE '' END AS AFB_GL_ACCT_ID
FROM SHOP_RESOURCE_SITE SRS
LEFT OUTER JOIN LIVESUPPLEMENTAL.dbo.SR_CHECK CK
ON SRS.RESOURCE_ID = CK.RESOURCE_ID
WHERE (SRS.[STATUS] IS NULL OR SRS.[STATUS] != 'O')
) X
WHERE CK_RESOURCEID IS NULL 
	OR USER_1 != ''
	OR USER_4 != ''
	OR USER_5 != ''
	OR USER_6 != ''
	OR USER_7 != ''
	OR USER_8 != ''
	OR USER_9 != ''
	OR USER_10 != ''
	OR SETUP_COST_PER_HR != ''
	OR RUN_COST_PER_HR != ''
	OR BUR_PERCENT_SETUP != ''
	OR BUR_PERCENT_RUN != ''
	OR AFC_GL_ACCT_ID != ''
	OR AFB_GL_ACCT_ID != ''

```
