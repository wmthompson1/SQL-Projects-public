# Parts Inventory with Trace2

```sql
use datawarehouse
go

WITH CTE_TRACE AS
(SELECT I.WAREHOUSE_ID
	, I.LOCATION_ID
	, I.PART_ID
	, P.[DESCRIPTION]
	, T.TRACE_ID
	, P.COMMODITY_CODE
	, P.STOCK_UM
	, PS.UNIT_MATERIAL_COST
	, SUM(T.QTY) AS QTY
	, PS.UNIT_MATERIAL_COST * SUM(T.QTY) AS TOTAL_COST
	, MAX(T.CREATE_DATE) AS LASTTRACEDATE
	, max(t.transaction_id) as lasttransxct
FROM PART P
	INNER JOIN PART_SITE PS
		ON P.ID = PS.PART_ID
	INNER JOIN TRACE_INV_TRANS T
		ON T.PART_ID = P.ID
	INNER JOIN INVENTORY_TRANS I
		ON T.TRANSACTION_ID = I.TRANSACTION_ID
		AND T.PART_ID = I.PART_ID
WHERE (P.COMMODITY_CODE IN ('DETAILS','RAW MATERIAL','STANDARDS')
	OR P.COMMODITY_CODE IS NULL)
	AND I.WAREHOUSE_ID <> '1'
	--AND I.WAREHOUSE_ID IN(@WAREHOUSE)
	--AND I.LOCATION_ID IN(@LOCATION)
GROUP BY I.WAREHOUSE_ID
	, I.LOCATION_ID
	, I.PART_ID
	, P.[DESCRIPTION]
	, T.TRACE_ID
	, P.COMMODITY_CODE
	, P.STOCK_UM
	, PS.UNIT_MATERIAL_COST
HAVING SUM(T.QTY) > 0
)

SELECT T.*
	, i2.[user_id]
	, I2.[DESCRIPTION]
FROM CTE_TRACE T
	INNER JOIN TRACE_INV_TRANS T2
		ON T.lasttransxct = T2.TRANSACTION_ID
		AND T.PART_ID = T2.PART_ID
		and t.trace_id = t2.trace_id
	LEFT OUTER JOIN INVENTORY_TRANS I2
		ON T.lasttransxct = I2.TRANSACTION_ID
		AND T2.PART_ID = I2.PART_ID
		AND T.WAREHOUSE_ID = I2.WAREHOUSE_ID 
		AND T.LOCATION_ID = I2.LOCATION_ID
--WHERE T.PART_ID = '182T7120-5'
ORDER BY WAREHOUSE_ID
	, LOCATION_ID
	, COMMODITY_CODE
	, PART_ID
	, TRACE_ID
	, STOCK_UM
	, UNIT_MATERIAL_COST

SELECT * FROM INVENTORY_TRANS WHERE PART_ID = '182T7120-5' ORDER BY CREATE_DATE
SELECT * FROM TRACE_INV_TRANS WHERE PART_ID = '182T7120-5' AND CREATE_DATE >= '1/1/2018' ORDER BY CREATE_DATE
```
