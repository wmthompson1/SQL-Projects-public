# IQM PM Listing Report (20210809)

```sql
USE SANDBOX_Baseline_App
GO

DECLARE	@RESULT NVARCHAR(25) = '_BLANK'
, @MAINTAINBY NVARCHAR(25) = '_BLANK'
, @STARTDATE DATETIME = '8/1/2021'
, @ENDDATE DATETIME = '12/31/2021'
, @EQUIPNAME NVARCHAR(25) = '_BLANK'
, @MAJOR NVARCHAR(25) = 'P2'
, @MINOR NVARCHAR(25) = '_BLANK'
, @BUSUNIT NVARCHAR(25) = '_BLANK'
, @EE NVARCHAR(25) = '_BLANK'
, @CUSTOMER NVARCHAR(25) = '_BLANK'
, @SUPPLIER NVARCHAR(25) = '_BLANK' 

SELECT P.PREVENTIVE
	, P.ERP_ID2
	, P.COMPONENT_NUMBER
	, EC.[NAME] AS PM_ACT_TYPE
	, CASE 
		WHEN P.RESULT = 'F' THEN 'Failed'
		WHEN P.RESULT = 'PA' THEN 'Passed'
		WHEN P.RESULT = 'PE' THEN 'Pending'
		WHEN P.RESULT = 'S' THEN 'Status'
		ELSE 'ISSUES'
	END AS RESULT
	, P.DUE_DATE
	, E.EQUIPMENT_ID
	, E.[NAME] AS EQUIP_NAME
	, E.MAJOR_LOCATION
	, MAJ.[DESCRIPTION] AS MAJOR_DESC
	, E.MINOR_LOCATION
	, MIL.[DESCRIPTION] AS MINOR_DESC
	, E.BUSINESS_UNIT_ID
	, E.COMMENTS
	, ISNULL(C.INSTRUCTIONS, EC.INSTRUCTIONS) AS INSTRUCTIONS
	, P.[MAINTAINED_BY]
	, P.[EMPLOYEE_REC_ID]
	, EE.FIRST_NAME + ' ' + EE.LAST_NAME AS EMPLOYEE
	, P.[CUSTOMER_REC_ID]
	, CUST.[NAME] AS CUSTOMER
	, P.[SUPPLIER_REC_ID]
	, S.[NAME] AS SUPPLIER
FROM [dbo].[VQ_PREVENTIVE] P
LEFT OUTER JOIN [dbo].[VQ_EQUIPMENT] E
	ON P.[EQUIPMENT_REC_ID] = E.[RECORD_ID]
LEFT OUTER JOIN [dbo].[VQ_EQUP_TYPE_COMP] EC
	ON E.[EQUIPMENT_TYPE_REC_ID] = EC.[EQUIPMENT_TYPE_REC_ID]
	AND P.COMPONENT_NUMBER = EC.COMPONENT_NUMBER
LEFT OUTER JOIN [dbo].[VQ_COMPONENTS] C
	ON P.[EQUIPMENT_REC_ID] = C.[EQUIPMENT_REC_ID]
	AND P.COMPONENT_NUMBER = C.COMPONENT_NUMBER
LEFT OUTER JOIN [dbo].[VQ_MAJOR_LOCATION] MAJ
	ON E.MAJOR_LOCATION = MAJ.MAJOR_LOCATION
LEFT OUTER JOIN [dbo].[VQ_MINOR_LOCATION] MIL
	ON E.MINOR_LOCATION = MIL.MINOR_LOCATION
LEFT OUTER JOIN [dbo].[VQ_EMPLOYEE] EE
	ON EE.RECORD_ID = P.[EMPLOYEE_REC_ID]
LEFT OUTER JOIN [dbo].[VQ_CUSTOMER] CUST
	ON CUST.RECORD_ID = P.[CUSTOMER_REC_ID]
LEFT OUTER JOIN [dbo].[VQ_SUPPLIER] S
	ON S.RECORD_ID = P.[SUPPLIER_REC_ID]
--WHERE P.PREVENTIVE IN (RIGHT('000000' + @PM, 7))
--WHERE ISNULL(EC.INSTRUCTIONS, '') != ISNULL(C.INSTRUCTIONS, '')
WHERE (P.DUE_DATE >= @STARTDATE AND P.DUE_DATE <= @ENDDATE)
	--AND CASE 
	--	WHEN P.RESULT = 'F' THEN 'Failed'
	--	WHEN P.RESULT = 'PA' THEN 'Passed'
	--	WHEN P.RESULT = 'PE' THEN 'Pending'
	--	WHEN P.RESULT = 'S' THEN 'Status'
	--	ELSE 'ISSUES'
	--END IN (@RESULT)
	--AND CASE 
	--	WHEN P.[MAINTAINED_BY] = 'I' THEN 'INTERNAL'
	--	WHEN P.[MAINTAINED_BY] = 'C' THEN 'CUSTOMER'
	--	WHEN P.[MAINTAINED_BY] = 'S' THEN 'SUPPLIER'
	--END IN (@MAINTAINBY)
	--AND ISNULL(E.[NAME], '_BLANK') IN (@EQUIPNAME)
	AND ISNULL(E.MAJOR_LOCATION, '_BLANK') IN (@MAJOR)
	--AND ISNULLISNULL(E.MINOR_LOCATION + '-' + MIL.[DESCRIPTION], '_BLANK') IN (@MINOR)
	--AND ISNULL(E.BUSINESS_UNIT_ID, '_BLANK') IN (@BUSUNIT)
	--AND ISNULL(EE.LAST_NAME + ', ' + EE.FIRST_NAME, '_BLANK') IN (@EE)
	--AND ISNULL(CUST.[NAME], '_BLANK') IN (@CUSTOMER)
	--AND ISNULL(S.[NAME], '_BLANK') IN (@SUPPLIER)


SELECT '_BLANK' AS BUSUNIT
UNION
SELECT ISNULL(E.BUSINESS_UNIT_ID, '_BLANK') AS BUSUNIT
FROM [dbo].[VQ_PREVENTIVE] P
LEFT OUTER JOIN [dbo].VQ_EQUIPMENT E
	ON P.[EQUIPMENT_REC_ID] = E.[RECORD_ID]
GROUP BY E.BUSINESS_UNIT_ID

SELECT '_BLANK' AS MINOR
UNION
SELECT ISNULL(E.MINOR_LOCATION, '_BLANK') AS MINOR
FROM [dbo].[VQ_PREVENTIVE] P
LEFT OUTER JOIN [dbo].VQ_EQUIPMENT E
	ON P.[EQUIPMENT_REC_ID] = E.[RECORD_ID]
GROUP BY E.MINOR_LOCATION


SELECT '_BLANK' AS EQUIPNAME
UNION
SELECT ISNULL(E.[NAME], '_BLANK') AS EQUIPNAME
FROM [dbo].[VQ_PREVENTIVE] P
LEFT OUTER JOIN [dbo].VQ_EQUIPMENT E
	ON P.[EQUIPMENT_REC_ID] = E.[RECORD_ID]
GROUP BY E.[NAME]

SELECT '_BLANK' AS EMPLOYEE
UNION
SELECT ISNULL(EE.LAST_NAME + ', ' + EE.FIRST_NAME, '_BLANK') AS EMPLOYEE
FROM [dbo].[VQ_PREVENTIVE] P
LEFT OUTER JOIN [dbo].[VQ_EMPLOYEE] EE
	ON EE.RECORD_ID = P.[EMPLOYEE_REC_ID]
GROUP BY EE.LAST_NAME + ', ' + EE.FIRST_NAME

SELECT '_BLANK' AS SUPPLIER
UNION
SELECT ISNULL([NAME], '_BLANK') AS SUPPLIER
FROM [dbo].[VQ_PREVENTIVE] P
LEFT OUTER JOIN [dbo].[VQ_SUPPLIER] S
	ON S.RECORD_ID = P.[SUPPLIER_REC_ID]
GROUP BY S.[NAME]

SELECT '_BLANK' AS CUSTOMER
UNION
SELECT ISNULL([NAME], '_BLANK') AS CUSTOMER
FROM [dbo].[VQ_PREVENTIVE] P
LEFT OUTER JOIN [dbo].[VQ_CUSTOMER] C
	ON C.RECORD_ID = P.[CUSTOMER_REC_ID]
GROUP BY C.[NAME]
```
