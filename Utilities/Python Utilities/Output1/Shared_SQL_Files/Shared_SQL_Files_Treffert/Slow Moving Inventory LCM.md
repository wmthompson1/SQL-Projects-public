# Slow Moving Inventory LCM

```sql
USE LIVE
GO

DECLARE --@PARTID NVARCHAR(50) = '140W9012-170006'
 @POSTINGDATE DATE = '9/30/2021'

SELECT IB.PART_ID
	, P.[DESCRIPTION]
	, P.COMMODITY_CODE 
	, IB.QTY
	, IB.MATERIAL_AMOUNT
	, IB.LABOR_AMOUNT
	, IB.BURDEN_AMOUNT
	, IB.SERVICE_AMOUNT
	, (IB.MATERIAL_AMOUNT + IB.LABOR_AMOUNT + IB.BURDEN_AMOUNT + IB.SERVICE_AMOUNT) AS TOTAL_AMOUNT
	, MAX(IT.TRANSACTION_ID) AS LAST_SALE_TRANS
	, MAX(IT.TRANSACTION_DATE) AS LAST_SALES_DATE
	, DATEDIFF(MM, MAX(IT.TRANSACTION_DATE), GETDATE()) AS MONTHS_SINCE_SALE
	, CASE	
		WHEN DATEDIFF(MM, MAX(IT.TRANSACTION_DATE), GETDATE()) < 18 THEN 0
		WHEN DATEDIFF(MM, MAX(IT.TRANSACTION_DATE), GETDATE()) < 25 THEN 0.20
		WHEN DATEDIFF(MM, MAX(IT.TRANSACTION_DATE), GETDATE()) < 37 THEN 0.5
		WHEN DATEDIFF(MM, MAX(IT.TRANSACTION_DATE), GETDATE()) >36 THEN 1
	END AS RESERVE_PCT
	, (IB.MATERIAL_AMOUNT + IB.LABOR_AMOUNT + IB.BURDEN_AMOUNT + IB.SERVICE_AMOUNT) * CASE	
		WHEN DATEDIFF(MM, MAX(IT.TRANSACTION_DATE), GETDATE()) < 18 THEN 0
		WHEN DATEDIFF(MM, MAX(IT.TRANSACTION_DATE), GETDATE()) < 25 THEN 0.20
		WHEN DATEDIFF(MM, MAX(IT.TRANSACTION_DATE), GETDATE()) < 37 THEN 0.5
		WHEN DATEDIFF(MM, MAX(IT.TRANSACTION_DATE), GETDATE()) > 36 THEN 1
	END AS RESERVE_AMOUNT
	, CPE.SELLING_UM
	, CPE.CUSTOMER_PART_ID AS CONTRACT
	--, MAX(PC.CUSTOMER_PART_ID) AS CONTRACT
	--, PC.CUSTOMER_ID
	, CO.CUSTOMER_ID
	
	, Case
		When P.COMMODITY_CODE in ('Details', 'End Items') Then CPE.DEFAULT_UNIT_PRICE
		Else 0
		End as 'Selling_Unit_Price'
	
	, (IB.Qty * 
		Case
			When P.COMMODITY_CODE in ('Details', 'End Items') Then CPE.DEFAULT_UNIT_PRICE
			Else 0
			End) - 
				(Case
					When P.COMMODITY_CODE in ('Details', 'End Items') Then (IB.MATERIAL_AMOUNT + IB.LABOR_AMOUNT + IB.BURDEN_AMOUNT + IB.SERVICE_AMOUNT) 
					Else 0
					End ) as 'Total_Price'
	
	, ((IB.Qty * 
		Case
			When P.COMMODITY_CODE in ('Details', 'End Items') Then CPE.DEFAULT_UNIT_PRICE
			Else 0
			End) - 
				(Case
					When P.COMMODITY_CODE in ('Details', 'End Items') Then (IB.MATERIAL_AMOUNT + IB.LABOR_AMOUNT + IB.BURDEN_AMOUNT + IB.SERVICE_AMOUNT) 
					Else 0
					End )) - 
						(Case
							When P.COMMODITY_CODE in ('Details', 'End Items') Then (IB.MATERIAL_AMOUNT + IB.LABOR_AMOUNT + IB.BURDEN_AMOUNT + IB.SERVICE_AMOUNT) 
							Else 0
							End ) as 'Change'


FROM INVENTORY_BALANCE IB
LEFT OUTER JOIN PART P
	ON P.ID = IB.PART_ID
LEFT OUTER JOIN INVENTORY_TRANS IT
	ON IT.PART_ID = P.ID
	AND IT.[TYPE] = 'O'
	AND IT.CLASS = 'I'
LEFT OUTER JOIN CUSTOMER_ORDER CO
	ON IT.CUST_ORDER_ID = CO.ID
LEFT OUTER JOIN CUST_PRICE_EFFECT CPE
	ON CPE.PART_ID = P.ID
	AND CPE.DISCONTINUE_DATE >= GETDATE()
	AND CO.CUSTOMER_ID = CPE.CUSTOMER_ID

WHERE
 IB.POSTING_DATE = @POSTINGDATE
	AND IT.[TYPE] = 'O'
	AND IT.CLASS = 'I'
	and P.COMMODITY_CODE <> 'CFM-Standards'
	and P.Commodity_Code <> 'CFM-Details'
	and P.Commodity_Code <> 'CFM-Raw Matl'
	and P.Commodity_Code <> 'Adhesives'
	and P.COMMODITY_CODE <> 'Gen-Controlled'
	and P.COMMODITY_CODE <> 'Non-Fly Away'

	and P.ID = '100T1430-87'
	--and P.ID = '113T1697-26'

GROUP BY IB.PART_ID
	, P.[DESCRIPTION]
	, P.COMMODITY_CODE 
	, IB.QTY
	, IB.MATERIAL_AMOUNT
	, IB.LABOR_AMOUNT
	, IB.BURDEN_AMOUNT
	, IB.SERVICE_AMOUNT
	, CPE.SELLING_UM
	, CPE.CUSTOMER_PART_ID
	, CO.CUSTOMER_ID
	, Case
		When P.COMMODITY_CODE in ('Details', 'End Items') Then CPE.DEFAULT_UNIT_PRICE
		Else 0
		End 
	, IB.Qty * Case
		When P.COMMODITY_CODE in ('Details', 'End Items') Then CPE.DEFAULT_UNIT_PRICE
		Else 0
		End
```
