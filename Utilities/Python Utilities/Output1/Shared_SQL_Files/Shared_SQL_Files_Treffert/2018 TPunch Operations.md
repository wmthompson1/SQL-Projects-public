# 2018 TPunch Operations

```sql
use DataWarehouse
go

select WORKORDER_TYPE
	, WORKORDER_BASE_ID
	, WORKORDER_LOT_ID
	, WORKORDER_SPLIT_ID
	, WORKORDER_SUB_ID
	, OPERATION_SEQ_NO
	, RESOURCE_ID
	, SUM(HOURS_WORKED) AS TOTAL_HOURS
	, MAX(TRANSACTION_DATE) AS TRANSDATE
from labor_ticket 
where resource_ID = 'tpunch' 
and TRANSACTION_DATE >= '1/1/2018'
AND [TYPE] = 'R'
group by WORKORDER_TYPE
	, WORKORDER_BASE_ID
	, WORKORDER_LOT_ID
	, WORKORDER_SPLIT_ID
	, WORKORDER_SUB_ID
	, OPERATION_SEQ_NO
	, RESOURCE_ID

```
