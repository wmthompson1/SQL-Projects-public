# Teshome's Jobs

```sql
USE LIVE
GO

WITH CTE_JOB AS
(SELECT WORKORDER_BASE_ID
FROM LABOR_TICKET
WHERE EMPLOYEE_ID = '1384'
AND TRANSACTION_DATE >= '4/12/2021'
AND TRANSACTION_DATE < '4/13/2021'
AND WORKORDER_BASE_ID IS NOT NULL
)

, CTE_DSL AS
(SELECT DEMAND_BASE_ID
, DEMAND_TYPE
, SUPPLY_BASE_ID
FROM CTE_JOB JOB
LEFT OUTER JOIN DEMAND_SUPPLY_LINK DSL
ON JOB.WORKORDER_BASE_ID = DSL.SUPPLY_BASE_ID
AND DSL.SUPPLY_TYPE = 'WO'
)
--SELECT * FROM CTE_DSL

, CTE_COS AS
(SELECT DEMAND_BASE_ID
, SUPPLY_BASE_ID
, CO.CUSTOMER_ID
, CO.SHIP_TO_ADDR_NO
, CA.NAME
, CA.ADDR_NO
FROM CTE_DSL CTE
LEFT OUTER JOIN CUSTOMER_ORDER CO
	ON CTE.DEMAND_BASE_ID = CO.ID
	AND CTE.DEMAND_TYPE = 'CO'
LEFT OUTER JOIN CUST_ADDRESS CA
	ON CO.SHIP_TO_ADDR_NO = CA.ADDR_NO
	AND CO.CUSTOMER_ID = CA.CUSTOMER_ID
WHERE CO.CUSTOMER_ID = 'BOE609'
)

SELECT * FROM CTE_COS
--ORDER BY NAME
```
