# Part_Maint_Data_Issues

```sql
use DataWarehouse
go

WITH CTE_PART AS
(SELECT P.ID AS PART_ID
	, P.[STATUS]
	, P.[DESCRIPTION]
	, P.PLANNER_USER_ID
	, P.BUYER_USER_ID
	, P.COMMODITY_CODE
FROM PART P
LEFT OUTER JOIN PLANNER_USER_ID PUI
	ON P.PLANNER_USER_ID = PUI.PLANNER_USER_ID
WHERE P.[STATUS] <> 'O'
AND (
	PUI.PLANNER_USER_ID IS NULL
	OR P.BUYER_USER_ID IS NULL
	OR P.[DESCRIPTION] IS NULL
	)
)
--select * from cte_part

SELECT P.PART_ID
	, P.[STATUS]
	, P.[DESCRIPTION]
	, P.PLANNER_USER_ID
	, P.BUYER_USER_ID
	, P.COMMODITY_CODE
	, MAX(ISNULL(COL.DESIRED_SHIP_DATE, CO.DESIRED_SHIP_DATE)) AS LAST_CO_DATE
	, MAX(WO.DESIRED_WANT_DATE) AS LAST_WO_DATE
	, MAX(IT.TRANSACTION_DATE) AS LAST_INV_DATE
FROM CTE_PART P
LEFT OUTER JOIN CUST_ORDER_LINE COL
	ON COL.PART_ID = P.PART_ID
LEFT OUTER JOIN CUSTOMER_ORDER CO
	ON COL.CUST_ORDER_ID = CO.ID
	AND CO.DESIRED_SHIP_DATE >= '1/1/2017'
LEFT OUTER JOIN DEMAND_SUPPLY_LINK DSL
	ON DSL.DEMAND_TYPE = 'CO'
	AND DSL.DEMAND_BASE_ID = CO.ID  
LEFT OUTER JOIN WORK_ORDER WO
	ON DSL.SUPPLY_TYPE = 'WO' 
	AND DSL.SUPPLY_BASE_ID = WO.BASE_ID
	AND DSL.SUPPLY_LOT_ID = WO.LOT_ID
	AND DSL.SUPPLY_SPLIT_ID = WO.SPLIT_ID
	AND DSL.SUPPLY_SUB_ID = WO.SUB_ID 
	AND P.PART_ID = WO.PART_ID
	AND WO.DESIRED_WANT_DATE >= '1/1/2017'
LEFT OUTER JOIN INVENTORY_TRANS IT
	ON P.PART_ID = IT.PART_ID
	--and WO.[TYPE] = IT.WORKORDER_TYPE
	--AND WO.BASE_ID = IT.WORKORDER_BASE_ID
	--AND WO.LOT_ID = IT.WORKORDER_LOT_ID
	--AND WO.SPLIT_ID = IT.WORKORDER_SPLIT_ID
	--AND WO.SUB_ID = IT.WORKORDER_SUB_ID
	AND IT.[TYPE] ='I'
	AND IT.CLASS = 'R'
	AND IT.TRANSACTION_DATE >= '1/1/2017'
--where p.part_id = '20240B-050 MIL' 
GROUP BY P.PART_ID
	, P.[STATUS]
	, P.[DESCRIPTION]
	, P.PLANNER_USER_ID
	, P.BUYER_USER_ID
	, P.COMMODITY_CODE
```
