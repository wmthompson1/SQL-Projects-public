CREATE TRIGGER [dbo].[SK_strINSERT_IBT_SHIP_LINE] 
   ON  [dbo].[IBT_SHIPPER_LINE] 
   AFTER INSERT
AS 
DECLARE	@nRcd INT,
		@IBT_TX_IN INT,
		@IBT_TX_OUT INT, 
		@IBT_NEW_TX_IN INT, 
		@IBT_NEW_TX_OUT INT,
		@PART_ID VARCHAR(30),
		@QTY DECIMAL(14,4),
		@USER_QTY DECIMAL(14,4),
		@IBT_ID VARCHAR(15),
		@IBT_LINE INT,
		@IBT_NEXT_ID INT,
		@IBT_NEXT_PRE VARCHAR(4),
		@IBT_NEXT_SUF VARCHAR(4),
		@IBT_NEXT_LEAD CHAR(1),
		@IBT_NEXT_DEC INT,
		@IBT_IN_WHS VARCHAR(15),
		@IBT_IN_LOC VARCHAR(15),
		@IBT_OUT_WHS VARCHAR(15),
		@IBT_OUT_LOC VARCHAR(15), 
		@IBT_SHIPPER_ID VARCHAR(15),
		@IBT_SHIPPER_LINE VARCHAR(15),
		@IBT_RECEIVER_ID VARCHAR(15),
		@IBT_MAT_AMT DECIMAL(14,4),
		@IBT_LAB_AMT DECIMAL(14,4),
		@IBT_BUR_AMT DECIMAL(14,4),
		@IBT_SER_AMT DECIMAL(14,4),
		@MSG VARCHAR(250)
BEGIN
	SELECT @nRcd = 0
	SET NOCOUNT ON
	DECLARE SK_IBT_SHIP_INS CURSOR LOCAL FOR 
		SELECT G.ALPHA_PREFIX, G.ALPHA_SUFFIX, G.DECIMAL_PLACES, G.NEXT_NUMBER, G.LEADING_ZEROS,
			I.IBT_ID, I.IBT_LINE_NO, I.IN_TRANSACTION_ID, I.OUT_TRANSACTION_ID, I.SHIPPED_QTY, I.USER_SHIPPED_QTY,
			IL.PART_ID, IBT.TO_WHSE_ID, I.SHIPPER_ID, I.LINE_NO, I.SHIPPED_QTY * P.UNIT_MATERIAL_COST, I.SHIPPED_QTY * P.UNIT_LABOR_COST, 
			I.SHIPPED_QTY * P.UNIT_BURDEN_COST, I.SHIPPED_QTY * P.UNIT_SERVICE_COST
		FROM INSERTED I JOIN NEXT_NUMBER_GEN G ON G.TABLE_NAME = 'IBT_RECEIVER'
		JOIN IBT_LINE IL ON I.IBT_ID = IL.IBT_ID AND I.IBT_LINE_NO = IL.LINE_NO
		JOIN IBT IBT ON IL.IBT_ID = IBT.ID
		JOIN PART_SITE P ON IL.PART_ID = P.PART_ID
	OPEN SK_IBT_SHIP_INS
	FETCH SK_IBT_SHIP_INS INTO @IBT_NEXT_PRE, @IBT_NEXT_SUF, @IBT_NEXT_DEC, @IBT_NEXT_ID, @IBT_NEXT_LEAD, @IBT_ID, @IBT_LINE, @IBT_TX_IN, @IBT_TX_OUT, @QTY, @USER_QTY, 
			@PART_ID, @IBT_IN_WHS, @IBT_SHIPPER_ID, @IBT_SHIPPER_LINE, @IBT_MAT_AMT, @IBT_LAB_AMT, @IBT_BUR_AMT, @IBT_SER_AMT
	
	BEGIN
		--Build next number
		SELECT @IBT_RECEIVER_ID = (SELECT ISNULL(@IBT_NEXT_PRE,'')+RIGHT(REPLICATE('0',@IBT_NEXT_DEC)+CAST(@IBT_NEXT_ID AS VARCHAR),@IBT_NEXT_DEC)+ISNULL(@IBT_NEXT_SUF,''))
		UPDATE NEXT_NUMBER_GEN SET NEXT_NUMBER = NEXT_NUMBER + 1 WHERE TABLE_NAME = 'IBT_RECEIVER'
		SELECT @IBT_OUT_WHS = (SELECT WAREHOUSE_ID FROM INVENTORY_TRANS WHERE TRANSACTION_ID = @IBT_TX_IN)
		SELECT @IBT_OUT_LOC = (SELECT LOCATION_ID FROM INVENTORY_TRANS WHERE TRANSACTION_ID = @IBT_TX_IN)
		--insert IBT Receiver initial transaction (out transfer trans)
		DECLARE @TXDATE DATETIME = (SELECT CAST(LEFT(CAST(GETDATE()AS VARCHAR),11)AS DATETIME))
		SELECT @IBT_IN_LOC = (SELECT TOP 1 ID FROM LOCATION WHERE WAREHOUSE_ID = @IBT_IN_WHS AND TYPE <> 'TRANSIT')
		SELECT @IBT_NEW_TX_OUT = (SELECT MAX(TRANSACTION_ID)+1 FROM INVENTORY_TRANS)
		SELECT @IBT_NEW_TX_IN = (SELECT @IBT_NEW_TX_OUT + 1)
		INSERT INTO INVENTORY_TRANS (TRANSACTION_ID, PART_ID, TYPE, CLASS, QTY, COSTED_QTY, TRANSACTION_DATE, WAREHOUSE_ID, LOCATION_ID, [USER_ID],POSTING_CANDIDATE,ACT_MATERIAL_COST,ACT_LABOR_COST,ACT_BURDEN_COST,ACT_SERVICE_COST, 
									 CREATE_DATE,TRANSFER_TRANS_ID,IBT_ID,IBT_LINE_NO,COUNT_SEQUENCE,SITE_ID)
					VALUES ( @IBT_NEW_TX_OUT,@PART_ID,'O','A',@QTY,0,@TXDATE,@IBT_OUT_WHS, @IBT_OUT_LOC,SYSTEM_USER,'Y',0,0,0,0,GETDATE(),@IBT_NEW_TX_IN,@IBT_ID,@IBT_LINE,1,'SK01')
		--insert IBT Receiver second transaction (in Transaction)
		
		INSERT INTO INVENTORY_TRANS (TRANSACTION_ID, PART_ID, TYPE, CLASS, QTY, COSTED_QTY, TRANSACTION_DATE, WAREHOUSE_ID, LOCATION_ID, [USER_ID],POSTING_CANDIDATE,ACT_MATERIAL_COST,ACT_LABOR_COST,ACT_BURDEN_COST,ACT_SERVICE_COST, 
									 CREATE_DATE,TRANSFER_TRANS_ID,IBT_ID,IBT_LINE_NO,COUNT_SEQUENCE,SITE_ID)
					VALUES ( @IBT_NEW_TX_IN,@PART_ID,'I','A',@QTY,0,@TXDATE,@IBT_IN_WHS, @IBT_IN_LOC,SYSTEM_USER,'Y',0,0,0,0,GETDATE(),@IBT_NEW_TX_OUT,@IBT_ID,@IBT_LINE,1,'SK01')
		--insert IBT receiver record
		INSERT INTO IBT_RECEIVER (ID, IBT_ID, RECEIVED_DATE,CREATE_DATE,[USER_ID],MARKED_FOR_PURGE,SHIPPER_ID,SITE_ID)
				VALUES (@IBT_RECEIVER_ID,@IBT_ID,@TXDATE,GETDATE(),SYSTEM_USER,'N',@IBT_SHIPPER_ID,'SK01')
		--insert IBT Receiver Line
		INSERT INTO IBT_RECEIVER_LINE (RECEIVER_ID, LINE_NO, IBT_ID, IBT_LINE_NO, USER_RECEIVED_QTY, RECEIVED_QTY, ACT_FREIGHT, WAREHOUSE_ID, LOCATION_ID, IN_TRANSACTION_ID, OUT_TRANSACTION_ID, SHIPPER_ID, SHIPPER_LINE_NO)
		VALUES (@IBT_RECEIVER_ID,@IBT_SHIPPER_LINE, @IBT_ID, @IBT_LINE, @USER_QTY, @QTY, 0.00, @IBT_IN_WHS, @IBT_IN_LOC,@IBT_NEW_TX_IN,@IBT_NEW_TX_OUT, @IBT_SHIPPER_ID, @IBT_SHIPPER_LINE)
		--Insert Trace Inv Trans records
		INSERT INTO TRACE_INV_TRANS (PART_ID, TRACE_ID, TRANSACTION_ID, QTY, CREATE_DATE, COSTED_QTY)
			SELECT PART_ID, TRACE_ID, @IBT_NEW_TX_OUT, QTY*-1, CREATE_DATE,0 
			FROM TRACE_INV_TRANS WHERE TRANSACTION_ID = @IBT_TX_IN
		INSERT INTO TRACE_INV_TRANS (PART_ID, TRACE_ID, TRANSACTION_ID, QTY, CREATE_DATE, COSTED_QTY)
			SELECT PART_ID, TRACE_ID, @IBT_NEW_TX_IN, QTY, CREATE_DATE,0 
			FROM TRACE_INV_TRANS WHERE TRANSACTION_ID = @IBT_TX_IN
		--update IBT Line
		UPDATE IBT_LINE SET USER_RECV_QTY = USER_RECV_QTY + @USER_QTY, RECEIVED_QTY = RECEIVED_QTY + @QTY WHERE IBT_ID = @IBT_ID AND LINE_NO = @IBT_LINE
		--create FIFO dist
		INSERT INTO INV_TRANS_DIST (IN_TRANS_ID, OUT_TRANS_ID, DIST_QTY, ACT_MATERIAL_COST, ACT_LABOR_COST, ACT_BURDEN_COST, ACT_SERVICE_COST)
		VALUES (@IBT_NEW_TX_IN, @IBT_NEW_TX_OUT, @QTY, @IBT_MAT_AMT, @IBT_LAB_AMT, @IBT_BUR_AMT, @IBT_SER_AMT)
	END

END
