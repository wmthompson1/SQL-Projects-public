# Planning and ECN Counts (20220203)

```sql
USE DATAWAREHOUSE
GO

--SELECT * FROM  LIVESupplemental.dbo.PlannersList P

DECLARE @STARTDATE DATE = '9/1/2021'
, @ENDDATE DATE = '1/31/2022'

SELECT WO.CREATE_DATE
	, WO.[TYPE]
	, WO.BASE_ID
	, WO.[STATUS]
	, dbo.ufnINITIALS(WO.ENGINEERED_BY) AS INITIALS
	, P.[LOCATION]
FROM WORK_ORDER WO 
LEFT OUTER JOIN LIVESupplemental.dbo.PlannersList P
	ON dbo.ufnINITIALS(WO.ENGINEERED_BY) = P.FILTER_STRING
WHERE WO.CREATE_DATE >= @STARTDATE
	AND WO.CREATE_DATE <= @ENDDATE
	AND WO.[TYPE] IN ('W')
	AND WO.[STATUS] IN ('R', 'C')
	AND WO.SUB_ID = '0'
	AND WO.CREATE_DATE >= '11/1/2021' 
	AND WO.CREATE_DATE <= '11/30/2021'
	AND ISNULL([LOCATION], '') NOT IN ('P1', 'BALLARD')



SELECT MONTH(CREATE_DATE) AS MONTH_DATE
	, DATEPART(WEEK, CREATE_DATE) AS WEEKNUM
	, CASE
		WHEN [LOCATION] IN ('P1', 'BALLARD') THEN 'FIN'
		ELSE 'MFG'
	END AS [LOCATION]
	, COUNT(*) AS WO_COUNT
FROM (SELECT WO.CREATE_DATE
	, WO.[TYPE]
	, WO.BASE_ID
	, WO.[STATUS]
	, dbo.ufnINITIALS(WO.ENGINEERED_BY) AS INITIALS
	, P.[LOCATION]
FROM WORK_ORDER WO 
LEFT OUTER JOIN LIVESupplemental.dbo.PlannersList P
	ON dbo.ufnINITIALS(WO.ENGINEERED_BY) = P.FILTER_STRING
WHERE WO.CREATE_DATE >= @STARTDATE
	AND WO.CREATE_DATE <= @ENDDATE
	AND WO.[TYPE] IN ('W')
	AND WO.[STATUS] IN ('R', 'C')
	AND WO.SUB_ID = '0'  -- eliminate non 0 legs as known issue for updating
--	AND CO.CUSTOMER_ID IN (@CUST_ID)
) X
GROUP BY MONTH(CREATE_DATE)
	, DATEPART(WEEK, CREATE_DATE) 
	, CASE
		WHEN [LOCATION] IN ('P1', 'BALLARD') THEN 'FIN'
		ELSE 'MFG'
	END
ORDER BY MONTH(CREATE_DATE)
	, DATEPART(WEEK, CREATE_DATE) 
	, CASE
		WHEN [LOCATION] IN ('P1', 'BALLARD') THEN 'FIN'
		ELSE 'MFG'
	END

SELECT MONTH(REQUEST_DATE) AS MONTH_DATE
	, REASON
	, COUNT(*) AS ECN_COUNT
FROM EC 
WHERE REQUEST_DATE >= @STARTDATE
	AND REQUEST_DATE <= @ENDDATE
--	AND ISNULL(EC.REASON, '_Blank') LIKE 'PLANNER%'
	AND EC.ID LIKE 'EC%'
GROUP BY MONTH(REQUEST_DATE), REASON
ORDER BY MONTH(REQUEST_DATE), REASON

```
