# Parts Inventory with Trace

```sql
USE DATAWAREHOUSE
GO

--VISUAL ENTERPRISE > PARTS > PARTS INVENTORY WITH TRACE

WITH CTE_TRACE AS
(SELECT I.WAREHOUSE_ID
	, I.LOCATION_ID
	, I.PART_ID
	, P.[DESCRIPTION]
	, T.TRACE_ID
	, P.COMMODITY_CODE
	, P.STOCK_UM
	, PS.UNIT_MATERIAL_COST
	, SUM(T.QTY) AS QTY
	, PS.UNIT_MATERIAL_COST * SUM(T.QTY) AS TOTAL_COST
	--, MAX(T.CREATE_DATE) AS LASTTRACEDATE
	, MAX(T.TRANSACTION_ID) AS LASTTRANSXCT
FROM PART P
	INNER JOIN PART_SITE PS
		ON P.ID = PS.PART_ID
	INNER JOIN TRACE_INV_TRANS T
		ON T.PART_ID = P.ID
	INNER JOIN INVENTORY_TRANS I
		ON T.TRANSACTION_ID = I.TRANSACTION_ID
		AND T.PART_ID = I.PART_ID
WHERE ISNULL(P.COMMODITY_CODE, 'DETAILS') IN ('DETAILS','RAW MATERIAL','STANDARDS')
	AND I.WAREHOUSE_ID <> '1'
GROUP BY I.WAREHOUSE_ID
	, I.LOCATION_ID
	, I.PART_ID
	, P.[DESCRIPTION]
	, T.TRACE_ID
	, P.COMMODITY_CODE
	, P.STOCK_UM
	, PS.UNIT_MATERIAL_COST
HAVING SUM(T.QTY) > 0
)

, CTE_PARTPRICE AS
(SELECT C.PART_ID
	, SL3.UNIT_PRICE
	, sl3.PACKLIST_ID
	, sl3.LINE_NO
FROM SHIPPER_LINE SL3
INNER JOIN 
	(SELECT b.PART_ID
		, b.PACKLIST_ID
		, MAX(SL2.LINE_NO) AS LINE_NO
	FROM	
		(SELECT COL.PART_ID
			, MAX(s1.PACKLIST_ID) AS PACKLIST_ID
		from
			(SELECT COL.PART_ID
				, MAX(shipped_date) AS Shipped_date
			FROM shipper s
				inner join SHIPPER_LINE SL0
					on s.PACKLIST_ID = sl0.PACKLIST_ID
				INNER JOIN CUST_ORDER_LINE COL
					ON SL0.CUST_ORDER_ID = COL.CUST_ORDER_ID
					AND SL0.CUST_ORDER_LINE_NO = COL.LINE_NO
					AND COL.PART_ID IS NOT NULL
			GROUP BY COL.PART_ID
			) A 
			inner join shipper s1
				on a.Shipped_date = s1.SHIPPED_DATE
			inner join SHIPPER_LINE SL1
				on s1.PACKLIST_ID = sl1.PACKLIST_ID
			INNER JOIN CUST_ORDER_LINE COL
				ON SL1.CUST_ORDER_ID = COL.CUST_ORDER_ID
				AND SL1.CUST_ORDER_LINE_NO = COL.LINE_NO
				AND COL.PART_ID = A.PART_ID
		GROUP BY COL.PART_ID
		) B
	INNER JOIN SHIPPER_LINE SL2
		ON SL2.PACKLIST_ID = b.PACKLIST_ID
	WHERE EXISTS
		(SELECT 1 
		FROM CUST_ORDER_LINE COL
		WHERE SL2.CUST_ORDER_ID = COL.CUST_ORDER_ID
			AND SL2.CUST_ORDER_LINE_NO = COL.LINE_NO
			AND COL.PART_ID = B.PART_ID
		)
	GROUP BY B.PART_ID
		, B.PACKLIST_ID 
	) C
		ON SL3.PACKLIST_ID = c.PACKLIST_ID
		AND SL3.LINE_NO = c.LINE_NO
)
--SELECT * FROM CTE_PARTPRICE

SELECT T.*
	, PP.UNIT_PRICE
	, T2.CREATE_DATE
	, I2.[USER_ID]
	, I2.[DESCRIPTION] AS NOTES
FROM CTE_TRACE T
	INNER JOIN TRACE_INV_TRANS T2
		ON T.LASTTRANSXCT = T2.TRANSACTION_ID
		AND T.PART_ID = T2.PART_ID
		AND T.TRACE_ID = T2.TRACE_ID
	LEFT OUTER JOIN INVENTORY_TRANS I2
		ON T.LASTTRANSXCT = I2.TRANSACTION_ID
		AND T2.PART_ID = I2.PART_ID
		AND T.WAREHOUSE_ID = I2.WAREHOUSE_ID 
		AND T.LOCATION_ID = I2.LOCATION_ID
	LEFT OUTER JOIN CTE_PARTPRICE PP
		ON T.PART_ID = PP.PART_ID

```
