# Production Run Times (Ann)

```sql
use live
go

SELECT WO.BASE_ID
	, WO.LOT_ID
	, WO.DESIRED_QTY
	, WO.RECEIVED_QTY
	, WO.CREATE_DATE
	, O.SEQUENCE_NO
	, O.RESOURCE_ID
	, O.SETUP_HRS
	, O.RUN
	, O.RUN_TYPE
	, O.RUN_HRS
	, IT.TRANSACTION_DATE
	, WO.PART_ID
	, IT.TYPE
	, IT.CLASS
	, SR.SKILLS_TYPE
	, O.ACT_SETUP_HRS
	, O.ACT_RUN_HRS
	, WO.SUB_ID
FROM INVENTORY_TRANS IT 
INNER JOIN WORK_ORDER WO 
	ON IT.WORKORDER_BASE_ID = WO.BASE_ID
	AND IT.WORKORDER_LOT_ID = WO.LOT_ID
	AND IT.WORKORDER_SPLIT_ID = WO.SPLIT_ID
	AND IT.WORKORDER_SUB_ID = WO.SUB_ID 
INNER JOIN OPERATION O
	ON WO.[TYPE] = O.WORKORDER_TYPE
	AND WO.BASE_ID = O.WORKORDER_BASE_ID
	AND WO.LOT_ID = O.WORKORDER_LOT_ID
	AND WO.SPLIT_ID = O.WORKORDER_SPLIT_ID
	AND WO.SUB_ID = O.WORKORDER_SUB_ID 
INNER JOIN SHOP_RESOURCE SR 
	ON O.RESOURCE_ID = SR.ID
WHERE IT.CLASS = 'R'
	AND IT.[TYPE] = 'I'
	AND SR.USER_1 = 'M'
	AND (IT.TRANSACTION_DATE >= '1/1/2020'
		AND IT.TRANSACTION_DATE < '10/1/2020')
ORDER BY WO.BASE_ID
```
