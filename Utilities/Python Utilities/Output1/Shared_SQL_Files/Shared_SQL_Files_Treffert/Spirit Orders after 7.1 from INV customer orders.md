# Spirit Orders after 7.1 from INV customer orders

```sql
USE LIVE
GO


SELECT R.INVOICE_ID
	, R.CUSTOMER_ID
	, R.INVOICE_DATE
	, COL.CUST_ORDER_ID
	, COL.CUSTOMER_PART_ID
FROM RECEIVABLE R
LEFT OUTER JOIN RECEIVABLE_LINE  RL
	ON RL.INVOICE_ID = R.INVOICE_ID
LEFT OUTER JOIN CUST_ORDER_LINE COL
	ON COL.CUST_ORDER_ID = RL.CUST_ORDER_ID
	AND COL.LINE_NO = RL.CUST_ORDER_LINE_NO
WHERE R.CUSTOMER_ID = 'SPIAER' 
	AND R.CREATE_DATE >= '7/1/2021' 
	AND TOTAL_AMOUNT > 0 
	AND COL.CUST_ORDER_ID LIKE '%INV%' 
	AND COL.CUSTOMER_PART_ID IS NOT NULL


```
