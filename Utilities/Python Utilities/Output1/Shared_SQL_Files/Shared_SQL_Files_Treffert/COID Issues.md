# COID Issues

```sql
use live
GO

SELECT *
FROM
(SELECT CO.CUSTOMER_ID
	, C.USER_9 AS PRODCODE
	, UDF.STRING_VAL AS BUSINESS_UNIT
	, C.SALESREP_ID
	, CA.ADDR_NO
	, CA.SALESREP_ID AS CA_SALESREP
	, BA.Business_Unit AS BA_BUSUNIT
	, REPLACE(CB.[USER_ID], '#', '') AS USERID
	, CO.ID CUST_ORDER_ID 
	, COL.LINE_NO
	, COL.PART_ID
	, COL.MISC_REFERENCE
	, COL.SERVICE_CHARGE_ID
	, COL.PRODUCT_CODE
	, CO.CUSTOMER_PO_REF
	, CO.CREATE_DATE
	, CO.DESIRED_SHIP_DATE
	, CO.[STATUS]
	, P.PLANNER_USER_ID
	, P.BUYER_USER_ID
	--, CO2.ID AS CO2
	, CASE
		WHEN CO.CUSTOMER_ID = 'BOE609' AND PO.[Valid PORefs] IS NULL THEN 'INVALID POREF'
		ELSE NULL
	END AS POREF
	, CASE
		WHEN CO.CUSTOMER_ID = 'BOE609' AND DSL.SUPPLY_BASE_ID IS NULL  THEN 'UNLINKED ORDER'
		ELSE NULL
	END AS UNLINKED
	, CASE
		WHEN COALESCE(COL.PART_ID, COL.MISC_REFERENCE, COL.SERVICE_CHARGE_ID, 'NOTHING') = 'NOTHING' THEN 'MISSING LINE DEF'
		WHEN COL.ORDER_QTY = 0 AND COL.LINE_STATUS = 'C' THEN NULL
		WHEN COL.ORDER_QTY = 0 THEN 'ORDER QTY = 0'
		ELSE NULL
	END AS LINEDEF
	, CASE
		WHEN LEN(CO.ID) != 6 
			AND CO.CUSTOMER_ID  NOT IN ('BOE605','BOE609', 'BOE610', 'BOE614','BOE619', 'BOEPOP', 'BOE602')
	--		AND CO.CUSTOMER_PO_REF NOT IN ('PO101930', 'P421099-01', '745247')    -- KNOWN ANOMALIES
			AND (RIGHT(CO.ID, 1) NOT IN ('A', 'B'))
			THEN 'INVALID STD LENGTH'
		WHEN CO.CUSTOMER_ID = 'BOE609' AND ISNUMERIC(REPLACE(REPLACE(CO.ID, 'A',''), 'F', '')) = 0 THEN 'POTENTIAL COID ISSUE'
		WHEN CO.CUSTOMER_ID = 'BOE609' AND LEN(REPLACE(CO.ID, 'A','')) != 9 THEN 'INVALID CMP LENGTH'
		ELSE NULL
	END AS IDLENGTH
	, CASE
		WHEN CO2.ID IS NOT NULL THEN 'POSSIBLE DUPLICATION'
		ELSE NULL
	END AS DUPLICATES
	, CASE 
		WHEN COL.PRODUCT_CODE IS NULL THEN 'NO COL PRODCODE'
		ELSE NULL
	END AS PRODCODES
	, CASE 
		WHEN C.USER_9 IS NULL AND C.ACTIVE_FLAG = 'Y' THEN 'NO CLIENT PRODCODE'
		ELSE NULL
	END AS CUSTPC
	, CASE
		WHEN C.ACTIVE_FLAG = 'Y' AND UDF.STRING_VAL IS NULL THEN 'MISSING CUST BUSUNIT'
		ELSE NULL
	END AS BUSUNIT
	, CASE
		WHEN C.ACTIVE_FLAG = 'Y' AND C.ID NOT IN ('BOE609', 'SYSTEC', 'PRIMUS', 'RAMMOU', 'CYGINC', 'VANSCH', 'MOTDUC') 
			AND UDF.STRING_VAL != BA.Business_Unit THEN 'SALESREP ISSUE'
		ELSE NULL
	END AS SALESREP
FROM CUSTOMER_ORDER CO
LEFT OUTER JOIN CUST_ORDER_LINE COL
	ON COL.CUST_ORDER_ID = CO.ID
LEFT OUTER JOIN CUSTOMER_BOOKINGS CB
	ON COL. CUST_ORDER_ID = CB.CUST_ORDER_ID
LEFT OUTER JOIN CUSTOMER C
	ON CO.CUSTOMER_ID = C.ID
LEFT OUTER JOIN CUST_ADDRESS CA
	ON C.ID = CA.CUSTOMER_ID
	AND CO.SHIP_TO_ADDR_NO = CA.ADDR_NO
LEFT OUTER JOIN DEMAND_SUPPLY_LINK DSL
	ON COL.CUST_ORDER_ID = DSL.DEMAND_BASE_ID
	AND COL.LINE_NO = DSL.DEMAND_SEQ_NO
LEFT OUTER JOIN CUSTOMER_ORDER CO2   -- DUPLICATES
	ON CO2.CUSTOMER_ID = 'BOE609'
	AND CO.ID = '0' + CO2.ID
LEFT OUTER JOIN USER_DEF_FIELDS UDF
	ON C.ID = UDF.DOCUMENT_ID
	AND UDF.ID = 'UDF-0000072'
LEFT OUTER JOIN PART P
	ON P.ID = COL.PART_ID
LEFT OUTER JOIN LIVESupplemental.dbo.buyer_assn BA
	ON C.SALESREP_ID = BA.Salesperson_ID
LEFT OUTER JOIN LIVESupplemental.dbo.[BOE609 PORefs] PO
	ON CO.CUSTOMER_PO_REF = PO.[Valid PORefs]
WHERE CO.CREATE_DATE >= DATEADD(DAY,1,(EOMONTH(DATEADD(MONTH,-2,GETDATE()))))  --> FOM
	AND CO.[STATUS] NOT IN ('C', 'X')
	AND NOT (RIGHT(CO.ID, 3) = 'INV' AND CO.CUSTOMER_ID = 'SPIAER')
) X
WHERE POREF IS NOT NULL
	OR UNLINKED IS NOT NULL
	OR LINEDEF IS NOT NULL
	OR IDLENGTH IS NOT NULL
	--OR DUPLICATES IS NOT NULL
	OR PRODCODES IS NOT NULL
	OR CUSTPC IS NOT NULL
	OR BUSUNIT IS NOT NULL
	OR SALESREP IS NOT NULL
GROUP BY CUSTOMER_ID
	, PRODCODE
	, BUSINESS_UNIT
	, SALESREP_ID
	, ADDR_NO
	, CA_SALESREP
	, BA_BUSUNIT
	, USERID
	, CUST_ORDER_ID 
	, LINE_NO
	, PART_ID
	, MISC_REFERENCE
	, SERVICE_CHARGE_ID
	, PRODUCT_CODE
	, CUSTOMER_PO_REF
	, CREATE_DATE
	, DESIRED_SHIP_DATE
	, [STATUS]
	, PLANNER_USER_ID
	, BUYER_USER_ID
	, POREF
	, UNLINKED
	, LINEDEF
	, IDLENGTH
	, DUPLICATES
	, PRODCODES
	, CUSTPC
	, BUSUNIT
	, SALESREP

```
