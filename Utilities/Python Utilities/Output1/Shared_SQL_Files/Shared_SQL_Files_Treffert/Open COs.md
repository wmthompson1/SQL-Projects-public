```sql
use live
go

select distinct 
	C.SALESREP_ID,
	C.ID,
	CO.ORDER_DATE,
	COL.CUST_ORDER_ID 
FROM	
	CUST_ORDER_LINE COL 
inner join 
	(select CUST_ORDER_ID, sum(ORDER_QTY) as OrdQty, sum(TOTAL_SHIPPED_QTY) as ShipQty
	from CUST_ORDER_LINE COL2
	group by CUST_ORDER_ID
	having sum(TOTAL_SHIPPED_QTY) - sum(ORDER_QTY) >= 0
	) COL2
	on col2.CUST_ORDER_ID = col.CUST_ORDER_ID
INNER JOIN 
	CUSTOMER_ORDER CO 
	ON  COL.CUST_ORDER_ID = CO.ID
INNER JOIN 
	CUSTOMER C
	ON  CO.CUSTOMER_ID = C.ID
LEFT OUTER JOIN  
	DEMAND_SUPPLY_LINK AS DSL 
	ON  DSL.DEMAND_BASE_ID = COL.CUST_ORDER_ID
	AND DSL.DEMAND_SEQ_NO = COL.LINE_NO
LEFT OUTER JOIN 
	WORK_ORDER WO 
	ON  DSL.SUPPLY_BASE_ID = WO.BASE_ID 
	AND DSL.SUPPLY_LOT_ID = WO.LOT_ID 
	AND DSL.SUPPLY_SPLIT_ID = WO.SPLIT_ID 
	AND DSL.SUPPLY_SUB_ID = WO.SUB_ID 
	AND DSL.SUPPLY_TYPE  = 'WO'
LEFT OUTER JOIN 
	WORK_ORDER RWO 
	ON  RWO.BASE_ID = 'RW' + CAST(WO.BASE_ID AS VARCHAR)
	AND RWO.LOT_ID = WO.LOT_ID
	AND RWO.SPLIT_ID = WO.SPLIT_ID
	AND RWO.SUB_ID = WO.SUB_ID
	AND RWO.TYPE  = 'WO'
LEFT OUTER JOIN 
	WORK_ORDER WOR 
	ON  WOR.BASE_ID = CAST(WO.BASE_ID AS VARCHAR) + 'R'
	AND WOR.LOT_ID = WO.LOT_ID
	AND WOR.SPLIT_ID = WO.SPLIT_ID
	AND WOR.SUB_ID = WO.SUB_ID
	AND WOR.TYPE  = 'WO'
LEFT OUTER JOIN 
	(select CUST_ORDER_ID, CUST_ORDER_LINE_NO, sum (qty) as InvoiceQty
	from RECEIVABLE_LINE RL
	group by CUST_ORDER_ID, CUST_ORDER_LINE_NO
	) RL
	ON  COL.CUST_ORDER_ID = RL.CUST_ORDER_ID
	AND COL.LINE_NO = RL.CUST_ORDER_LINE_NO
	and rl.InvoiceQty > 0
WHERE 
	RL.CUST_ORDER_ID IS NOT NULL
	AND CO.STATUS = 'R' 
	AND COL.PART_ID IS NOT NULL
	and ShipQty - OrdQty >= 0
	and WO.[status] = 'C'
	and (RWO.[status] = 'C' or RWO.[status] is NULL)
	and (WOR.[status] = 'C' or WOR.[status] is NULL)
--  AND NOT (CO.CUSTOMER_ID in ('boedef', 'boepop'))
--	and CO.ORDER_DATE >= '2015-01-01'
--	AND CO.ORDER_DATE < '2016-12-31' 
--	AND COL.DESIRED_SHIP_DATE IS NULL
--	AND CO.DESIRED_SHIP_DATE <= getdate()-7
--	and COL.ORDER_QTY - COL.TOTAL_SHIPPED_QTY <= 0
order by 
	C.SALESREP_ID,
	C.ID,
	CO.ORDER_DATE,
	COL.CUST_ORDER_ID



```
