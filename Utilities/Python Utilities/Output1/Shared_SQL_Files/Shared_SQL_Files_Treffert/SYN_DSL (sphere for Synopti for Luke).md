```sql
USE [LIVE]
GO

/****** Object:  View [dbo].[SYN_DSL]    Script Date: 9/4/2020 10:55:33 AM ******/
SET ANSI_NULLS ON
GO

SET QUOTED_IDENTIFIER ON
GO

ALTER VIEW [dbo].[SYN_DSL] AS 

SELECT CO.ID AS CO_ID
, CO.CUSTOMER_ID
, CO.ORDER_DATE AS CO_ORDER_DATE
, WO.BASE_ID
, WO.LOT_ID
, WO.SPLIT_ID
, WO.SUB_ID
, WO.PART_ID
, WO.CREATE_DATE AS WO_CREATE_DATE
, WO.CLOSE_DATE AS WO_CLOSE_DATE
, WO.[STATUS] AS WO_STATUS
,  CASE   
	WHEN SR.USER_1 = 'B' THEN 'FINISH'
	WHEN SR.USER_1 = 'A' THEN 'FINISH'
	WHEN SR.USER_1 = 'M' THEN 'MANUFACTURING'
	WHEN SR.USER_1 = 'O' THEN 'BUSINESS SOLUTIONS'
END AS BUSINESS_UNIT
, LT.TRANSACTION_DATE AS LT_TRANSDATE
, LT.HOURS_WORKED
FROM [LIVE].[dbo].WORK_ORDER WO
LEFT OUTER JOIN [LIVE].[dbo].OPERATION O
	ON O.WORKORDER_BASE_ID = WO.BASE_ID
	AND O.WORKORDER_LOT_ID = WO.LOT_ID
	AND O.WORKORDER_SPLIT_ID = WO.SPLIT_ID
	AND O.WORKORDER_SUB_ID = WO.SUB_ID
	AND O.WORKORDER_TYPE = WO.[TYPE]
LEFT OUTER JOIN [LIVE].[dbo].SHOP_RESOURCE SR
	ON SR.ID = O.RESOURCE_ID
LEFT OUTER JOIN [LIVE].[dbo].[DEMAND_SUPPLY_LINK] DSL
	ON DSL.[SUPPLY_TYPE] = 'WO'
	AND DSL.SUPPLY_BASE_ID = WO.BASE_ID
	AND DSL.SUPPLY_LOT_ID = WO.LOT_ID
	AND DSL.SUPPLY_SPLIT_ID = WO.SPLIT_ID
	AND DSL.SUPPLY_SUB_ID = WO.SUB_ID
--	AND DSL.SUPPLY_SEQ_NO = O.SEQUENCE_NO
--LEFT OUTER JOIN [LIVE].[dbo].CUST_ORDER_LINE COL
--	ON DSL.DEMAND_TYPE = 'CO'
--	AND DSL.DEMAND_BASE_ID = COL.CUST_ORDER_ID
--	AND DSL.DEMAND_SEQ_NO = COL.LINE_NO
--	AND COL.PART_ID IS NOT NULL
LEFT OUTER JOIN [LIVE].[dbo].CUSTOMER_ORDER CO
	ON DSL.DEMAND_BASE_ID = CO.ID
	AND CO.ORDER_DATE >= '1/1/2019'
LEFT OUTER JOIN [LIVE].[dbo].LABOR_TICKET LT
	ON LT.WORKORDER_BASE_ID = O.WORKORDER_BASE_ID
	AND LT.WORKORDER_LOT_ID = O.WORKORDER_LOT_ID
	AND LT.WORKORDER_SPLIT_ID = O.WORKORDER_SPLIT_ID
	AND LT.WORKORDER_SUB_ID = O.WORKORDER_SUB_ID
	AND LT.WORKORDER_TYPE = O.WORKORDER_TYPE
	AND LT.OPERATION_SEQ_NO = O.SEQUENCE_NO
	AND LT.TRANSACTION_DATE >= '1/1/2019'
WHERE WO.CREATE_DATE >= '1/1/2019'
	AND WO.[TYPE] = 'W'
	AND WO.[STATUS] NOT IN ('F', 'U')

GO
```
