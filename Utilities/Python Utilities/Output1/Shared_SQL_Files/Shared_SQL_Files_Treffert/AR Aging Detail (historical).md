# AR Aging Detail (historical)

```sql
USE LIVE
GO

--> NEED TO CHECK OUT FOR WHEN PAID AND/OR VOIDED
--> PURC_MATL_DETAIL AND PURCHASE_DIST

IF OBJECT_ID('tempdb..#T1') IS NOT NULL DROP TABLE #T1

SET DATEFIRST 7; -- Set's sunday as first day of week, won't work otherwise

DECLARE @STARTDATE DATE = '01/01/2021'
, @ENDDATE DATETIME = '12/31/2022'
, @CURRENTDATE DATETIME

DECLARE @TABLEOFDATES TABLE(DATEVALUE DATETIME)

SET @CURRENTDATE = @STARTDATE

WHILE @CURRENTDATE <= @ENDDATE
	BEGIN
		INSERT INTO @TABLEOFDATES(DATEVALUE) VALUES (@CURRENTDATE)
		SET @CURRENTDATE = DATEADD(DAY, 1, @CURRENTDATE)
	END

SELECT * 
INTO #T1
FROM @TABLEOFDATES 
WHERE DATEPART(WEEKDAY, DATEVALUE) = 2

--SELECT * FROM #T1

SELECT T.DATEVALUE
	, DATEDIFF(DD, DATEADD(DD, PERIOD_DAYS, INVOICE_DATE), T.DATEVALUE) AS CATEGORY
	, CASE
		WHEN DATEDIFF(DD, DATEADD(DD, PERIOD_DAYS, INVOICE_DATE), T.DATEVALUE) <= 0 THEN TOTAL_AMOUNT
		ELSE 0
	END AS CURRENT_AMOUNT
	, CASE
		WHEN DATEDIFF(DD, DATEADD(DD, PERIOD_DAYS, INVOICE_DATE), T.DATEVALUE) BETWEEN 1 AND 30 THEN TOTAL_AMOUNT
		ELSE 0
	END AS [30 DAYS PAST DUE]
	, CASE
		WHEN DATEDIFF(DD, DATEADD(DD, PERIOD_DAYS, INVOICE_DATE), T.DATEVALUE) BETWEEN 31 AND 45 THEN TOTAL_AMOUNT
		ELSE 0
	END AS [45 DAYS PAST DUE]
	, CASE
		WHEN DATEDIFF(DD, DATEADD(DD, PERIOD_DAYS, INVOICE_DATE), T.DATEVALUE) BETWEEN 46 AND 60 THEN TOTAL_AMOUNT
		ELSE 0
	END AS [60 DAYS PAST DUE]
	, CASE
		WHEN DATEDIFF(DD, DATEADD(DD, PERIOD_DAYS, INVOICE_DATE), T.DATEVALUE) > 60 THEN TOTAL_AMOUNT
		ELSE 0
	END AS [90 DAYS PAST DUE]
	, X.*
FROM (SELECT C.ID AS CUSTOMER_ID
	, C.[NAME] AS CUSTOMER_NAME
	, R.INVOICE_ID
	, R.INVOICE_DATE
	, R.CREATE_DATE
	, CR.POSTING_DATE
	, CASE
		WHEN R.TERMS_DESCRIPTION = 'Due on Receipt' THEN 0
		ELSE R.TERMS_NET_DAYS
	END AS PERIOD_DAYS
	, R.TERMS_NET_DAYS
	, R.TERMS_DESCRIPTION
	, R.TOTAL_AMOUNT
	, R.PAID_AMOUNT
	, R.TOTAL_AMOUNT - R.PAID_AMOUNT AS AMOUNT_OUTSTANDING
FROM RECEIVABLE R
LEFT OUTER JOIN CUSTOMER C
	ON R.CUSTOMER_ID = C.ID
LEFT OUTER JOIN CASH_RECEIPT_LINE CRL
	ON CRL.INVOICE_ID = R.INVOICE_ID
LEFT OUTER JOIN CASH_RECEIPT CR
	ON CRL.CHECK_ID = CR.CHECK_ID
	AND CRL.CUSTOMER_ID = CRL.CUSTOMER_ID
WHERE R.[TYPE] NOT IN ('V')
	AND R.[STATUS] NOT IN ('X')

) X
--LEFT OUTER JOIN #T1 T
INNER JOIN #T1 T
ON T.DATEVALUE >= X.CREATE_DATE
AND T.DATEVALUE <= X.POSTING_DATE

ORDER BY INVOICE_DATE, INVOICE_ID, T.DATEVALUE


```
