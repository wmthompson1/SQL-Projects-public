# WO Specs Report

```sql
USE LIVE
GO 

DECLARE @TYPE NVARCHAR(10) = 'M'
, @PLANNER NVARCHAR(50) = 'MFG-MRP'
, @PART NVARCHAR(100) = '100T%'
, @STATUS NVARCHAR(10) = 'U'
, @STARTDATE DATE = '1/1/2020'

--> Home > Visual Enterprise Reports > Part Reports > WO Specs
SELECT WO.BASE_ID
	, WO.LOT_ID
	, WO.SPLIT_ID
	, P.ID
	, CASE
		WHEN WO.[TYPE] = 'M' THEN 'MASTERS'
		ELSE 'WORKORDER'
	END AS WO_TYPE
	, P.[DESCRIPTION]
	, P.PLANNER_USER_ID
	, P.BUYER_USER_ID
	, WO.CREATE_DATE
	, WO.[DESIRED_WANT_DATE]
	, WO.[DESIRED_RLS_DATE]
	, WO.CLOSE_DATE
	, CAST(CAST(WB.BITS AS VARBINARY(MAX)) AS NVARCHAR(MAX)) AS WO_SPECS
	, WO.[STATUS]
FROM WORK_ORDER WO
INNER JOIN PART P
	ON WO.PART_ID = P.ID
INNER JOIN WORKORDER_BINARY WB
	ON WO.BASE_ID = WB.WORKORDER_BASE_ID
	AND WO.LOT_ID = WB.WORKORDER_LOT_ID
	AND WO.SPLIT_ID = WB.WORKORDER_SPLIT_ID
	AND WO.SUB_ID = WB.WORKORDER_SUB_ID
	AND WO.[TYPE] = WB.WORKORDER_TYPE
WHERE WO.[TYPE] = @TYPE
	AND P.PLANNER_USER_ID IN (@PLANNER)
	AND P.ID LIKE @PART
	AND WO.[STATUS] IN (@STATUS)
	AND ISNULL(WO.CLOSE_DATE, '12/31/2018') >= '12/31/2018'
	AND (WO.[TYPE] = 'M' 
		OR WO.[TYPE] = 'W' AND ISNULL(WO.CREATE_DATE, GETDATE()) >= @STARTDATE
		)
```
