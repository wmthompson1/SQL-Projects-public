# Trace Inventory Report

```sql
USE LIVE
GO

SELECT PL.PART_ID 
, P.[DESCRIPTION]
, PL.WAREHOUSE_ID 
, PL.LOCATION_ID 
, PL.QTY
, TP.NUMBERING_ID
FROM PART P
INNER JOIN PART_LOCATION PL
	ON PL.PART_ID = P.ID
LEFT OUTER JOIN TRACE_PROFILE TP
	ON P.ID = TP.PART_ID

WHERE  PL.QTY <> 0
--AND PL.WAREHOUSE_ID = @WAREHOUSE
--AND PL.LOCATION_ID = @LOCATION
AND PL.WAREHOUSE_ID = 'STDS'
AND PL.LOCATION_ID = 'ADHESIVE'


```
