# Open Purchase Orders Without Notes

```sql
USE LIVE
GO

SELECT 
	PO.VENDOR_ID, 
	PO.ID as Purchase_Order, 
	PL.LINE_NO,
	PL.PART_ID, 
	P.COMMODITY_CODE,
	PO.STATUS, 
	PL.LINE_STATUS, 
	CONVERT(VARCHAR(11),PO.ORDER_DATE,101) as PO_OrderDate, 
	CASE 
		WHEN PL.DESIRED_RECV_DATE IS NULL  THEN CONVERT(VARCHAR(11),PO.DESIRED_RECV_DATE,101)
		ELSE CONVERT(VARCHAR(11),PL.DESIRED_RECV_DATE,101)
	END AS Desired_Recv_Date,
	wo.[BASE_ID] as Work_Order_Base,
	wo.[LOT_ID] as Work_Order_Lot

FROM 
	LIVE.DBO.PURC_ORDER_LINE PL 

INNER JOIN 
	LIVE.DBO.PURCHASE_ORDER PO 
		ON PL.PURC_ORDER_ID = PO.ID

LEFT OUTER JOIN 
	[DBO].[PURC_LINE_BINARY] PLB
		ON PL.PURC_ORDER_ID = PLB.PURC_ORDER_ID
		AND  PL.LINE_NO = PLB.PURC_ORDER_LINE_NO

LEFT OUTER JOIN 
	LIVE.DBO.PART P
		ON PL.PART_ID = P.ID

left outer join 
	[dbo].[REQUIREMENT] r
		ON P.ID = r.PART_ID

left outer join 
	[dbo].[WORK_ORDER] wo
		ON r.[WORKORDER_TYPE] = wo.[TYPE]
		and r.[WORKORDER_BASE_ID] = wo.[BASE_ID]
		and r.[WORKORDER_LOT_ID] = wo.[LOT_ID]
		and r.[WORKORDER_SPLIT_ID] = wo.[SPLIT_ID]
		and r.[WORKORDER_SUB_ID] = wo.[SUB_ID]

WHERE 
	PL.LINE_STATUS='A' 
	AND NOT (PO.STATUS='C' OR PO.STATUS='X')
	AND PLB.PURC_ORDER_ID IS NULL
	AND P.COMMODITY_CODE IN ('RAW MATERIAL', 'STANDARDS')
	and wo.status = 'R'

ORDER BY 
	PO.VENDOR_ID,  
	PO.ID,  
	PL.LINE_NO
```
