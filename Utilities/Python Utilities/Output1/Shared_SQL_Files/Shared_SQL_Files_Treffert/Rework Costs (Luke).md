USE DataWarehouse

	RELEASED AND COMPLETE/CLOSED WORKORDERS ONLY
	COSTS FROM OPERATIONS OF REWORKS ONLY (NOT THE TOTAL COST OF WO W/ LEGS)

SET TRANSACTION ISOLATION LEVEL READ UNCOMMITTED
SET DEADLOCK_PRIORITY LOW

IF OBJECT_ID('TEMPDB..#TEMP1') IS NOT NULL DROP TABLE #TEMP1
IF OBJECT_ID('TEMPDB..#TEMP2') IS NOT NULL DROP TABLE #TEMP2
IF OBJECT_ID('TEMPDB..#RESULTS') IS NOT NULL DROP TABLE #RESULTS
IF OBJECT_ID('TEMPDB..#REQUIRESET') IS NOT NULL DROP TABLE #REQUIRESET
IF OBJECT_ID('TEMPDB..#DATASETNCM') IS NOT NULL DROP TABLE #DATASETNCM

DECLARE @IDX NVARCHAR(30); ---- = '1549524'; 

WITH REQUIRESETX AS 
(SELECT WO.BASE_ID
	, IDX = managedData.fn_Parse_Numeric(BASE_ID)
	, WO.LOT_ID
	, WO.SUB_ID
	, WO.SPLIT_ID
--	, WO.[TYPE]
	, WO.PART_ID
	, DESCRIPT = CASE 
		WHEN  ((WO.SUB_ID > 0) AND (WO.PART_ID LIKE '%REWORK%') AND (WO.[TYPE] LIKE 'W')) THEN 'LEG'
		WHEN  ((WO.SUB_ID = 0) AND ( WO.BASE_ID LIKE '%R%') AND (WO.[TYPE] LIKE 'W')) THEN 'HEADERITEM'
	END				
FROM WORK_ORDER WO WITH (NOLOCK)
WHERE (1=1)
	AND (WO.SUB_ID > 0)	
	AND (WO.PART_ID LIKE '%REWORK%') 
	AND (WO.[TYPE] = 'W')
	AND (WO.CREATE_DATE > '2019-01-01')	
	AND (WO.[STATUS] IN ('C', 'R'))
UNION ALL
	SELECT WO.BASE_ID
		, IDX = managedData.fn_Parse_Numeric(BASE_ID)
		, WO.LOT_ID
		, WO.SUB_ID
		, WO.SPLIT_ID
--		, WO.[TYPE]
		, WO.PART_ID
		, DESCRIPT = CASE 
			WHEN  ((WO.SUB_ID > 0) AND (WO.PART_ID LIKE '%REWORK%') AND (WO.[TYPE] LIKE 'W')) THEN 'LEG'
			WHEN  ((WO.SUB_ID = 0) AND (WO.BASE_ID LIKE '%R%') AND (WO.[TYPE] LIKE 'W')) THEN 'HEADERITEM'
		END				
	FROM WORK_ORDER WO WITH (NOLOCK)
	WHERE (1=1)
		AND (WO.SUB_ID = 0) 
		AND (WO.BASE_ID LIKE '%R%') 
		AND (WO.[TYPE] LIKE 'W')
		AND (WO.CREATE_DATE > '2019-01-01')	
		AND (WO.[STATUS] IN ('C', 'R'))
)

  --REMOVE IF PARSE INCONSISTENT
, REQUIRESET AS 
(SELECT * 
FROM REQUIRESETX WO
WHERE (PART_ID IS NOT NULL)
	AND (ISNULL(PART_ID, '') != 'TEST')
	AND NOT(ISNULL(BASE_ID, '') IN ('R/T968273', 'R/T1005967', 'R/T1003608', 'RW/1/1019429', 'RW/2/1016996')
	)
)
--SELECT * FROM REQUIRESET

, DATASETWO AS
(SELECT IDX = managedData.fn_Parse_Numeric(WO.BASE_ID)
	, WO.BASE_ID
	, WO.LOT_ID
	, WO.SPLIT_ID
	, WO.SUB_ID
--	, WO.[TYPE]
	--, WO.ACT_LABOR_COST
	--, WO.ACT_BURDEN_COST
	--, WO.ACT_SERVICE_COST
	--, WO.ACT_MATERIAL_COST
	, WO.CREATE_DATE
	, WO.PART_ID
	, DESCRIPT = CASE 
		WHEN  ((WO.SUB_ID > 0) AND (WO.PART_ID LIKE '%REWORK%') AND (WO.[TYPE] LIKE 'W')) THEN 'LEG'
		WHEN  ((WO.SUB_ID = 0) AND ( WO.BASE_ID LIKE '%R%') AND (WO.[TYPE] LIKE 'W')) THEN 'HEADERITEM'
	END		
FROM WORK_ORDER WO
WHERE EXISTS (SELECT 1 
	FROM REQUIRESET RS
	WHERE RS.BASE_ID = WO.BASE_ID
	AND RS.LOT_ID = WO.LOT_ID
	AND RS.SPLIT_ID = WO.SPLIT_ID
	AND RS.SUB_ID = WO.SUB_ID
--	AND RS.[TYPE] = WO.[TYPE]
	)
)

SELECT *
, CHRONORDER = ROW_NUMBER() OVER (PARTITION BY IDX ORDER BY CREATE_DATE ASC)
INTO #REQUIRESET
FROM DATASETWO
WHERE (IDX = @IDX OR @IDX IS NULL)
ORDER BY CREATE_DATE
;

WITH DATASETNCM AS
(SELECT IDX = managedData.fn_Parse_Numeric(PO.ALT_ID_2)
	-- PROJECTION PO --  >
	, PO.ALT_ID_1  -- > TYPE	
	, PO.ALT_ID_2  -- > BASE_ID
	, PO.ALT_ID_3  -- > LOT_ID
	, PO.ALT_ID_4  -- > SPLIT_ID 
	, PO.ALT_ID_5  -- > SUB_ID
	, PO.DISPLAY_ID
	, PO.RECORD_ID
	-- PROJECTION NCM --  >
	, NCM.ACCEPTED
	, NCM.BUSINESS_UNIT_ID
	, NCM.DISPOSITION
	, NCM.DISPOSITION_TYPE
	, NCM.DISP_DATE
	, NCM.DUE_DATE
	, NCM.LOT_SIZE
	, NCM.NCM_DATE
	, NCM.NCM_ID
	, NCM.NCM_TYPE
	, NCM.PO_NUMBER -- AS VENDOR_PO_NUM
	, NCM.REFERENCE_1 --AS [SPEC_SECT]
	, NCM.REFERENCE_10 --AS [ASIS_QTY]
	, NCM.REFERENCE_2 --AS [DRAW_REF]
	, NCM.REFERENCE_3 --AS [ASIS_DIM]
	, NCM.REFERENCE_4 --AS [SHOULD_BE_DIM]
	, NCM.REFERENCE_6 --AS [REWORK_QTY]
	, NCM.REFERENCE_7 --AS [SCRAP_QTY]
	, NCM.REFERENCE_8 --AS [OVERAGE_QTY]
	, NCM.REFERENCE_9 --AS [RETURN_QTY]
	, NCM.REJECTED
	, NCM.SK_FIN_NCM_PROCESS
	, NCM.SK_FIN_NCM_SUBPROCESS
	, NCM.SK_MFG_NCM_PROCESS
	, NCM.SK_MFG_NCM_SUBPROCESS
	, NCM.USER_DEFINED_3 --AS [CATEGORY]
	, NCM.USER_DEFINED_4
	, NCM.USER_DEFINED_7
	--- PROJECTION NCM /
	--- PROJECTION P -- >
	, P.PRODUCT_ID
	--- PROJECTION P /
	-- PROJECTION INFORMATIONAL -- >
	, COALESCE(NCM.SK_FIN_NCM_PROCESS, NCM.SK_MFG_NCM_PROCESS) PROCESSX
	, ISPAINT = CASE COALESCE(NCM.SK_FIN_NCM_PROCESS, NCM.SK_MFG_NCM_PROCESS)
		WHEN 'PAINT' THEN 1 ELSE 0 
	END
-- PROJECTION INFORMATIONAL / 
FROM VQ_PRODUCTION_ORDER PO WITH (NOLOCK)
JOIN VQ_NONCONFORMANCE NCM WITH (NOLOCK)
	ON PO.RECORD_ID = NCM.PROD_ORDER_REC_ID
LEFT JOIN [VQ_PRODUCT] P WITH (NOLOCK)
	ON PO.PRODUCT_REC_ID = P.RECORD_ID
WHERE (1=1)
	AND NCM.DISPOSITION_TYPE NOT IN ('OVERAGE', 'NONDISCREPANT', 'NON-DISCREPANT', 'SCRAP')	
	AND COALESCE(NCM.SK_FIN_NCM_PROCESS, NCM.SK_MFG_NCM_PROCESS, '') <> 'PENETRANT'
	AND COALESCE(NCM.SK_FIN_NCM_PROCESS, NCM.SK_MFG_NCM_PROCESS, '') <> 'NDT'
) 

SELECT *
	, CHRONORDER = ROW_NUMBER() OVER (PARTITION BY IDX ORDER BY ISPAINT DESC, NCM_DATE DESC)
INTO #DATASETNCM
FROM DATASETNCM
WHERE (IDX = @IDX OR @IDX IS NULL)
ORDER BY NCM_DATE

--SELECT * FROM #REQUIRESET W 
--SELECT * FROM #DATASETNCM P

SELECT WO.IDX
	, WO.CHRONORDER
	, WO.DESCRIPT
	, WO.CREATE_DATE
	, WO.PART_ID
	, WO.BASE_ID
	, WO.LOT_ID
	, WO.SPLIT_ID
	, WO.SUB_ID
--	, WO.[TYPE]
	, SUM(OP.ACT_ATL_LAB_COST) AS ATL_LABOR_COST
	, SUM(OP.ACT_ATL_BUR_COST) AS ATL_BURDEN_COST
	, SUM(OP.ACT_ATL_SER_COST) AS ATL_SERVICE_COST
	, SUM(OP.ACT_TTL_MAT_COST) AS TTL_MATERIAL_COST
	, SUM(OP.ACT_TTL_LAB_COST) AS TTL_LABOR_COST
	, SUM(OP.ACT_TTL_BUR_COST) AS TTL_BURDEN_COST
	, SUM(OP.ACT_TTL_SER_COST) AS TTL_SERVICE_COST
	, SUM(OP.ACT_ATL_LAB_COST) + SUM(OP.ACT_ATL_BUR_COST) + SUM(OP.ACT_ATL_SER_COST) 
		+ SUM(OP.ACT_TTL_MAT_COST) + SUM(OP.ACT_TTL_LAB_COST) + SUM(OP.ACT_TTL_BUR_COST) 
		+ SUM(OP.ACT_TTL_SER_COST) AS [TOTAL OP COST]
--	, SUM(ISNULL(OP.ACT_ATL_BUR_COST,0) + ISNULL(OP.ACT_ATL_LAB_COST,0) + ISNULL(OP.ACT_ATL_SER_COST,0)) AS [TOTAL OPERATION COST]
--	, MAX(WO.ACT_BURDEN_COST + WO.ACT_SERVICE_COST + WO.ACT_LABOR_COST + WO.ACT_MATERIAL_COST) AS [TOTAL WO COST]
	, NCM.DISPLAY_ID AS PRODUCTION_ORDER_ID
	, NCM.ALT_ID_1 AS WORKORDER_TYPE
	, NCM.ALT_ID_2 AS WORKORDER_BASE_ID
	, NCM.ALT_ID_3 AS WORKORDER_LOT_ID
	, NCM.ALT_ID_4 AS WORKORDER_SPLIT_ID
	, NCM.ALT_ID_5 AS WORKORDER_SUB_ID
	, NCM.PO_NUMBER AS VENDOR_PO_NUM
	, NCM.LOT_SIZE
	, NCM.ACCEPTED
	, NCM.REJECTED
	, NCM.NCM_DATE
	, NCM.DUE_DATE
	, NCM.REFERENCE_1 AS [SPEC_SECT]
	, NCM.REFERENCE_2 AS [DRAW_REF]
	, NCM.REFERENCE_3 AS [ASIS_DIM]
	, NCM.REFERENCE_4 AS [SHOULD_BE_DIM]
	, COALESCE(NCM.SK_FIN_NCM_PROCESS, NCM.SK_MFG_NCM_PROCESS) AS [PROCESS]
	, COALESCE(NCM.SK_FIN_NCM_SUBPROCESS, NCM.SK_MFG_NCM_SUBPROCESS) AS [SUBPROCESS]
	, NCM.USER_DEFINED_3 AS [CATEGORY]
	, ISNULL(NCM.USER_DEFINED_4, NCM.USER_DEFINED_7) AS [DEFECT]
	, NCM.DISPOSITION_TYPE
	, NCM.DISP_DATE
	, NCM.REFERENCE_6 AS [REWORK_QTY]
	, NCM.REFERENCE_7 AS [SCRAP_QTY]
	, NCM.REFERENCE_8 AS [OVERAGE_QTY]
	, NCM.REFERENCE_9 AS [RETURN_QTY]
	, NCM.REFERENCE_10 AS [ASIS_QTY]
	, NCM.DISPOSITION
	, NCM.BUSINESS_UNIT_ID
	, DATEPART(WEEK,NCM.NCM_DATE) AS WEEK_NUM
	, NCM.PROCESSX AS PROCESSX
INTO #RESULTS
FROM #REQUIRESET WO 
LEFT OUTER JOIN [DBO].[OPERATION] OP WITH (NOLOCK)
	ON OP.WORKORDER_BASE_ID = WO.BASE_ID
	AND OP.WORKORDER_LOT_ID = WO.LOT_ID
	AND OP.WORKORDER_SUB_ID = WO.SUB_ID
	AND OP.WORKORDER_SPLIT_ID = WO.SPLIT_ID
--	AND OP.WORKORDER_TYPE = WO.[TYPE]
LEFT JOIN #DATASETNCM NCM 
	ON WO.CHRONORDER = NCM.CHRONORDER
	AND WO.IDX = NCM.IDX
	AND WO.LOT_ID = NCM.ALT_ID_3
	AND WO.SPLIT_ID = NCM.ALT_ID_4 
WHERE (1=1)
GROUP BY WO.IDX
	, WO.CHRONORDER
	, WO.DESCRIPT
	, NCM.NCM_ID 
	, WO.BASE_ID
	, WO.LOT_ID
	, WO.SPLIT_ID
	, WO.SUB_ID
--	, WO.[TYPE]
	, WO.CREATE_DATE
	, WO.PART_ID
	, NCM.DISPLAY_ID
	, NCM.ALT_ID_1 
	, NCM.ALT_ID_2 
	, NCM.ALT_ID_3 
	, NCM.ALT_ID_4 
	, NCM.ALT_ID_5 
	, NCM.PO_NUMBER 
	, NCM.LOT_SIZE
	, NCM.ACCEPTED
	, NCM.REJECTED
	, NCM.NCM_DATE
	, NCM.DUE_DATE
	, NCM.REFERENCE_1 
	, NCM.REFERENCE_2 
	, NCM.REFERENCE_3 
	, NCM.REFERENCE_4 
	, NCM.SK_FIN_NCM_PROCESS
	, NCM.SK_MFG_NCM_PROCESS 
	, NCM.SK_FIN_NCM_SUBPROCESS 
	, NCM.SK_MFG_NCM_SUBPROCESS
	, NCM.USER_DEFINED_3 
	, NCM.USER_DEFINED_4
	, NCM.USER_DEFINED_7 
	, NCM.DISPOSITION_TYPE
	, NCM.DISP_DATE
	, NCM.REFERENCE_6 
	, NCM.REFERENCE_7 
	, NCM.REFERENCE_8 
	, NCM.REFERENCE_9 
	, NCM.REFERENCE_10 
	, NCM.DISPOSITION
	, NCM.BUSINESS_UNIT_ID
	, NCM.PROCESSX 
ORDER BY WO.IDX
	, WO.CHRONORDER

SELECT * FROM #RESULTS WITH (NOLOCK)
ORDER BY IDX
	, CHRONORDER
