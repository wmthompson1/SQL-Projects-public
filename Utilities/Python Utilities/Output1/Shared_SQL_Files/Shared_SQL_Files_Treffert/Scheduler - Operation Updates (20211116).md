# Scheduler - Operation Updates (20211116)

```sql
USE DataWarehouse
GO

; WITH CTE_BASE AS
(SELECT WO.BASE_ID + '/' + WO.LOT_ID + CASE WHEN WO.SPLIT_ID = 0 THEN '' ELSE '.' + WO.SPLIT_ID END AS JOB_ID
	, WO.BASE_ID 
	, WO.LOT_ID 
	, WO.SPLIT_ID
	, WO.SUB_ID
	, WO.[STATUS] AS WO_STATUS
	, dbo.sfn_current_operation(WO.[TYPE], WO.BASE_ID, WO.LOT_ID, WO.SPLIT_ID, WO.SUB_ID) AS CURRENT_OP
--	, MAX(O.SEQUENCE_NO) AS CURRENTOP
FROM WORK_ORDER WO (NOLOCK) 
WHERE WO.[STATUS] IN ('F','U','R')
)

SELECT JOB_ID
	, BASE_ID 
	, LOT_ID 
	, SPLIT_ID
	, SUB_ID
	, WO_STATUS
	, CURRENT_OP
	, SEQUENCE_NO
	, O.[STATUS] AS OP_STATUS
FROM CTE_BASE CTE
INNER JOIN OPERATION O
	ON O.WORKORDER_TYPE = 'W'
	AND CTE.BASE_ID = O.WORKORDER_BASE_ID
	AND CTE.LOT_ID = O.WORKORDER_LOT_ID
	AND CTE.SPLIT_ID = O.WORKORDER_SPLIT_ID
	AND CTE.SUB_ID = O.WORKORDER_SUB_ID
	AND CTE.CURRENT_OP > O.SEQUENCE_NO
WHERE O.[STATUS] NOT IN ('C', 'X')
	AND EXISTS (SELECT 1 FROM OPERATION 
		WHERE WORKORDER_TYPE = O.WORKORDER_TYPE 
		AND WORKORDER_BASE_ID = O.WORKORDER_BASE_ID 
		AND WORKORDER_LOT_ID = O.WORKORDER_LOT_ID 
		AND WORKORDER_SPLIT_ID = O.WORKORDER_SPLIT_ID 
		AND WORKORDER_SUB_ID = O.WORKORDER_SUB_ID 
		AND SEQUENCE_NO < O.SEQUENCE_NO 
		AND [STATUS] IN ('C')
		) 
ORDER BY JOB_ID
	, BASE_ID 
	, LOT_ID 
	, SPLIT_ID
	, SUB_ID
	, SEQUENCE_NO

```
