SELECT ENGINEERED_BY
, COUNT(*)
FROM LIVE.dbo.WORK_ORDER
GROUP BY ENGINEERED_BY
ORDER BY ENGINEERED_BY

SELECT BASE_ID, ENGINEERED_BY 
FROM WORK_ORDER
WHERE ENGINEERED_BY LIKE 'FEO/ %'
AND [TYPE] = 'W'
--AND BASE_ID LIKE '16%'

SELECT ENGINEERED_BY, count(*)
FROM WORK_ORDER
WHERE ENGINEERED_BY LIKE '% %'
AND [TYPE] = 'W'
group by ENGINEERED_BY
