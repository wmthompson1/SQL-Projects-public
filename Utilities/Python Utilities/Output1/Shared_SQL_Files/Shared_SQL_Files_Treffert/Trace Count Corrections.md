# Trace Count Corrections

```sql
DECLARE @PART_ID NVARCHAR(50) = '20243C-071 CMPN'

; WITH CTE_TRACE AS
(SELECT PART_ID
	, ID AS TRACE_ID
	, SUM(IN_QTY - OUT_QTY) AS TRACE_QTY
FROM TRACE
--WHERE PART_ID = @PART_ID
GROUP BY PART_ID
, ID
)
--SELECT * FROM CTE_TRACE T
-- 542972 > -198

, CTE_TRACE_TOT AS
(SELECT PART_ID
	, SUM(IN_QTY - OUT_QTY) AS TRACE_TOT_QTY
FROM TRACE
--WHERE PART_ID = @PART_ID
GROUP BY PART_ID
)
--SELECT * FROM CTE_TRACE_TOT TT

, CTE_XCT AS
(SELECT PART_ID
	, TRACE_ID
	, SUM(QTY) AS XCT_QTY
FROM TRACE_INV_TRANS
--WHERE PART_ID = @PART_ID
GROUP BY PART_ID
	, TRACE_ID
)
--SELECT * FROM CTE_XCT X
-- 542972 > -198


, CTE_XCT_TOT AS
(SELECT PART_ID
	, SUM(QTY) AS XCT_TOT_QTY
FROM TRACE_INV_TRANS
--WHERE PART_ID = @PART_ID
GROUP BY PART_ID
)
--SELECT * FROM CTE_XCT_TOT C

, CTE_INV AS
(SELECT PART_ID
	, SUM(CASE
		WHEN TYPE = 'I' AND CLASS = 'R' THEN 1
		WHEN TYPE = 'O' AND CLASS = 'I' THEN -1
		WHEN TYPE = 'O' AND CLASS = 'A' THEN -1
		WHEN TYPE = 'I' AND CLASS = 'A' THEN 1
		WHEN TYPE = 'I' AND CLASS = 'I' THEN 1
		WHEN TYPE = 'O' AND CLASS = 'R' THEN -1
		ELSE 0
	END * QTY) AS INV_QTY
FROM INVENTORY_TRANS
--WHERE PART_ID = @PART_ID
GROUP BY PART_ID
)
--SELECT * FROM CTE_INV I

SELECT P.ID
	, P.[STATUS]
	, I.INV_QTY
	, C.XCT_TOT_QTY
	, TT.TRACE_TOT_QTY
	, T.TRACE_ID
	, T.TRACE_QTY
	, X.XCT_QTY
	, I.INV_QTY
	, ISNULL(I.INV_QTY, 0) AS INV_QTY_ISNULL
	, C.XCT_TOT_QTY
	, ISNULL(C.XCT_TOT_QTY, 0) AS XCT_TOT_ISNULL
	, TT.TRACE_TOT_QTY
	, ISNULL(TT.TRACE_TOT_QTY, 0) AS TRACE_TOT_ISNULL
	, T.TRACE_ID
	, ISNULL(T.TRACE_ID, '') AS TRACE_ID_ISNULL
	, T.TRACE_QTY
	, ISNULL(T.TRACE_QTY, 0) AS TRACE_QTY_ISNULL
	, X.XCT_QTY
	, ISNULL(X.XCT_QTY, 0) AS XCT_QTY_ISNULL
FROM PART P
LEFT OUTER JOIN TRACE_PROFILE TP
	ON P.ID = TP.PART_ID
LEFT OUTER JOIN CTE_INV I
	ON P.ID = I.PART_ID
LEFT OUTER JOIN CTE_TRACE T
	ON P.ID = T.PART_ID
LEFT OUTER JOIN CTE_XCT X
	ON P.ID = X.PART_ID
	AND X.TRACE_ID = T.TRACE_ID
LEFT OUTER JOIN CTE_XCT_TOT C
	ON P.ID = C.PART_ID
LEFT OUTER JOIN CTE_TRACE_TOT TT
	ON P.ID = TT.PART_ID
WHERE (P.[STATUS] IS NULL OR P.[STATUS] != 'O')
	AND TP.PART_ID IS NOT NULL
	AND (CASE
		WHEN T.TRACE_QTY < 0 THEN 'NEGATIVE QTY'
		WHEN ISNULL(TT.TRACE_TOT_QTY, 0) > ISNULL(C.XCT_TOT_QTY, 0) THEN 'TRACE > XCT_TOT' 
		WHEN ISNULL(TT.TRACE_TOT_QTY, 0) < ISNULL(C.XCT_TOT_QTY, 0) THEN 'TRACE < XCT_TOT' 
		WHEN ISNULL(I.INV_QTY, 0) != ISNULL(TT.TRACE_TOT_QTY, 0) THEN 'INV != TRACE_TOT'
		WHEN ISNULL(I.INV_QTY, 0) != ISNULL(C.XCT_TOT_QTY, 0) THEN 'INV != XCT_TOT' 
		ELSE ''
	END) != ''
--AND P.ID = @PART_ID
```
