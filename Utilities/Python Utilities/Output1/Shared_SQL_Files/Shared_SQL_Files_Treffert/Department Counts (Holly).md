# Department Counts (Holly)

```sql
USE DataWarehouse
GO

; WITH CTE_BASE AS 
(SELECT WORKORDER_BASE_ID
, WORKORDER_LOT_ID
, WORKORDER_SPLIT_ID
, WORKORDER_SUB_ID
--, O.CLOSE_DATE
, SR.USER_5
, SR.USER_6
, SR.USER_8
, COUNT(*) AS COUNTS
FROM OPERATION O
INNER JOIN SHOP_RESOURCE SR
ON O.RESOURCE_ID = SR.ID
WHERE O.CLOSE_DATE >= '2/1/2021'
GROUP BY WORKORDER_BASE_ID
, WORKORDER_LOT_ID
, WORKORDER_SPLIT_ID
, WORKORDER_SUB_ID
, SR.USER_5
, SR.USER_6
, SR.USER_8
)

SELECT USER_5
, USER_6
, USER_8
, COUNT(*) 
FROM CTE_BASE
GROUP BY USER_5
, USER_6
, USER_8
ORDER BY USER_5
, USER_6
, USER_8
```
