# Voucher Audit Report (20210806)

```sql
USE LIVE
GO

SELECT P.VENDOR_ID
	, V.[NAME] AS VENDOR_NAME
	, P.VOUCHER_ID
	, PL.LINE_NO AS VOUCHER_LINE
	, P.CREATE_DATE AS PAYABLE_DATE
	, PL.REFERENCE
	, P.INVOICE_ID
	, PL.PURC_ORDER_ID
	, PL.PURC_ORDER_LINE_NO
FROM LIVE.dbo.PAYABLE P
INNER JOIN LIVE.dbo.PAYABLE_LINE PL
ON P.VOUCHER_ID = PL.VOUCHER_ID
LEFT OUTER JOIN PURC_ORDER_LINE POL
ON PL.PURC_ORDER_ID = POL.PURC_ORDER_ID
AND PL.PURC_ORDER_LINE_NO = POL.LINE_NO
LEFT OUTER JOIN PURCHASE_ORDER PO
ON PL.PURC_ORDER_ID = PO.ID
LEFT OUTER JOIN VENDOR V
ON P.VENDOR_ID = V.ID
WHERE P.CREATE_DATE >= '1/1/2020'
	AND P.INVOICE_ID NOT LIKE '%CLR'
	AND P.VOUCHER_ID NOT IN (SELECT CAST(F.Voucher# AS NVARCHAR(25)) AS VOUCHER_ID
		FROM dbRMS.dbo.TBLFILES F
		WHERE ISNULL(F.Voucher#, '0') != '0'
			--AND F.SECTION = 'INVOICE'
		)
GROUP BY P.VOUCHER_ID
	, PL.LINE_NO
	, PL.REFERENCE
	, P.CREATE_DATE
	, P.INVOICE_ID
	, PL.PURC_ORDER_ID
	, PL.PURC_ORDER_LINE_NO
	, P.VENDOR_ID
	, V.[NAME]
ORDER BY P.VOUCHER_ID
	, PL.LINE_NO

```
