# Labor Ticket Find

```sql
USE LIVE
GO

SELECT WO.BASE_ID
, WO.LOT_ID
, WO.SPLIT_ID
, WO.SUB_ID
, WO.DESIRED_QTY
, O.SEQUENCE_NO
, O.RESOURCE_ID
, LT.EMPLOYEE_ID
, LT.CLOCK_IN
, LT.CLOCK_OUT

FROM LIVE.dbo.WORK_ORDER WO
INNER JOIN LIVE.dbo.OPERATION O
	ON WO.BASE_ID = O.WORKORDER_BASE_ID
	AND WO.LOT_ID = O.WORKORDER_LOT_ID
	AND WO.SPLIT_ID = O.WORKORDER_SPLIT_ID
	AND WO.SUB_ID = O.WORKORDER_SUB_ID
	AND WO.[TYPE] = O.WORKORDER_TYPE
	AND O.RESOURCE_ID IN ('P1F1-CP-PM', 'P1F1-PM', 'P1F2-PM-CMP', 'P1F2-TKDWN-CMP', 'P1F1-TKDWN')
--	AND O.RESOURCE_ID IN ('P1F2-TKDWN-CMP') 
INNER JOIN LIVE.dbo.[DEMAND_SUPPLY_LINK] DSL
	ON DSL.[SUPPLY_TYPE] = 'WO'
	AND DSL.SUPPLY_BASE_ID = WO.BASE_ID
	AND DSL.SUPPLY_LOT_ID = WO.LOT_ID
	AND DSL.SUPPLY_SPLIT_ID = WO.SPLIT_ID
	AND DSL.SUPPLY_SUB_ID = WO.SUB_ID
INNER JOIN LIVE.dbo.CUST_ORDER_LINE COL
	ON DSL.DEMAND_TYPE = 'CO'
	AND DSL.DEMAND_BASE_ID = COL.CUST_ORDER_ID
	AND DSL.DEMAND_SEQ_NO = COL.LINE_NO
INNER JOIN LIVE.dbo.CUSTOMER_ORDER CO
	ON COL.CUST_ORDER_ID = CO.ID
	AND CO.CUSTOMER_ID = 'BOE609'
LEFT OUTER JOIN LIVE.dbo.LABOR_TICKET LT
	ON WO.BASE_ID = LT.WORKORDER_BASE_ID
	AND WO.LOT_ID = LT.WORKORDER_LOT_ID
	AND WO.SPLIT_ID = LT.WORKORDER_SPLIT_ID
	AND WO.SUB_ID = LT.WORKORDER_SUB_ID
	AND WO.[TYPE] = LT.WORKORDER_TYPE
	AND LT.OPERATION_SEQ_NO = O.SEQUENCE_NO
WHERE WO.[TYPE] = 'W'
AND LT.TRANSACTION_ID IS NULL
AND WO.CLOSE_DATE >= '9/25/2020'
AND WO.DESIRED_QTY = 2
AND O.SEQUENCE_NO = 400
--AND O.RESOURCE_ID IN ('P1F2-TKDWN-CMP')
AND O.RESOURCE_ID IN ('P1F1-CP-PM', 'P1F2-PM-CMP')
--AND O.RESOURCE_ID IN ('P1F1-CP-PM', 'P1F2-PM-CMP', 'P1F2-TKDWN-CMP')
--AND ISNULL(WO.CLOSE_DATE, '10/1/2020') >= '9/25/2020'
--AND O.RESOURCE_ID IN ('P1F1-CP-PM', 'P1F1-PM', 'P1F2-PM-CMP', 'P1F2-TKDWN-CMP', 'P1F1-TKDWN')
--AND O.RESOURCE_ID IN ('P1F2-TKDWN-CMP')

ORDER BY WO.DESIRED_QTY, WO.BASE_ID, O.SEQUENCE_NO
```
