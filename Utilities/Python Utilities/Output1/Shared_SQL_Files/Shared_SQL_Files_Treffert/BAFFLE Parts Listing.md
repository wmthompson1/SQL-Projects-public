# BAFFLE Parts Listing

```sql
USE DataWarehouse
GO

; WITH CTE_PARTS AS
(SELECT ID AS PART_ID
, [DESCRIPTION]
, [STATUS]
FROM PART P
WHERE [DESCRIPTION] LIKE '%BAFFLE%'
)

, CTE_RLS AS
(SELECT WO.PART_ID
, MAX(WO.DESIRED_RLS_DATE) AS LAST_RELEASE_DATE
FROM WORK_ORDER WO
INNER JOIN CTE_PARTS CTE
ON WO.PART_ID = CTE.PART_ID
WHERE WO.[TYPE] = 'W'
GROUP BY WO.PART_ID
)
--select * from cte_rls where part_id = '112U5885-115'

SELECT P.*
, WO.BASE_ID
, WO.LOT_ID
, WO.SPLIT_ID
, WO.SUB_ID
, WO.DESIRED_RLS_DATE
FROM CTE_PARTS P
INNER JOIN CTE_RLS D
ON P.PART_ID = D.PART_ID
INNER JOIN WORK_ORDER WO
ON D.LAST_RELEASE_DATE = WO.DESIRED_RLS_DATE
AND D.PART_ID = WO.PART_ID
ORDER BY WO.PART_ID
```
