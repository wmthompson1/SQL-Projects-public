USE LIVE

DECLARE @STARTDATE DATETIME = '1/1/2021'

; WITH CTE_WO_COUNT AS
(SELECT PART_ID
	, COUNT(*) AS WO_COUNT
FROM WORK_ORDER WO
WHERE WO.[STATUS] IN ('R', 'C')
	AND WO.CREATE_DATE >= @STARTDATE
	AND BASE_ID NOT LIKE '%R%'
	AND PART_ID NOT LIKE '%REWORK%'
GROUP BY PART_ID
)

, CTE_LT AS
(SELECT WO.PART_ID
	, LT.TRANSACTION_ID
	, LT.WORKORDER_BASE_ID
	, LT.WORKORDER_LOT_ID
	, LT.WORKORDER_SPLIT_ID
	, LT.WORKORDER_SUB_ID
	, LT.OPERATION_SEQ_NO
	, LT.RESOURCE_ID
	, LT.HOURS_WORKED
	, CASE
		WHEN WO.DESIRED_QTY != 0 THEN WO.DESIRED_QTY
		WHEN LT.GOOD_QTY = 0 THEN 1
		ELSE LT.GOOD_QTY
	END AS DESIRED_QTY
FROM LABOR_TICKET LT
LEFT OUTER JOIN WORK_ORDER WO
	ON LT.WORKORDER_BASE_ID = WO.BASE_ID
	AND LT.WORKORDER_LOT_ID = WO.LOT_ID
	AND LT.WORKORDER_SPLIT_ID = WO.SPLIT_ID
	AND LT.WORKORDER_SUB_ID = WO.SUB_ID
	AND LT.WORKORDER_TYPE = WO.[TYPE]
WHERE LT.TRANSACTION_DATE >= @STARTDATE
	AND WO.[STATUS] IN ('R', 'C')
)
--> SELECT * FROM CTE_LT WHERE DESIRED_QTY <= 0

, CTE_LT_AVG AS
(SELECT PART_ID 
	, WORKORDER_BASE_ID
	, WORKORDER_LOT_ID
	, WORKORDER_SPLIT_ID
	, WORKORDER_SUB_ID
	, OPERATION_SEQ_NO
	, RESOURCE_ID
	, DESIRED_QTY
	, SUM(HOURS_WORKED) AS TOTAL_HRS_WORKED
	, SUM(HOURS_WORKED) / DESIRED_QTY AS PIECE_HRS_WORKED
FROM CTE_LT
GROUP BY PART_ID 
	, WORKORDER_BASE_ID
	, WORKORDER_LOT_ID
	, WORKORDER_SPLIT_ID
	, WORKORDER_SUB_ID
	, OPERATION_SEQ_NO
	, RESOURCE_ID
	, DESIRED_QTY
)

, CTE_LT_RANGE AS
(SELECT CTE.PART_ID
	, P.[STATUS] AS PART_STATUS
	, RESOURCE_ID
	, SR.USER_1 AS SKILLS_TYPE
	, SR.USER_5 AS FACILITY
	, SR.USER_6 AS PROCESS
	, SR.USER_7 AS SUBPROCESS
	, SR.USER_8 AS DEPARTMENT
	, ISNULL(WC.WO_COUNT, 1) AS WO_COUNT
	, MIN(PIECE_HRS_WORKED) AS MIN_HRS_WORKED
	, MAX(PIECE_HRS_WORKED) AS MAX_HRS_WORKED
	, MAX(PIECE_HRS_WORKED) - MIN(PIECE_HRS_WORKED) AS DELTA
FROM CTE_LT_AVG CTE
LEFT OUTER JOIN PART P
	ON CTE.PART_ID = P.ID
LEFT OUTER JOIN SHOP_RESOURCE SR
	ON CTE.RESOURCE_ID = SR.ID
LEFT OUTER JOIN CTE_WO_COUNT WC
	ON WC.PART_ID = CTE.PART_ID
WHERE WORKORDER_BASE_ID NOT LIKE '%R%'
	AND CTE.PART_ID NOT LIKE '%REWORK%'
GROUP BY CTE.PART_ID
	, P.[STATUS]
	, RESOURCE_ID
	, SR.USER_1
	, SR.USER_5
	, SR.USER_6
	, SR.USER_7
	, SR.USER_8
	, WC.WO_COUNT
)
--> SELECT * FROM CTE_LT_RANGE ORDER BY PART_ID, PROCESS, RESOURCE_ID

SELECT * 
FROM CTE_LT_RANGE 
WHERE RESOURCE_ID IN (@RESOURCEID)
	AND SKILLS_TYPE IN (@SKTYPE)
	AND FACILITY IN (@FACILITY)
	AND PROCESS IN (@PROCESS)
	AND SUBPROCESS IN (@SUBPROCESS)
	AND DEPARTMENT IN (@DEPARTMENT)
	AND PART_ID LIKE REPLACE(@PART_ID + '%', '%%', '%')
ORDER BY PART_ID
	, PROCESS
	, RESOURCE_ID
