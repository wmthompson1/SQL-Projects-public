# Trace Updates from CO to WO

```sql
USE LIVE
GO

DECLARE @PART_ID NVARCHAR(50) = '453T7027-2'
DECLARE @CORRECT_TRACEID NVARCHAR(50) = '1652825/1'
DECLARE @TRANSID NVARCHAR(20) = '2946665'
DECLARE @IN_QTY NVARCHAR(20) = '1'
DECLARE @OUT_QTY NVARCHAR(20) = '1'
DECLARE @INV_DATE DATETIME = (SELECT TRANSACTION_DATE FROM INVENTORY_TRANS WHERE TRANSACTION_ID = @TRANSID)

--> CREATE CORRECT TRACE_ID
INSERT INTO LIVE.dbo.TRACE (PART_ID, ID, NUMBERING_ID, OUT_QTY, IN_QTY, REPORTED_QTY, ASSIGNED_QTY, APROPERTY_1, APROPERTY_2, APROPERTY_3, APROPERTY_4, APROPERTY_5, NPROPERTY_1, NPROPERTY_2, NPROPERTY_3, NPROPERTY_4, NPROPERTY_5, COMMENTS, EXPIRATION_DATE, CREATE_DATE, DISP_IN_QTY, DISP_OUT_QTY, UNAVAILABLE_QTY, OWNER_ID, LOT_ID, SERIAL_ID, COMMITTED_QTY, PRODUCTION_DATE, RECEIVE_BY_DATE, AVAILABLE_DATE, SHIP_BY_DATE, SITE_ID)
	VALUES (@PART_ID, @CORRECT_TRACEID, @PART_ID, @OUT_QTY, @IN_QTY, 0, 0, NULL, NULL, NULL, NULL, NULL, NULL, NULL, NULL, NULL, NULL, NULL, NULL, @INV_DATE, 0, 0, 0, NULL, NULL, NULL, NULL, NULL, NULL, NULL, NULL, NULL)

--SELECT * FROM TRACE WHERE PART_ID = '453T7027-2'

--> CORRECT TRACE TRANSACTION RECORD
UPDATE TRACE_INV_TRANS
SET TRACE_ID = @CORRECT_TRACEID
WHERE TRANSACTION_ID IN (@TRANSID)
AND PART_ID = @PART_ID

SELECT * FROM TRACE_INV_TRANS WHERE PART_ID = @PART_ID
```
