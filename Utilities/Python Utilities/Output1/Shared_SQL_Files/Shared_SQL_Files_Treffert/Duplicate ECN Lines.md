# Duplicate ECN Lines

```sql
USE LIVE
GO

IF EXISTS
(SELECT EL.EC_ID
, EL.LINE_NO
, EL.[TYPE]
, EL.DOC_ID
, EC.CREATE_DATE
, EC.CREATED_BY
, EC.ORIGINATOR
, EC.[STATUS]
, EC.UPDATE_LEVEL
FROM EC_LINE EL
INNER JOIN EC 
ON EC.ID = EL.EC_ID
INNER JOIN (SELECT DOC_ID
	FROM EC_LINE EL
	INNER JOIN EC 
		ON EC.ID = EL.EC_ID
	WHERE EC.[STATUS] NOT IN ('C', 'X')
		AND EL.[TYPE] = 'D'
	GROUP BY EL.[TYPE], DOC_ID
	HAVING COUNT(*) > 1
	) CTE
	ON EL.DOC_ID = CTE.DOC_ID
WHERE EC.[STATUS] NOT IN ('C', 'X')
	AND EL.[TYPE] = 'D'
)
	SELECT 1 AS Result
ELSE 
	SELECT 0 AS Result
```
