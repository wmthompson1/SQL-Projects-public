# PPD Operations and Labor Tickets (PowerBI - MarkH)

```sql
SELECT 
--> Operation data
O.[WORKORDER_TYPE]
, O.[WORKORDER_BASE_ID]
, O.[WORKORDER_LOT_ID]
, O.[WORKORDER_SPLIT_ID]
, O.[WORKORDER_SUB_ID]
, O.[SEQUENCE_NO]
, O.[RESOURCE_ID]
, O.[SERVICE_ID]
, O.[COMPLETED_QTY]
, O.[DEVIATED_QTY]
, O.[STATUS]
, O.[SETUP_COMPLETED]
, O.[SERVICE_BEGIN_DATE]
, O.[CLOSE_DATE]
, O.[OPERATION_TYPE]
, O.[VENDOR_ID]
, O.[VENDOR_SERVICE_ID]
, O.[SERVICE_PART_ID]

--> Resource Assignments and Shop Resources data
, RA.Dept
, RA.WORK_CELL
, RA.PPD_Process

, SR.USER_2
, SR.USER_3 
, SR.USER_4 AS SUPERVISOR
, SR.USER_2 AS PLANT_LOCATION

--> Labor Ticket data
, LT.[EMPLOYEE_ID]
, LT.[CLOCK_IN]
, LT.[CLOCK_OUT]
, LT.[HOURS_WORKED]
, LT.[GOOD_QTY]
, LT.[BAD_QTY]
, LT.[TRANSACTION_DATE]
, LT.[DESCRIPTION]
, LT.[ACT_LABOR_COST]
, LT.[ACT_BURDEN_COST]
, LT.[UNADJ_CLOCK_IN]
, LT.[UNADJ_CLOCK_OUT]

FROM OPERATION O
INNER JOIN LABOR_TICKET LT
	ON  O.WORKORDER_BASE_ID = LT.WORKORDER_BASE_ID
	AND O.WORKORDER_LOT_ID = LT.WORKORDER_LOT_ID
	AND O.WORKORDER_SPLIT_ID = LT.WORKORDER_SPLIT_ID
	AND O.WORKORDER_SUB_ID = LT.WORKORDER_SUB_ID
	AND O.SEQUENCE_NO = LT.OPERATION_SEQ_NO

INNER JOIN SHOP_RESOURCE SR
	ON O.RESOURCE_ID = SR.ID
INNER JOIN LIVESUPPLEMENTAL.DBO.ResourceID_Assignments RA
	ON SR.ID = RA.RESOURCE_ID

WHERE RA.Resource_Description NOT LIKE '%DO NOT USE%'
AND EXISTS
	(SELECT 1
	FROM  TBLNONCONFORMANCES PPD
	WHERE O.WORKORDER_BASE_ID = PPD.WORKORDER
	AND O.WORKORDER_LOT_ID = PPD.Lot
	AND O.WORKORDER_SPLIT_ID = PPD.Split
	AND O.WORKORDER_SUB_ID = PPD.Sub
	AND O.SEQUENCE_NO = PPD.Opp
	AND PPD.IntDate >= '1/1/2017'
	)



```
