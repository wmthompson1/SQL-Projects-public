FROM PURC_ORDER_LINE POL
INNER JOIN dbo.PURCHASE_ORDER PO
	ON PO.ID = POL.PURC_ORDER_ID
INNER JOIN dbo.VENDOR V
	ON V.ID = PO.VENDOR_ID
INNER JOIN dbo.PART P
	ON P.ID = POL.PART_ID
INNER JOIN dbo.RECEIVER_LINE RL
	ON POL.PURC_ORDER_ID = RL.PURC_ORDER_ID
	AND POL.LINE_NO = RL.PURC_ORDER_LINE_NO
INNER JOIN dbo.RECEIVER R
	ON R.ID = RL.RECEIVER_ID
WHERE RL.INSPECT_QTY > 0
	AND R.RECEIVED_DATE >= @STARTDATE
	AND R.RECEIVED_DATE <= @ENDDATE

	AND P.ID = '2128-01'

	, V.[NAME]
