```sql
SELECT 
	i.WAREHOUSE_ID,
	i.LOCATION_ID,
	t.PART_ID,
	P.DESCRIPTION,
	sum(t.QTY) as qty,
	MAX(t.CREATE_DATE) as lasttracedate, 
	t.TRACE_ID,
	P.COMMODITY_CODE,
	p.STOCK_UM
FROM 
	TRACE_INV_TRANS T
INNER JOIN 
	INVENTORY_TRANS I
	ON t.TRANSACTION_ID = i.TRANSACTION_ID
INNER JOIN 
	PART P
	ON i.PART_ID = P.ID
WHERE 
(P.COMMODITY_CODE in ('Details','Raw Material','Standards')
	or p.COMMODITY_CODE is null)
and i.WAREHOUSE_ID <> '1'
--and p.id = '15-5PHP-750X3.75X9.50'
AND I.WAREHOUSE_ID IN ('Auburn')
AND I.LOCATION_ID IN ('Insp')
group by t.PART_ID,
t.TRACE_ID,
-- t.TRANSACTION_ID,
i.WAREHOUSE_ID,
i.LOCATION_ID,
P.COMMODITY_CODE,
p.STOCK_UM,
P.DESCRIPTION
having sum(t.QTY) <> 0
ORDER BY i.WAREHOUSE_ID,
i.LOCATION_ID,
P.COMMODITY_CODE,
t.PART_ID,
t.TRACE_ID

SELECT *
	--i.WAREHOUSE_ID,

	--MAX(t.CREATE_DATE) as lasttracedate, 
	--t.TRACE_ID,
	--P.COMMODITY_CODE,
	--p.STOCK_UM
FROM 
	INVENTORY_TRANS I
left outer JOIN 
	TRACE_INV_TRANS T
	ON t.TRANSACTION_ID = i.TRANSACTION_ID
inner JOIN 
	PART P
	ON i.PART_ID = P.ID
WHERE 
	p.id = '15-5PHP-750X3.75X9.50'

order by i.transaction_id
```
