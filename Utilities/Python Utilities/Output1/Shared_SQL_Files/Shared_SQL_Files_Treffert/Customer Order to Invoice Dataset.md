# Customer Order to Invoice Dataset

```sql
use datawarehouse
go

--> VISUAL ENTERPRISE > SHIPPING REPORTS > CUST_ORDER_AND_SHIPMENTS DATASET
DECLARE @StartDate DATE = '1/1/'+CAST(YEAR(GETDATE())-1 AS NVARCHAR(4)); 

SELECT CO.CUSTOMER_ID 
	, CO.ID AS CUST_ORDER_ID
	, COL.LINE_NO AS COL_LINE_NO
	, CO.CUSTOMER_PO_REF
	, CO.SALESREP_ID
	, CO.ORDER_DATE
	, ISNULL(COL.DESIRED_SHIP_DATE, CO.DESIRED_SHIP_DATE) AS DESIRED_SHIP_DATE
	, CO.[STATUS] AS CO_STATUS
	, CO.TOTAL_AMT_ORDERED
	, CO.TOTAL_AMT_SHIPPED
	, CO.ENTERED_BY AS CO_USERID
	, COL.PART_ID
	, COL.CUSTOMER_PART_ID
	, COL.LINE_STATUS AS SHIP_SHORT
	, COL.ORDER_QTY
	, COL.USER_ORDER_QTY
	, COL.UNIT_PRICE
	, COL.SELLING_UM
	, COL.MISC_REFERENCE
	, COL.PRODUCT_CODE
	, COL.COMMODITY_CODE
	, COL.GL_REVENUE_ACCT_ID
	, COL.TOTAL_ACT_FREIGHT
	, COL.TOTAL_SHIPPED_QTY
	, COL.TOTAL_USR_SHIP_QTY
	, COL.TOTAL_AMT_SHIPPED AS COL_AMT_SHIPPED
	, COL.TOTAL_AMT_ORDERED AS COL_TOTAL_ORDERED
	, COL.WAREHOUSE_ID
	, COL.ALLOCATED_QTY
	, COL.FULFILLED_QTY
	, COL.ENTERED_BY
	, COL.USER_2 AS LINE
	, P.[DESCRIPTION]
	, P.PRODUCT_CODE AS PART_PROD_CODE
	, P.COMMODITY_CODE AS PART_COMM_CODE
	, P.FABRICATED
	, P.PURCHASED
	, P.STOCKED
	, P.DETAIL_ONLY
	, P.PLANNER_USER_ID
	, P.BUYER_USER_ID
	, P.USER_1 AS PROCESS
	, P.USER_2 AS SPECIAL_NOTES
	, P.USER_3 AS CERTS_REQD
	, P.USER_4 AS DEF_VENDOR_ONLY
	, P.USER_5 AS SCD
	, P.USER_6 AS CBO
	, P.USER_7 AS PLANE_MODEL
	, P.USER_8 AS AWARD_DATE
	, P.USER_9 AS PART_FAMILY
	, P.USER_10 AS PURCH_INFO
	, UDF.STRING_VAL AS BUS_STREAM
	, UDF1.STRING_VAL AS MATERIAL_TYPE
	, SUM(SL.SHIPPED_QTY) AS SHIPPED_QTY
	, MAX(S.SHIPPED_DATE) AS SHIPPED_DATE
	, SUM(RL.QTY) AS INVOICE_QTY
	, SUM(RL.AMOUNT) AS INVOICE_AMT
FROM CUST_ORDER_LINE COL
INNER JOIN CUSTOMER_ORDER CO
	ON CO.ID = COL.CUST_ORDER_ID
LEFT OUTER JOIN PART P
	ON COL.PART_ID = P.ID
LEFT OUTER JOIN SHIPPER_LINE SL
	ON COL.CUST_ORDER_ID = SL.CUST_ORDER_ID
	AND COL.LINE_NO = SL.CUST_ORDER_LINE_NO
LEFT OUTER JOIN SHIPPER S
	ON SL.PACKLIST_ID = S.PACKLIST_ID
	AND CO.ID = S.CUST_ORDER_ID
LEFT OUTER JOIN RECEIVABLE_LINE RL
	ON SL.PACKLIST_ID = RL.PACKLIST_ID
	AND SL.LINE_NO = RL.PACKLIST_LINE_NO
LEFT OUTER JOIN USER_DEF_FIELDS UDF
	ON UDF.DOCUMENT_ID = CO.CUSTOMER_ID
	AND PROGRAM_ID = 'VMCUSMNT'
	AND UDF.ID = 'UDF-0000072'
LEFT OUTER JOIN USER_DEF_FIELDS UDF1
	ON UDF1.DOCUMENT_ID = P.ID
	AND UDF1.PROGRAM_ID = 'VMPRTMNT'
	AND UDF1.ID = 'UDF-0000035'
WHERE ISNULL(COL.DESIRED_SHIP_DATE, CO.DESIRED_SHIP_DATE) >= @StartDate
	AND CO.[STATUS] IN ('H', 'R', 'C')
	AND (SL.USER_SHIPPED_QTY <> 0 AND SL.SHIPPED_QTY <>0)
GROUP BY CO.CUSTOMER_ID 
	, CO.ID
	, COL.LINE_NO
	, CO.CUSTOMER_PO_REF
	, CO.SALESREP_ID
	, CO.ORDER_DATE
	, CO.DESIRED_SHIP_DATE
	, COL.DESIRED_SHIP_DATE
	, CO.[STATUS]
	, CO.TOTAL_AMT_ORDERED
	, CO.TOTAL_AMT_SHIPPED
	, CO.ENTERED_BY
	, COL.USER_2
	, COL.PART_ID
	, COL.CUSTOMER_PART_ID
	, COL.LINE_STATUS
	, COL.ORDER_QTY
	, COL.USER_ORDER_QTY
	, COL.UNIT_PRICE
	, COL.SELLING_UM
	, COL.MISC_REFERENCE
	, COL.PRODUCT_CODE
	, COL.COMMODITY_CODE
	, COL.GL_REVENUE_ACCT_ID
	, COL.LAST_SHIPPED_DATE
	, COL.TOTAL_ACT_FREIGHT
	, COL.TOTAL_SHIPPED_QTY
	, COL.TOTAL_USR_SHIP_QTY
	, COL.TOTAL_AMT_SHIPPED
	, COL.TOTAL_AMT_ORDERED
	, COL.WAREHOUSE_ID
	, COL.ALLOCATED_QTY
	, COL.FULFILLED_QTY
	, COL.ENTERED_BY
	, P.[DESCRIPTION]
	, P.PRODUCT_CODE
	, P.COMMODITY_CODE
	, P.FABRICATED
	, P.PURCHASED
	, P.STOCKED
	, P.DETAIL_ONLY
	, P.PLANNER_USER_ID
	, P.BUYER_USER_ID
	, P.USER_1
	, P.USER_2
	, P.USER_3
	, P.USER_4
	, P.USER_5
	, P.USER_6
	, P.USER_7
	, P.USER_8
	, P.USER_9
	, P.USER_10
	, UDF.STRING_VAL
	, UDF1.STRING_VAL
```
