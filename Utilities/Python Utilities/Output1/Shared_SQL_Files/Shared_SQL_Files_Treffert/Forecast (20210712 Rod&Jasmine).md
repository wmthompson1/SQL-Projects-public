```sql
USE DataWarehouse
GO	
	
SELECT WO.PART_ID	
	, BASE_ID
	, DESIRED_WANT_DATE
	, WO.[STATUS]
	, WO.CREATE_DATE
	, CLOSE_DATE
	, DESIRED_QTY
	, [ACT_MATERIAL_COST] + [ACT_LABOR_COST] + [ACT_BURDEN_COST] + [ACT_SERVICE_COST] AS TotalActualCost
	, ([ACT_MATERIAL_COST] + [ACT_LABOR_COST] + [ACT_BURDEN_COST] + [ACT_SERVICE_COST])/DESIRED_QTY AS AvgActualCost
	, CPE.EFFECTIVE_DATE
	, CPE.DISCONTINUE_DATE
	, CPE.DEFAULT_UNIT_PRICE
	, CPE2.CURR_DISC_DATE
--select * 
FROM WORK_ORDER WO
INNER JOIN LIVESupplemental.dbo.[JasminePart20210712] R	
	ON WO.PART_ID = R.PART_ID
LEFT OUTER JOIN CUST_PRICE_EFFECT CPE
	ON CPE.PART_ID = WO.PART_ID
	AND CPE.CUSTOMER_ID = 'GKNSOU'
	AND CPE.EFFECTIVE_DATE <= WO.CREATE_DATE
	AND CPE.DISCONTINUE_DATE >= WO.CREATE_DATE
LEFT OUTER JOIN (SELECT PART_ID
	, MAX(DISCONTINUE_DATE) AS CURR_DISC_DATE
	FROM CUST_PRICE_EFFECT CPE
	WHERE CPE.CUSTOMER_ID = 'GKNSOU'
	GROUP BY PART_ID
	) CPE2
	ON CPE2.PART_ID = WO.PART_ID
WHERE WO.[TYPE] = 'W'	
	AND WO.DESIRED_QTY > 0
ORDER BY WO.PART_ID	
, WO.CREATE_DATE DESC
```
