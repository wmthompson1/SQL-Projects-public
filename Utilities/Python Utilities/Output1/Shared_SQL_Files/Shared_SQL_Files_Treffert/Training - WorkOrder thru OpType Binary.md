# Training - WorkOrder thru OpType Binary

```sql
--> SQL-BI-1
USE DataWarehouse
GO

--> ALL WORKORDER INFORMATION
SELECT * FROM WORK_ORDER WHERE CREATE_DATE > '4/1/2021' ORDER BY CREATE_DATE DESC

--> MASTERS 
SELECT * FROM WORK_ORDER WHERE CREATE_DATE > '4/1/2021' AND [TYPE] = 'M' ORDER BY CREATE_DATE DESC

--> WORKORDERS
SELECT * FROM WORK_ORDER WHERE CREATE_DATE > '4/1/2021' AND [TYPE] = 'W' AND BASE_ID = '1653308' ORDER BY CREATE_DATE DESC

--> WORKORDER BINARY INFORMATION, INCLUDING THE CONVERSION (CAST)
SELECT * 
	, CAST(CAST(WB.BITS AS VARBINARY(MAX)) AS NVARCHAR(MAX)) AS [WO_BIN]
FROM WORKORDER_BINARY WB
WHERE WORKORDER_TYPE = 'W' AND WORKORDER_BASE_ID = '1653308' 

--> ALL OPERATIONS
SELECT * FROM OPERATION WHERE WORKORDER_TYPE = 'W' AND WORKORDER_BASE_ID = '1653308' ORDER BY WORKORDER_SUB_ID, SEQUENCE_NO

--> OPERATION BINARY, INCLUDING THE CONVERSION (CAST)
SELECT * 
	, CAST(CAST(OB.BITS AS VARBINARY(MAX)) AS NVARCHAR(MAX)) AS [OP_BIN]
FROM OPERATION_BINARY OB
WHERE WORKORDER_TYPE = 'W' 
	AND WORKORDER_BASE_ID = '1653308' 
	AND SEQUENCE_NO = '60'
	AND WORKORDER_SUB_ID = '1'

--> RESOURCE_IDS
SELECT * FROM SHOP_RESOURCE WHERE [STATUS] = 'A'
SELECT * FROM SHOP_RESOURCE WHERE [STATUS] != 'O' AND STATUS IS NOT NULL

SELECT * FROM OPERATION_TYPE WHERE ID = 'F22.05-706'

--> OPER_TYPE_BINARY, INCLUDING THE CONVERSION (CAST)
SELECT * 
	, CAST(CAST(OB.BITS AS VARBINARY(MAX)) AS NVARCHAR(MAX)) AS [OT_BIN]
FROM OPER_TYPE_BINARY OB
WHERE WORKORDER_TYPE = 'W' 
	AND WORKORDER_BASE_ID = '1653308' 
	AND SEQUENCE_NO = '60'
	AND WORKORDER_SUB_ID = '1'
```
