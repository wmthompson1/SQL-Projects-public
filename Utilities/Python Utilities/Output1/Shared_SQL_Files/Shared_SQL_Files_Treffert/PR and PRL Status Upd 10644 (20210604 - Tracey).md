# PR and PRL Status Upd 10644 (20210604 - Tracey)

```sql
USE LIVE
GO

SELECT * FROM PURC_REQUISITION PR
INNER JOIN PURC_REQ_LINE PRL
ON PR.ID = PRL.PURC_REQ_ID
WHERE PRL.PURC_REQ_ID IN ('10951')



UPDATE PURC_REQ_LINE
SET [LINE_STATUS] = 'C'
WHERE PURC_REQ_ID IN ('10951')
AND LINE_NO = '2'

UPDATE PURC_REQUISITION 
SET [STATUS] = 'C'
WHERE ID IN ('10951')
```
