# OT Labor Tickets

```sql
-- Visual Enterprise Reports > Labor Reports > Labor TIckets Search
DECLARE @datadate datetime = CAST('1/1/'+ CONVERT(NVARCHAR(4),(YEAR(GETDATE()) - 1)) AS DATE)
--SELECT @datadate

SELECT [Reports To]
	, LT.EMPLOYEE_ID
	, E.FIRST_NAME
	, E.LAST_NAME
	, SHIFT_ID
	, ISNULL(CO.CUSTOMER_ID,'') AS CUSTOMERID
	, COL.CUST_ORDER_ID
	, LT.WORKORDER_BASE_ID
	, LT.WORKORDER_LOT_ID
	, LT.DEPARTMENT_ID
	, LT.RESOURCE_ID
	, WO.PART_ID
	, WO.DESIRED_QTY
	, LT.TRANSACTION_ID
	, LT.CLOCK_IN
	, LT.CLOCK_OUT
	, LT.HOURS_WORKED
	, OT.[Year]
	, OT.[WeekNo]
	, OT.WeeklyHours
	, case
		when OT.WeeklyHours < 40 then OT.WeeklyHours
		else 40
	end as Regular_Hours
	, case
		when OT.WeeklyHours > 40 then OT.WeeklyHours - 40
		else 0
	end as OT_Hours
	, CASE
		WHEN WO.BASE_ID IS NULL THEN 'INDIRECT'
		WHEN CO.CUSTOMER_ID IS NULL THEN 'UNLINKED'
		ELSE 'LINKED'
	END AS TICKET_TYPE
FROM LABOR_TICKET LT
	inner join EMPLOYEE E
		ON LT.EMPLOYEE_ID = E.ID
	LEFT OUTER JOIN WORK_ORDER WO
		ON  WO.BASE_ID = LT.WORKORDER_BASE_ID
		AND WO.LOT_ID = LT.WORKORDER_LOT_ID
		AND WO.SPLIT_ID = LT.WORKORDER_SPLIT_ID
		AND WO.SUB_ID = LT.WORKORDER_SUB_ID
		AND WO.[TYPE] = LT.WORKORDER_TYPE
	LEFT OUTER JOIN LIVESUPPLEMENTAL.DBO.EMP_ADP ADP
		ON E.ID = ADP.EMPID
	LEFT OUTER JOIN DEMAND_SUPPLY_LINK DSL
		ON  DSL.SUPPLY_BASE_ID = WO.BASE_ID
		AND DSL.SUPPLY_LOT_ID = WO.LOT_ID
		AND DSL.SUPPLY_SPLIT_ID = WO.SPLIT_ID
		AND DSL.SUPPLY_SUB_ID = WO.SUB_ID
		AND DSL.SUPPLY_TYPE = 'WO'
	LEFT OUTER JOIN CUST_ORDER_LINE COL
		ON  COL.CUST_ORDER_ID = DSL.DEMAND_BASE_ID
		AND COL.LINE_NO = DSL.DEMAND_SEQ_NO 
		AND DSL.DEMAND_TYPE = 'CO'
	LEFT OUTER JOIN CUSTOMER_ORDER CO
		ON COL.CUST_ORDER_ID = CO.ID
	LEFT OUTER JOIN 
		(SELECT EMPLOYEE_ID
			, DATEPART(YYYY, CLOCK_IN) AS [Year]
			, DATEPART(wk, CLOCK_IN) AS [WeekNo]
			, SUM(HOURS_WORKED) as WeeklyHours
		FROM LABOR_TICKET LT
		GROUP BY EMPLOYEE_ID
			, DATEPART(YYYY, CLOCK_IN)
			, DATEPART(wk, CLOCK_IN)
		) OT
		ON LT.EMPLOYEE_ID = OT.EMPLOYEE_ID
		AND OT.WeekNo = DATEPART(wk, LT.CLOCK_IN)
where lt.CLOCK_IN >= @datadate



```
