# Sample PRs

```sql
USE LIVE
GO

SELECT *
FROM USER_DEF_FIELDS UDF6
WHERE UDF6.PROGRAM_ID = 'VMPRTMNT'
AND UDF6.ID = 'UDF-0000101'
AND STRING_VAL IS NOT NULL

SELECT top 50 * FROM PURC_requisition order by REQUISITION_DATE desc

SELECT top 50 * FROM PURC_requisition PR
INNER JOIN PURC_REQ_LINE PRL
ON PR.ID = PRL.PURC_REQ_ID
--> tt20210512 : UDF VENDOR QC NOTES
LEFT OUTER JOIN USER_DEF_FIELDS UDF1
	ON UDF1.DOCUMENT_ID = PR.VENDOR_ID
	AND UDF1.PROGRAM_ID = 'VMVNDMNT'
	AND UDF1.ID = 'UDF-0000102'
--> tt20210512 : UDF PART QC NOTES
LEFT OUTER JOIN USER_DEF_FIELDS UDF6
	ON UDF6.DOCUMENT_ID = PRL.PART_ID
	AND UDF6.PROGRAM_ID = 'VMPRTMNT'
	AND UDF6.ID = 'UDF-0000101'

WHERE PRL.PART_ID IN (SELECT DOCUMENT_ID
	FROM USER_DEF_FIELDS UDF6
	WHERE UDF6.PROGRAM_ID = 'VMPRTMNT'
	AND UDF6.ID = 'UDF-0000101'
	AND STRING_VAL IS NOT NULL
	GROUP BY DOCUMENT_ID
	)


order by REQUISITION_DATE desc



USE LIVE
GO

SELECT DOCUMENT_ID
	, STRING_VAL AS CONTROLLED
FROM USER_DEF_FIELDS 
WHERE ID = 'UDF-0000082' 
AND DOCUMENT_ID IS NOT NULL
AND ISNULL(STRING_VAL, 'Not Controlled') != 'Not Controlled'


SELECT STRING_VAL 
, COUNT(*)
FROM USER_DEF_FIELDS 
WHERE ID = 'UDF-0000082' 
AND DOCUMENT_ID IS NOT NULL
GROUP BY STRING_VAL 
```
