# Missing Labor Tickets

```sql
use DATAWAREHOUSE
go

-- Visual Enterprise Reports > Labor Reports > Open Labor Tickets

SELECT A.[Reports To]
	, T.EMPLOYEE_ID
	, T.TRANSACTION_DATE
	, E.LAST_NAME
	, E.FIRST_NAME
	, E.DEPARTMENT_ID
	, T.WORKORDER_BASE_ID 
	, T.WORKORDER_LOT_ID
	, T.WORKORDER_SPLIT_ID 
	, T.WORKORDER_SUB_ID
	, T.RESOURCE_ID
	, T.HOURS_WORKED
	, E.SHIFT_ID
	, T.TRANSACTION_ID
	, T.ACT_CLOCK_OUT
	, T.ACT_CLOCK_IN
	, CASE
		WHEN T.WORKORDER_BASE_ID IS NULL THEN 'Indirect'
		ELSE 'Direct'
	END AS [Ticket Type]
FROM LABOR_TICKET AS T 
	INNER JOIN EMPLOYEE AS E 
		ON T.EMPLOYEE_ID = E.ID
	LEFT OUTER JOIN LIVESupplemental.DBO.emp_ADP AS A
		ON T.EMPLOYEE_ID = A.EmpID
WHERE 
	T.ACT_CLOCK_OUT IS NULL
	AND T.ACT_CLOCK_IN IS NOT NULL
	AND T.HOURS_WORKED IS NULL
	AND (T.TRANSACTION_DATE >= '1/1/2014'
	AND T.TRANSACTION_DATE < '9/16/2019')
	and T.WORKORDER_BASE_ID IS NULL 
--	AND A.[Reports To] IN (@ReportsTo)

SELECT COUNT(*) FROM LABOR_TICKET
--15712557 (LAB1)
--15714144

SELECT 15712557 - 15714144

INTO #T1
FROM [SQL-BI-1].DATAWAREHOUSE.DBO.LABOR_TICKET LT1
LEFT OUTER JOIN [SQL-LAB-2].live.DBO.LABOR_TICKET LT2
ON lt1.TRANSACTION_ID = lt2.TRANSACTION_ID
where lt2.TRANSACTION_ID is null

select * from #t1
```
