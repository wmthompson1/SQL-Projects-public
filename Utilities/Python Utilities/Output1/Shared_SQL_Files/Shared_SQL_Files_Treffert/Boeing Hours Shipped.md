# Boeing Hours Shipped

```sql
USE LIVE
GO

DECLARE @SINCEDATE DATE = '12/1/2020'
, @CUSTID NVARCHAR(50) = 'BOE609'

SELECT CUSTOMER_ID
	, SHIPPED_DATE
	, HOURSTYPE
	, SUM(HOURS_WORKED) AS HOURS_WORKED
FROM (SELECT CO.CUSTOMER_ID
	, S.SHIPPED_DATE
	, LT.HOURS_WORKED
	, CASE 
		WHEN SR.USER_1 = 'M' THEN 'MFG HOURS'
		WHEN SR.USER_1 IN ('A', 'B') THEN 'FIN HOURS'
		ELSE 'OTHER HOURS'
	END AS HOURSTYPE
FROM SHIPPER S
INNER JOIN SHIPPER_LINE SL
	ON S.PACKLIST_ID = SL.PACKLIST_ID
LEFT OUTER JOIN RECEIVABLE_LINE RL
	ON RL.PACKLIST_ID = SL.PACKLIST_ID
	AND RL.PACKLIST_LINE_NO = SL.LINE_NO
LEFT OUTER JOIN CUST_ORDER_LINE COL
	ON SL.CUST_ORDER_ID = COL.CUST_ORDER_ID
	AND SL.CUST_ORDER_LINE_NO = COL.LINE_NO
LEFT OUTER JOIN CUSTOMER_ORDER CO
	ON CO.ID = COL.CUST_ORDER_ID
LEFT OUTER JOIN DEMAND_SUPPLY_LINK DSL
	ON DSL.DEMAND_BASE_ID = COL.CUST_ORDER_ID
	AND DSL.DEMAND_TYPE = 'CO'
LEFT OUTER JOIN LABOR_TICKET LT
	ON DSL.SUPPLY_BASE_ID = LT.WORKORDER_BASE_ID
	AND DSL.SUPPLY_LOT_ID = LT.WORKORDER_LOT_ID
	--AND DSL.SUPPLY_SPLIT_ID = LT.WORKORDER_SPLIT_ID
	--AND DSL.SUPPLY_SUB_ID = LT.WORKORDER_SUB_ID
	AND LT.WORKORDER_TYPE = 'W'
	AND DSL.SUPPLY_TYPE = 'WO'
LEFT OUTER JOIN SHOP_RESOURCE SR
	ON SR.ID = LT.RESOURCE_ID
WHERE (SR.USER_9 != 'BOE BILLING EXCLUDE' OR SR.USER_9 IS NULL)
	AND CO.CUSTOMER_ID = @CUSTID
	AND S.SHIPPED_DATE >= @SINCEDATE
) X
GROUP BY CUSTOMER_ID
	, SHIPPED_DATE
	, HOURSTYPE
ORDER BY CUSTOMER_ID
	, SHIPPED_DATE
	, HOURSTYPE



```
