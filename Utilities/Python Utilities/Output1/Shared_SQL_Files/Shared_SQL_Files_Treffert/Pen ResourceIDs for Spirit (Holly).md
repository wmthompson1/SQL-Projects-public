# Pen ResourceIDs for Spirit (Holly)

```sql
USE LIVE
GO

SELECT WO.BASE_ID
, WO.LOT_ID
, WO.SPLIT_ID
, WO.SUB_ID
, WO.PART_ID
, WO.DESIRED_QTY
, WO.CREATE_DATE
, WO.CLOSE_DATE
, WO.[STATUS]
, O.SEQUENCE_NO
, O.RESOURCE_ID
, O.CLOSE_DATE
, LT.CLOCK_IN
, LT.CLOCK_OUT
, LT.EMPLOYEE_ID
, EE.FIRST_NAME
, EE.LAST_NAME
--, COL.CUST_ORDER_ID
--, COL.LINE_NO
--, COL.PART_ID
--, COL.ORDER_QTY
--, COL.DESIRED_SHIP_DATE
--, CO.CUSTOMER_ID
--, CO.ORDER_DATE
--, CO.[STATUS]
FROM LIVE.dbo.WORK_ORDER WO

INNER JOIN OPERATION O
	ON WO.BASE_ID = O.WORKORDER_BASE_ID
	AND WO.LOT_ID = O.WORKORDER_LOT_ID
	AND WO.SPLIT_ID = O.WORKORDER_SPLIT_ID
	AND WO.SUB_ID = O.WORKORDER_SUB_ID
	AND WO.[TYPE] = O.WORKORDER_TYPE
	AND O.RESOURCE_ID IN ('AUB-pen', 'BAL-pen', 'P1F1-PEN')

INNER JOIN LABOR_TICKET LT
	ON WO.BASE_ID = LT.WORKORDER_BASE_ID
	AND WO.LOT_ID = LT.WORKORDER_LOT_ID
	AND WO.SPLIT_ID = LT.WORKORDER_SPLIT_ID
	AND WO.SUB_ID = LT.WORKORDER_SUB_ID
	AND WO.[TYPE] = LT.WORKORDER_TYPE
	AND LT.OPERATION_SEQ_NO = O.SEQUENCE_NO

INNER JOIN EMPLOYEE EE
	ON LT.EMPLOYEE_ID = EE.ID

LEFT OUTER JOIN LIVE.dbo.PART_SITE PS
ON WO.PART_ID = PS.PART_ID
--> PUT INTO INVENTORY
LEFT OUTER JOIN LIVE.dbo.INVENTORY_TRANS IT
	ON IT.WORKORDER_BASE_ID = WO.BASE_ID
	AND IT.WORKORDER_LOT_ID = WO.LOT_ID
	AND IT.WORKORDER_SPLIT_ID = WO.SPLIT_ID
	AND IT.WORKORDER_SUB_ID = WO.SUB_ID
	AND IT.WORKORDER_TYPE = WO.[TYPE]

LEFT OUTER JOIN LIVE.dbo.INV_TRANS_DIST ITD
	ON IT.TRANSACTION_ID = ITD.IN_TRANS_ID

--> TAKEN OUT OF INVENTORY, ISSUED TO CUSTOMER ORDER
LEFT OUTER JOIN LIVE.dbo.INVENTORY_TRANS OT
	ON ITD.OUT_TRANS_ID = OT.TRANSACTION_ID

LEFT OUTER JOIN LIVE.dbo.SHIPPER_LINE SL
	ON SL.TRANSACTION_ID = OT.TRANSACTION_ID
LEFT OUTER JOIN LIVE.dbo.SHIPPER S
	ON SL.PACKLIST_ID = S.PACKLIST_ID
LEFT OUTER JOIN LIVE.dbo.CUST_ORDER_LINE COL
	ON SL.CUST_ORDER_ID = COL.CUST_ORDER_ID
	AND SL.CUST_ORDER_LINE_NO = COL.LINE_NO
	AND COL.PART_ID IS NOT NULL
LEFT OUTER JOIN LIVE.dbo.CUSTOMER_ORDER CO
	ON COL.CUST_ORDER_ID = CO.ID



WHERE WO.PART_ID != 'REWORK MFG'
--AND WO.CREATE_DATE >= '1/1/2019'
--AND WO.[STATUS] = 'C'

AND CO.CUSTOMER_ID = 'SPIAER'

GROUP BY WO.BASE_ID
, WO.LOT_ID
, WO.SPLIT_ID
, WO.SUB_ID
, WO.PART_ID
, WO.DESIRED_QTY
, WO.CREATE_DATE
, WO.CLOSE_DATE
, WO.[STATUS]
, O.SEQUENCE_NO
, O.RESOURCE_ID
, O.CLOSE_DATE
, LT.CLOCK_IN
, LT.CLOCK_OUT
, LT.EMPLOYEE_ID
, EE.FIRST_NAME
, EE.LAST_NAME
--, COL.CUST_ORDER_ID
--, COL.LINE_NO
--, COL.PART_ID
--, COL.ORDER_QTY
--, COL.DESIRED_SHIP_DATE
--, CO.CUSTOMER_ID
--, CO.ORDER_DATE
--, CO.[STATUS]

ORDER BY WO.BASE_ID
, WO.LOT_ID
, WO.SPLIT_ID
, WO.SUB_ID


```
