# OpType and Resource_ID Listing (20210727)

```sql
SELECT O.WORKORDER_TYPE AS [TYPE]
	, O.WORKORDER_BASE_ID AS [BASE_ID]
	, O.WORKORDER_LOT_ID AS [LOT]
	, O.WORKORDER_SPLIT_ID AS [SPLIT]
	, O.WORKORDER_SUB_ID AS [SUB]
	, O.SEQUENCE_NO AS [SEQ_NO]
	, O.OPERATION_TYPE AS [OPTYPE]
	, O.RESOURCE_ID
	, OB.[SPECS]
FROM OPERATION AS O
INNER JOIN SHOP_RESOURCE AS SR
	ON O.RESOURCE_ID = SR.ID
INNER JOIN [SKILL_OPERATION_BINARY] OB
	ON OB.WORKORDER_TYPE = O.WORKORDER_TYPE
	AND OB.WORKORDER_BASE_ID= O.WORKORDER_BASE_ID
	AND OB.WORKORDER_LOT_ID= O.WORKORDER_LOT_ID
	AND OB.WORKORDER_SPLIT_ID= O.WORKORDER_SPLIT_ID
	AND OB.WORKORDER_SUB_ID = O.WORKORDER_SUB_ID
	AND OB.SEQUENCE_NO = O.SEQUENCE_NO
WHERE O.[WORKORDER_TYPE] = 'M'
	AND (ISNULL(O.OPERATION_TYPE, '_Blank') IN (@OPTYPE)
			AND ISNULL(O.RESOURCE_ID, '_Blank') IN (@RESOURCE_ID)
		)


```
