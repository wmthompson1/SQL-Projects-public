# SCLEAN Report (Shannen)

```sql
USE DataWarehouse
GO

; WITH CTE_SLCT AS
(SELECT WORKORDER_BASE_ID
, WORKORDER_LOT_ID
, WORKORDER_SPLIT_ID
, WORKORDER_SUB_ID
, SEQUENCE_NO
, RESOURCE_ID
, OPERATION_TYPE
, SR.USER_6 
, SR.USER_7 
, SR.USER_8 
FROM OPERATION O
INNER JOIN SHOP_RESOURCE SR
ON O.RESOURCE_ID = SR.ID
WHERE WORKORDER_TYPE = 'M'
	AND 
		(OPERATION_TYPE = 'SCLEAN'
		OR SR.USER_6 = 'MASKING'
		OR SR.USER_7 = 'MASKING'
		OR SR.USER_8 = 'MASKING'
		)
)

SELECT SC.WORKORDER_BASE_ID
, SC.WORKORDER_LOT_ID
, SC.WORKORDER_SPLIT_ID
, SC.WORKORDER_SUB_ID
, CASE
	WHEN SC.OPERATION_TYPE = 'SCLEAN' THEN SC.SEQUENCE_NO
	ELSE NULL
END AS SCLEAN_SEQ
, CASE
	WHEN SC.OPERATION_TYPE = 'SCLEAN' THEN SC.RESOURCE_ID
	ELSE NULL
END AS SCLEAN_RESOUCEID
, CASE
	WHEN SC.OPERATION_TYPE = 'SCLEAN' THEN SC.OPERATION_TYPE
	ELSE NULL
END AS SCLEAN_OPTYPE
, CASE
	WHEN MSK.USER_6 = 'MASKING' OR MSK.USER_7 = 'MASKING' OR MSK.USER_8 = 'MASKING' THEN MSK.SEQUENCE_NO
	ELSE NULL
END AS MASK_SEQ
, CASE
	WHEN MSK.USER_6 = 'MASKING' OR MSK.USER_7 = 'MASKING' OR MSK.USER_8 = 'MASKING' THEN MSK.RESOURCE_ID
	ELSE NULL
END AS MASK_RESOURCEID
FROM CTE_SLCT SC
LEFT OUTER JOIN CTE_SLCT MSK
ON SC.WORKORDER_BASE_ID = MSK.WORKORDER_BASE_ID
AND SC.WORKORDER_LOT_ID = MSK.WORKORDER_LOT_ID
AND SC.WORKORDER_SPLIT_ID = MSK.WORKORDER_SPLIT_ID
AND SC.WORKORDER_SUB_ID = MSK.WORKORDER_SUB_ID
AND (MSK.USER_6 = 'MASKING' OR MSK.USER_7 = 'MASKING' OR MSK.USER_8 = 'MASKING')
WHERE SC.OPERATION_TYPE = 'SCLEAN'
AND SC.SEQUENCE_NO < MSK.SEQUENCE_NO

ORDER BY SC.WORKORDER_BASE_ID
, SC.WORKORDER_LOT_ID
, SC.WORKORDER_SPLIT_ID
, SC.WORKORDER_SUB_ID
```
