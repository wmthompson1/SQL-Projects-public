USE DataWarehouse

DECLARE @ENDDATE DATE = CAST(DATEADD(DD, -DAY(GETDATE()), GETDATE()) AS DATE)
, @STARTDATE DATE 

SELECT @STARTDATE = CAST(DATEADD(YEAR, -1, @ENDDATE) AS DATE)

; WITH CTE_PO AS
(SELECT PO.VENDOR_ID
	, POL.PURC_ORDER_ID
	, ISNULL(COALESCE(POL.PART_ID, POL.USER_1, POL.VENDOR_PART_ID), '') AS PART_ID
	, PO.[STATUS] AS PO_STATUS
	, SUM(ISNULL(POL.ORDER_QTY, 0)) AS ORDER_QTY
FROM PURC_ORDER_LINE POL
INNER JOIN PURCHASE_ORDER PO
	ON PO.ID = POL.PURC_ORDER_ID
WHERE EXISTS (SELECT 1 FROM RECEIVER_LINE RL
	INNER JOIN RECEIVER R
		ON R.ID = RL.RECEIVER_ID
	WHERE RL.PURC_ORDER_ID = POL.PURC_ORDER_ID
	AND RL.PURC_ORDER_LINE_NO = POL.LINE_NO
	AND R.RECEIVED_DATE >= @STARTDATE
	AND R.RECEIVED_DATE <= @ENDDATE
	)
GROUP BY PO.VENDOR_ID
	, POL.PURC_ORDER_ID
	, ISNULL(COALESCE(POL.PART_ID, POL.USER_1, POL.VENDOR_PART_ID), '')
	, PO.[STATUS]
)
--SELECT * FROM CTE_PO WHERE PURC_ORDER_ID = '144553'

, CTE_RECEIVED AS
(SELECT RL.PURC_ORDER_ID
	, ISNULL(COALESCE(POL.PART_ID, POL.USER_1, POL.VENDOR_PART_ID), '') AS PART_ID
	, SUM(ISNULL(RL.RECEIVED_QTY, 0)) AS RECEIVED_QTY
FROM RECEIVER R
LEFT OUTER JOIN RECEIVER_LINE RL
	ON RL.RECEIVER_ID = R.ID
LEFT OUTER JOIN PURC_ORDER_LINE POL
	ON RL.PURC_ORDER_ID = POL.PURC_ORDER_ID
	AND RL.PURC_ORDER_LINE_NO = POL.LINE_NO
WHERE R.RECEIVED_DATE >= @STARTDATE
	AND R.RECEIVED_DATE <= @ENDDATE
GROUP BY RL.PURC_ORDER_ID
	, ISNULL(COALESCE(POL.PART_ID, POL.USER_1, POL.VENDOR_PART_ID), '')
)
--SELECT * FROM CTE_RECEIVED WHERE PURC_ORDER_ID = '144553'

, CTE_ONTIME AS
(SELECT X.*
	, CONVERT(DATE, DESIRED_RECV_DATE + 5, 101) AS DUE_DATE
	, CASE 
		WHEN RECEIVED_DATE <= DESIRED_RECV_DATE + 5 THEN 1
		ELSE 0
	END AS ONTIME
	, CASE 
		WHEN RECEIVED_DATE <= DESIRED_RECV_DATE + 5 THEN 0
		ELSE 1
	END AS LATE
	, 1 AS TOTALS
FROM (SELECT RL.PURC_ORDER_ID
		, RL.PURC_ORDER_LINE_NO
		, ISNULL(COALESCE(POL.PART_ID, POL.USER_1, POL.VENDOR_PART_ID), '') AS PART_ID
		, MAX(ISNULL(POL.DESIRED_RECV_DATE, PO.DESIRED_RECV_DATE)) AS DESIRED_RECV_DATE
		, MAX(R.RECEIVED_DATE) AS RECEIVED_DATE
	FROM RECEIVER R
	LEFT OUTER JOIN RECEIVER_LINE RL
		ON RL.RECEIVER_ID = R.ID
	LEFT OUTER JOIN PURC_ORDER_LINE POL
		ON RL.PURC_ORDER_ID = POL.PURC_ORDER_ID
		AND RL.PURC_ORDER_LINE_NO = POL.LINE_NO
	LEFT OUTER JOIN PURCHASE_ORDER PO
		ON PO.ID = POL.PURC_ORDER_ID
	WHERE R.RECEIVED_DATE >= @STARTDATE
		AND R.RECEIVED_DATE <= @ENDDATE
	GROUP BY RL.PURC_ORDER_ID
		, RL.PURC_ORDER_LINE_NO
		, ISNULL(COALESCE(POL.PART_ID, POL.USER_1, POL.VENDOR_PART_ID), '')
	) X
)
--SELECT * FROM CTE_ONTIME

, CTE_TOTALS AS 
(SELECT PURC_ORDER_ID
	, PART_ID
	, SUM(ONTIME) AS TTL_ONTIME
	, SUM(LATE) AS TTL_LATE
	, SUM(TOTALS) AS TOTAL_LINES
FROM CTE_ONTIME CTE
GROUP BY PURC_ORDER_ID
	, PART_ID
)
--SELECT * FROM CTE_TOTALS

, CTE_NCM AS
(SELECT CASE 
		WHEN CHARINDEX('/', NCM.PO_NUMBER,1) = 0 THEN NCM.PO_NUMBER
		ELSE SUBSTRING(NCM.PO_NUMBER, 1, CHARINDEX('/', NCM.PO_NUMBER, 1) -1)
	END AS PURC_ORDER_ID
	, NCM.PO_NUMBER
	, ISNULL(VP.PRODUCT_ID, '') AS PART_ID
	, CASE 
		WHEN VPO.ALT_ID_2 IS NOT NULL THEN 'WO CONTRACTOR'
		ELSE 'PURCHASE'
	END AS NCM_SOURCE
	, SUM(NCM.REJECTED) AS REJECTED_QTY
FROM VQ_NONCONFORMANCE NCM
LEFT OUTER JOIN VQ_PRODUCT VP
	ON VP.RECORD_ID = NCM.PRODUCT_REC_ID
LEFT OUTER JOIN VQ_PRODUCTION_ORDER VPO
	ON VPO.RECORD_ID = NCM.PROD_ORDER_REC_ID
WHERE NCM.NCM_TYPE = 'SUPPLIER - MAJOR'
GROUP BY CASE 
		WHEN CHARINDEX('/', NCM.PO_NUMBER,1) = 0 THEN NCM.PO_NUMBER
		ELSE SUBSTRING(NCM.PO_NUMBER, 1, CHARINDEX('/', NCM.PO_NUMBER, 1) -1)
	END
	, NCM.PO_NUMBER
	, ISNULL(VP.PRODUCT_ID, '')
	, CASE 
		WHEN VPO.ALT_ID_2 IS NOT NULL THEN 'WO CONTRACTOR'
		ELSE 'PURCHASE'
	END
)

SELECT PO.VENDOR_ID
	, PO.PURC_ORDER_ID
	, PO.PART_ID
	, PO.PO_STATUS
	, PO.ORDER_QTY
	, RECEIVED_QTY
	, ISNULL(REJECTED_QTY, 0) AS REJECTED_QTY
	, NCM_SOURCE
	, TTL_ONTIME
	, TTL_LATE
	, TOTAL_LINES
FROM CTE_PO PO
LEFT OUTER JOIN CTE_RECEIVED R
	ON PO.PURC_ORDER_ID = R.PURC_ORDER_ID
	AND PO.PART_ID = R.PART_ID
LEFT OUTER JOIN CTE_TOTALS T
	ON PO.PURC_ORDER_ID = T.PURC_ORDER_ID
	AND PO.PART_ID = T.PART_ID
LEFT OUTER JOIN CTE_NCM N
	ON N.PURC_ORDER_ID = R.PURC_ORDER_ID
	AND N.PART_ID = R.PART_ID
WHERE R.PURC_ORDER_ID = '144553'
