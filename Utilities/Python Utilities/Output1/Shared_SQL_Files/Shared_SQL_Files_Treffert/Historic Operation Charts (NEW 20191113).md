# Historic Operation Charts (NEW 20191113)

```sql
use datamart
go

DECLARE @Facility NVARCHAR(50) = 'AP1'
--, @Process NVARCHAR(50) = 'CHEMICAL PROCESSING'
, @Department NVARCHAR(50) = 'CHEMICAL PROCESSING'
, @STARTDATE DATE = '11/4/2019'
, @ENDDATE DATE = '11/10/2019'
, @TIMEFRAME NVARCHAR(50) = 'LAST WEEK'
--Visual Enterprise Reports>Work Order Reports>Historic Operation Charts
set datefirst 1;

SELECT O.WORKORDER_BASE_ID
	, O.WORKORDER_LOT_ID
	, O.WORKORDER_SPLIT_ID
	, O.WORKORDER_SUB_ID
	, O.WORKORDER_TYPE
	, O.SEQUENCE_NO
	, O.RESOURCE_ID
	, O.OPERATION_TYPE
	, O.CLOSE_DATE
	, CASE
		WHEN SR.USER_1 = 'A' THEN 'FIN'
		WHEN SR.USER_1 = 'M' THEN 'MFG'
		WHEN SR.USER_1 = 'O' THEN 'OUTREACH'
		WHEN SR.USER_1 = 'B' THEN 'BALLARD'
		WHEN SR.USER_1 = 'C' THEN 'CLOSED'
	END AS BUSUNIT
	, SR.USER_5 AS [FACILITY]
	, SR.USER_7 AS [PROCESS]
	, SR.USER_8 AS [REPTGROUP]
	, CASE
		WHEN @TIMEFRAME IN ('LAST WEEK', 'CURRENT WEEK') THEN DATENAME(DD, O.CLOSE_DATE)
		WHEN @TIMEFRAME IN ('LAST MONTH', 'CURRENT MONTH') THEN CONVERT(VARCHAR(20), DATEADD(wk, 0, DATEADD(DAY, 1-DATEPART(WEEKDAY, O.CLOSE_DATE), DATEDIFF(dd, 0, O.CLOSE_DATE))), 101)
		ELSE DATENAME(MONTH, O.CLOSE_DATE)
	END CHARTSORT
	, CASE
		WHEN @TIMEFRAME IN ('LAST WEEK', 'CURRENT WEEK') THEN LEFT(DATENAME(WEEKDAY, O.CLOSE_DATE), 3)  + '-' + CONVERT(NVARCHAR, O.CLOSE_DATE,1)
		WHEN @TIMEFRAME IN ('LAST MONTH', 'CURRENT MONTH') THEN CONVERT(VARCHAR(20), DATEADD(wk, 0, DATEADD(DAY, 1-DATEPART(WEEKDAY, O.CLOSE_DATE), DATEDIFF(dd, 0, O.CLOSE_DATE))), 101)
		ELSE DATENAME(MONTH, O.CLOSE_DATE)
	END AS TIMEFRAME
FROM OPERATION O
LEFT OUTER JOIN SHOP_RESOURCE_SITE SR
	ON O.RESOURCE_ID = SR.RESOURCE_ID
WHERE SR.USER_5 IN (@Facility)
	AND SR.USER_8 IN (@Department)
	--AND O.CLOSE_DATE >= @STARTDATE
	--AND O.CLOSE_DATE <= @ENDDATE
	AND	(
		(@TIMEFRAME = 'LAST YEAR' AND YEAR(O.CLOSE_DATE) = YEAR(GETDATE()) - 1
			)
		OR (@TIMEFRAME = 'CURRENT YEAR' AND YEAR(O.CLOSE_DATE) = YEAR(GETDATE())
			)
		OR (@TIMEFRAME = 'CURRENT WEEK' AND YEAR(O.CLOSE_DATE) = YEAR(GETDATE()) AND DATEPART(WW, O.CLOSE_DATE) = DATEPART(WW, GETDATE())
			)
		OR (@TIMEFRAME = 'CURRENT MONTH' AND YEAR(O.CLOSE_DATE) = YEAR(GETDATE()) AND DATEPART(MM, O.CLOSE_DATE) = DATEPART(MM, GETDATE())
			)
	---- in same year
		OR (@TIMEFRAME = 'LAST WEEK' AND DATEPART(WW, GETDATE()) > 1 AND YEAR(O.CLOSE_DATE) = YEAR(GETDATE()) AND DATEPART(WW, O.CLOSE_DATE) = DATEPART(WW, GETDATE()) - 1
			)
	---- in different year, over the gap
		OR (@TIMEFRAME = 'LAST WEEK' AND DATEPART(WW, GETDATE()) = 1 AND YEAR(O.CLOSE_DATE) = YEAR(GETDATE()) - 1 AND DATEPART(WW, O.CLOSE_DATE) >= 52
			)
	---- in same year
		OR (@TIMEFRAME = 'LAST MONTH' AND DATEPART(MM, GETDATE()) > 1 AND YEAR(O.CLOSE_DATE) = YEAR(GETDATE()) AND DATEPART(MM, O.CLOSE_DATE) = DATEPART(MM, GETDATE()) - 1
	--		)
	---- in different year, over the gap
		OR (@TIMEFRAME = 'LAST MONTH'  AND DATEPART(MM, GETDATE()) = 1 AND YEAR(O.CLOSE_DATE) = YEAR(GETDATE()) - 1 AND DATEPART(MM, O.CLOSE_DATE) = 12
			)
	---- in same year
		OR (@TIMEFRAME = 'LAST QUARTER'  AND DATEPART(QQ, GETDATE()) > 1 AND YEAR(O.CLOSE_DATE) = YEAR(GETDATE()) AND DATEPART(QQ, O.CLOSE_DATE) = DATEPART(QQ, GETDATE()) - 1
			)
	---- in different year, over the gap
		OR (@TIMEFRAME = 'LAST QUARTER' AND DATEPART(QQ, GETDATE()) = 1 AND YEAR(O.CLOSE_DATE) = YEAR(GETDATE()) -1 AND DATEPART(QQ, O.CLOSE_DATE) = 4
			)
		 )
	 )
```
