SET TRANSACTION ISOLATION LEVEL READ UNCOMMITTED

IF OBJECT_ID('tempdb..#Results') IS NOT NULL DROP TABLE #Results
IF OBJECT_ID('tempdb..#TEMP1') IS NOT NULL DROP TABLE #TEMP1

SELECT 'Test.1' descr
	, PR.ID Requisition_ID
	, PRL.LINE_NO RequisitionLine
	, Specs =  IIF(DR.DOCUMENT_ID IS NULL,'', DR.DOCUMENT_ID + '   ' ) + IIF(D.REVISION_ID IS NULL,'', 'Rev: ' + d.REVISION_ID)
INTO #Results
FROM PURC_REQUISITION PR
INNER JOIN PURC_REQ_LINE PRL
	ON PR.ID = PRL.PURC_REQ_ID
LEFT JOIN PART P
	ON P.ID = PRL.PART_ID
LEFT OUTER JOIN DOCUMENT_REFERENCE DR
	ON DR.ID = PRL.PART_ID
LEFT OUTER JOIN DOCUMENT D
	ON D.ID = DR.DOCUMENT_ID
	AND DR.SOURCE_TYPE = 'P'
	AND D.SITE_ID = 'SK01'
--WHERE PR.ID = @requisition_id--'120'

SELECT 'Level to PR Line' Descr
	, CAT.Requisition_ID
	, CAT.RequisitionLine
	, NULLIF(REPLACE(
		STUFF((SELECT '~' + SUB.Specs AS [text()]
		FROM #Results SUB
		WHERE
		SUB.Requisition_ID = CAT.Requisition_ID
		AND SUB.RequisitionLine = CAT.RequisitionLine
		FOR XML PATH('') -- Select it as XML
		), 1, 1, '' )
		,'~', CHAR(13) + CHAR(10))
		, '') AS [SpecsList]
INTO #TEMP1
FROM #Results CAT

SELECT Requisition_ID
, RequisitionLine
, [SpecsList]
INTO #SPECS
FROM #TEMP1
GROUP BY Requisition_ID
, RequisitionLine
, [SpecsList]

; WITH CTE_PR AS 
(SELECT PR.ID AS REQUISITION_ID
	, PR.TERMS_ID
	, CASE
		WHEN PR.[STATUS] = 'O' THEN 'Ordered'
		WHEN PR.[STATUS] = 'I' THEN 'In Process'
		WHEN PR.[STATUS] = 'V' THEN 'Approved'
		WHEN PR.[STATUS] = 'C' THEN 'Closed'
		WHEN PR.[STATUS] = 'X' THEN 'Cancelled'
		ELSE 'Draft'
	END AS [STATUS]
	, PR.REQUISITION_DATE
	, COALESCE(PRL.DESIRED_RECV_DATE,PR.DESIRED_RECV_DATE) AS DATE_REQUIRED
	, PR.REQUESTOR
	, E.FIRST_NAME + ' ' + E.LAST_NAME AS REQUESTOR_NAME
	, PR.PREPARER
	, PRE.FIRST_NAME + ' ' + PRE.LAST_NAME AS PREPARER_NAME
	, PR.BUYER
	, B.FIRST_NAME + ' ' + B.LAST_NAME AS BUYER_NAME
	, PR.VENDOR_ID
	, PR.CONTACT_PHONE AS VENDOR_PHONE
	, PR.CONTACT_FAX AS VENDOR_FAX
	, V.[NAME] AS VENDOR_NAME
	, V.ADDR_1 AS VENDOR_ADDR_1
	, V.ADDR_3 AS VENDOR_ADDR_3
	, V.CITY AS VENDOR_CITY
	, V.[STATE] AS VENDOR_STATE
	, V.ZIPCODE AS VENDOR_ZIPCODE
	, V.COUNTRY AS VENDOR_COUNTRY
	, PR.SHIP_VIA
	, PRL.LINE_NO
	, PRL.ORDER_QTY
	, PRL.PURCHASE_UM
	, PRL.UNIT_PRICE
	, PRL.USER_ORDER_QTY * PRL.UNIT_PRICE AS TOTAL_LINE_AMT
	, COALESCE(PRL.PART_ID, PRL.USER_1) AS PART_ID
	, PRL.VENDOR_PART_ID
	, P.[DESCRIPTION] AS PART_DESCRIPTION
	, PRL.USER_1
	, PRL.USER_2
	, PRL.GL_EXPENSE_ACCT_ID
	, PRL.MFG_NAME
	, PRL.MFG_PART_ID
	, CASE 
		WHEN ISNULL(UDF2.STRING_VAL, 'Not Controlled') != 'Not Controlled' THEN UDF2.STRING_VAL 
		ELSE NULL
	END	AS CONTROLLED
	, UDF3.STRING_VAL AS CERT_REQ
	, UDF4.STRING_VAL AS USED_ON
	, UDF5.STRING_VAL AS VENDOR_QCNOTES
	, UDF6.STRING_VAL AS PART_QCNOTES
	, STA.[NAME] AS SHIPTO_NAME
	, STA.ADDR_1 AS SHIPTO_ADDR_1
	, STA.ADDR_2 AS SHIPTO_ADDR_2
	, STA.ADDR_3 AS SHIPTO_ADDR_3
	, STA.CITY AS SHIPTO_CITY
	, STA.[STATE] AS SHIPTO_STATE
	, STA.ZIPCODE AS SHIPTO_ZIPCODE
	, STA.COUNTRY AS SHIPTO_COUNTRY
FROM PURC_REQUISITION PR
INNER JOIN PURC_REQ_LINE PRL
	ON PR.ID = PRL.PURC_REQ_ID
LEFT JOIN PART P
	ON P.ID = PRL.PART_ID
LEFT OUTER JOIN EMPLOYEE E
	ON PR.REQUESTOR = E.ID
LEFT OUTER JOIN EMPLOYEE PRE
	ON PR.PREPARER = PRE.USER_ID
LEFT OUTER JOIN EMPLOYEE B
	ON PR.BUYER = B.USER_ID
LEFT OUTER JOIN DBO.VENDOR V
	ON PR.VENDOR_ID = V.ID
LEFT OUTER JOIN USER_DEF_FIELDS U1
	ON U1.PROGRAM_ID = 'VMREQENT'
	AND U1.ID = 'UDF-0000064'
--	AND U1.DOCUMENT_ID = @requisition_id
	AND U1.LINE_NO = PRL.LINE_NO
-- GET ITAR INFO
LEFT OUTER JOIN USER_DEF_FIELDS AS UDF
	ON UDF.DOCUMENT_ID = PRL.PART_ID
	AND UDF.PROGRAM_ID = 'VMPRTMNT'
	AND UDF.ID = 'UDF-0000040'
	AND UDF.BOOL_VAL = 1
-- GET CONTROLLED INFO
LEFT OUTER JOIN USER_DEF_FIELDS AS UDF2
	ON UDF2.DOCUMENT_ID = PRL.PART_ID
	AND UDF2.PROGRAM_ID = 'VMPRTMNT'
	AND UDF2.ID = 'UDF-0000082'
-- GET CERT INFO
LEFT OUTER JOIN USER_DEF_FIELDS AS UDF3
	ON UDF3.DOCUMENT_ID = PRL.PART_ID
	AND UDF3.PROGRAM_ID = 'VMPRTMNT'
	AND UDF3.ID = 'UDF-0000084'
-- GET USED ON INFO
LEFT OUTER JOIN USER_DEF_FIELDS AS UDF4
	ON UDF4.DOCUMENT_ID = PRL.PART_ID
	AND UDF4.PROGRAM_ID = 'VMPRTMNT'
	AND UDF4.ID = 'UDF-0000085'
--> tt20210512 : UDF PART QC NOTES
LEFT OUTER JOIN USER_DEF_FIELDS UDF6
	ON UDF6.DOCUMENT_ID = PRL.PART_ID
	AND UDF6.PROGRAM_ID = 'VMPRTMNT'
	AND UDF6.ID = 'UDF-0000101'
--> tt20210512 : UDF VENDOR QC NOTES
LEFT OUTER JOIN USER_DEF_FIELDS UDF5
	ON UDF5.DOCUMENT_ID = PR.VENDOR_ID
	AND UDF5.PROGRAM_ID = 'VMVNDMNT'
	AND UDF5.ID = 'UDF-0000102'
LEFT OUTER JOIN SHIPTO_ADDRESS STA
	ON STA.ADDR_NO = PR.SHIPTO_ADDR_NO
--WHERE  PR.ID = @requisition_id--'120'
)

SELECT 1 AS [Singleton]
	, CTE.REQUISITION_ID	-- join sets 1
	, TERMS_ID	
	, [STATUS]
	, REQUISITION_DATE	
	, DATE_REQUIRED	
	, REQUESTOR	
	, REQUESTOR_NAME	
	, PREPARER	
	, PREPARER_NAME	
	, BUYER	
	, BUYER_NAME	
	, VENDOR_ID	
	, VENDOR_PHONE	
	, VENDOR_FAX	
	, VENDOR_NAME	
	, VENDOR_ADDR_1	
	, VENDOR_ADDR_3	
	, VENDOR_CITY	
	, VENDOR_STATE	
	, VENDOR_ZIPCODE	
	, VENDOR_COUNTRY	
	, SHIP_VIA	
	, LINE_NO	--
	, ORDER_QTY	
	, PURCHASE_UM	
	, UNIT_PRICE	
	, TOTAL_LINE_AMT	
	, PART_ID	
	, VENDOR_PART_ID	
	, PART_DESCRIPTION	
	, USER_1	
	, USER_2	
	, GL_EXPENSE_ACCT_ID	
	, MFG_NAME	
	, MFG_PART_ID	
	, CERT_REQ	-- Update 5/29/2020
	, USED_ON	
	, CAST(CAST(PRB.BITS AS VARBINARY(MAX)) AS NVARCHAR(MAX)) AS PR_SPECS
	, CAST(CAST(PLB.BITS AS VARBINARY(MAX)) AS NVARCHAR(MAX)) AS PRL_SPECS
	, CAST(CAST(N.NOTE AS VARBINARY(MAX)) AS NVARCHAR(MAX)) AS PR_NOTES
	, S.SpecsList
	, lblCert_Req = iif(CERT_REQ is not null, 'Certifications Required:  ',	NULL)
	, lblSpecsList = iif(SpecsList is not null, 'Must meet the reqmts of:  ',	NULL)
	, VENDOR_QCNOTES
	, PART_QCNOTES
	, CTE.CONTROLLED
	, CTE.SHIPTO_NAME
	, CTE.SHIPTO_ADDR_1
	, CTE.SHIPTO_ADDR_2
	, CTE.SHIPTO_ADDR_3
	, CTE.SHIPTO_CITY
	, CTE.SHIPTO_STATE
	, CTE.SHIPTO_ZIPCODE
	, CTE.SHIPTO_COUNTRY
FROM   CTE_PR CTE
JOIN #SPECS S
	ON CTE.REQUISITION_ID = S.Requisition_ID
	AND CTE.LINE_NO  = S.RequisitionLine
LEFT OUTER JOIN PURC_REQ_BINARY PRB
	ON CTE.REQUISITION_ID = PRB.PURC_REQ_ID
LEFT OUTER JOIN PURC_REQ_LN_BINARY PLB
	ON CTE.REQUISITION_ID = PLB.PURC_REQ_ID
	AND CTE.LINE_NO = PLB.PURC_REQ_LINE_NO
LEFT OUTER JOIN NOTATION N
	ON CTE.REQUISITION_ID = N.OWNER_ID
	AND N.[TYPE] = 'PR'
--	AND N.OWNER_ID = @requisition_id
	
;
