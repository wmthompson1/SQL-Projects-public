USE DATAWAREHOUSE
GO

--> VISUAL ENTERPRISE > WORK ORDER REPORTS > JOB COUNTS
SET DATEFIRST 1;

WITH CTE_BASE AS
(SELECT O.WORKORDER_BASE_ID AS BASE_ID
	, O.WORKORDER_LOT_ID AS LOT_ID
	, O.WORKORDER_SPLIT_ID AS SPLIT_ID
	, O.WORKORDER_SUB_ID AS SUB_ID
	, O.SEQUENCE_NO
	, O.RESOURCE_ID
	, O.OPERATION_TYPE
	, O.CLOSE_DATE AS OPS_CLOSEDATE
 	, SR.USER_1 AS BUSUNIT
	, CASE
		WHEN SR.USER_1 = 'A' THEN 'FIN'
		WHEN SR.USER_1 = 'M' THEN 'MFG'
		WHEN SR.USER_1 = 'O' THEN 'OUTREACH'
		WHEN SR.USER_1 = 'B' THEN 'BALLARD'
		WHEN SR.USER_1 = 'C' THEN 'CLOSED'
	END AS BUSINESS_UNIT
	, SR.USER_2 AS [PRIMARY]
	, SR.USER_3 AS [SECONDARY]
	, SR.USER_4 AS [SUPERVISOR]
	, SR.USER_5 AS [FACILITY]
	, SR.USER_7 AS [PROCESS]
	, SR.USER_8 AS [DEPARTMENT]
	, CASE
		WHEN ISNULL(CO.CUSTOMER_ID, '') = '' THEN 'UNLINKED'
		ELSE CO.CUSTOMER_ID
	END AS CUSTOMER_ID
	, O.[STATUS] AS OPS_STATUS
	, WO.[STATUS] AS WO_STATUS
	, CO.[STATUS] AS CO_STATUS
	, WO.CLOSE_DATE AS WO_CLOSEDATE
FROM WORK_ORDER WO
LEFT OUTER JOIN OPERATION O
	ON WO.[TYPE] = O.WORKORDER_TYPE
	AND WO.BASE_ID = O.WORKORDER_BASE_ID
	AND WO.LOT_ID = O.WORKORDER_LOT_ID
	AND WO.SPLIT_ID = O.WORKORDER_SPLIT_ID
	AND WO.SUB_ID = O.WORKORDER_SUB_ID
LEFT OUTER JOIN SHOP_RESOURCE SR
	ON SR.ID = O.RESOURCE_ID
LEFT OUTER JOIN DEMAND_SUPPLY_LINK DSL
	ON WO.BASE_ID = DSL.SUPPLY_BASE_ID
	AND WO.LOT_ID = DSL.SUPPLY_LOT_ID
LEFT OUTER JOIN CUST_ORDER_LINE COL
	ON COL.CUST_ORDER_ID = DSL.DEMAND_BASE_ID
	AND COL.LINE_NO = DSL.DEMAND_SEQ_NO
LEFT OUTER JOIN CUSTOMER_ORDER CO
	ON CO.ID = COL.CUST_ORDER_ID
WHERE O.CLOSE_DATE >= '10/1/2019'
	AND O.[STATUS] != 'X'
	AND WO.[STATUS] != 'X'
)

--SELECT CONVERT(NVARCHAR(4), YEAR(OPS_CLOSEDATE)) + '-' + CONVERT(NVARCHAR(2), MONTH(OPS_CLOSEDATE)) AS [YYYY-MM]

--FROM CTE_BASE

SELECT CONVERT(NVARCHAR(4), YEAR(WO_CLOSEDATE)) + '-' + CONVERT(NVARCHAR(2), MONTH(WO_CLOSEDATE)) AS [YYYY-MM]
	, COUNT(BASE_ID) AS WO_TOTALS
FROM 
	(SELECT BASE_ID
		, LOT_ID
		, SPLIT_ID
		, SUB_ID
		, WO_CLOSEDATE
	FROM CTE_BASE
	WHERE WO_STATUS = 'C'
	GROUP BY BASE_ID
		, LOT_ID
		, SPLIT_ID
		, SUB_ID
		, WO_CLOSEDATE
	) X
GROUP BY CONVERT(NVARCHAR(4), YEAR(WO_CLOSEDATE)) + '-' + CONVERT(NVARCHAR(2), MONTH(WO_CLOSEDATE))
ORDER BY CONVERT(NVARCHAR(4), YEAR(WO_CLOSEDATE)) + '-' + CONVERT(NVARCHAR(2), MONTH(WO_CLOSEDATE))
