# IQM Report - NCM Type

```sql
use SANDBOX_Baseline_App
go

--IQM Report > NCM Type
SELECT NCM.DISPOSITION_TYPE
	, NCM.NCM_TYPE
	, NCM_ID
	, NCM.MADE_BY_EMP_REC_ID
	, MADE_EE.EMPLOYEE_ID AS MADE_BY_EEID
	, MADE_EE.FIRST_NAME AS MADE_BY_EE_FNAME
	, MADE_EE.LAST_NAME AS MADE_BY__LNAME
	, NCM.NCM_DATE
	, NCM.DUE_DATE
	, P.PRODUCT_ID
	, PO.DISPLAY_ID AS PRODUCTION_ORDER_ID
	, O.DISPLAY_ID AS OPERATION_ID
	, NCM.LOT_SIZE
	, NCM.ACCEPTED
	, NCM.REJECTED
	, NCM.DISPOSITION_TYPE
	, NCM.DISP_DATE
	, NCM.REFERENCE_7 AS [SCRAP_QTY]
	, NCM.REFERENCE_8 AS [OVERAGE_QTY]
	, NCM.REFERENCE_10 AS [ASIS_QTY]
FROM VQ_NONCONFORMANCE NCM
LEFT OUTER JOIN VQ_PRODUCT P
	ON P.RECORD_ID = NCM.PRODUCT_REC_ID
LEFT OUTER JOIN VQ_PRODUCTION_ORDER PO
	ON PO.RECORD_ID = NCM.PROD_ORDER_REC_ID
LEFT OUTER JOIN VQ_OPERATION O
	ON O.RECORD_ID = NCM.OPERATION_REC_ID
LEFT OUTER JOIN VQ_EMPLOYEE MADE_EE
	ON MADE_EE.RECORD_ID = NCM.MADE_BY_EMP_REC_ID
WHERE NCM.DISPOSITION_TYPE IN (@DispType)
	AND NCM.NCM_DATE >= @StartDate
	AND NCM.NCM_DATE <= @EndDate




```
