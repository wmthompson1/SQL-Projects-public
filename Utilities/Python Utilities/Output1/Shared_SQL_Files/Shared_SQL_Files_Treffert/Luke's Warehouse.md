# Luke's Warehouse

```sql
SELECT 
	P.ID, 
	P.DESCRIPTION, 
	CCP.COUNT_FREQ, 
	CCP.LAST_COUNT_DATE, 
	PL.LOCATION_ID, 
	PL.QTY, 
	PL.WAREHOUSE_ID, 
	P.STOCK_UM, 
	P.PURCHASED, 
	PS.UNIT_MATERIAL_COST, 
	PS.UNIT_LABOR_COST, 
	PS.UNIT_SERVICE_COST,
	PS.UNIT_MATERIAL_COST + PS.UNIT_LABOR_COST + PS.UNIT_SERVICE_COST as [UnitCost],
	PS.UNIT_MATERIAL_COST + PS.UNIT_LABOR_COST + PS.UNIT_SERVICE_COST * PL.QTY as [TotalCost]
FROM   
	PART_SITE PS
INNER JOIN 
	PART_LOCATION PL 
	ON PS.PART_ID = PL.PART_ID
LEFT OUTER JOIN 
	CYCLE_COUNT_PART CCP 
	ON PL.PART_ID = CCP.PART_ID 
	AND PL.WAREHOUSE_ID = CCP.WAREHOUSE_ID
INNER JOIN 
	PART P 
	ON CCP.PART_ID = P.ID
WHERE  
	PL.QTY <> 0 
	AND PL.WAREHOUSE_ID = 'FGA'
ORDER BY 
	PL.LOCATION_ID


```
