# Historic Operation Charts DataDriven Subscription (NEW 20191113)

```sql
use datawarehouse
go

WITH CTE_OPERATIONS AS
(SELECT SR.USER_1 AS [BUS_UNIT]
	, SR.USER_5 AS [FACILITY]
	, SR.USER_8 AS [DEPARTMENT]
FROM OPERATION O
LEFT OUTER JOIN SHOP_RESOURCE_SITE SR
	ON O.RESOURCE_ID = SR.RESOURCE_ID
WHERE O.CLOSE_DATE >= DATEADD(dd,  0, DATEADD(ww, DATEDIFF(ww, 0, DATEADD(dd, -1, GETDATE())) - 1, 0))
	AND O.CLOSE_DATE <= DATEADD(dd,  6, DATEADD(ww, DATEDIFF(ww, 0, DATEADD(dd, -1, GETDATE())) - 1, 0))
	AND ISNULL(NULLIF(SR.USER_8, 'NULL'), '') NOT IN ('MAINTENANCE', 'ARCHIVED', 'PAINT', 'BALLARD', 'OUTREACH') 
GROUP BY SR.USER_1
	, SR.USER_5 
	, SR.USER_8 
)
--SELECT * FROM CTE_OPERATIONS

SELECT CASE
		WHEN SR.USER_1 = 'A' THEN 'FIN'
		WHEN SR.USER_1 = 'M' THEN 'MFG'
		WHEN SR.USER_1 = 'O' THEN 'OUTREACH'
		WHEN SR.USER_1 = 'B' THEN 'BALLARD'
		WHEN SR.USER_1 = 'C' THEN 'CLOSED'
	END AS BUSUNIT
	, USER_5 AS FACILITY
	, USER_8 AS DEPARTMENT
	, USER_4 AS SUPERVISOR
	, ISNULL(EE.EMAIL_ADDR, N'TRACIT@SKILLSINC.COM') AS EMAIL_ADDR
--	, *
FROM SHOP_RESOURCE SR
LEFT OUTER JOIN EMPLOYEE EE
	ON SR.USER_4 = EE.[FIRST_NAME] + ' ' + EE.[LAST_NAME]
	AND EE.ACTIVE = 'Y'
WHERE ([STATUS] IS NULL OR STATUS != 'O')
	AND ISNULL(NULLIF(RTRIM(LTRIM(SR.USER_8)), 'NULL'), '') NOT IN ('', 'MAINTENANCE', 'ARCHIVED', 'PAINT', 'BALLARD', 'OUTREACH') 
	AND EXISTS
		(SELECT 1
		FROM CTE_OPERATIONS CTE
		WHERE CTE.[BUS_UNIT] = SR.USER_1
			AND CTE.[FACILITY] = SR.USER_5
			AND CTE.[DEPARTMENT] = SR.USER_8
		)
GROUP BY SR.USER_1
	, SR.USER_5 
	, SR.USER_4 
	, SR.USER_8 
	, EE.EMAIL_ADDR
```
