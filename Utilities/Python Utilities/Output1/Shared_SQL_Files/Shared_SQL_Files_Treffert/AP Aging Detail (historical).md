# AP Aging Detail (historical)

```sql
USE LIVE
GO

IF OBJECT_ID('tempdb..#T1') IS NOT NULL DROP TABLE #T1

SET DATEFIRST 7; -- Set's sunday as first day of week, won't work otherwise

DECLARE @STARTDATE DATE = '01/01/2021'
, @ENDDATE DATETIME = '12/31/2022'
, @CURRENTDATE DATETIME

DECLARE @TABLEOFDATES TABLE(DATEVALUE DATETIME)

SET @CURRENTDATE = @STARTDATE

WHILE @CURRENTDATE <= @ENDDATE
	BEGIN
		INSERT INTO @TABLEOFDATES(DATEVALUE) VALUES (@CURRENTDATE)
		SET @CURRENTDATE = DATEADD(DAY, 1, @CURRENTDATE)
	END

SELECT * 
INTO #T1
FROM @TABLEOFDATES 
WHERE DATEPART(WEEKDAY, DATEVALUE) = 2

--SELECT * FROM #T1

SELECT T.DATEVALUE
	, DATEDIFF(DD, DATEADD(DD, PERIOD_DAYS, INVOICE_DATE), T.DATEVALUE) AS CATEGORY
	, CASE
		WHEN DATEDIFF(DD, DATEADD(DD, PERIOD_DAYS, INVOICE_DATE), T.DATEVALUE) <= 0 THEN TOTAL_AMOUNT
		ELSE 0
	END AS CURRENT_AMOUNT
	, CASE
		WHEN DATEDIFF(DD, DATEADD(DD, PERIOD_DAYS, INVOICE_DATE),T.DATEVALUE) BETWEEN 1 AND 30 THEN TOTAL_AMOUNT
		ELSE 0
	END AS [30 DAYS PAST DUE]
	, CASE
		WHEN DATEDIFF(DD, DATEADD(DD, PERIOD_DAYS, INVOICE_DATE), T.DATEVALUE) BETWEEN 31 AND 60 THEN TOTAL_AMOUNT
		ELSE 0
	END AS [60 DAYS PAST DUE]
	, CASE
		WHEN DATEDIFF(DD, DATEADD(DD, PERIOD_DAYS, INVOICE_DATE), T.DATEVALUE) BETWEEN 61 AND 90 THEN TOTAL_AMOUNT
		ELSE 0
	END AS [90 DAYS PAST DUE]
	, CASE
		WHEN DATEDIFF(DD, DATEADD(DD, PERIOD_DAYS, INVOICE_DATE), T.DATEVALUE) > 90 THEN TOTAL_AMOUNT
		ELSE 0
	END AS [120 DAYS PAST DUE]
	, *
FROM (SELECT VENDOR_ID
	, V.[NAME] AS VENDOR_NAME
	, P.INVOICE_ID
	, P.INVOICE_DATE
	, P.PAYMENT_DATE
	, P.CREATE_DATE
	, CASE
		WHEN P.TERMS_DESCRIPTION = 'Due on Receipt' THEN 0
		ELSE P.TERMS_NET_DAYS
	END AS PERIOD_DAYS
	, P.TERMS_NET_DAYS
	, P.TERMS_DESCRIPTION
	, P.TOTAL_AMOUNT
	, P.PAID_AMOUNT
	, P.TOTAL_AMOUNT - P.PAID_AMOUNT AS AMOUNT_OUTSTANDING
FROM PAYABLE P
INNER JOIN VENDOR V
	ON P.VENDOR_ID = V.ID
WHERE P.PAY_STATUS IN ('H', 'P')
	AND P.INVOICE_DATE >= '1/1/2019'
	AND P.INVOICE_DATE <= DATEADD(DAY, 1, EOMONTH(GETDATE(), 0))
) X
INNER JOIN #T1 T
ON T.DATEVALUE >= X.CREATE_DATE
AND T.DATEVALUE <= X.PAYMENT_DATE

ORDER BY INVOICE_DATE, INVOICE_ID, T.DATEVALUE


```
