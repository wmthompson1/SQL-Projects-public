# Scratchpad

```sql
SELECT * FROM CUSTOMER WHERE ID LIKE 'A&%'

SELECT * FROM WORK_ORDER WHERE CLOSE_DATE > '4/1/2021'

SELECT * 
FROM CUSTOMER_ORDER CO
LEFT OUTER JOIN DEMAND_SUPPLY_LINK DSL
ON CO.ID = DSL.DEMAND_BASE_ID
AND DSL.[DEMAND_TYPE] = 'CO'
WHERE CUSTOMER_ID = 'A&GMAC'
AND DSL.DEMAND_BASE_ID IS NULL
AND STATUS != 'X'
```
