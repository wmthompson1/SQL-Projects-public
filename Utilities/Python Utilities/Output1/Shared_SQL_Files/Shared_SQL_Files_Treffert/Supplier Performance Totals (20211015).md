# Supplier Performance Totals (20211015)

```sql
USE DataWarehouse
GO

--WHERE POL.PURC_ORDER_ID = '144553'

; WITH VENDORMONTHS AS
(SELECT A.VENDOR_ID
	, A.VENDOR_NAME
	, A.ACTIVE_FLAG
	, MTHS.YEARMTH
FROM (SELECT ID AS VENDOR_ID 
		, [NAME] AS VENDOR_NAME
		, ACTIVE_FLAG
	FROM VENDOR 
	GROUP BY ID
		, [NAME] 
		, ACTIVE_FLAG) A
CROSS APPLY LIVESupplemental.dbo.sfnMonths(CAST(DATEADD(MONTH, 0, DATEADD(YEAR, -1, DATEADD(DD, -DAY(GETDATE())+1, GETDATE()))) AS DATE), CAST(DATEADD(DD, -DAY(GETDATE()), GETDATE()) AS DATE)) MTHS
)
--SELECT * FROM VENDORMONTHS

, CTE_PO AS
(SELECT PO.VENDOR_ID
	, POL.PURC_ORDER_ID
	, POL.LINE_NO AS POL_LINE
	, RL.RECEIVER_ID
	, RL.LINE_NO AS RCV_LINE
	, ISNULL(COALESCE(POL.PART_ID, POL.USER_1, POL.VENDOR_PART_ID), '') AS PART_ID
	, PO.[STATUS] AS PO_STATUS
	, ISNULL(POL.DESIRED_RECV_DATE, PO.DESIRED_RECV_DATE) AS DESIRED_RECV_DATE
	, R.RECEIVED_DATE
	, RL.RECEIVED_QTY
FROM RECEIVER R
LEFT OUTER JOIN RECEIVER_LINE RL
	ON R.ID = RL.RECEIVER_ID
LEFT OUTER JOIN PURC_ORDER_LINE POL
	ON RL.PURC_ORDER_ID = POL.PURC_ORDER_ID
	AND RL.PURC_ORDER_LINE_NO = POL.LINE_NO
LEFT OUTER JOIN PURCHASE_ORDER PO
	ON PO.ID = POL.PURC_ORDER_ID
WHERE R.RECEIVED_DATE >= CAST(DATEADD(MONTH, 0, DATEADD(YEAR, -1, DATEADD(DD, -DAY(GETDATE())+1, GETDATE()))) AS DATE)
	AND R.RECEIVED_DATE <= CAST(DATEADD(DD, -DAY(GETDATE()), GETDATE()) AS DATE)
)
--SELECT * FROM CTE_PO WHERE PURC_ORDER_ID = '117306'

, CTE_RECEIVED AS
(SELECT VENDOR_ID
	, PURC_ORDER_ID
	, POL_LINE
	, PART_ID
	, MAX(DESIRED_RECV_DATE) AS DESIRED_RECV_DATE
	, MAX(RECEIVED_DATE) AS RECEIVED_DATE
	, SUM(RECEIVED_QTY) AS RECEIVED_QTY
FROM CTE_PO
GROUP BY VENDOR_ID
	, PURC_ORDER_ID
	, POL_LINE
	, PART_ID
)
--SELECT * FROM CTE_RECEIVED

, CTE_ONTIME AS
(SELECT VENDOR_ID
	, PURC_ORDER_ID
	, POL_LINE
	, PART_ID
	, DESIRED_RECV_DATE
	, RECEIVED_DATE
	, RECEIVED_QTY
	, CONVERT(DATE, DESIRED_RECV_DATE + 5, 101) AS DUE_DATE
	, CASE 
		WHEN RECEIVED_DATE <= DESIRED_RECV_DATE + 5 THEN 1
		ELSE 0
	END AS ONTIME
	, CASE 
		WHEN RECEIVED_DATE <= DESIRED_RECV_DATE + 5 THEN 0
		ELSE 1
	END AS LATE
	, 1 AS LINE_TOTALS
FROM CTE_RECEIVED
)
--SELECT * FROM CTE_ONTIME WHERE PURC_ORDER_ID = '144553' ORDER BY PURC_ORDER_ID, POL_LINE

, CTE_POS AS
(SELECT PO.ID AS PURC_ORDER_ID
	, POL.PART_ID
	, PO.VENDOR_ID
	, MIN(POL.LINE_NO) AS POL_LINE
FROM PURC_ORDER_LINE POL
INNER JOIN PURCHASE_ORDER PO
	ON PO.ID = POL.PURC_ORDER_ID
WHERE PO.CREATE_DATE >= '1/1/2019'
GROUP BY PO.ID
	, PART_ID
	, PO.VENDOR_ID
)

, CTE_NCM AS
(SELECT CASE 
		WHEN CHARINDEX('/', NCM.PO_NUMBER,1) = 0 THEN NCM.PO_NUMBER
		ELSE SUBSTRING(NCM.PO_NUMBER, 1, CHARINDEX('/', NCM.PO_NUMBER, 1) -1)
	END AS PURC_ORDER_ID
	, PO.POL_LINE
	, ISNULL(VP.PRODUCT_ID, '') AS PART_ID
	, PO.VENDOR_ID
	, NCM.NCM_ID + ' ('+ NCM.PO_NUMBER + ') ' + CASE 
		WHEN VPO.ALT_ID_2 IS NOT NULL THEN 'WO CONTRACTOR'
		ELSE 'PURCHASE'
	END AS NCM_INFO
	, NCM.NCM_DATE
	, ISNULL(NCM.REJECTED, 0) AS REJECTED_QTY
FROM VQ_NONCONFORMANCE NCM
LEFT OUTER JOIN VQ_PRODUCT VP
	ON VP.RECORD_ID = NCM.PRODUCT_REC_ID
LEFT OUTER JOIN VQ_PRODUCTION_ORDER VPO
	ON VPO.RECORD_ID = NCM.PROD_ORDER_REC_ID
LEFT OUTER JOIN CTE_POS PO
	ON PO.PURC_ORDER_ID = CASE 
		WHEN CHARINDEX('/', NCM.PO_NUMBER,1) = 0 THEN NCM.PO_NUMBER
		ELSE SUBSTRING(NCM.PO_NUMBER, 1, CHARINDEX('/', NCM.PO_NUMBER, 1) -1)
	END
	AND PO.PART_ID = VP.PRODUCT_ID
WHERE NCM.NCM_TYPE = 'SUPPLIER - MAJOR'
	AND PO.PURC_ORDER_ID IS NOT NULL
	AND NCM.PO_NUMBER IS NOT NULL
)
--SELECT * FROM CTE_NCM

, CTE_MERGE AS
(SELECT VENDOR_ID
	, CONVERT(DATE, DESIRED_RECV_DATE + 5, 101) AS DUE_DATE
	, CONVERT(NVARCHAR,LEFT(CONVERT(VARCHAR, DESIRED_RECV_DATE + 5, 112), 6)) AS YYYYMM
	, PURC_ORDER_ID
	, POL_LINE
	, PART_ID
	, DESIRED_RECV_DATE
	, RECEIVED_DATE
	, RECEIVED_QTY
	, NULL AS NCM_INFO
	, NULL AS NCM_DATE
	, 0 AS REJECTED_QTY
	, SUM(ONTIME) AS ONTIME
	, SUM(LATE) AS LATE
	, SUM(LINE_TOTALS) AS LINE_TOTALS
FROM CTE_ONTIME
GROUP BY VENDOR_ID
	, CONVERT(DATE, DESIRED_RECV_DATE + 5, 101)
	, CONVERT(NVARCHAR,LEFT(CONVERT(VARCHAR, DESIRED_RECV_DATE + 5, 112), 6))
	, PURC_ORDER_ID
	, POL_LINE
	, PART_ID
	, DESIRED_RECV_DATE
	, RECEIVED_DATE
	, RECEIVED_QTY
UNION 
SELECT VENDOR_ID
	, CONVERT(DATE, NCM_DATE, 101) AS DUE_DATE
	, CONVERT(NVARCHAR,LEFT(CONVERT(VARCHAR, NCM_DATE, 112), 6)) AS YYYYMM
	, PURC_ORDER_ID
	, POL_LINE
	, PART_ID
	, NULL AS DESIRED_RECV_DATE
	, NULL AS RECEIVED_DATE
	, 0 AS RECEIVED_QTY
	, NCM_INFO
	, NCM_DATE
	, SUM(REJECTED_QTY) AS REJECTED_QTY
	, 0 AS ONTIME
	, 0 AS LATE
	, 0 AS LINE_TOTALS
FROM CTE_NCM
WHERE PURC_ORDER_ID IS NOT NULL
GROUP BY VENDOR_ID
	, CONVERT(DATE, NCM_DATE, 101)
	, CONVERT(NVARCHAR,LEFT(CONVERT(VARCHAR, NCM_DATE, 112), 6))
	, PURC_ORDER_ID
	, POL_LINE
	, PART_ID
	, NCM_INFO
	, NCM_DATE
)

, CTE_ANNUAL AS
(SELECT VENDOR_ID
	, YYYYMM
	, SUM(REJECTED_QTY) AS REJECTED_QTY
	, SUM(RECEIVED_QTY) AS RECEIVED_QTY
	, SUM(ONTIME) AS ONTIME
	, SUM(LATE) AS LATE
	, SUM(LINE_TOTALS) AS LINE_TOTALS
FROM CTE_MERGE 
WHERE YYYYMM >= CONVERT(NVARCHAR,LEFT(CONVERT(VARCHAR, GETDATE() - 365, 112), 6))
GROUP BY VENDOR_ID
	, YYYYMM
)
--SELECT * FROM CTE_ANNUAL

, CTE_RESULTS AS
(SELECT V.VENDOR_ID
	, V.VENDOR_NAME
	, V.ACTIVE_FLAG
	, V.YEARMTH AS YRMONTH
	, ISNULL(SUM(T.ONTIME), 0) AS ONTIME_QTY
	, ISNULL(SUM(T.LATE), 0) AS LATE_QTY
	, ISNULL(SUM(T.LINE_TOTALS), 0) AS TOTAL_QTY
	, CAST(SUM(ISNULL(T.REJECTED_QTY,0)) AS NUMERIC) REJECTED_QTY
	, CAST(SUM(ISNULL(T.RECEIVED_QTY,0)) AS NUMERIC) RECEIVED_QTY
	, CAST(SUM(ISNULL(T.RECEIVED_QTY,0)) - SUM(ISNULL(T.REJECTED_QTY,0)) AS NUMERIC) ACCEPTED_QTY
	, SUM(ISNULL(SUM(T.ONTIME), 0)) OVER (PARTITION BY V.VENDOR_ID ORDER BY V.YEARMTH) AS ONTIME_RUNTOTAL
	, SUM(ISNULL(SUM(T.LATE), 0)) OVER (PARTITION BY V.VENDOR_ID ORDER BY V.YEARMTH) AS LATE_RUNTOTAL
	, SUM(ISNULL(SUM(T.LINE_TOTALS), 0)) OVER (PARTITION BY V.VENDOR_ID ORDER BY V.YEARMTH) AS TOTAL_RUNTOTAL
	, SUM(CAST(SUM(ISNULL(T.REJECTED_QTY,0)) AS NUMERIC)) OVER (PARTITION BY V.VENDOR_ID ORDER BY V.YEARMTH) AS REJECTED_RUNTOTAL
	, SUM(CAST(SUM(ISNULL(T.RECEIVED_QTY,0)) AS NUMERIC)) OVER (PARTITION BY V.VENDOR_ID ORDER BY V.YEARMTH) AS RECEIVED_RUNTOTAL
	, SUM(CAST(SUM(ISNULL(T.RECEIVED_QTY,0)) - SUM(ISNULL(T.REJECTED_QTY,0)) AS NUMERIC)) OVER (PARTITION BY V.VENDOR_ID ORDER BY V.YEARMTH) AS ACCEPTED_RUNTOTAL
FROM VENDORMONTHS V
LEFT JOIN CTE_ANNUAL T
	ON V.VENDOR_ID = T.VENDOR_ID 
	AND V.YEARMTH = T.YYYYMM
GROUP BY V.VENDOR_ID
	, V.VENDOR_NAME
	, V.ACTIVE_FLAG
	, V.YEARMTH
)
--SELECT * FROM FINALRESULTS WHERE VENDOR_ID = 'KLXAER'

, CTE_TOTALS AS
(SELECT F.VENDOR_ID
	, F.VENDOR_NAME
	, F.ACTIVE_FLAG
	, F.YRMONTH
	, F.ONTIME_QTY
	, F.LATE_QTY
	, F.TOTAL_QTY
	, F.REJECTED_QTY
	, F.RECEIVED_QTY
	, F.ACCEPTED_QTY
	, F.ONTIME_RUNTOTAL
	, F.LATE_RUNTOTAL
	, F.TOTAL_RUNTOTAL
	, F.REJECTED_RUNTOTAL
	, F.RECEIVED_RUNTOTAL
	, F.ACCEPTED_RUNTOTAL
	, CAST(CASE
		WHEN ISNULL(F.TOTAL_QTY, 0) != 0 THEN CONVERT(NVARCHAR(10), F.ONTIME_QTY) + N' / ' + CONVERT(NVARCHAR(10), F.TOTAL_QTY)
		ELSE N'0 / 0'
	END AS NVARCHAR(20)) AS ONTIME_DISPLAY
	, CAST(CASE
		WHEN ISNULL(F.RECEIVED_QTY, 0) != 0 THEN CONVERT(NVARCHAR(10), F.ACCEPTED_QTY) + N' / ' + CONVERT(NVARCHAR(10), F.RECEIVED_QTY)
		ELSE N'0 / 0'
	END AS NVARCHAR(20)) AS ACCEPTANCE_DISPLAY

	, CAST(CASE
		WHEN ISNULL(F.TOTAL_RUNTOTAL, 0) != 0 THEN CONVERT(NVARCHAR(10), F.ONTIME_RUNTOTAL) + N' / ' + CONVERT(NVARCHAR(10), F.TOTAL_RUNTOTAL)
		ELSE N'0 / 0'
	END AS NVARCHAR(20)) AS RUNNING_ONTIME_DISPLAY
	, CAST(CASE
		WHEN ISNULL(F.RECEIVED_RUNTOTAL, 0) != 0 THEN CONVERT(NVARCHAR(10), F.ACCEPTED_RUNTOTAL) + N' / ' + CONVERT(NVARCHAR(10), F.RECEIVED_RUNTOTAL)
		ELSE N'0 / 0'
	END AS NVARCHAR(20)) AS RUNNING_ACCEPTANCE_DISPLAY

	, CAST(CASE
		WHEN ISNULL(F.TOTAL_QTY, 0) != 0 THEN CONVERT(FLOAT, F.ONTIME_QTY) / CONVERT(FLOAT, F.TOTAL_QTY)
		ELSE 0
	END AS NUMERIC (5, 2)) AS ONTIME_PCT
	, CAST(CASE
		WHEN ISNULL(F.RECEIVED_QTY, 0) != 0 THEN CONVERT(FLOAT, F.ACCEPTED_QTY) / CONVERT(FLOAT, F.RECEIVED_QTY)
		ELSE 0
	END AS NUMERIC (5, 2)) AS ACCEPTANCE_PCT

	, CAST(CASE
		WHEN ISNULL(F.TOTAL_RUNTOTAL, 0) != 0 THEN CONVERT(FLOAT, F.ONTIME_RUNTOTAL) / CONVERT(FLOAT, F.TOTAL_RUNTOTAL)
		ELSE 0
	END AS NUMERIC (5, 2)) AS RUNNING_ONTIME_PCT
	, CAST(CASE
		WHEN ISNULL(F.RECEIVED_RUNTOTAL, 0) != 0 THEN CONVERT(FLOAT, F.ACCEPTED_RUNTOTAL) / CONVERT(FLOAT, F.RECEIVED_RUNTOTAL)
		ELSE 0
	END AS NUMERIC (5, 2)) AS RUNNING_ACCEPTANCE_PCT
	, GETDATE() AS PRINTDATE
FROM CTE_RESULTS F
)
--SELECT * FROM CTE_TOTALS T

SELECT *
FROM (SELECT VENDOR_ID
	, YRMONTH
	, COL
	, [VALUE]
FROM CTE_TOTALS T
CROSS APPLY (SELECT 'ONTIME_DISPLAY', CAST(ONTIME_DISPLAY AS NVARCHAR(20))
	UNION ALL
	SELECT 'ACCEPTANCE_DISPLAY', CAST(ACCEPTANCE_DISPLAY AS NVARCHAR(20))
	) X (COL, [VALUE])
) AS PVT_DATA
PIVOT (MAX([VALUE]) 
	FOR YRMONTH IN ([202010], [202011], [202012], [202101], [202102], [202103], [202104], [202105], [202106], [202107], [202108], [202109])
) AS TBL_PIVOT1
--PIVOT (MAX([VALUE]) 
--	FOR COL IN ([ONTIME_DISPLAY], [ACCEPTANCE_DISPLAY])
--) AS TBL_PIVOT2
ORDER BY VENDOR_ID

;
```
