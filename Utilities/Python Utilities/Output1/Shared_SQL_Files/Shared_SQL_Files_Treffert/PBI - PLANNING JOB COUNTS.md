# PBI - PLANNING JOB COUNTS

```sql
/*   PLANNING JOB COUNTS
want to expand to include FIN in addition to MFG, and all customers, not just BOE609
One potential issue though. The way this report is currently coded is to check for specific ResourceIDs for the 
last 2 counts (planned and released), without any checking for the Customer ID, the Planner User ID or the 
Buyer User ID. I am guessing the assumption is the Resource IDs are properly coded for all workorders.  
Some day I would like to potentially change that to be more definitive.

*/

; WITH CTE_CO_BOE609 AS 
--> TOTAL NUMBER OF JOBS CREATED (RECEIVED)
(SELECT CO.ID AS CUST_ORDER_ID
	, CONVERT(DATE, CO.CREATE_DATE) AS CREATE_DATE
	, CO.CUSTOMER_ID
	, P.PLANNER_USER_ID 
FROM CUST_ORDER_LINE COL
INNER JOIN CUSTOMER_ORDER CO
	ON CO.ID = COL.CUST_ORDER_ID
INNER JOIN PART P
	ON P.ID = COL.PART_ID
WHERE CO.[STATUS] NOT IN ('C', 'X')
	AND COL.PART_ID IS NOT NULL
)

, CTE_P2_REVIEW_CLOSED AS 
--> TOTAL NUMBER OF CMP JOBS PLANNED (REVIEW OP CLOSE DATE) AND RESOURCE_ID IN ('P2M1-REVIEW-CMP', 'P2M2-REVIEWCM','P2M1-REVIEWCM')
(SELECT WO.BASE_ID
	, CONVERT(DATE, O.CLOSE_DATE) AS CLOSE_DATE
	, CO.CUSTOMER_ID
	, P.PLANNER_USER_ID 
FROM WORK_ORDER WO
INNER JOIN OPERATION O
	ON WO.TYPE = O.WORKORDER_TYPE
	AND WO.BASE_ID = O.WORKORDER_BASE_ID
	AND WO.LOT_ID = O.WORKORDER_LOT_ID
	AND WO.SPLIT_ID = O.WORKORDER_SPLIT_ID
	AND WO.SUB_ID = O.WORKORDER_SUB_ID
INNER JOIN DEMAND_SUPPLY_LINK DSL
	ON WO.BASE_ID = DSL.SUPPLY_BASE_ID
	AND WO.LOT_ID = DSL.SUPPLY_LOT_ID
INNER JOIN CUST_ORDER_LINE COL
	ON COL.CUST_ORDER_ID = DSL.DEMAND_BASE_ID
	AND COL.LINE_NO = DSL.DEMAND_SEQ_NO
INNER JOIN CUSTOMER_ORDER CO
	ON CO.ID = COL.CUST_ORDER_ID
INNER JOIN PART P
	ON P.ID = COL.PART_ID
WHERE WO.[STATUS] NOT IN ('C', 'X')
	AND WO.[TYPE] = 'W'
	AND O.RESOURCE_ID IN ('P2M1-REVIEW-CMP', 'P2M2-REVIEWCM', 'P2M1-REVIEWCM')
	AND O.STATUS = 'C'
)

, CTE_P2_STAGING_CLOSED AS 
--> TOTAL NUMBER OF CMP JOBS THROUGH STAGING (STAGING OPS COMPLETED)
(SELECT WO.BASE_ID
	, CONVERT(DATE, O.CLOSE_DATE) AS CLOSE_DATE
	, CO.CUSTOMER_ID	
	, P.PLANNER_USER_ID 
FROM WORK_ORDER WO
INNER JOIN OPERATION O
	ON WO.TYPE = O.WORKORDER_TYPE
	AND WO.BASE_ID = O.WORKORDER_BASE_ID
	AND WO.LOT_ID = O.WORKORDER_LOT_ID
	AND WO.SPLIT_ID = O.WORKORDER_SPLIT_ID
	AND WO.SUB_ID = O.WORKORDER_SUB_ID
INNER JOIN DEMAND_SUPPLY_LINK DSL
	ON WO.BASE_ID = DSL.SUPPLY_BASE_ID
	AND WO.LOT_ID = DSL.SUPPLY_LOT_ID
INNER JOIN CUST_ORDER_LINE COL
	ON COL.CUST_ORDER_ID = DSL.DEMAND_BASE_ID
	AND COL.LINE_NO = DSL.DEMAND_SEQ_NO
INNER JOIN CUSTOMER_ORDER CO
	ON CO.ID = COL.CUST_ORDER_ID
INNER JOIN PART P
	ON P.ID = COL.PART_ID
WHERE WO.STATUS NOT IN('C', 'X')
	AND WO.TYPE = 'W'
	AND RESOURCE_ID IN ('P2M1-STAGNG-CMP', 'P2M2-RLSMTL-CMP')
	AND O.STATUS = 'C'
)

SELECT 1 AS SORT
	, 'TOTAL JOBS RECEIVED' AS TITLE
	, CUSTOMER_ID	
	, PLANNER_USER_ID
	, COUNT(CUST_ORDER_ID) AS CNT
	, CONVERT(DATE, CO.CREATE_DATE) AS [DATE]
FROM CTE_CO_BOE609 CO
WHERE CONVERT(DATE, CO.CREATE_DATE) > GETDATE()-8
GROUP BY CREATE_DATE
	, CUSTOMER_ID	
	, PLANNER_USER_ID
UNION ALL
SELECT 2 AS SORT
	, 'TOTAL JOBS PLANNED' AS TITLE
	, CUSTOMER_ID	
	, PLANNER_USER_ID
	, COUNT(BASE_ID) AS CNT
	, CONVERT(DATE, WO.CLOSE_DATE) AS [DATE]
FROM CTE_P2_REVIEW_CLOSED WO
WHERE CONVERT(DATE, WO.CLOSE_DATE) > GETDATE()-8
GROUP BY CLOSE_DATE
	, CUSTOMER_ID	
	, PLANNER_USER_ID
UNION ALL
SELECT 3 AS SORT
	, 'TOTAL JOBS RELEASED' AS TITLE
	, CUSTOMER_ID	
	, PLANNER_USER_ID
	, COUNT(BASE_ID) AS CNT
	, CONVERT(DATE, WO.CLOSE_DATE) AS [DATE]
FROM CTE_P2_STAGING_CLOSED WO
WHERE CONVERT(DATE, WO.CLOSE_DATE) > GETDATE()-8
GROUP BY CLOSE_DATE
	, CUSTOMER_ID	
	, PLANNER_USER_ID

```
