; WITH CTE_ALL AS
--> PULL ALL OPERATIONS FOR THE MASTERS
(SELECT O.WORKORDER_TYPE
	, O.WORKORDER_BASE_ID
	, O.WORKORDER_LOT_ID
	, O.WORKORDER_SPLIT_ID
	, O.WORKORDER_SUB_ID
	, O.SEQUENCE_NO

	AND O.WORKORDER_LOT_ID = U.LOT_ID
	AND O.WORKORDER_SPLIT_ID = U.SPLIT_ID
	AND O.WORKORDER_SUB_ID = U.SUB_ID
GROUP BY O.WORKORDER_TYPE
	, O.WORKORDER_BASE_ID
	, O.WORKORDER_LOT_ID
	, O.WORKORDER_SPLIT_ID
	, O.WORKORDER_SUB_ID
	, O.SEQUENCE_NO
	, O.RESOURCE_ID
	, O.OPERATION_TYPE

, CTE_NEXT AS
--> FIND THE NEXT OPERATION SEQUENCE_NO FOR EACH OPERATION SEQUENCE_NO

	, WORKORDER_BASE_ID
	, WORKORDER_LOT_ID
	, WORKORDER_SPLIT_ID
	, WORKORDER_SUB_ID
	, SEQUENCE_NO

FROM CTE_ALL
)

	, A.OPERATION_TYPE
	, A.NEXT_OP
	, O.RESOURCE_ID
	, O.OPERATION_TYPE
FROM CTE_NEXT A

	AND A.WORKORDER_BASE_ID = U.BASE_ID

	AND A.WORKORDER_SPLIT_ID = O.WORKORDER_SPLIT_ID
	AND A.WORKORDER_SUB_ID = O.WORKORDER_SUB_ID
	AND A.NEXT_OP = O.SEQUENCE_NO
