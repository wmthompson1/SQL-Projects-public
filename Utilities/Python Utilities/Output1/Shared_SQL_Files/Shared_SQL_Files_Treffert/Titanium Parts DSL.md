/*
DROP TABLE #TITANIUM
DROP TABLE #ALL_TITANIUM 
DROP TABLE #ALL_TITANIUM_PARTS
DROP TABLE #ALL_TITANIUM_PARTS_DISTINCT
*/

SELECT 
	P.ID as part_ID
INTO	#TITANIUM
FROM	PART AS P 
INNER JOIN 
	PART_SITE AS ps 
	ON P.ID = ps.PART_ID 
INNER JOIN 
	USER_DEF_FIELDS AS UDF 
	ON P.ID = UDF.DOCUMENT_ID
WHERE 
	UDF.ID = 'UDF-0000035' 
	AND UDF.STRING_VAL LIKE '%titanium%'
	AND P.STATUS IS NULL

SELECT	t.part_ID
INTO	#ALL_TITANIUM 
FROM	#TITANIUM t
UNION ALL
 --Parent requirements Masters only - parent = titanium
SELECT	R.PART_ID
FROM	#TITANIUM T
INNER JOIN
	REQUIREMENT R
	ON T.part_id = R.WORKORDER_BASE_ID
INNER JOIN 
	PART P 
	ON R.PART_ID=P.ID
INNER JOIN
	PART P_1 
	ON R.WORKORDER_BASE_ID = P_1.ID
WHERE	
	P.STATUS IS  NULL  
	AND P_1.STATUS IS  NULL 
UNION ALL
SELECT	R.WORKORDER_BASE_ID
FROM	#TITANIUM T
INNER JOIN 
	REQUIREMENT R 
	ON T.part_id = R.PART_ID
INNER JOIN 
	PART P 
	ON R.WORKORDER_BASE_ID=P.ID
WHERE	
	R.WORKORDER_TYPE='M' 
	AND P.STATUS IS  NULL 
UNION ALL
--work order part is titanium; child is anything; includes masters and workorders
SELECT	t.PART_ID
FROM	#TITANIUM T
INNER JOIN	
	WORK_ORDER WO
	ON T.part_id = WO.PART_ID
INNER JOIN 
	REQUIREMENT R 
	ON WO.TYPE=R.WORKORDER_TYPE
	AND WO.BASE_ID=R.WORKORDER_BASE_ID
	AND WO.LOT_ID=R.WORKORDER_LOT_ID
	AND WO.SPLIT_ID=R.WORKORDER_SPLIT_ID 
	AND WO.SUB_ID=R.WORKORDER_SUB_ID
INNER JOIN
	PART P_1 
	ON WO.PART_ID=P_1.ID
INNER JOIN
	PART P 
	ON R.PART_ID=P.ID
WHERE
	P.STATUS IS  NULL  
	AND P_1.STATUS IS  NULL 
UNION ALL
--child is titanium and parent is anything
SELECT	t.PART_ID
FROM	#TITANIUM T
INNER JOIN
	WORK_ORDER WO
	ON T.part_id = WO.PART_ID
INNER JOIN 
	REQUIREMENT R 
	ON WO.TYPE=R.WORKORDER_TYPE 
	AND WO.BASE_ID=R.WORKORDER_BASE_ID 
	AND WO.LOT_ID=R.WORKORDER_LOT_ID 
	AND WO.SPLIT_ID=R.WORKORDER_SPLIT_ID 
	AND WO.SUB_ID=R.WORKORDER_SUB_ID
INNER JOIN
	PART P 
	ON R.PART_ID=P.ID
WHERE	P.STATUS IS  NULL 
UNION ALL
SELECT	PMR.REQUIRED_PART_ID
FROM	#TITANIUM T
INNER JOIN
	PLANNED_MATL_REQ PMR 
	ON T.part_id = PMR.PARENT_PART_ID
INNER JOIN
	PART P 
	ON PMR.REQUIRED_PART_ID=P.ID
WHERE	P.STATUS IS  NULL 
UNION ALL
SELECT	PMR.PARENT_PART_ID
FROM	#TITANIUM T
INNER JOIN
	PLANNED_MATL_REQ PMR 
	ON T.part_id = PMR.REQUIRED_PART_ID
INNER JOIN
	PART P 
	ON PMR.PARENT_PART_ID=P.ID
WHERE	P.STATUS IS  NULL 

--BOM.rpt
SELECT DISTINCT R.PART_ID
INTO	#ALL_TITANIUM_PARTS
FROM	#ALL_TITANIUM T
INNER JOIN 
	REQUIREMENT R 
	ON T.part_id = R.WORKORDER_BASE_ID
INNER JOIN 
	PART P 
	ON R.WORKORDER_BASE_ID=P.ID
	AND P.STATUS IS  NULL 
INNER JOIN
	PART P_1 
	ON R.PART_ID=P_1.ID
	AND P_1.STATUS IS  NULL 
UNION ALL
SELECT DISTINCT T.PART_ID
FROM	#ALL_TITANIUM T

SELECT DISTINCT part_id 
INTO #ALL_TITANIUM_PARTS_DISTINCT
FROM #ALL_TITANIUM_PARTS
WHERE
	PART_ID NOT LIKE '%REWORK%'
	AND PART_ID NOT LIKE '%TOOLING%'

select * from #ALL_TITANIUM_PARTS_DISTINCT where part_id = '6al4v-032'
